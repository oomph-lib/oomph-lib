61  /// \short Initialise storage for source elements and their
62  /// associated local coordinates for the specified number of interactions
63  //========================================================================
64  void ElementWithExternalElement::initialise_external_element_storage()
65  {
66  //Find the number of interactions
67  const unsigned n_interaction = this->Ninteraction;
68  //Find the number of integration points
69  const unsigned n_intpt = this->integral_pt()->nweight();
70 
71  //Work out the required storage
72  const unsigned n_external_element_storage = n_interaction*n_intpt;
73 
74  //If we have not allocated the correct amount of memory,
75  //then do so
76  if(Nexternal_element_storage != n_external_element_storage)
77  {
78  //Allocate storage for the pointers to the elements
79  //Delete old storage, if needed
80  if(External_element_pt) {delete[] External_element_pt;}
81  //Allocate new memory
82  External_element_pt = new FiniteElement*[n_external_element_storage];
83 
84  //Initialise all new pointers to zero
85  for(unsigned i=0;i<n_external_element_storage;i++)
86  {External_element_pt[i] = 0;}
87 
88  //Alloacte storage for local coordinates
89  //Delete old storage, if needed
90  if(External_element_local_coord) {delete[] External_element_local_coord;}
91  //Allocate the new memory
92  External_element_local_coord =
93  new Vector<double>[n_external_element_storage];
94 
95  //Finally record how much memory we have allocated
96  Nexternal_element_storage = n_external_element_storage;
97  Nintpt = n_intpt;
98  }
99  }

212 /// This function determines the all Data in external elemetns that
213 /// affects the residuals of the element
214 /// and adds their global equation numbers to the
215 /// local-to-global look-up scheme. Note that we only include
216 /// Data items into the element's External_interaction_data
217 /// if they are not already

221 void ElementWithExternalElement::
222 assign_external_interaction_data_local_eqn_numbers(
223  const bool &store_local_dof_pt)
224 {
225  //Reset number of stored field data to zero
226  Nexternal_interaction_field_data=0;
227  //Clear all the internal field data storage, if it's been allocated
228  if(External_interaction_field_data_pt)
229  {
230  delete[] External_interaction_field_data_pt;
231  External_interaction_field_data_pt=0;
232  }
233  if(External_interaction_field_data_index)
234  {
235  delete[] External_interaction_field_data_index;
236  External_interaction_field_data_index=0;
237  }
238  if(External_interaction_field_data_local_eqn)
239  {
240  delete[] External_interaction_field_data_local_eqn;
241  External_interaction_field_data_local_eqn=0;
242  }
243 
244  //Reset number of stored geometric data to zero
245  Nexternal_interaction_geometric_data=0;
246  //Clear all internal external data storage, if it's been allocated
247  if(External_interaction_geometric_data_pt)
248  {
249  delete[] External_interaction_geometric_data_pt;
250  External_interaction_geometric_data_pt=0;
251  }
252  if(External_interaction_geometric_data_index)
253  {
254  delete[] External_interaction_geometric_data_index;
255  External_interaction_geometric_data_index=0;
256  }
257  if(External_interaction_geometric_data_local_eqn)
258  {
259  delete[] External_interaction_geometric_data_local_eqn;
260  External_interaction_geometric_data_local_eqn=0;
261  }
262 
263  //Only bother with non-halo elements
264 #ifdef OOMPH_HAS_MPI
265  if(!this->is_halo())
266 #endif
267  {
268  //If desired, determine the Data that affects the interactions but is not
269  //already included in elements other generic Data.
270  //The conditional test is done here, so that the vectors are cleared
271  //and not re-filled if Add_external_interaction_data is false
272  if(Add_external_interaction_data)
273  {
274  //Number of interactions
275  const unsigned n_interaction = this->ninteraction();
276  // Number of integration points
277  const unsigned n_intpt = integral_pt()->nweight();
278 
279  //Sets of all (external) FiniteElements that affect the interactions
280  //One set per interaction
281  Vector<std::set<FiniteElement*> >
282  external_interaction_elements_pt(n_interaction);
283 
284  //Loop over the interactions
285  for (unsigned i=0;i<n_interaction;i++)
286  {
287  //Loop over the integration points and the adjacent element at
288  //each integration point to the set
289  for(unsigned ipt=0;ipt<n_intpt;ipt++)
290  {
291  //Add the element adjacent to the element into the set
292  external_interaction_elements_pt[i].
293  insert(external_element_pt(i,ipt));
294  }
295  //For safety erase any null pointers
296  external_interaction_elements_pt[i].erase(0);
297  }
298 
299  // Storage for a pairs of interaction data (pointer to Data and the index
300  // of the value within this Data object) affecting the element.
301  std::set<std::pair<Data*,unsigned> > paired_field_data;
302 
303  //Determine the field data that affects the external interactions
304  //for all sets of external elements
305  identify_all_field_data_for_external_interaction(
306  external_interaction_elements_pt,paired_field_data);
307 
308  //It's just possible that some of the field data could be internal
309  //or nodal data, so we should remove it if that's the case
310  //Loop over all internal data
311  const unsigned n_internal = this->ninternal_data();
312  for(unsigned n=0;n<n_internal;n++)
313  {
314  //Cache the data pointer
315  Data* const dat_pt = this->internal_data_pt(n);
316  //Find the number of data values stored in the data object
317  const unsigned n_value = dat_pt->nvalue();
318  //Add the index of each data value and the pointer to the set
319  //of pairs
320  for(unsigned i=0;i<n_value;i++)
321  {
322  paired_field_data.erase(std::make_pair(dat_pt,i));
323  }
324  }
325 
326  //Loop over all the nodes
327  const unsigned n_node = this->nnode();
328  for(unsigned n=0;n<n_node;n++)
329  {
330  //Find the node point
331  Node* const nod_pt = this->node_pt(n);
332  //Find the number of values stored at the node
333  const unsigned n_value = nod_pt->nvalue();
334  //Loop over values and erase all pairs from the set
335  for(unsigned i=0;i<n_value;i++)
336  {
337  paired_field_data.erase(std::make_pair(nod_pt,i));
338  }
339  SolidNode* solid_nod_pt=dynamic_cast<SolidNode*>(nod_pt);
340  if (solid_nod_pt!=0)
341  {
342  Data* pos_data_pt=solid_nod_pt->variable_position_pt();
343  //Find the number of positional values stored at the node
344  const unsigned n_value = pos_data_pt->nvalue();
345  //Loop over values and erase all pairs from the set
346  for(unsigned i=0;i<n_value;i++)
347  {
348  paired_field_data.erase(std::make_pair(pos_data_pt,i));
349  }
350  }
351  }
352 
353  //Now allocate storage for the external field data
354  //associated indices and local equation numbers
355  const unsigned n_external_interaction_field_data = paired_field_data.size();
356  Nexternal_interaction_field_data = n_external_interaction_field_data;
357  External_interaction_field_data_pt =
358  new Data*[n_external_interaction_field_data];
359  External_interaction_field_data_index =
360  new unsigned[n_external_interaction_field_data];
361 
362  //Add the pairs of data to the field data vectors
363  {
364  unsigned count=0;
365  for(std::set<std::pair<Data*,unsigned> >::iterator it=
366  paired_field_data.begin();
367  it != paired_field_data.end();it++)
368  {
369  External_interaction_field_data_pt[count] = it->first;
370  External_interaction_field_data_index[count] = it->second;
371  ++count;
372  }
373  }
374 
375 
376  //Only bother to add geometric data if we're told to
377  if(Add_external_geometric_data)
378  {
379  //Storage for a set of external geometric Data affecting the element
380  std::set<Data*> external_geometric_data_pt;
381 
382  //Determine the geometric data that affects the external interactions
383  //for all sets of external elements
384  identify_all_geometric_data_for_external_interaction(
385  external_interaction_elements_pt, external_geometric_data_pt);
386 
387  // Now loop over any geometric data of the Element itself
388  // and erase them from the external_geometric_data_pt
389  // because these data are actually intrinsic
390  // data of this element and are counted and numbered elsewhere
391  unsigned n_geom_data = ngeom_data();
392  for(unsigned j=0;j<n_geom_data;j++)
393  {
394  external_geometric_data_pt.erase(geom_data_pt(j));
395  }
396 
397  //It is possible that the geometric data may have already been added as
398  //external data. We should erase any common entries from the
399  //External_interaction_data
400  //but not touch the external data that has been set up by a
401  //"knowledgeable" user
402  unsigned n_external = nexternal_data();
403  for(unsigned j=0;j<n_external;j++)
404  {
405  external_geometric_data_pt.erase(external_data_pt(j));
406  }
407 
408  //Loop over all the nodes of present element to avoid double counting
409  const unsigned n_node = this->nnode();
410  for(unsigned n=0;n<n_node;n++)
411  {
412  Node* const nod_pt = this->node_pt(n);
413  external_geometric_data_pt.erase(nod_pt);
414 
415  SolidNode* solid_nod_pt=dynamic_cast<SolidNode*>(nod_pt);
416  if (solid_nod_pt!=0)
417  {
418  Data* pos_data_pt=solid_nod_pt->variable_position_pt();
419 // std::ostringstream junk;
420 // junk << "Erasing ";
421 // unsigned nval=pos_data_pt->nvalue();
422 // for (unsigned i=0;i<nval;i++)
423 // {
424 // junk << pos_data_pt->eqn_number(i) << " ";
425 // }
426 // oomph_info << junk.str() << std::endl;
427  external_geometric_data_pt.erase(pos_data_pt);
428  }
429  }
430 
431 
432  //Next allocate storage for the geometric field data
433  //Find out how many individual data we have
434  unsigned n_external_interaction_geometric_data = 0;
435  for(std::set<Data*>::iterator it=external_geometric_data_pt.begin();
436  it != external_geometric_data_pt.end(); it++)
437  {
438  //Add the number of values stored in each geometric datum
439  n_external_interaction_geometric_data += (*it)->nvalue();
440  }
441 
442  //Now allocate storage
443  Nexternal_interaction_geometric_data =
444  n_external_interaction_geometric_data;
445  External_interaction_geometric_data_pt =
446  new Data*[n_external_interaction_geometric_data];
447  External_interaction_geometric_data_index =
448  new unsigned[n_external_interaction_geometric_data];
449 
450  // Now we can add all the geometric data to the geometric data vectors
451  {
452  unsigned count=0;
453  for(std::set<Data*>::iterator it=external_geometric_data_pt.begin();
454  it != external_geometric_data_pt.end(); it++)
455  {
456  //Find the number of values stored in the geometric data
457  unsigned n_value = (*it)->nvalue();
458  //Loop over the values
459  for(unsigned j=0;j<n_value;j++)
460  {
461  //Add data to the external geometric data
462  External_interaction_geometric_data_pt[count] = *it;
463  External_interaction_geometric_data_index[count] = j;
464  ++count;
465  }
466  }
467  }
468  }
469  }
470 
471  //All external interaction data has now been specified
472 
473  //Find the number of external field data
474  const unsigned n_external_field_data = nexternal_interaction_field_data();
475 
476  //If there are interaction data fill in the internal storage
477  if(n_external_field_data > 0)
478  {
479 
480  //Allocate storage for the local equation numbers associated with
481  //external field data
482  External_interaction_field_data_local_eqn =
483  new int[n_external_field_data];
484 
485  //Find the number of local equations assigned so far
486  unsigned local_eqn_number = ndof();
487 
488  //A local queue to store the global equation numbers
489  std::deque<unsigned long> global_eqn_number_queue;
490 
491  //Now loop over the field data again to assign local equation numbers
492  for(unsigned i=0;i<n_external_field_data;i++)
493  {
494  //Get the GLOBAL equation number
495  long eqn_number =
496  External_interaction_field_data_pt[i]->
497  eqn_number(External_interaction_field_data_index[i]);
498 
499  //If the GLOBAL equation number is positive (i.e. not pinned)
500  if(eqn_number >= 0)

502 // std::ostringstream junk;

506 // junk << "not";
507 // }

509

519

529  }
530  }

532  add_global_eqn_numbers(global_eqn_number_queue,
533  GeneralisedElement::Dof_pt_deque);
534  //Clear the memory used in the deque
535  if(store_local_dof_pt)
536  {std::deque<double*>().swap(GeneralisedElement::Dof_pt_deque);}
537  }
538 
539  //Find the number of external geometric data
540  unsigned n_external_geom_data = nexternal_interaction_geometric_data();
541 
542  //If there are external geometric data fill in the internal storage
543  if(n_external_geom_data > 0)
544  {
545 
546  //Allocate storage for the local equation numbers associated with
547  //external geometric data
548  External_interaction_geometric_data_local_eqn =
549  new int[n_external_geom_data];
550 
551  //Find the number of local equations assigned so far

553

600 /// This function calculates the entries of Jacobian matrix, used in
601 /// the Newton method, associated with the external interaction
602 /// degrees of freedom for external fields.
603 /// It does this using finite differences,
604 /// rather than an analytical formulation, so can be done in total generality.
605 //==========================================================================
606  void ElementWithExternalElement::
607  fill_in_jacobian_from_external_interaction_field_by_fd(
608  Vector<double> &residuals, DenseMatrix<double> &jacobian)
609  {
610  //Locally cache the number of data
611  const unsigned n_external_interaction_field_data =
612  nexternal_interaction_field_data();
613 
614  //If there is no such data return
615  if(n_external_interaction_field_data==0) {return;}
616 
617  //Call the update function to ensure that the element is in a
618  //consistent state before finite differencing
619  update_before_external_interaction_field_fd();
620 
621  //Find the number of dofs in the element
622  const unsigned n_dof = ndof();
