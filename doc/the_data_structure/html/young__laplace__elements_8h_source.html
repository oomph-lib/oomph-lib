49 /// A class for all isoparametric elements that solve the
50 /// YoungLaplace equations.
51 /// \f[
52 /// div (\frac{1}{W} \nabla u) = \kappa
53 /// \f]
54 /// with
55 /// \f[
56 /// W^2=1+\|\nabla u\|^2
57 /// \f]
58 /// These equations can either be solved in the above (cartesian)
59 /// form, or in a parametric representation using the method
60 /// of spines. See the theory write-up in the documentation for
61 /// details.
62 /// This class contains the generic maths. Shape functions, geometric
63 /// mapping etc. must get implemented in derived class.
64 //=============================================================

207  /// Get flux: flux[i] = du/dx_i: Mainly used for error estimation.
208  void get_flux(const Vector<double>& s, Vector<double>& flux) const
209  {
210  //Find out how many nodes there are in the element

212 
213  //Set up memory for the shape (same as test functions)
214  Shape psi(n_node);
215  DShape dpsidx(n_node,2);
216 
217  //Call the derivatives of the shape (same as test functions)
218  dshape_eulerian(s,psi,dpsidx);
219 
220  //Initialise to zero
221  for(unsigned j=0;j<2;j++)
222  {
223  flux[j] = 0.0;
224  }
225 
226  // Loop over nodes
227  for(unsigned l=0;l<n_node;l++)
228  {

309  inline double interpolated_u(const Vector<double> &s) const

311  //Find number of nodes

565 /// Height control element for YoungLaplace equations: Prescribe
566 /// displacement along a spine (i.e. the "height of the meniscus"
567 /// in exchange for treating the curvature as an unknown. Very
568 /// similar to the DisplacementControlElement used in solid
569 /// mechanics problems.

578  /// node.

580  double* prescribed_height_pt) : GeneralisedElement()
581  {

584  Prescribed_height_pt=prescribed_height_pt;

592  // the pointer to it...).
593  Curvature_data_index=add_internal_data(new Data(1));

596  add_external_data(static_cast<Data*>(control_node_pt));
