34  /// Check mesh integrity -- performs some internal consistency checks
35  /// and throws error if violated.

37  void TriangleScaffoldMesh::check_mesh_integrity()
38  {
39  // Check that all nodes are attached to elements
40  std::map<Node*, bool> done;
41 
42  //Number of nodes per element
43  unsigned nnod_el = finite_element_pt(0)->nnode();
44 
45  // Loop over elements to visit their nodes
46  unsigned nelem=nelement();
47  for (unsigned e = 0; e < nelem; e++)
48  {
49  // Loop over all nodes in element
50  for (unsigned j = 0; j < nnod_el; j++)
51  {
52  // Pointer to node in the scaffold mesh
53  Node* node_pt = finite_element_pt(e)->node_pt(j);
54 
55  done[node_pt]=true;
56  }
57  }
58 
59  // Now check if all nodes have been visited
60  std::ostringstream error_stream;
61  bool broken=false;
62  unsigned nnod=nnode();
63  for (unsigned j=0;j<nnod;j++)
64  {
65  if (!done[Node_pt[j]])
66  {
67  error_stream
68  << "Scaffold node " << j
69  << " does not seem to be attached to any element";
70  broken=true;
71  }
72  }
73  if (broken)
74  {
75  throw OomphLibError(error_stream.str(),
76  OOMPH_CURRENT_FUNCTION,
77  OOMPH_EXCEPTION_LOCATION);
78  }

378  }
379  }

384  for(unsigned i=0;i<n_node_poly;i++)
385  {
386  poly_file>>dummy;
387  poly_file>>dummy;
388  poly_file>>dummy;
389  poly_file>>dummy;
390  poly_file>>dummy;
391  }
392  }

395  for(unsigned i=0;i<n_node_poly;i++)
396  {
397  poly_file>>dummy;
398  poly_file>>dummy;
399  poly_file>>dummy;

406  // Now extract the segment information
407  // Segements are lines that lie on boundaries of the domain
408  //----------------------------------------------------------
409 
410  // Number of segments
411  unsigned n_segment;
412  poly_file>>n_segment;
413 
414  // Boundary marker flag
415  poly_file>>boundary_markers_flag;
416 
417  // Storage for the global node numbers (in the triangle 1-based
418  // numbering scheme!) of the first and second node in each segment
419  Vector<unsigned> first_node(n_segment);
420  Vector<unsigned> second_node(n_segment);
421 
422  // Storage for the boundary marker for each segment
423  Vector<unsigned> segment_boundary(n_segment);
424 
425  // Dummy for global segment number
426  unsigned dummy_segment_number;
427 
428  // Storage for the edges associated with each node. Nodes are indexed
429  // using the Triangle 1-based index which is why there is a +1 here.
430  Vector<std::set<unsigned> > node_on_edges(n_node+1);
431 
432  // Extract information for each segment
433  for(unsigned i=0;i<n_segment;i++)
434  {
435  poly_file >> dummy_segment_number;
436  poly_file >> first_node[i];
437  poly_file >> second_node[i];
438  poly_file >> segment_boundary[i];
439  //Check that we don't have a higher segment boundary number
440  if(segment_boundary[i] > n_bound) {n_bound = segment_boundary[i];}
441  //Add the segment index to each node
442  node_on_edges[first_node[i]].insert(i);
443  node_on_edges[second_node[i]].insert(i);
444  }
445 
446  // Extract hole center information
447  unsigned nhole=0;
448  poly_file>>nhole;
449  if(nhole!=0)
450  {
451  Hole_centre.resize(nhole);
452 
453  // Dummy for hole number
454  unsigned dummy_hole;
455  // Loop over the holes to get centre coords
456  for(unsigned ihole=0;ihole<nhole;ihole++)
457  {
458  Hole_centre[ihole].resize(2);
459 
460  // Read the centre value

462  poly_file >> Hole_centre[ihole][0];

479  // Create the elements
480  //--------------------
481 
482  // Counter for nodes in the vector that lists
483  // the global node numbers of the elements' local nodes
484  unsigned counter=0;
485  for(unsigned e=0;e<n_element;e++)
486  {
487  Element_pt[e]=new TElement<2,2>;
488  for(unsigned j=0;j<n_local_node;j++)
489  {
490  unsigned global_node_number=Global_node[counter];
491  if(done[global_node_number-1]==false) //... -1 because node number
492  // begins at 1 in triangle
493  {
494  //If we are on the boundary
495  if((boundary_markers_flag==1) &&
496  (bound[global_node_number-1]>0))
497  {
498  //Construct a boundary node
499  Node_pt[global_node_number-1] =
500  finite_element_pt(e)->construct_boundary_node(j);
501  //Add to the boundary node look-up scheme
502  add_boundary_node(bound[global_node_number-1]-1,
503  Node_pt[global_node_number-1]);
504  }
505  //Otherwise make an ordinary node
506  else
507  {
508  Node_pt[global_node_number-1]=finite_element_pt(e)->construct_node(j);
509  }

512  Node_pt[global_node_number-1]->x(1)=y_node[global_node_number-1];
513  }
514  else

517  }
518  counter++;
519  }

522  // Resize the "matrix" that stores the boundary id for each
523  // edge in each element.
524  Edge_boundary.resize(n_element);
525  Edge_index.resize(n_element);
526 
527  // Storage for the global node numbers (in triangle's 1-based
528  // numbering scheme) for the zero-th, 1st, and 2nd node in each
529  // triangle.
530  unsigned glob_num[3]={0,0,0};
531 
532  //0-based index used to construct a global index-based lookup scheme
533  //for each edge that will be used to uniquely construct mid-side
534  //nodes.
535  //The segments (edges that lie on boundaries) have already
536  //been added to the scheme, so we start with the number of segments.
537  Nglobal_edge=n_segment;
538 
539  // Loop over the elements
540  for(unsigned e=0;e<n_element;e++)
541  {
542  // Each element has three edges
543  Edge_boundary[e].resize(3);
544  Edge_index[e].resize(3);
545  // By default each edge is NOT on a boundary
546  for(unsigned i=0;i<3;i++)
547  {
548  Edge_boundary[e][i]=0;
549  }
550 
551  //Read out the global node numbers from the triangle data structure
552  const unsigned element_offset = e*n_local_node;
553  for(unsigned i=0;i<3;i++)
554  {
555  glob_num[i] = Global_node[element_offset+i];
556  }
557 
558  // Now we know the global node numbers of the elements' three nodes
559  // in triangle's 1-based numbering.
560 
561  // Determine whether any of the elements edges have already been
562  // allocated an index. This may be because they are on boundaries
563  // (segments) or because they have already occured.
564  // The global index for the i-th edge will be stored in edge_index[i]
565  for(unsigned i=0;i<3;i++)
566  {
567  std::vector<unsigned> local_edge_index;
568 
569  //Find the common global edge index for the nodes on
570  //the i-th element edge (note the use of moular arithmetic here)
571  std::set_intersection(node_on_edges[glob_num[i]].begin(),
572  node_on_edges[glob_num[i]].end(),
573  node_on_edges[glob_num[(i+1)%3]].begin(),
574  node_on_edges[glob_num[(i+1)%3]].end(),
575  std::insert_iterator<std::vector<unsigned> >(
576  local_edge_index,local_edge_index.begin()));
577 
578  //If the nodes share more than one global edge index, then
579  //we have a problem
580  if(local_edge_index.size() > 1)
581  {
582  throw OomphLibError(
583  "Nodes in scaffold mesh share more than one global edge",
584  OOMPH_CURRENT_FUNCTION,
585  OOMPH_EXCEPTION_LOCATION);
586  }
587 
588  //If the element's edge is not already allocated, the intersection
589  //will be empty
590  if(local_edge_index.size()==0)
591  {
592  //Allocate the next global index
593  Edge_index[e][i] = Nglobal_edge;
594  //Associate the new edge index with the nodes
595  node_on_edges[glob_num[i]].insert(Nglobal_edge);
596  node_on_edges[glob_num[(i+1)%3]].insert(Nglobal_edge);
597  //Increment the global edge index
598  ++Nglobal_edge;
599  }
600  //Otherwise we already have an edge
601  else if(local_edge_index.size()==1)
602  {
603  //Set the edge index
604  Edge_index[e][i] = local_edge_index[0];
605  //Allocate the boundary index, if it is a segment
606  if(local_edge_index[0] < n_segment)
607  {
608  Edge_boundary[e][i] = segment_boundary[local_edge_index[0]];
609  //Add the nodes to the boundary look-up scheme in
610  //oomph-lib (0-based) index
611  add_boundary_node(segment_boundary[local_edge_index[0]]-1,
612  Node_pt[glob_num[i]-1]);
613  add_boundary_node(segment_boundary[local_edge_index[0]]-1,
614  Node_pt[glob_num[(i+1)%3]-1]);
615 
616  }
617  }
618  }
619 
620  }
