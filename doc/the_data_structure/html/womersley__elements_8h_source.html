152 //=============================================================
153 /// A class for all isoparametric elements that solve the
154 /// Womersley (parallel flow) equations.
155 /// \f[
156 /// Re St \frac{\partial u}{\partial t} = - g +
157 /// \frac{\partial^2 u}{\partial x_i^2}
158 /// \f]
159 /// which may be derived from the full Navier-Stokes equations
160 /// (with a viscous scaling of the pressure) under the assumption of
161 /// parallel flow in the z direction. u then represents the axial
162 /// velocity and g is the (spatially constant) axial component of
163 /// the pressure gradient.
164 ///
165 /// This class contains the generic maths. Shape functions, geometric
166 /// mapping etc. must get implemented in derived class.
167 /// Note that this class assumes an isoparametric formulation, i.e. that
168 /// the scalar unknown is interpolated using the same shape functions
169 /// as the position.
170 ///
171 /// Generally, the instantaneous value of the pressure gradient, g,
172 /// is prescribed (and specified via a pointer to a single-valued
173 /// Data object whose current (pinned) value contains the pressure.
174 ///
175 /// It is also possible to prescribe the flow rate through
176 /// a mesh of Womersley elements and to determine the pressure
177 /// gradient required to achieve this flow rate as an unknown.
178 /// In that case the external pressure is treated as
179 /// an external Data object that an associated
180 /// ImposeFluxForWomersleyElement is in charge of. Note that only
181 /// the ImposeFluxForWomersleyElement can set the pressure gradient
182 /// Data object as external Data. This is because (counter to
183 /// general practice) the WomersleyEquations make contributions
184 /// to the residuals of the ImposeFluxForWomersleyElements in order
185 /// to keep the elemental Jacobians as small as possible.
186 //=============================================================
187 template<unsigned DIM>
188 class WomersleyEquations : public virtual FiniteElement
189 {
190 public:
191 /// Constructor: Initialises the Pressure_gradient_data_pt to null
192 WomersleyEquations() : Pressure_gradient_data_pt(0)
193 {
194 ReSt_pt = &Default_ReSt_value;
195 }

254 /// du/dt at local node n.
255 /// Uses suitably interpolated value for hanging nodes.
256 double du_dt_womersley(const unsigned& n) const
257 {
258 // Get the data's timestepper
259 TimeStepper* time_stepper_pt = this->node_pt(n)->time_stepper_pt();
260
261 // Initialise dudt
262 double dudt = 0.0;
263
264 // Loop over the timesteps, if there is a non Steady timestepper
265 if (!time_stepper_pt->is_steady())
266 {
267 // Find the index at which the variable is stored
268 const unsigned u_nodal_index = u_index_womersley();
269
270 // Number of timsteps (past & present)
271 const unsigned n_time = time_stepper_pt->ntstorage();
272
273 // Add the contributions to the time derivative
274 for (unsigned t = 0; t < n_time; t++)
275 {
276 dudt +=
277 time_stepper_pt->weight(1, t) * nodal_value(t, n, u_nodal_index);
278 }
279 }
280 return dudt;
281 }

344 /// Get flux: flux[i] = du/dx_i
345 void get_flux(const Vector<double>& s, Vector<double>& flux) const
346 {
347 // Find out how many nodes there are in the element
348 unsigned n_node = nnode();
349
350 // Find the index at which the variable is stored
351 unsigned u_nodal_index = u_index_womersley();
352
353 // Set up memory for the shape and test functions
354 Shape psi(n_node);
355 DShape dpsidx(n_node, DIM);
356
357 // Call the derivatives of the shape and test functions
358 dshape_eulerian(s, psi, dpsidx);
359
360 // Initialise to zero
361 for (unsigned j = 0; j < DIM; j++)
362 {
363 flux[j] = 0.0;
364 }
365
366 // Loop over nodes
367 for (unsigned l = 0; l < n_node; l++)
368 {
369 // Loop over derivative directions
370 for (unsigned j = 0; j < DIM; j++)
371 {
372 flux[j] += nodal_value(l, u_nodal_index) * dpsidx(l, j);
373 }
374 }
375 }

397 /// Return FE representation of function value u(s) at local coordinate s
398 inline double interpolated_u_womersley(const Vector<double>& s) const
399 {
400 // Find number of nodes
401 unsigned n_node = nnode();
402
403 // Find the index at which the variable is stored
404 unsigned u_nodal_index = u_index_womersley();
405
406 // Local shape function
407 Shape psi(n_node);
408
409 // Find values of shape function
410 shape(s, psi);
411
412 // Initialise value of u
413 double interpolated_u = 0.0;
414
415 // Loop over the local nodes and sum
416 for (unsigned l = 0; l < n_node; l++)
417 {
418 interpolated_u += nodal_value(l, u_nodal_index) * psi[l];
419 }
420
421 return (interpolated_u);
422 }

495 //========================================================================
496 /// Element to impose volume flux through collection of Womersley
497 /// elements, in exchange for treating the pressure gradient
498 /// as an unknown. The pressure gradient is created (as a single-valued
499 /// Data item) in the constructor for this element which also
500 /// takes a pointer to the Mesh containing the Womersley elements whose
501 /// total flux is being controlled. While doing this we tell them
502 /// that their pressure gradient is now an unknown and must be
503 /// treated as external Data.
504 //========================================================================
505 template<unsigned DIM>
506 class ImposeFluxForWomersleyElement : public virtual GeneralisedElement
507 {
508 public:
509 /// Constructor: Pass pointer to mesh that contains the
510 /// Womersley elements whose volume flux is controlled, and pointer to
511 /// double that contains the instantaneous value of the
512 /// prescribed flux
513 ImposeFluxForWomersleyElement(Mesh* womersley_mesh_pt,
514 double* prescribed_flux_pt)
515 : Prescribed_flux_pt(prescribed_flux_pt)
516 {
517 // Store the mesh of the flux-controlled Womerersley elements
518 Womersley_mesh_pt = womersley_mesh_pt;
519
520 // Create the pressure gradient Data
521 Pressure_gradient_data_pt = new Data(1);
522 Pressure_gradient_data_pt->set_value(0, 0.0);
523
524 // Pressure gradient is internal data of this element
525 add_internal_data(Pressure_gradient_data_pt);
526
527 // Find number of elements in the mesh of Womersley elements
528 // whose total flux is controlled
529 unsigned n_element = womersley_mesh_pt->nelement();
530
531 // Loop over the elements to tell them that the pressure
532 // gradient is given by the newly created Data object
533 // which is to be treated as external Data.
534 for (unsigned e = 0; e < n_element; e++)
535 {
536 // Upcast from FiniteElement to the present element
537 WomersleyEquations<DIM>* el_pt = dynamic_cast<WomersleyEquations<DIM>*>(
538 womersley_mesh_pt->element_pt(e));
539
540 // Set the pressure gradient function pointer
541 el_pt->set_pressure_gradient_and_add_as_external_data(
542 Pressure_gradient_data_pt);
543 }
544 }
545
546 /// Read-only access to the single-valued Data item that
547 /// stores the pressure gradient (to be determined via the
548 /// flux control)
549 Data* pressure_gradient_data_pt()
550 {
551 return Pressure_gradient_data_pt;
552 }
553
554
555 /// Get volume flux through all Womersley elements
556 double total_volume_flux()
557 {
558 // Initialise
559 double flux = 0.0;
560
561 // Assemble contributions from elements
562 unsigned nelem = Womersley_mesh_pt->nelement();
563 for (unsigned e = 0; e < nelem; e++)
564 {
565 WomersleyEquations<DIM>* el_pt = dynamic_cast<WomersleyEquations<DIM>*>(
566 Womersley_mesh_pt->element_pt(e));
567 if (el_pt != 0) flux += el_pt->get_volume_flux();
568 }
569
570 // Return total volume flux
571 return flux;
572 }

575 /// Compute residual vector: the volume flux constraint
576 /// determines this element's one-and-only internal Data which represents
577 /// the pressure gradient
578 void get_residuals(Vector<double>& residuals)
579 {
580 // Local equation number of volume flux constraint -- associated
581 // with the internal data (the unknown pressure gradient)
582 int local_eqn = internal_local_eqn(0, 0);
583 if (local_eqn >= 0)
584 {
585 residuals[local_eqn] += total_volume_flux() - (*Prescribed_flux_pt);
586 }
587 }
588
589
590 /// Compute element residual Vector and element Jacobian matrix
591 /// Note: Jacobian is zero because the derivatives w.r.t. to
592 /// velocity dofs are added by the Womersley elements; the current
593 /// element's internal Data (the pressure gradient) does not feature
594 /// in the volume constraint.
595 void get_jacobian(Vector<double>& residuals, DenseMatrix<double>& jacobian)
596 {
597 // Initialise Jacobian
598 unsigned n_dof = ndof();
599 for (unsigned i = 0; i < n_dof; i++)
600 {
601 for (unsigned j = 0; j < n_dof; j++)
602 {
603 jacobian(i, j) = 0.0;
604 }
605 }
606 // Get residuals
607 get_residuals(residuals);
608 }

614 /// Data item whose one and only value contains the pressure
615 /// gradient
616 Data* Pressure_gradient_data_pt;
617
618 /// Pointer to current value of prescribed flux
619 double* Prescribed_flux_pt;
620 };

1201 /// Set up the Womersley tubes so that a subsequent call
1202 /// to get_response(...) computes the inlet pressure for the currently
1203 /// prescribed instantaneous flow rate, assuming that at all previous times
1204 /// the tube conveyed steady, fully-developed flow with flowrate q_initial.
1205 /// dt specifies the timestep for the subsequent time integration.
1206 /// Specify: Womersley number, (constant) timestep, the initial volume
1207 /// flux (from which the subsequent impulsive start is performed) and,
1208 /// optionally the pointer to the timestepper to be used in the Womersley
1209 /// elements (defaults to BDF<2>).
1210 void setup(double* re_st_pt,
1211 const double& dt,
1212 const double& q_initial,
1213 TimeStepper* time_stepper_pt = 0)
1214 {
1215 // Create timestepper if none specified so far
1216 if (time_stepper_pt == 0)
1217 {
1218 time_stepper_pt = new BDF<2>;
1219 }
1220
1221 // Build mesh and apply bcs
1222 Mesh* my_mesh_pt =
1223 build_mesh_and_apply_boundary_conditions(time_stepper_pt);
1224
1225 // Build problem
1226 Womersley_problem_pt = new WomersleyProblem<ELEMENT, DIM>(
1227 re_st_pt, Current_volume_flux_pt, time_stepper_pt, my_mesh_pt);
1228
1229 /// By default, we do want to suppress the output from the
1230 /// Newton solver
1231 Womersley_problem_pt->disable_info_in_newton_solve();

1233 << " Newton solver in WomersleyImpedanceTubeBase. "
1234 << std::endl;

1237 // impedance traction elements (if they're used to specify the inflow
1238 if ((!Using_flux_control_elements) &&
1239 (Navier_stokes_outflow_mesh_pt != 0))
1240 {
1241 precompute_aux_integrals();
1242 }
1243
1244 // Initialise timestep -- also sets the weights for all timesteppers
1245 // in the problem.
1246 Womersley_problem_pt->initialise_dt(dt);
1247
1248 // Set currently imposed flux
1249 *Current_volume_flux_pt = q_initial;
1250
1251 // Assign steady initial solution for this flux
1252 Womersley_problem_pt->steady_newton_solve();
1253
1254 // Allow for resolve
1255 Womersley_problem_pt->linear_solver_pt()->enable_resolve();
1256

1258 Womersley_problem_pt->enable_jacobian_reuse();
1259
1260 // Shut up
1261 Womersley_problem_pt->linear_solver_pt()->disable_doc_time();
1262
1263 // Do a dummy solve with time-dependent terms switched on
1264 // to generate (and store) the Jacobian. (We're not using
1265 // a Newton solve because the initial residual may be zero
1266 // in which case the Jacobian would never be computed!)
1267 unsigned n_dof = Womersley_problem_pt->ndof();
1268
1269 // Local scope to make sure dx goes out of scope
1270 {
1271 DoubleVector dx;
1272 Womersley_problem_pt->linear_solver_pt()->solve(Womersley_problem_pt,
1273 dx);
1274 }

1278

1280 LinearAlgebraDistribution dist(
1281 Womersley_problem_pt->communicator_pt(), n_dof, false);
1282 DoubleVector drdq(&dist, 0.0);
1283 DoubleVector dxdq(&dist, 0.0);
1284
1285 // What's the global equation number of the equation that
1286 // determines the pressure gradient
1287 unsigned g_eqn =
1288 Womersley_problem_pt->pressure_gradient_data_pt()->eqn_number(0);
1289
1290 // Derivative of volume constraint residual w.r.t. prescribed
1291 // instantaenous volume flux (in ImposeFluxForWomersleyElement)

1295 // instantaenous volume flux (in ImposeFluxForWomersleyElement)
1296 Womersley_problem_pt->linear_solver_pt()->resolve(drdq, dxdq);
1297

1299 // volume flux
1300 Dp_in_dq = dxdq[g_eqn] * Length;

1374 /// Compute inlet pressure, p_in, required to achieve the currently
1375 /// imposed, instantaneous volume flux q prescribed by
1376 /// total_volume_flux_into_impedance_tube(), and its
1377 /// derivative, dp_in/dq.
1378 void get_response(double& p_in, double& dp_in_dq)
1379 {
1380 // Set currently imposed flux
1381 *Current_volume_flux_pt = total_volume_flux_into_impedance_tube();
1382
1383 // Do a Newton solve to compute the pressure gradient
1384 // required to achieve the imposed instantaneous flow rate
1385 Womersley_problem_pt->newton_solve();
1386
1387 // Compute inflow pressure based on computed pressure gradient,
1388 // the length of tube, and the outlet pressure
1389 p_in =

1602 /// Mesh of Womersley elements whose topology, nodal position etc.
1603 /// matches that of a given mesh of face elements in the outflow
1604 /// cross-section of a full Navier-Stokes mesh.
1605 //====================================================================
1606 template<class WOMERSLEY_ELEMENT>
1607 class WomersleyMesh : public virtual Mesh,
1608 public virtual TemplateFreeWomersleyMeshBase
1609 {
1610 public:
1611 /// Constructor: Pass pointer to mesh of face elements in the
1612 /// outflow cross-section of a full Navier-Stokes mesh, the timestepper to
1613 /// be used for the Womersley elements, the coordinate (in the
1614 /// higher-dimensional Navier-Stokes mesh) that is constant
1615 /// in the outflow cross-section and the velocity component
1616 /// (in terms of the nodal index) that represents the outflow
1617 /// component -- the latter is used to automatically apply
1618 /// the boundary conditions for the Womersley problem.
1619 WomersleyMesh(Mesh* n_st_outflow_mesh_pt,
1620 TimeStepper* time_stepper_pt,
1621 const unsigned& fixed_coordinate,
1622 const unsigned& w_index)
1623 {
1624 /// How many elements and nodes are there in the original mesh?
1625 unsigned nelem = n_st_outflow_mesh_pt->nelement();
1626
1627 // Navier-Stokes outflow mesh may not have any nodes stored (it usually
1628 // just acts as a container for traction elements) -->
1629 // Count number of distinct Navier-Stokes nodes by adding
1630 // the elements' nodes to a set
1631 std::set<Node*> n_st_nodes;
1632 for (unsigned e = 0; e < nelem; e++)
1633 {
1634 FiniteElement* el_pt = n_st_outflow_mesh_pt->finite_element_pt(e);
1635 unsigned nnod_el = el_pt->nnode();
1636 for (unsigned j = 0; j < nnod_el; j++)
1637 {
1638 n_st_nodes.insert(el_pt->node_pt(j));
1639
1640 // Careful: It there are hanging nodes this won't work!
1641 if (el_pt->node_pt(j)->is_hanging())
1642 {
1643 throw OomphLibError(
1644 "Cannot build WomersleyMesh from mesh with hanging nodes!",
1645 OOMPH_CURRENT_FUNCTION,
1646 OOMPH_EXCEPTION_LOCATION);
1647 }
1648 }
1649 }
1650
1651 // Extract size then wipe
1652 unsigned nnode_n_st = n_st_nodes.size();
1653 n_st_nodes.clear();
1654
1655 // Create enough storage
1656 Node_pt.resize(nnode_n_st);
1657
1658 /// Create new elements
1659 for (unsigned e = 0; e < nelem; e++)
1660 {
1661 add_element_pt(new WOMERSLEY_ELEMENT);
1662#ifdef PARANOID
1663 if (finite_element_pt(e)->nnode() !=
1664 n_st_outflow_mesh_pt->finite_element_pt(e)->nnode())
1665 {
1666 throw OomphLibError(
1667 "Number of nodes in existing and new elements don't match",
1668 OOMPH_CURRENT_FUNCTION,
1669 OOMPH_EXCEPTION_LOCATION);
1670 }
1671#endif
1672 }
1673
1674 // Map to record which Navier-Stokes nodes have been visited (default
1675 // return is false)
1676 std::map<Node*, bool> n_st_node_done;
1677
1678 // Map to store the Womersley node that corresponds to a
1679 // Navier Stokes node
1680 std::map<Node*, Node*> equivalent_womersley_node_pt;
1681
1682 // Initialise count of newly created nodes
1683 unsigned node_count = 0;
1684
1685
1686 // This is awkward do diagnose: We're assuming that
1687 // the boundary conditions have been applied for the
1688 // underlying Navier-Stokes problem before calling
1689 // this function (otherwise it's really tricky to
1690 // apply the right boundary conditions here), but it's
1691 // hard to police. Issue definite (but suppressable)
1692 // warning if nothing has been pinned at all.
1693 unsigned n_pinned_nodes = 0;
1694
1695 // Loop over nst and womersley elements in tandem to sort out
1696 // which new nodes are required
1697 for (unsigned e = 0; e < nelem; e++)
1698 {
1699 FiniteElement* n_st_el_pt = n_st_outflow_mesh_pt->finite_element_pt(e);
1700 unsigned nnod_el = n_st_el_pt->nnode();
1701 for (unsigned j = 0; j < nnod_el; j++)
1702 {
1703 // Has the Navier Stokes node been done yet?
1704 Node* n_st_node_pt = n_st_el_pt->node_pt(j);
1705
1706 // Hasn't been done: Create new node in Womersley element
1707 if (!n_st_node_done[n_st_node_pt])
1708 {
1709 // Create a new node in the Womersley element
1710 Node* new_node_pt =
1711 finite_element_pt(e)->construct_node(j, time_stepper_pt);
1712
1713 // Add newly created node
1714 Node_pt[node_count] = new_node_pt;
1715 node_count++;
1716
1717
1718 // Set coordinates
1719 unsigned dim = n_st_node_pt->ndim();
1720 unsigned icount = 0;
1721 for (unsigned i = 0; i < dim; i++)
1722 {
1723 if (i != fixed_coordinate)
1724 {
1725 new_node_pt->x(icount) = n_st_node_pt->x(i);
1726 icount++;
1727 }
1728 }
1729
1730 // Set pin status
1731 if (n_st_node_pt->is_pinned(w_index))
1732 {
1733 new_node_pt->pin(0);
1734 n_pinned_nodes++;
1735 }
1736 else
1737 {
1738 // shouldn't need this, but...
1739 new_node_pt->unpin(0);
1740 }
1741
1742 // Record which Womersley node the
1743 // Navier Stokes node is associated with
1744 equivalent_womersley_node_pt[n_st_node_pt] = new_node_pt;
1745
1746 // Now the Navier-Stokes node has been done
1747 n_st_node_done[n_st_node_pt] = true;
1748 }
1749 // The node has already been done -- set pointer to existing
1750 // Womersley node
1751 else
1752 {
1753 finite_element_pt(e)->node_pt(j) =
1754 equivalent_womersley_node_pt[n_st_node_pt];
1755 }
1756 }
1757
1758 bool passed = true;
1759 finite_element_pt(e)->check_J_eulerian_at_knots(passed);
1760 if (!passed)
1761 {
1762 // Reverse the nodes
1763 unsigned nnod = finite_element_pt(e)->nnode();
1764 Vector<Node*> orig_nod_pt(nnod);
1765 for (unsigned j = 0; j < nnod; j++)
1766 {
1767 orig_nod_pt[j] = finite_element_pt(e)->node_pt(j);
1768 }
1769 for (unsigned j = 0; j < nnod; j++)
1770 {
1771 finite_element_pt(e)->node_pt(j) = orig_nod_pt[nnod - j - 1];
1772 }
1773 bool passed = true;
1774 finite_element_pt(e)->check_J_eulerian_at_knots(passed);
1775 if (!passed)
1776 {
1777 throw OomphLibError("Element remains inverted even after reversing "
1778 "the local node numbers",
1779 OOMPH_CURRENT_FUNCTION,
1780 OOMPH_EXCEPTION_LOCATION);
1781 }
1782 }
1783 }
1784
1785
1786#ifdef PARANOID
1787 if (!Suppress_warning_about_unpinned_nst_dofs)
1788 {
1789 if (n_pinned_nodes == 0)
1790 {
1791 std::stringstream bla;
1792 bla << "Boundary conditions must be applied in Navier-Stokes\n"
1793 << "problem before attaching impedance elements.\n"
1794 << "Note: This warning can be suppressed by setting the\n"
1795 << "global static boolean\n\n"
1796 << " "
1797 "TemplateFreeWomersleyMeshBase::Suppress_warning_about_"
1798 "unpinned_nst_dofs\n\n"
1799 << "to true\n";
1800 OomphLibWarning(
1801 bla.str(), OOMPH_CURRENT_FUNCTION, OOMPH_EXCEPTION_LOCATION);
1802 }
1803 }
1804#endif
1805
1806#ifdef PARANOID
1807 if (nnode() != nnode_n_st)
1808 {
1809 throw OomphLibError(
1810 "Number of nodes in the new mesh don't match that in the old one",
1811 OOMPH_CURRENT_FUNCTION,
1812 OOMPH_EXCEPTION_LOCATION);
1813 }
1814#endif
1815 }
1816 };

2421 //============================================================================

2423 class WOMERSLEY_ELEMENT,
2424 unsigned DIM>
2425 void NavierStokesImpedanceTractionElement<BULK_NAVIER_STOKES_ELEMENT,
2426 WOMERSLEY_ELEMENT,
2427 DIM>::
2428 fill_in_generic_residual_contribution_fluid_traction(
2429 Vector<double>& residuals, DenseMatrix<double>& jacobian, unsigned flag)
2430 {
2431 // Find out how many nodes there are
2432 unsigned n_node = nnode();
2433
2434 // Set up memory for the shape and test functions
2435 Shape psif(n_node), testf(n_node);
2436
2437 // Set the value of n_intpt
2438 unsigned n_intpt = integral_pt()->nweight();
2439
2440 // Integers to store local equation numbers
2441 int local_eqn = 0;
2442 int local_unknown = 0;
2443
2444 // Loop over the integration points
2445 for (unsigned ipt = 0; ipt < n_intpt; ipt++)
2446 {
2447 // Get the integral weight
2448 double w = integral_pt()->weight(ipt);
2449
2450 // Find the shape and test functions and return the Jacobian
2451 // of the mapping
2452 double J = shape_and_test_at_knot(ipt, psif, testf);
2453
2454 // Premultiply the weights and the Jacobian
2455 double W = w * J;
2456
2457 // Traction vector
2458 Vector<double> traction(DIM + 1);
2459
2460 // Initialise response
2461 double p_in = 0.0;
2462 double dp_in_dq = 0.0;
2463
2464 // Traction= outer unit normal x pressure at upstream end of
2465 // impedance tube
2466 if (Navier_stokes_outflow_mesh_pt != 0)
2467 {
2468 // Get response of the impedance tube:
2469 Impedance_tube_pt->get_response(p_in, dp_in_dq);

2472 // Get outer unit normal at current integration point
2473 Vector<double> unit_normal(DIM + 1);
2474 outer_unit_normal(ipt, unit_normal);
2475
2476 // Loop over the directions
2477 for (unsigned i = 0; i < DIM + 1; i++)
2478 {
2479 traction[i] = -unit_normal[i] * p_in;
2480 }
2481
2482
2483 // Loop over the test functions
2484 for (unsigned l = 0; l < n_node; l++)
2485 {
2486 // Loop over the velocity components
2487 for (unsigned i = 0; i < DIM + 1; i++)
2488 {
2489 local_eqn = u_local_eqn(l, i);
2490 /*IF it's not a boundary condition*/

2492 {
2493 // Add the user-defined traction terms
2494 residuals[local_eqn] += traction[i] * testf[l] * W;
2495
2496 // Compute the Jacobian too?
2497 if (flag && (Navier_stokes_outflow_mesh_pt != 0))
2498 {
2499 // Loop over the nodes
2500 for (unsigned j = 0; j < n_node; j++)
2501 {
2502 // Get pointer to Node
2503 Node* nod_pt = node_pt(j);
2504
2505 // Loop over the velocity components
2506 for (unsigned ii = 0; ii < DIM + 1; ii++)
2507 {
2508 local_unknown = u_local_eqn(j, ii);
2509
2510 /*IF it's not a boundary condition*/
2511 if (local_unknown >= 0)
2512 {
2513 // Get corresponding global unknown number
2514 unsigned global_unknown = nod_pt->eqn_number(ii);
2515
2516 // Add contribution
2517 jacobian(local_eqn, local_unknown) -=
2518 (*Aux_integral_pt)[global_unknown] * psif[l] *

2521 }
2522 }
2523

2534 {
2535 // Get local unknown number

2537
2538 // Real dof or bc?

2540 {
2541 // Get corresponding global unknown number

2543
2544 // Add contribution
2545 jacobian(local_eqn, local_unknown) -=
2546 (*Aux_integral_pt)[global_unknown] * psif[l] *
2547 unit_normal[i] * dp_in_dq * W;
2548 }
2549 }
2550 }
2551 } // end of computation of Jacobian terms
2552 }
2553 } // End of loop over dimension
2554 } // End of loop over shape functions
2555 }
