300  /// \short Runtime checked dynamic cast. This is the safe but slightly slower
301  /// cast. Use it in any of these cases:
302  /// - You aren't entirely sure the cast is always safe.
303  /// - You have strange inheritance structures (e.g. the "Diamond of Death" in
304  /// element inheritance).
305  /// - Efficiency is not critical.
306  /// Note that if you just want to check if a pointer can be converted to
307  /// some type you will need to use a plain dynamic_cast. Adapted from
308  /// polymorphic_cast in boost/cast.hpp, see
309  /// http://www.boost.org/doc/libs/1_52_0/libs/conversion/cast.htm for more
310  /// details.
311  template <class Target, class Source>
312  inline Target checked_dynamic_cast(Source* x)
313  {
314  Target tmp = dynamic_cast<Target>(x);
315 #ifdef PARANOID
316  if(tmp == 0)
317  {
318  throw OomphLibError("Bad cast.", OOMPH_CURRENT_FUNCTION,
319  OOMPH_EXCEPTION_LOCATION);
320  }
321 #endif
322  return tmp;
323  }
324 
325  /// \short Checked static cast. Only use this cast if ALL of these are true:
326  /// - You are sure that the cast will always succeed.
327  /// - You aren't using any strange inheritance structures (e.g. the "Diamond of
328  /// Death" in element inheritance, if you aren't sure just try compiling).
329  /// - You need efficiency.
330  /// Adapted from polymorphic_downcast in boost/cast.hpp, See
331  /// http://www.boost.org/doc/libs/1_52_0/libs/conversion/cast.htm for more
332  /// details.
333  template <class Target, class Source>
334  inline Target checked_static_cast(Source* x)
335  {
336 #ifdef PARANOID
337  // Check that the cast will work as expected.
338  if(dynamic_cast<Target>(x) != x)
339  {
340  throw OomphLibError("Bad cast.", OOMPH_CURRENT_FUNCTION,
341  OOMPH_EXCEPTION_LOCATION);
342  }
343 #endif
344  return static_cast<Target>(x);
345  }
