175 /// Helper function to create a HYPRE_IJVector and HYPRE_ParVector.
176 /// + If no MPI then serial vectors are created
177 /// + If MPI and serial input vector then distributed hypre vectors are created
178 /// + If MPI and distributed input vector the distributed output vectors
179 /// are created.
180 //========================================================================
181  void create_HYPRE_Vector(const DoubleVector &oomph_vec, const
182  LinearAlgebraDistribution* dist_pt,
183  HYPRE_IJVector& hypre_ij_vector,
184  HYPRE_ParVector& hypre_par_vector)
185  {
186  // the lower and upper row of the vector on this processor
187  unsigned lower = dist_pt->first_row();
188  unsigned upper = dist_pt->first_row() +
189  dist_pt->nrow_local() - 1;
190 
191  // number of local rows
192  unsigned nrow_local = dist_pt->nrow_local();
193 
194  // initialize Hypre vector
195 #ifdef OOMPH_HAS_MPI
196  HYPRE_IJVectorCreate(oomph_vec.distribution_pt()->
197  communicator_pt()->mpi_comm(),
198  lower,upper, &hypre_ij_vector);
199 #else
200  HYPRE_IJVectorCreate(MPI_COMM_WORLD,
201  lower,upper, &hypre_ij_vector);
202 #endif
203  HYPRE_IJVectorSetObjectType(hypre_ij_vector, HYPRE_PARCSR);
204  HYPRE_IJVectorInitialize(hypre_ij_vector);
205 
206  // set up array containing indices
207  int* indices = new int[nrow_local];
208  double* values = new double[nrow_local];
209  const unsigned hypre_first_row = dist_pt->first_row();
210  unsigned j = 0;
211  if (!oomph_vec.distributed() && dist_pt->distributed())
212  {
213  j = hypre_first_row;
214  }
215  const double* o_pt = oomph_vec.values_pt();
216  for (unsigned i=0; i<nrow_local; i++)
217  {
218  indices[i] = hypre_first_row + i;
219  values[i] = o_pt[j+i];
220  }
221 
222  // insert values
223  HYPRE_IJVectorSetValues(hypre_ij_vector,nrow_local,indices,values);
224 
225  // assemble vectors
226  HYPRE_IJVectorAssemble(hypre_ij_vector);
227  HYPRE_IJVectorGetObject(hypre_ij_vector, (void **) &hypre_par_vector);
228 
229  // clean up
230  delete[] indices;
231  delete[] values;
232  }

276  HYPRE_ParCSRMatrix& hypre_par_matrix,
277  LinearAlgebraDistribution* dist_pt)
278  {
279 #ifdef PARANOID

281  if ( !oomph_matrix->built() )
282  {

284  error_message << "The matrix has not been built";
285  throw OomphLibError(error_message.str(),
286  OOMPH_CURRENT_FUNCTION,
287  OOMPH_EXCEPTION_LOCATION);
288  }
289  // check the matrix is square
290  if ( oomph_matrix->nrow() != oomph_matrix->ncol() )
291  {
292  std::ostringstream error_message;
293  error_message << "create_HYPRE_Matrix require a square matrix. "
294  << "Matrix is " << oomph_matrix->nrow()
295  << " by " << oomph_matrix->ncol() << std::endl;
296  throw OomphLibError(error_message.str(),
297  OOMPH_CURRENT_FUNCTION,
298  OOMPH_EXCEPTION_LOCATION);
299  }
300 #endif
301 
302 #ifdef OOMPH_HAS_MPI
303  //Trap the case when we have compiled with MPI,
304  //but are running in serial
305  if(!MPI_Helpers::mpi_has_been_initialised())
306  {
307  std::ostringstream error_stream;
308  error_stream << "Oomph-lib has been compiled with MPI support and "
309  << "you are using HYPRE.\n"
310  <<
311  "For this combination of flags, MPI must be initialised.\n"
312  << "Call MPI_Helpers::init() in the "
313  << "main() function of your driver code\n";
314  throw OomphLibError(error_stream.str(),
315  OOMPH_CURRENT_FUNCTION,
316  OOMPH_EXCEPTION_LOCATION);
317  }
318 #endif
319 
320  // find number of rows/columns
321  const unsigned nrow = int(oomph_matrix->nrow());
322 
323  // get pointers to the matrix
324 
325  // column indices of matrix
326  const int* matrix_cols = oomph_matrix->column_index();
327 
328  // entries of matrix
329  const double* matrix_vals = oomph_matrix->value();
330 
331  // row starts
332  const int* matrix_row_start = oomph_matrix->row_start();
333 
334  // build the distribution
335  if (oomph_matrix->distribution_pt()->distributed())
336  {
337  dist_pt->build(oomph_matrix->distribution_pt());
338  }
339  else
340  {
341  bool distributed = true;
342  if (oomph_matrix->distribution_pt()->communicator_pt()->nproc() == 1)
343  {
344  distributed = false;
345  }
346  dist_pt->build
347  (oomph_matrix->distribution_pt()->communicator_pt(),nrow,distributed);
348  }
349 
350  // initialize hypre matrix
351  unsigned lower = dist_pt->first_row();
352  unsigned upper = lower + dist_pt->nrow_local() - 1;
353 
354 #ifdef OOMPH_HAS_MPI
355  HYPRE_IJMatrixCreate(dist_pt->communicator_pt()->mpi_comm(),
356  lower,
357  upper,
358  lower,
359  upper,
360  &hypre_ij_matrix);
361 #else
362  HYPRE_IJMatrixCreate(MPI_COMM_WORLD,
363  lower,
364  upper,
365  lower,
366  upper,
367  &hypre_ij_matrix);
368 #endif
369  HYPRE_IJMatrixSetObjectType(hypre_ij_matrix, HYPRE_PARCSR);
370  HYPRE_IJMatrixInitialize(hypre_ij_matrix);
371 
372  // set up a row map
373  // and first row / nrow_local
374  const unsigned hypre_nrow_local = dist_pt->nrow_local();
375  const unsigned hypre_first_row = dist_pt->first_row();
376  int* ncols_per_row = new int[hypre_nrow_local];
377  int* row_map = new int[hypre_nrow_local];
378  for (unsigned i = 0; i < hypre_nrow_local; i++)
379  {
380  unsigned j = i;
381  if (!oomph_matrix->distributed() && dist_pt->distributed())
382  {
383  j += hypre_first_row;
384  }
385  ncols_per_row[i] = matrix_row_start[j+1] - matrix_row_start[j];
386  row_map[i] = hypre_first_row + i;
387  }
388 
389  // put values in HYPRE matrix
390  int local_start = 0;
391  if (!oomph_matrix->distributed() && dist_pt->distributed())
392  {
393  local_start += matrix_row_start[hypre_first_row];
394  }
395 
396 
397  HYPRE_IJMatrixSetValues(hypre_ij_matrix,
398  hypre_nrow_local,
399  ncols_per_row,
400  row_map,
401  matrix_cols+local_start,
402  matrix_vals+local_start);
403 
404  // assemble matrix
405  HYPRE_IJMatrixAssemble(hypre_ij_matrix); // hierher leak?
406  HYPRE_IJMatrixGetObject(hypre_ij_matrix, (void **) &hypre_par_matrix);
407 
408  // tidy up memory
409  delete[] ncols_per_row;
410  delete[] row_map;
411  }

1103  DoubleVector &solution)
1104  {
1105  // Record time
1106  double t_start = TimingHelpers::timer();
1107 
1108  // Set up hypre vectors
1109  // --------------------
1110 
1111  // Hypre vector for rhs values
1112  HYPRE_IJVector rhs_ij;
1113  HYPRE_ParVector rhs_par;
1114 
1115  // Hypre vector for solution
1116  HYPRE_IJVector solution_ij;
1117  HYPRE_ParVector solution_par;
1118 
1119  // set up rhs_values and vec_indices
1120  HypreHelpers::create_HYPRE_Vector(rhs,
1121  Hypre_distribution_pt,
1122  rhs_ij,
1123  rhs_par);
1124 
1125  HypreHelpers::create_HYPRE_Vector(Hypre_distribution_pt,
1126  solution_ij,
1127  solution_par);
1128 
1129  // check error flag
1130  if (Hypre_error_messages)
1131  {
1132  std::ostringstream message;
1133  int err = HypreHelpers::check_HYPRE_error_flag(message);
1134  if (err)
1135  {
1136  OomphLibWarning(message.str(),
1137  "HypreSolver::hypre_solve()",
1138  OOMPH_EXCEPTION_LOCATION);
1139  }
1140  }
1141 
1142  // solve
1143  // -----
1144 
1145  // for solver stats
1146  int iterations=0;
1147  double norm=0;
1148 
1149  // Get the norm of rhs
1150  const double rhs_norm = rhs.norm();
1151  bool do_solving = false;
1152  if (rhs_norm > 0.0)
1153  {
1154  do_solving = true;
1155  }
1156 
1157 #ifdef OOMPH_HAS_MPI
1158  // We need to check whether any processor requires to solve, if that
1159  // is the case then do the solving
1160  if (MPI_Helpers::mpi_has_been_initialised())
1161  {
1162  if (MPI_Helpers::communicator_pt()->nproc()>1)
1163  {
1164  unsigned this_processor_do_solving = 0;
1165  unsigned all_processors_do_solving = 0;
1166  if (do_solving)
1167  {
1168  this_processor_do_solving = 1;
1169  }
1170  // Get the communicator
1171  OomphCommunicator* comm_pt = MPI_Helpers::communicator_pt();
1172  // Communicate with all procesoors
1173  MPI_Allreduce(&this_processor_do_solving,&all_processors_do_solving,
1174  1, MPI_UNSIGNED,MPI_SUM,comm_pt->mpi_comm());
1175  if (all_processors_do_solving > 0)
1176  {
1177  do_solving = true;
1178  }
1179  }
1180  }
1181 #endif
1182 
1183  if (do_solving)
1184  {
1185  if (Existing_solver==BoomerAMG)
1186  {
1187  HYPRE_BoomerAMGSolve(Solver, Matrix_par, rhs_par, solution_par);
1188  HYPRE_BoomerAMGGetNumIterations(Solver, &iterations);
1189  HYPRE_BoomerAMGGetFinalRelativeResidualNorm(Solver, &norm);
1190  }
1191  else if (Existing_solver==CG)
1192  {
1193  HYPRE_PCGSolve(Solver,
1194  (HYPRE_Matrix) Matrix_par,
1195  (HYPRE_Vector) rhs_par,
1196  (HYPRE_Vector) solution_par);
1197  HYPRE_PCGGetNumIterations(Solver, &iterations);
1198  HYPRE_PCGGetFinalRelativeResidualNorm(Solver, &norm);
1199  }
1200  else if (Existing_solver==GMRES)
1201  {
1202  HYPRE_GMRESSolve(Solver,
1203  (HYPRE_Matrix) Matrix_par,
1204  (HYPRE_Vector) rhs_par,
1205  (HYPRE_Vector) solution_par);
1206  HYPRE_GMRESGetNumIterations(Solver, &iterations);
1207  HYPRE_GMRESGetFinalRelativeResidualNorm(Solver, &norm);
1208  }
1209  else if (Existing_solver==BiCGStab)
1210  {
1211  HYPRE_BiCGSTABSolve(Solver,
1212  (HYPRE_Matrix) Matrix_par,
1213  (HYPRE_Vector) rhs_par,
1214  (HYPRE_Vector) solution_par);
1215  HYPRE_BiCGSTABGetNumIterations(Solver, &iterations);
1216  HYPRE_BiCGSTABGetFinalRelativeResidualNorm(Solver, &norm);
1217  }
1218  else if (Existing_solver==Euclid)
1219  {
1220  HYPRE_EuclidSolve(Solver, Matrix_par, rhs_par, solution_par);
1221  }
1222  else if (Existing_solver==ParaSails)
1223  {
1224  HYPRE_ParaSailsSolve(Solver, Matrix_par, rhs_par, solution_par);
1225  }
1226 
1227  // output any error message
1228  if (Hypre_error_messages)
1229  {
1230  std::ostringstream message;
1231  int err = HypreHelpers::check_HYPRE_error_flag(message);
1232  if (err)
1233  {
1234  OomphLibWarning(message.str(),
1235  "HypreSolver::hypre_solve()",
1236  OOMPH_EXCEPTION_LOCATION);
1237  }
1238  }
1239 
1240  } // if (do_solving)
1241 
1242  // Copy result to solution
1243  unsigned nrow_local = Hypre_distribution_pt->nrow_local();
1244  unsigned first_row = Hypre_distribution_pt->first_row();
1245  int* indices = new int[nrow_local];
1246  for (unsigned i = 0; i < nrow_local; i++)
1247  {
1248  indices[i] = first_row + i;
1249  }
1250  LinearAlgebraDistribution* soln_dist_pt;
1251  if (solution.built())
1252  {
1253  soln_dist_pt = new LinearAlgebraDistribution(solution.distribution_pt());
1254  }
1255  else
1256  {
1257  soln_dist_pt = new LinearAlgebraDistribution(rhs.distribution_pt());
1258  }
1259  solution.build(Hypre_distribution_pt,0.0);
1260  HYPRE_IJVectorGetValues(solution_ij,
1261  nrow_local,
1262  indices,
1263  solution.values_pt());
1264  solution.redistribute(soln_dist_pt);
1265  delete[] indices;
1266  delete soln_dist_pt;
1267 
1268  // output any error message
1269  if (Hypre_error_messages)
1270  {
1271  std::ostringstream message;
1272  int err = HypreHelpers::check_HYPRE_error_flag(message);
1273  if (err)
1274  {
1275  OomphLibWarning(message.str(),
1276  "HypreSolver::hypre_solve()",
1277  OOMPH_EXCEPTION_LOCATION);
1278  }
1279  }
1280 
1281  // deallocation
1282  HYPRE_IJVectorDestroy(solution_ij);
1283  HYPRE_IJVectorDestroy(rhs_ij);
1284 
1285  // Record time
1286  double solve_time=0;
1287  if (Output_info)
1288  {
1289  double t_end = TimingHelpers::timer();
1290  solve_time = t_end-t_start;
1291  }
1292 
1293  // output timings and info
1294  if (Output_info)
1295  {
1296  oomph_info << "Time for HYPRE solve [s] : "
1297  << solve_time << std::endl;
1298  }
1299 
1300  // for iterative solvers output iterations and final norm
1301  if ((Hypre_method>=CG) && (Hypre_method<=BoomerAMG))

1303  if (iterations>1)
1304  {

1306  << iterations << std::endl;

1308  << norm << std::endl;
1309  }
1310  }
