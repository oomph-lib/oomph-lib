48 // Set iterations to 1
49 hypre_preconditioner_pt->set_amg_iterations(1);

53

63 hypre_preconditioner_pt->amg_strength() = 0.25;
64
65 // Coarsening type
66 hypre_preconditioner_pt->amg_coarsening() = 0;
67 }

130 /// 0 = CLJP (parallel coarsening using independent sets)

178 //========================================================================
179 /// Helper function to create a HYPRE_IJVector and HYPRE_ParVector.
180 /// + If no MPI then serial vectors are created
181 /// + If MPI and serial input vector then distributed hypre vectors are
182 /// created
183 /// + If MPI and distributed input vector the distributed output vectors
184 /// are created.
185 //========================================================================
186 void create_HYPRE_Vector(const DoubleVector& oomph_vec,
187 const LinearAlgebraDistribution* dist_pt,
188 HYPRE_IJVector& hypre_ij_vector,
189 HYPRE_ParVector& hypre_par_vector)
190 {
191 // the lower and upper row of the vector on this processor
192 unsigned lower = dist_pt->first_row();
193 unsigned upper = dist_pt->first_row() + dist_pt->nrow_local() - 1;
194
195 // number of local rows
196 unsigned nrow_local = dist_pt->nrow_local();
197
198 // initialize Hypre vector
199#ifdef OOMPH_HAS_MPI
200 HYPRE_IJVectorCreate(
201 oomph_vec.distribution_pt()->communicator_pt()->mpi_comm(),
202 lower,
203 upper,
204 &hypre_ij_vector);
205#else
206 HYPRE_IJVectorCreate(MPI_COMM_WORLD, lower, upper, &hypre_ij_vector);
207#endif
208 HYPRE_IJVectorSetObjectType(hypre_ij_vector, HYPRE_PARCSR);
209 HYPRE_IJVectorInitialize(hypre_ij_vector);
210
211 // set up array containing indices
212 int* indices = new int[nrow_local];
213 double* values = new double[nrow_local];
214 const unsigned hypre_first_row = dist_pt->first_row();
215 unsigned j = 0;
216 if (!oomph_vec.distributed() && dist_pt->distributed())
217 {
218 j = hypre_first_row;
219 }
220 const double* o_pt = oomph_vec.values_pt();
221 for (unsigned i = 0; i < nrow_local; i++)
222 {
223 indices[i] = hypre_first_row + i;
224 values[i] = o_pt[j + i];
225 }
226
227 // insert values
228 HYPRE_IJVectorSetValues(hypre_ij_vector, nrow_local, indices, values);
229
230 // assemble vectors
231 HYPRE_IJVectorAssemble(hypre_ij_vector);
232 HYPRE_IJVectorGetObject(hypre_ij_vector, (void**)&hypre_par_vector);
233
234 // clean up
235 delete[] indices;
236 delete[] values;
237 }

272 //========================================================================
273 /// Helper function to create a serial HYPRE_IJMatrix and
274 /// HYPRE_ParCSRMatrix from a CRDoubleMatrix
275 /// NOTE: dist_pt is rebuilt to match the distribution of the hypre solver
276 /// which is not necassarily the same as the oomph lib matrix
277 //========================================================================
278 void create_HYPRE_Matrix(CRDoubleMatrix* oomph_matrix,
279 HYPRE_IJMatrix& hypre_ij_matrix,
280 HYPRE_ParCSRMatrix& hypre_par_matrix,
281 LinearAlgebraDistribution* dist_pt)
282 {
283#ifdef PARANOID
284 // check that the matrix is built
285 if (!oomph_matrix->built())
286 {
287 std::ostringstream error_message;
288 error_message << "The matrix has not been built";
289 throw OomphLibError(error_message.str(),
290 OOMPH_CURRENT_FUNCTION,
291 OOMPH_EXCEPTION_LOCATION);
292 }
293 // check the matrix is square
294 if (oomph_matrix->nrow() != oomph_matrix->ncol())
295 {
296 std::ostringstream error_message;
297 error_message << "create_HYPRE_Matrix require a square matrix. "
298 << "Matrix is " << oomph_matrix->nrow() << " by "
299 << oomph_matrix->ncol() << std::endl;
300 throw OomphLibError(error_message.str(),
301 OOMPH_CURRENT_FUNCTION,
302 OOMPH_EXCEPTION_LOCATION);
303 }
304#endif
305
306#ifdef OOMPH_HAS_MPI
307 // Trap the case when we have compiled with MPI,
308 // but are running in serial
309 if (!MPI_Helpers::mpi_has_been_initialised())
310 {
311 std::ostringstream error_stream;
312 error_stream
313 << "Oomph-lib has been compiled with MPI support and "
314 << "you are using HYPRE.\n"
315 << "For this combination of flags, MPI must be initialised.\n"
316 << "Call MPI_Helpers::init() in the "
317 << "main() function of your driver code\n";
318 throw OomphLibError(
319 error_stream.str(), OOMPH_CURRENT_FUNCTION, OOMPH_EXCEPTION_LOCATION);
320 }
321#endif
322
323 // find number of rows/columns
324 const unsigned nrow = int(oomph_matrix->nrow());
325
326 // get pointers to the matrix
327
328 // column indices of matrix
329 const int* matrix_cols = oomph_matrix->column_index();
330
331 // entries of matrix
332 const double* matrix_vals = oomph_matrix->value();
333
334 // row starts
335 const int* matrix_row_start = oomph_matrix->row_start();
336
337 // build the distribution
338 if (oomph_matrix->distribution_pt()->distributed())
339 {
340 dist_pt->build(oomph_matrix->distribution_pt());
341 }
342 else
343 {
344 bool distributed = true;
345 if (oomph_matrix->distribution_pt()->communicator_pt()->nproc() == 1)
346 {
347 distributed = false;
348 }
349 dist_pt->build(oomph_matrix->distribution_pt()->communicator_pt(),
350 nrow,
351 distributed);
352 }
353
354 // initialize hypre matrix
355 unsigned lower = dist_pt->first_row();
356 unsigned upper = lower + dist_pt->nrow_local() - 1;
357
358#ifdef OOMPH_HAS_MPI
359 int i_err = 0;
360 i_err = HYPRE_IJMatrixCreate(dist_pt->communicator_pt()->mpi_comm(),
361 lower,
362 upper,
363 lower,
364 upper,
365 &hypre_ij_matrix);
366 if (i_err)
367 {
368 throw OomphLibError("Error in HYPRE_IJMatrixCreate()",
369 OOMPH_CURRENT_FUNCTION,
370 OOMPH_EXCEPTION_LOCATION);
371 }
372#else
373 int i_err = 0;
374 i_err = HYPRE_IJMatrixCreate(
375 MPI_COMM_WORLD, lower, upper, lower, upper, &hypre_ij_matrix);
376 if (i_err)
377 {
378 throw OomphLibError("Error in HYPRE_IJMatrixCreate()",
379 OOMPH_CURRENT_FUNCTION,
380 OOMPH_EXCEPTION_LOCATION);
381 }
382#endif
383
384 i_err = HYPRE_IJMatrixSetObjectType(hypre_ij_matrix, HYPRE_PARCSR);
385 if (i_err)
386 {
387 throw OomphLibError("Error in HYPRE_IJMatrixSetObjectType()",
388 OOMPH_CURRENT_FUNCTION,
389 OOMPH_EXCEPTION_LOCATION);
390 }
391
392 i_err = HYPRE_IJMatrixInitialize(hypre_ij_matrix);
393 if (i_err)
394 {
395 throw OomphLibError("Error in HYPRE_IJMatrixInitialize()",
396 OOMPH_CURRENT_FUNCTION,
397 OOMPH_EXCEPTION_LOCATION);
398 }
399 // set up a row map
400 // and first row / nrow_local
401 const unsigned hypre_nrow_local = dist_pt->nrow_local();
402 const unsigned hypre_first_row = dist_pt->first_row();
403 int* ncols_per_row = new int[hypre_nrow_local];
404 int* row_map = new int[hypre_nrow_local];
405 for (unsigned i = 0; i < hypre_nrow_local; i++)
406 {
407 unsigned j = i;
408 if (!oomph_matrix->distributed() && dist_pt->distributed())
409 {
410 j += hypre_first_row;
411 }
412 ncols_per_row[i] = matrix_row_start[j + 1] - matrix_row_start[j];
413 row_map[i] = hypre_first_row + i;
414 }
415
416 // put values in HYPRE matrix
417 int local_start = 0;
418 if (!oomph_matrix->distributed() && dist_pt->distributed())
419 {
420 local_start += matrix_row_start[hypre_first_row];
421 }
422
423
424 HYPRE_IJMatrixSetValues(hypre_ij_matrix,
425 hypre_nrow_local,
426 ncols_per_row,
427 row_map,
428 matrix_cols + local_start,
429 matrix_vals + local_start);
430
431 // assemble matrix
432 HYPRE_IJMatrixAssemble(hypre_ij_matrix); // leak?
433 HYPRE_IJMatrixGetObject(hypre_ij_matrix, (void**)&hypre_par_matrix);
434
435 // tidy up memory
436 delete[] ncols_per_row;
437 delete[] row_map;
438 }

587 //=============================================================================
588 /// Sets up the solver data required for use in an oomph-lib
589 /// LinearSolver or Preconditioner, once the Hypre matrix has been
590 /// generated using hypre_matrix_setup(...).
591 //=============================================================================
592 void HypreInterface::hypre_solver_setup()
593 {
594 // Store time
595 double t_start = TimingHelpers::timer();
596 double t_end = 0;
597
598
599 // reset Hypre's global error flag
600 hypre__global_error.error_flag = 0;
601
602 // create dummy Hypre vectors which are required for setup
603 HYPRE_IJVector dummy_sol_ij;
604 HYPRE_ParVector dummy_sol_par;
605 HYPRE_IJVector dummy_rhs_ij;
606 HYPRE_ParVector dummy_rhs_par;
607 HypreHelpers::create_HYPRE_Vector(
608 Hypre_distribution_pt, dummy_sol_ij, dummy_sol_par);
609 HypreHelpers::create_HYPRE_Vector(
610 Hypre_distribution_pt, dummy_rhs_ij, dummy_rhs_par);
611
612 // Set up internal preconditioner for CG, GMRES or BiCGSTAB
613 // --------------------------------------------------------
614 if ((Hypre_method >= CG) && (Hypre_method <= BiCGStab))
615 {
616 // AMG preconditioner
617 if (Internal_preconditioner == BoomerAMG)
618 {
619 // set up BoomerAMG
620 HYPRE_BoomerAMGCreate(&Preconditioner);
621 HYPRE_BoomerAMGSetPrintLevel(Preconditioner, AMG_print_level);
622 HYPRE_BoomerAMGSetOldDefault(Preconditioner);
623 HYPRE_BoomerAMGSetMaxLevels(Preconditioner, AMG_max_levels);
624 HYPRE_BoomerAMGSetMaxIter(Preconditioner, 1);
625 HYPRE_BoomerAMGSetTol(Preconditioner, 0.0);
626 HYPRE_BoomerAMGSetCoarsenType(Preconditioner, AMG_coarsening);
627 HYPRE_BoomerAMGSetStrongThreshold(Preconditioner, AMG_strength);
628 HYPRE_BoomerAMGSetMaxRowSum(Preconditioner, AMG_max_row_sum);
629 HYPRE_BoomerAMGSetTruncFactor(Preconditioner, AMG_truncation);
630
631 if (AMG_using_simple_smoothing)
632 {
633 HYPRE_BoomerAMGSetRelaxType(Preconditioner, AMG_simple_smoother);
634
635 // New parameter. From HYPRE_parcsr_ls.h:
636 // (Optional) Defines in which order the points are relaxed. There
637 // are the following options for \e relax_order:
638 //
639 // - 0 : the points are relaxed in natural or lexicographic order
640 // on each processor
641 // - 1 : CF-relaxation is used, i.e on the fine grid and the down
642 // cycle the
643 // coarse points are relaxed first, followed by the fine
644 // points; on the up cycle the F-points are relaxed first,
645 // followed by the C-points. On the coarsest level, if an
646 // iterative scheme is used, the points are relaxed in
647 // lexicographic order.
648 //
649 // The default is 0.
650 // NOTE: Using option causes non-convergence in parallel mode.
651 HYPRE_BoomerAMGSetRelaxOrder(Preconditioner, 1);
652
653 // Number of smoother iterations
654 HYPRE_BoomerAMGSetNumSweeps(Preconditioner, AMG_smoother_iterations);
655
656
657 // Relax weight/damping for all levels
658 HYPRE_BoomerAMGSetRelaxWt(Preconditioner, AMG_damping);
659 }
660 else
661 {
662 HYPRE_BoomerAMGSetSmoothType(Preconditioner, AMG_complex_smoother);
663 HYPRE_BoomerAMGSetSmoothNumLevels(Preconditioner, AMG_max_levels);
664 HYPRE_BoomerAMGSetSmoothNumSweeps(Preconditioner,
665 AMG_smoother_iterations);
666
667 // If we are using Euclid then set up additional Euclid only options
668 if (AMG_complex_smoother == 9)
669 {
670 HypreHelpers::euclid_settings_helper(
671 AMGEuclidSmoother_use_block_jacobi,
672 AMGEuclidSmoother_use_row_scaling,
673 AMGEuclidSmoother_use_ilut,
674 AMGEuclidSmoother_level,
675 AMGEuclidSmoother_drop_tol,
676 AMGEuclidSmoother_print_level,
677 Preconditioner);
678 }
679 }
680
681 Existing_preconditioner = BoomerAMG;
682 }
683
684 // Euclid preconditioner
685 else if (Internal_preconditioner == Euclid)
686 {
687#ifdef OOMPH_HAS_MPI
688 HYPRE_EuclidCreate(Hypre_distribution_pt->communicator_pt()->mpi_comm(),
689 &Preconditioner);
690#else
691 HYPRE_EuclidCreate(MPI_COMM_WORLD, &Preconditioner);
692#endif
693
694 // Set parameters
695 HypreHelpers::euclid_settings_helper(Euclid_using_BJ,
696 Euclid_rowScale,
697 Euclid_using_ILUT,
698 Euclid_level,
699 Euclid_droptol,
700 Euclid_print_level,
701 Preconditioner);
702
703 Existing_preconditioner = Euclid;
704 }

1145 //===================================================================
1146 /// Helper function performs a solve if solver data has been set
1147 /// up using hypre_solver_setup(...).
1148 //====================================================================
1149 void HypreInterface::hypre_solve(const DoubleVector& rhs,
1150 DoubleVector& solution)
1151 {
1152 // Record time
1153 double t_start = TimingHelpers::timer();
1154
1155 // Set up hypre vectors
1156 // --------------------
1157
1158 // Hypre vector for rhs values
1159 HYPRE_IJVector rhs_ij;
1160 HYPRE_ParVector rhs_par;
1161
1162 // Hypre vector for solution
1163 HYPRE_IJVector solution_ij;
1164 HYPRE_ParVector solution_par;
1165
1166 // set up rhs_values and vec_indices
1167 HypreHelpers::create_HYPRE_Vector(
1168 rhs, Hypre_distribution_pt, rhs_ij, rhs_par);
1169
1170 HypreHelpers::create_HYPRE_Vector(
1171 Hypre_distribution_pt, solution_ij, solution_par);
1172
1173 // check error flag
1174 if (Hypre_error_messages)
1175 {
1176 std::ostringstream message;
1177 int err = HypreHelpers::check_HYPRE_error_flag(message);
1178 if (err)
1179 {
1180 OomphLibWarning(message.str(),
1181 "HypreSolver::hypre_solve()",
1182 OOMPH_EXCEPTION_LOCATION);
1183 }
1184 }
1185
1186 // solve
1187 // -----
1188
1189 // for solver stats
1190 int iterations = 0;
1191 double norm = 0;
1192
1193 // Get the norm of rhs
1194 const double rhs_norm = rhs.norm();
1195 bool do_solving = false;
1196 if (rhs_norm > 0.0)
1197 {
1198 do_solving = true;
1199 }
1200
1201#ifdef OOMPH_HAS_MPI
1202 // We need to check whether any processor requires to solve, if that
1203 // is the case then do the solving
1204 if (MPI_Helpers::mpi_has_been_initialised())
1205 {
1206 if (MPI_Helpers::communicator_pt()->nproc() > 1)
1207 {
1208 unsigned this_processor_do_solving = 0;
1209 unsigned all_processors_do_solving = 0;
1210 if (do_solving)
1211 {
1212 this_processor_do_solving = 1;
1213 }
1214 // Get the communicator
1215 OomphCommunicator* comm_pt = MPI_Helpers::communicator_pt();
1216 // Communicate with all procesoors
1217 MPI_Allreduce(&this_processor_do_solving,
1218 &all_processors_do_solving,
1219 1,
1220 MPI_UNSIGNED,
1221 MPI_SUM,
1222 comm_pt->mpi_comm());
1223 if (all_processors_do_solving > 0)
1224 {
1225 do_solving = true;
1226 }
1227 }
1228 }
1229#endif
1230
1231 if (do_solving)
1232 {
1233 if (Existing_solver == BoomerAMG)
1234 {
1235 HYPRE_BoomerAMGSolve(Solver, Matrix_par, rhs_par, solution_par);
1236 HYPRE_BoomerAMGGetNumIterations(Solver, &iterations);
1237 HYPRE_BoomerAMGGetFinalRelativeResidualNorm(Solver, &norm);
1238 }
1239 else if (Existing_solver == CG)
1240 {
1241 HYPRE_PCGSolve(Solver,
1242 (HYPRE_Matrix)Matrix_par,
1243 (HYPRE_Vector)rhs_par,
1244 (HYPRE_Vector)solution_par);
1245 HYPRE_PCGGetNumIterations(Solver, &iterations);
1246 HYPRE_PCGGetFinalRelativeResidualNorm(Solver, &norm);
1247 }
1248 else if (Existing_solver == GMRES)
1249 {
1250 HYPRE_GMRESSolve(Solver,
1251 (HYPRE_Matrix)Matrix_par,
1252 (HYPRE_Vector)rhs_par,
1253 (HYPRE_Vector)solution_par);
1254 HYPRE_GMRESGetNumIterations(Solver, &iterations);
1255 HYPRE_GMRESGetFinalRelativeResidualNorm(Solver, &norm);
1256 }
1257 else if (Existing_solver == BiCGStab)
1258 {
1259 HYPRE_BiCGSTABSolve(Solver,
1260 (HYPRE_Matrix)Matrix_par,
1261 (HYPRE_Vector)rhs_par,
1262 (HYPRE_Vector)solution_par);
1263 HYPRE_BiCGSTABGetNumIterations(Solver, &iterations);
1264 HYPRE_BiCGSTABGetFinalRelativeResidualNorm(Solver, &norm);
1265 }
1266 else if (Existing_solver == Euclid)
1267 {
1268 HYPRE_EuclidSolve(Solver, Matrix_par, rhs_par, solution_par);
1269 }
1270 else if (Existing_solver == ParaSails)
1271 {
1272 HYPRE_ParaSailsSolve(Solver, Matrix_par, rhs_par, solution_par);
1273 }
1274
1275 // output any error message
1276 if (Hypre_error_messages)
1277 {
1278 std::ostringstream message;
1279 int err = HypreHelpers::check_HYPRE_error_flag(message);
1280 if (err)
1281 {
1282 OomphLibWarning(message.str(),
1283 "HypreSolver::hypre_solve()",
1284 OOMPH_EXCEPTION_LOCATION);
1285 }
1286 }
1287
1288 } // if (do_solving)
1289
1290 // Copy result to solution
1291 unsigned nrow_local = Hypre_distribution_pt->nrow_local();
1292 unsigned first_row = Hypre_distribution_pt->first_row();
1293 int* indices = new int[nrow_local];
1294 for (unsigned i = 0; i < nrow_local; i++)
1295 {
1296 indices[i] = first_row + i;
1297 }
1298 LinearAlgebraDistribution* soln_dist_pt;
1299 if (solution.built())
1300 {
1301 soln_dist_pt = new LinearAlgebraDistribution(solution.distribution_pt());
1302 }
1303 else
1304 {
1305 soln_dist_pt = new LinearAlgebraDistribution(rhs.distribution_pt());
1306 }
1307 solution.build(Hypre_distribution_pt, 0.0);
1308 HYPRE_IJVectorGetValues(
1309 solution_ij, nrow_local, indices, solution.values_pt());
1310 solution.redistribute(soln_dist_pt);
1311 delete[] indices;
1312 delete soln_dist_pt;
1313
1314 // output any error message
1315 if (Hypre_error_messages)
1316 {
1317 std::ostringstream message;
1318 int err = HypreHelpers::check_HYPRE_error_flag(message);
1319 if (err)
1320 {
1321 OomphLibWarning(message.str(),
1322 "HypreSolver::hypre_solve()",
1323 OOMPH_EXCEPTION_LOCATION);
1324 }
1325 }
1326
1327 // deallocation
1328 HYPRE_IJVectorDestroy(solution_ij);
1329 HYPRE_IJVectorDestroy(rhs_ij);
1330
1331 // Record time
1332 double solve_time = 0;
1333 if (Output_info)
1334 {
1335 double t_end = TimingHelpers::timer();
1336 solve_time = t_end - t_start;
1337 }
1338
1339 // output timings and info
1340 if (Output_info)
1341 {
1342 oomph_info << "Time for HYPRE solve [s] : " << solve_time << std::endl;
1343 }
1344
1345 // for iterative solvers output iterations and final norm
1346 if ((Hypre_method >= CG) && (Hypre_method <= BoomerAMG))
1347 {
1348 if (iterations > 1)
1349 {
1350 if (Output_info)
1351 oomph_info << "Number of iterations : " << iterations
1352 << std::endl;
1353 if (Output_info)
1354 oomph_info << "Final Relative Residual Norm : " << norm << std::endl;
1355 }
1356 }
1357 }
