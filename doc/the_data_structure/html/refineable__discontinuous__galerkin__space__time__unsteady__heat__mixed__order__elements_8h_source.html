84  /// \short Get 'flux' for Z2 error recovery:
85  /// Different to the get_flux function in the base class as we also
86  /// have to include du/dt as we're doing temporal adaptivity too
87  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)
88  {
89  // Find out how many nodes there are in the element
90  unsigned n_node = nnode();
91 
92  // Find the index at which the variable is stored
93  unsigned u_nodal_index = this->u_index_ust_heat();
94 
95  // Set up memory for the shape and test functions
96  Shape psi(n_node);
97 
98  // Set up memory for the derivatives of the shape and test functions
99  DShape dpsidx(n_node, SPATIAL_DIM + 1);
100 
101  // Call the derivatives of the shape and test functions
102  dshape_eulerian(s, psi, dpsidx);
103 
104  // Loop over the entries of the flux vector
105  for (unsigned j = 0; j < SPATIAL_DIM + 1; j++)
106  {
107  // Initialise j-th flux entry to zero
108  flux[j] = 0.0;
109  }
110 
111  // Loop over nodes
112  for (unsigned l = 0; l < n_node; l++)
113  {
114  // Loop over derivative directions
115  for (unsigned j = 0; j < SPATIAL_DIM + 1; j++)
116  {
117  // Update the flux value
118  flux[j] += this->nodal_value(l, u_nodal_index) * dpsidx(l, j);
119  }
120  } // for (unsigned l=0;l<n_node;l++)
121  } // End of get_Z2_flux

124  /// \short Get the function value u in Vector.
125  /// Note: Given the generality of the interface (this function is usually
126  /// called from black-box documentation or interpolation routines), the
127  /// values Vector sets its own size in here.
128  void get_interpolated_values(const Vector<double>& s,
129  Vector<double>& values)
130  {
131  // Set the size of the vector u
132  values.resize(1);
133 
134  // Find the number of nodes
135  unsigned n_node = nnode();
136 
137  // Find the nodal index at which the unknown is stored
138  unsigned u_nodal_index = this->u_index_ust_heat();
139 
140  // Local shape function
141  Shape psi(n_node);
142 
143  // Find values of shape function
144  shape(s, psi);
145 
146  // Initialise the value of u
147  values[0] = 0.0;
148 
149  // Loop over the local nodes and sum
150  for (unsigned l = 0; l < n_node; l++)
151  {
152  // Update the solution value
153  values[0] += this->nodal_value(l, u_nodal_index) * psi[l];
154  }
155  } // End of get_interpolated_values
