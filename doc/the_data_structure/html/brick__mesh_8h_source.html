111  /// \short Compute vector of FE interpolated local coordinate in tet,
112  /// s_tet, evaluated at local coordinate s in current element.
113  void interpolated_s_tet(const Vector<double> &s, Vector<double>& s_tet) const
114  {
115  //Find number of nodes
116  unsigned n_node = nnode();
117 
118  //Local shape function
119  Shape psi(n_node);
120 
121  //Find values of shape function
122  shape(s,psi);
123 
124  for (unsigned i=0;i<3;i++)
125  {
126  //Initialise value of u
127  s_tet[i] = 0.0;
128 
129  //Loop over the local nodes and sum
130  for(unsigned l=0;l<n_node;l++)
131  {
132  s_tet[i] += nodal_value(l,i)*psi[l];
133  }
134  }
135  }
136
