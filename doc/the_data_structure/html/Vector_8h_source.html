52 /// we can include "graceful" array range checks if the RANGE_CHECKING

283  /// \short Probably not always best/fastest because not optimised for
284  /// dimension but useful...
285  inline double dot(const Vector<double>& a, const Vector<double>& b)
286  {
287  check_lengths_match(a,b);
288  double temp = 0;

313  inline void cross(const Vector<double>& A, const Vector<double>& B,

333  }
