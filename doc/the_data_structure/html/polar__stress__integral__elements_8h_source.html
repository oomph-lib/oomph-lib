200 /// Function that returns the shear stress
201 //============================================================================
202 template<class ELEMENT>
203 double PolarStressIntegralElement<ELEMENT>::
204 get_shear_stress()
205 {
206  //Storage for shear stress
207  double dudphi,shear_contribution=0.0;
208 
209  //Set the value of n_intpt
210  unsigned n_intpt = integral_pt()->nweight();
211 
212  //Storage for local coordinate
213  Vector<double> s_local(1);
214  //Storage for local coordinate in bulk
215  Vector<double> s_bulk(2);
216 
217  //Find out how many nodes there are
218  unsigned n_node = nnode();
219 
220  //Set up memory for the shape and test functions
221  Shape psif(n_node), testf(n_node);
222 
223  //Loop over the integration points
224  for(unsigned ipt=0;ipt<n_intpt;ipt++)
225  {
226  //Get the integral weight
227  double w = integral_pt()->weight(ipt);
228 
229  //Find the shape and test functions and return the Jacobian
230  //of the mapping
231  double J = shape_and_test_at_knot(ipt,psif,testf);
232 
233  //Premultiply the weights and the Jacobian
234  double W = w*J;
235 
236  //Need to find position to feed into Traction function
237  Vector<double> interpolated_x(Dim);
238 
239  //Initialise to zero
240  for(unsigned i=0;i<Dim;i++)
241  {
242  interpolated_x[i] = 0.0;
243  }
244 
245  //Calculate velocities and derivatives
246  for(unsigned l=0;l<n_node;l++)
247  {
248  //Loop over velocity components
249  for(unsigned i=0;i<Dim;i++)
250  {
251  interpolated_x[i] += nodal_position(l,i)*psif[l];
252  }
253  }
254 
255  //Get the local coordinate
256  s_local[0] = integral_pt()->knot(ipt,0);
257 
258  //Get bulk coordinate
259  s_bulk = this->local_coordinate_in_bulk(s_local);
260 
261  // Upcast from GeneralisedElement to the present element
262  ELEMENT* el_pt = dynamic_cast<ELEMENT*>(this->Bulk_element_pt);
263 
264  //Get du_dphi from bulk element
265  dudphi = el_pt->interpolated_dudx_pnst(s_bulk,0,1);
266 
267  //The contribution to the unweighted shear stress
268  shear_contribution += dudphi*W;
269  //dudphi*interpolated_x[0]*W;
270 
271  } //End of loop over integration points
272 
273  return shear_contribution;
274 
275 } //End of get_shear_stress
