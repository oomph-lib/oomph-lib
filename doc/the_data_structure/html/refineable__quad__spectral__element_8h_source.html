85  /// The only thing to add is rebuild from sons
86  void rebuild_from_sons(Mesh* &mesh_pt)
87  {
88  //The timestepper should be the same for all nodes and node 0 should
89  //never be deleted.
90  if(this->node_pt(0)==0)
91  {
92  throw OomphLibError("The Corner node (0) does not exist",
93  OOMPH_CURRENT_FUNCTION,
94  OOMPH_EXCEPTION_LOCATION);
95  }
96 
97  TimeStepper* time_stepper_pt = this->node_pt(0)->time_stepper_pt();
98  unsigned ntstorage = time_stepper_pt->ntstorage();
99 
100  unsigned jnod=0;
101  Vector<double> s_fraction(2), s(2);
102  //Loop over the nodes in the element
103  unsigned n_p = this->nnode_1d();
104  for(unsigned i0=0;i0<n_p;i0++)
105  {
106  //Get the fractional position of the node
107  s_fraction[0] = this->local_one_d_fraction_of_node(i0,0);
108  //Local coordinate
109  s[0] = -1.0 + 2.0*s_fraction[0];
110 
111  for(unsigned i1=0;i1<n_p;i1++)
112  {
113  //Get the fractional position of the node in the direction of s[1]
114  s_fraction[1] = this->local_one_d_fraction_of_node(i1,1);
115  // Local coordinate in father element
116  s[1] = -1.0 + 2.0*s_fraction[1];
117 
118  //Set the local node number
119  jnod = i0 + n_p*i1;
120 
121  //If the node has not been built
122  if(this->node_pt(jnod)==0)
123  {
124  //Has the node been created by one of its neighbours
125  bool is_periodic = false;
126  Node* created_node_pt =
127  this->node_created_by_neighbour(s_fraction,is_periodic);
128 
129  //If it has set the pointer
130  if(created_node_pt!=0)
131  {
132  //If the node is periodic
133  if(is_periodic)
134  {
135  throw OomphLibError(
136  "Cannot handle periodic nodes in refineable spectral elements yet",
137  OOMPH_CURRENT_FUNCTION,
138  OOMPH_EXCEPTION_LOCATION);
139  }
140  //Non-periodic case, just set the pointer
141  else
142  {
143  this->node_pt(jnod) = created_node_pt;
144  }
145  }
146  //Otherwise, we need to build it
147  else
148  {
149  //First, find the son element in which the node should live
150 
151  //Find coordinates in the sons
152  Vector<double> s_in_son(2);
153  using namespace QuadTreeNames;
154  int son=-10;
155  //If negative on the west side
156  if(s_fraction[0] < 0.5)
157  {
158  //On the south side
159  if(s_fraction[1] < 0.5)
160  {
161  //It's the southwest son
162  son = SW;
163  s_in_son[0] = -1.0 + 4.0*s_fraction[0];
164  s_in_son[1] = -1.0 + 4.0*s_fraction[1];
165  }
166  //On the north side
167  else
168  {
169  //It's the northwest son
170  son = NW;
171  s_in_son[0] = -1.0 + 4.0*s_fraction[0];
172  s_in_son[1] = -1.0 + 4.0*(s_fraction[1]-0.5);
173  }
174  }
175  else
176  {
177  //On the south side
178  if(s_fraction[1] < 0.5)
179  {
180  //It's the southeast son
181  son = SE;
182  s_in_son[0] = -1.0 + 4.0*(s_fraction[0]-0.5);
183  s_in_son[1] = -1.0 + 4.0*s_fraction[1];
184  }
185  //On the north side
186  else
187  {
188  //It's the northeast son
189  son = NE;
190  s_in_son[0] = -1.0 + 4.0*(s_fraction[0]-0.5);
191  s_in_son[1] = -1.0 + 4.0*(s_fraction[1]-0.5);
192  }
193  }
194 
195  //Get the pointer to the son element in which the new node
196  //would live
197  RefineableQSpectralElement<2>* son_el_pt =
198  dynamic_cast<RefineableQSpectralElement<2>*>(
199  this->tree_pt()->son_pt(son)->object_pt());
200 
201  //If we are rebuilding, then worry about the boundary conditions
202  //Find the boundary of the node
203  //Initially none
204  int boundary=Tree::OMEGA;
205  //If we are on the western boundary
206  if(i0==0) {boundary = W;}
207  //If we are on the eastern boundary
208  else if(i0==n_p-1) {boundary = E;}
209 
210  //If we are on the southern boundary
211  if(i1==0)
212  {
213  //If we already have already set the boundary, we're on a corner
214  switch(boundary)
215  {
216  case W:
217  boundary = SW;
218  break;
219  case E:
220  boundary = SE;
221  break;
222  //Boundary not set
223  default:
224  boundary = S;
225  break;
226  }
227  }
228  //If we are the northern bounadry
229  else if(i1==n_p-1)
230  {
231  //If we already have a boundary
232  switch(boundary)
233  {
234  case W:
235  boundary = NW;
236  break;
237  case E:
238  boundary = NE;
239  break;
240  default:
241  boundary = N;
242  break;
243  }
244  }
245 
246  // set of boundaries that this edge in the son lives on
247  std::set<unsigned> boundaries;
248 
249  //Now get the boundary conditions from the son
250  //The boundaries will be common to the son because there can be
251  //no rotations here
252  if(boundary!=Tree::OMEGA)
253  {
254  son_el_pt->get_boundaries(boundary,boundaries);
255  }
256 
257  // If the node lives on a boundary:
258  // Construct a boundary node,
259  // Get boundary conditions and
260  // update all lookup schemes
261  if(boundaries.size()>0)
262  {
263  //Construct the new node
264  this->node_pt(jnod) =
265  this->construct_boundary_node(jnod,time_stepper_pt);
266 
267  //Get the boundary conditions from the son
268  Vector<int> bound_cons(ncont_interpolated_values());
269  son_el_pt->get_bcs(boundary,bound_cons);
270 
271  //Loop over the values and pin if necessary
272  unsigned nval = this->node_pt(jnod)->nvalue();
273  for(unsigned k=0;k<nval;k++)
274  {
275  if(bound_cons[k]) {this->node_pt(jnod)->pin(k);}
276  }
277 
278  // Solid node? If so, deal with the positional boundary
279  // conditions:
280  SolidNode* solid_node_pt =
281  dynamic_cast<SolidNode*>(this->node_pt(jnod));
282  if (solid_node_pt!=0)
283  {
284  //Get the positional boundary conditions from the father:
285  unsigned n_dim = this->node_pt(jnod)->ndim();
286  Vector<int> solid_bound_cons(n_dim);
287  RefineableSolidQElement<2>* son_solid_el_pt=
288  dynamic_cast<RefineableSolidQElement<2>*>(son_el_pt);
289 #ifdef PARANOID
290  if (son_solid_el_pt==0)
291  {
292  std::string error_message =
293  "We have a SolidNode outside a refineable SolidElement\n";
294  error_message +=
295  "during mesh refinement -- this doesn't make sense\n";
296 
297  throw OomphLibError(
298  error_message,
299  OOMPH_CURRENT_FUNCTION,
300  OOMPH_EXCEPTION_LOCATION);
301  }
302 #endif
303  son_solid_el_pt->get_solid_bcs(boundary,solid_bound_cons);
304 
305  //Loop over the positions and pin, if necessary
306  for(unsigned k=0;k<n_dim;k++)
307  {
308  if (solid_bound_cons[k]) {solid_node_pt->pin_position(k);}
309  }
310  }
311 
312  //Next we update the boundary look-up schemes
313  //Loop over the boundaries stored in the set
314  for(std::set<unsigned>::iterator it = boundaries.begin();
315  it != boundaries.end(); ++it)
316  {
317  //Add the node to the boundary
318  mesh_pt->add_boundary_node(*it,this->node_pt(jnod));
319 
320  //If we have set an intrinsic coordinate on this
321  //mesh boundary then it must also be interpolated on
322  //the new node
323  //Now interpolate the intrinsic boundary coordinate
324  if(mesh_pt->boundary_coordinate_exists(*it)==true)
325  {
326  Vector<double> zeta(1);
327  son_el_pt->interpolated_zeta_on_edge(*it,boundary,
328  s_in_son,zeta);
329 
330  this->node_pt(jnod)
331  ->set_coordinates_on_boundary(*it,zeta);
332  }
333  }
334  }
335  //Otherwise the node is not on a Mesh boundary
336  //and we create a normal "bulk" node
337  else
338  {
339  //Construct the new node
340  this->node_pt(jnod) = this->construct_node(jnod,time_stepper_pt);
341  }
342 
343  //Now we set the position and values at the newly created node
344 
345  // In the first instance use macro element or FE representation
346  // to create past and present nodal positions.
347  // (THIS STEP SHOULD NOT BE SKIPPED FOR ALGEBRAIC
348  // ELEMENTS AS NOT ALL OF THEM NECESSARILY IMPLEMENT
349  // NONTRIVIAL NODE UPDATE FUNCTIONS. CALLING
350  // THE NODE UPDATE FOR SUCH ELEMENTS/NODES WILL LEAVE
351  // THEIR NODAL POSITIONS WHERE THEY WERE (THIS IS APPROPRIATE
352  // ONCE THEY HAVE BEEN GIVEN POSITIONS) BUT WILL
353  // NOT ASSIGN SENSIBLE INITIAL POSITONS!
354 
355  // Loop over # of history values
356  //Loop over # of history values
357  for(unsigned t=0;t<ntstorage;t++)
358  {
359  using namespace QuadTreeNames;
360  //Get the position from the son
361  Vector<double> x_prev(2);
362 
363  //Now let's fill in the value
364  son_el_pt->get_x(t,s_in_son,x_prev);
365  for(unsigned i=0;i<2;i++)
366  {
367  this->node_pt(jnod)->x(t,i) = x_prev[i];
368  }
369  }
370 
371  // Now set up the values
372  // Loop over all history values
373  for(unsigned t=0;t<ntstorage;t++)
374  {
375  // Get values from father element
376  // Note: get_interpolated_values() sets Vector size itself.
377  Vector<double> prev_values;
378  son_el_pt->get_interpolated_values(t,s_in_son,prev_values);
379 
380  //Initialise the values at the new node
381  for(unsigned k=0;k<this->node_pt(jnod)->nvalue();k++)
382  {
383  this->node_pt(jnod)->set_value(t,k,prev_values[k]);
384  }
385  }
386 
387  //Add the node to the mesh
388  mesh_pt->add_node_pt(this->node_pt(jnod));
389 
390  } //End of the case when we build the node ourselvesx
391 
392  //Algebraic stuff here
393  //Check whether the element is an algebraic element
394  AlgebraicElementBase* alg_el_pt =
395  dynamic_cast<AlgebraicElementBase*>(this);
396 
397  //If we do have an algebraic element
398  if(alg_el_pt!=0)
399  {
400  std::string error_message =
401  "Have not implemented rebuilding from sons for";
402  error_message +=
403  "Algebraic Spectral elements yet\n";
404 
405  throw
406  OomphLibError(error_message,
407  "RefineableQSpectralElement::rebuild_from_sons()",
408  OOMPH_EXCEPTION_LOCATION);
409  }
410 
411  }
412  }
413  }
414  }
