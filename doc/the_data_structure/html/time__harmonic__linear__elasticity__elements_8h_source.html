55 /// A base class for elements that solve the equations of time-harmonic linear
56 /// elasticity in Cartesian coordinates.
57 /// Combines a few generic functions that are shared by
58 /// TimeHarmonicLinearElasticityEquations

67  /// \short Return the index at which the i-th real or imag unknown
68  /// displacement component is stored. The default value is appropriate for
69  /// single-physics problems:
70  virtual inline std::complex<unsigned>
71  u_index_time_harmonic_linear_elasticity(const unsigned i) const
72  {
73  return std::complex<unsigned>(i,i+DIM);
74  }

76  /// Compute vector of FE interpolated displacement u at local coordinate s
77  void interpolated_u_time_harmonic_linear_elasticity(
78  const Vector<double> &s,
79  Vector<std::complex<double> >& disp)
80  const
81  {
82  //Find number of nodes
83  unsigned n_node = nnode();
84 
85  //Local shape function
86  Shape psi(n_node);
87 
88  //Find values of shape function
89  shape(s,psi);
90 
91  for (unsigned i=0;i<DIM;i++)
92  {
93  //Index at which the nodal value is stored
94  std::complex<unsigned> u_nodal_index =
95  u_index_time_harmonic_linear_elasticity(i);
96 
97  //Initialise value of u
98  disp[i] = std::complex<double>(0.0,0.0);
99 
100  //Loop over the local nodes and sum
101  for(unsigned l=0;l<n_node;l++)
102  {
103  const std::complex<double> u_value(
104  nodal_value(l,u_nodal_index.real()),
105  nodal_value(l,u_nodal_index.imag()));
106 
107  disp[i] += u_value*psi[l];
108  }
109  }
110  }
111 
112  /// Return FE interpolated displacement u[i] at local coordinate s
113  std::complex<double> interpolated_u_time_harmonic_linear_elasticity(
114  const Vector<double> &s,
115  const unsigned &i) const
116  {
117  //Find number of nodes
118  unsigned n_node = nnode();
119 
120  //Local shape function
121  Shape psi(n_node);
122 
123  //Find values of shape function
124  shape(s,psi);
125 
126  //Get nodal index at which i-th velocity is stored
127  std::complex<unsigned> u_nodal_index =
128  u_index_time_harmonic_linear_elasticity(i);
129 
130  //Initialise value of u
131  std::complex<double> interpolated_u(0.0,0.0);
132 
133  //Loop over the local nodes and sum
134  for(unsigned l=0;l<n_node;l++)
135  {
136  const std::complex<double> u_value(
137  nodal_value(l,u_nodal_index.real()),
138  nodal_value(l,u_nodal_index.imag()));
139 
140  interpolated_u += u_value*psi[l];
141  }
142 
143  return(interpolated_u);
144  }

194  /// \short Evaluate body force at Eulerian coordinate x at present time
195  /// (returns zero vector if no body force function pointer has been set)
196  inline void body_force(const Vector<double>& x,
197  Vector<std::complex<double> >& b) const
198  {
199  //If no function has been set, return zero vector
200  if(Body_force_fct_pt==0)
201  {
202  // Get spatial dimension of element
203  unsigned n=dim();
204  for (unsigned i=0;i<n;i++)
205  {
206  b[i] = std::complex<double>(0.0,0.0);
207  }
208  }
209  else
210  {
211  // Get time from timestepper of first node (note that this must
212  // work -- body force only makes sense for elements that can be
213  // deformed and thefore store displacements (at their nodes)
214  double time=node_pt(0)->time_stepper_pt()->time_pt()->time();
215 
216  // Get body force
217  (*Body_force_fct_pt)(time,x,b);
218  }
219  }

221  /// \short The number of "DOF types" that degrees of freedom in this element
222  /// are sub-divided into: for now lump them all into one DOF.
223  /// Can be adjusted later
224  unsigned ndof_types() const
225  {
226  return 1;
227  }
228 
229  /// \short Create a list of pairs for all unknowns in this element,
230  /// so that the first entry in each pair contains the global equation
231  /// number of the unknown, while the second one contains the number
232  /// of the "DOF type" that this unknown is associated with.
233  /// (Function can obviously only be called if the equation numbering
234  /// scheme has been set up.)
235  void get_dof_numbers_for_unknowns(
236  std::list<std::pair<unsigned long,unsigned> >& dof_lookup_list) const
237  {
238 
239  // temporary pair (used to store DOF lookup prior to being added
240  // to list)
241  std::pair<unsigned long,unsigned> dof_lookup;
242 
243  // number of nodes
244  const unsigned n_node = this->nnode();
245 
246  //Integer storage for local unknown
247  int local_unknown=0;
248 
249  //Loop over the nodes
250  for(unsigned n=0;n<n_node;n++)
251  {
252  //Loop over dimension (real and imag)
253  for(unsigned i=0;i<2*DIM;i++)
254  {
255  //If the variable is free
256  local_unknown = nodal_local_eqn(n,i);
257 
258  // ignore pinned values
259  if (local_unknown >= 0)
260  {
261  // store DOF lookup in temporary pair: First entry in pair
262  // is global equation number; second entry is DOF type
263  dof_lookup.first = this->eqn_number(local_unknown);
264  dof_lookup.second = 0;
265 
266  // add to list
267  dof_lookup_list.push_front(dof_lookup);
268 
269  }
270  }
271  }
272  }

287  static double Default_omega_sq_value;

501 /// Time-harmonic linear elasticity upgraded to become projectable
502 //==========================================================
503  template<class TIME_HARMONIC_LINEAR_ELAST_ELEMENT>
504  class ProjectableTimeHarmonicLinearElasticityElement :
505  public virtual ProjectableElement<TIME_HARMONIC_LINEAR_ELAST_ELEMENT>
506  {
507 
508  public:
509 
510  /// \short Constructor [this was only required explicitly
511  /// from gcc 4.5.2 onwards...]
512  ProjectableTimeHarmonicLinearElasticityElement(){}
513 
514 
515  /// \short Specify the values associated with field fld.
516  /// The information is returned in a vector of pairs which comprise
517  /// the Data object and the value within it, that correspond to field fld.
518  /// In the underlying time-harmonic linear elasticity elements the
519  /// real and complex parts of the displacements are stored
520  /// at the nodal values
521  Vector<std::pair<Data*,unsigned> > data_values_of_field(const unsigned& fld)
522  {
523  // Create the vector
524  Vector<std::pair<Data*,unsigned> > data_values;
525 
526  // Loop over all nodes and extract the fld-th nodal value
527  unsigned nnod=this->nnode();
528  for (unsigned j=0;j<nnod;j++)
529  {
530  // Add the data value associated with the displacement components
531  data_values.push_back(std::make_pair(this->node_pt(j),fld));
532  }
533 
534  // Return the vector
535  return data_values;
536 
537  }
538 
539  /// \short Number of fields to be projected: 2*dim, corresponding to
540  /// real and imag parts of the displacement components
541  unsigned nfields_for_projection()
542  {
543  return 2*this->dim();
544  }
545 
546  /// \short Number of history values to be stored for fld-th field.
547  /// (includes present value!)
548  unsigned nhistory_values_for_projection(const unsigned &fld)
549  {
550 #ifdef PARANOID
551  if (fld>3)
552  {
553  std::stringstream error_stream;
554  error_stream
555  << "Elements only store four fields so fld can't be"
556  << " " << fld << std::endl;
557  throw OomphLibError(
558  error_stream.str(),
559  OOMPH_CURRENT_FUNCTION,
560  OOMPH_EXCEPTION_LOCATION);
561  }
562 #endif
563  return this->node_pt(0)->ntstorage();
564  }
565 
566  ///\short Number of positional history values: Read out from
567  /// positional timestepper
568  /// (Note: count includes current value!)
569  unsigned nhistory_values_for_coordinate_projection()
570  {
571  return this->node_pt(0)->position_time_stepper_pt()->ntstorage();
572  }

574  /// \short Return Jacobian of mapping and shape functions of field fld
575  /// at local coordinate s
576  double jacobian_and_shape_of_field(const unsigned &fld,
577  const Vector<double> &s,
578  Shape &psi)
579  {
580  unsigned n_dim=this->dim();
581  unsigned n_node=this->nnode();
582  DShape dpsidx(n_node,n_dim);
583 
584  // Call the derivatives of the shape functions and return
585  // the Jacobian
586  return this->dshape_eulerian(s,psi,dpsidx);
587  }

589 
590 
591  /// \short Return interpolated field fld at local coordinate s, at time level
592  /// t (t=0: present; t>0: history values)
593  double get_field(const unsigned &t,
594  const unsigned &fld,
595  const Vector<double>& s)
596  {
597  unsigned n_node=this->nnode();
598 
599 #ifdef PARANOID
600  unsigned n_dim=this->node_pt(0)->ndim();
601 #endif
602 
603  //Local shape function
604  Shape psi(n_node);
605 
606  //Find values of shape function
607  this->shape(s,psi);
608 
609  //Initialise value of u
610  double interpolated_u = 0.0;
611 
612  //Sum over the local nodes at that time
613  for(unsigned l=0;l<n_node;l++)
614  {
615 #ifdef PARANOID
616  unsigned nvalue=this->node_pt(l)->nvalue();
617  if (nvalue!=2*n_dim)
618  {
619  std::stringstream error_stream;
620  error_stream
621  << "Current implementation only works for non-resized nodes\n"
622  << "but nvalue= " << nvalue << "!= 2 dim = " << 2*n_dim << std::endl;
623  throw OomphLibError(
624  error_stream.str(),
625  OOMPH_CURRENT_FUNCTION,
626  OOMPH_EXCEPTION_LOCATION);
627  }
628 #endif
629  interpolated_u += this->nodal_value(t,l,fld)*psi[l];
630  }
631  return interpolated_u;
632  }

637  {
638  return this->nnode();
639  }
