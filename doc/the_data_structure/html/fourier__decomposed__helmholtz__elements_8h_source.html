72  //=============================================================
73  /// A class for all isoparametric elements that solve the
74  /// Helmholtz equations.
75  /// \f[
76  /// \nabla^2 U + k^2 U = f
77  /// \f]
78  /// in Fourier decomposed form (cylindrical polars):
79  /// \f[
80  /// U(r,\varphi,z) = \Re( u^{(n)}(r,z) \exp(-i n \varphi))
81  /// \f]
82  /// We are solving for \f$ u^{(n)}(r,z)\f$ for given parameters
83  /// \f$ k^2 \f$ and \f$ n \f$ .
84  /// This contains the generic maths. Shape functions, geometric
85  /// mapping etc. must get implemented in derived class.
86  //=============================================================

289  /// Get flux: flux[i] = du/dx_i for real and imag part
290  void get_flux(const Vector<double>& s,
291  Vector<std::complex<double>>& flux) const
292  {
293  // Find out how many nodes there are in the element
294  const unsigned n_node = nnode();
295 
296  // Set up memory for the shape and test functions
297  Shape psi(n_node);
298  DShape dpsidx(n_node, 2);
299 
300  // Call the derivatives of the shape and test functions
301  dshape_eulerian(s, psi, dpsidx);
302 
303  // Initialise to zero
304  const std::complex<double> zero(0.0, 0.0);
305  for (unsigned j = 0; j < 2; j++)
306  {
307  flux[j] = zero;
308  }
309 
310  // Loop over nodes
311  for (unsigned l = 0; l < n_node; l++)
312  {
313  // Cache the complex value of the unknown
314  const std::complex<double> u_value(
315  this->nodal_value(l, u_index_fourier_decomposed_helmholtz().real()),
316  this->nodal_value(l, u_index_fourier_decomposed_helmholtz().imag()));
317 
318  // Loop over derivative directions
319  for (unsigned j = 0; j < 2; j++)
320  {
321  flux[j] += u_value * dpsidx(l, j);
322  }
323  }
324  }

348  /// \short Return FE representation of function value u(s)
349  /// at local coordinate s
350  inline std::complex<double> interpolated_u_fourier_decomposed_helmholtz(
351  const Vector<double>& s) const
352  {
353  // Find number of nodes
354  const unsigned n_node = nnode();
355 
356  // Local shape function
357  Shape psi(n_node);
358 
359  // Find values of shape function
360  shape(s, psi);
361 
362  // Initialise value of u
363  std::complex<double> interpolated_u(0.0, 0.0);
364 
365  // Get the index at which the helmholtz unknown is stored
366  const unsigned u_nodal_index_real =
367  u_index_fourier_decomposed_helmholtz().real();
368  const unsigned u_nodal_index_imag =
369  u_index_fourier_decomposed_helmholtz().imag();
370 
371  // Loop over the local nodes and sum
372  for (unsigned l = 0; l < n_node; l++)
373  {
374  // Make a temporary complex number from the stored data
375  const std::complex<double> u_value(
376  this->nodal_value(l, u_nodal_index_real),
377  this->nodal_value(l, u_nodal_index_imag));
378  // Add to the interpolated value
379  interpolated_u += u_value * psi[l];
380  }
381  return interpolated_u;
382  }
