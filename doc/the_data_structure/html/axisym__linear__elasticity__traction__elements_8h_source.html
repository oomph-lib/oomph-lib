315  interpolated_x(s,x);

331 /// Return the residuals for the
332 /// AxisymmetricLinearElasticityTractionElement equations
333 //=====================================================================
334 template<class ELEMENT>
335  void AxisymmetricLinearElasticityTractionElement<ELEMENT>::
336  fill_in_contribution_to_residuals_axisymmetric_linear_elasticity_traction(
337  Vector<double> &residuals)
338  {
339 
340  //Find out how many nodes there are
341  unsigned n_node = nnode();
342 
343  // Get continuous time from timestepper of first node
344  double time=node_pt(0)->time_stepper_pt()->time_pt()->time();
345 
346  #ifdef PARANOID
347  //Find out how many positional dofs there are
348  unsigned n_position_type = this->nnodal_position_type();
349  if(n_position_type != 1)
350  {
351  throw OomphLibError(
352  "AxisymmetricLinearElasticity is not yet implemented for more than one position type",
353  OOMPH_CURRENT_FUNCTION,
354  OOMPH_EXCEPTION_LOCATION);
355  }
356 #endif
357 
358  //Find out the dimension of the node
359  unsigned n_dim = this->nodal_dimension();
360 
361  //Cache the nodal indices at which the displacement components are stored
362  unsigned u_nodal_index[n_dim+1];
363  for(unsigned i=0;i<n_dim+1;i++)
364  {
365  u_nodal_index[i]=this->U_index_axisymmetric_linear_elasticity_traction[i];
366  }
367 
368  //Integer to hold the local equation number
369  int local_eqn=0;
370 
371  //Set up memory for the shape functions
372  //Note that in this case, the number of lagrangian coordinates is always
373  //equal to the dimension of the nodes
374  Shape psi(n_node);
375  DShape dpsids(n_node,n_dim-1);
376 
377  //Set the value of n_intpt
378  unsigned n_intpt = integral_pt()->nweight();
379 
380  //Loop over the integration points
381  for(unsigned ipt=0;ipt<n_intpt;ipt++)
382  {
383  //Get the integral weight
384  double w = integral_pt()->weight(ipt);
385 
386  //Only need to call the local derivatives
387  dshape_local_at_knot(ipt,psi,dpsids);
388 
389  //Calculate the Eulerian and Lagrangian coordinates
390  Vector<double> interpolated_x(n_dim,0.0);
391 
392  //Also calculate the surface Vectors (derivatives wrt local coordinates)
393  DenseMatrix<double> interpolated_A(n_dim-1,n_dim,0.0);
394 
395  //Calculate displacements and derivatives
396  for(unsigned l=0;l<n_node;l++)
397  {
398  //Loop over directions
399  for(unsigned i=0;i<n_dim;i++)
400  {
401  //Calculate the Eulerian coords
402  const double x_local = nodal_position(l,i);
403  interpolated_x[i] += x_local*psi(l);
404 
405  //Loop over LOCAL derivative directions, to calculate the tangent(s)
406  for(unsigned j=0;j<n_dim-1;j++)
407  {
408  interpolated_A(j,i) += x_local*dpsids(l,j);
409  }
410  }
411  }
412 
413  //Now find the local metric tensor from the tangent Vectors
414  DenseMatrix<double> A(n_dim-1);
415  for(unsigned i=0;i<n_dim-1;i++)
416  {
417  for(unsigned j=0;j<n_dim-1;j++)
418  {
419  //Initialise surface metric tensor to zero
420  A(i,j) = 0.0;
421 
422  //Take the dot product
423  for(unsigned k=0;k<n_dim;k++)
424  {
425  A(i,j) += interpolated_A(i,k)*interpolated_A(j,k);
426  }
427  }
428  }
429 
430  //Get the outer unit normal
431  Vector<double> interpolated_normal(n_dim);
432  outer_unit_normal(ipt,interpolated_normal);
433 
434  //Find the determinant of the metric tensor
435  double Adet =0.0;
436  switch(n_dim)
437  {
438  case 2:
439  Adet = A(0,0);
440  break;
441  case 3:
442  Adet = A(0,0)*A(1,1) - A(0,1)*A(1,0);
443  break;
444  default:
445  throw
446  OomphLibError(
447  "Wrong dimension in AxisymmetricLinearElasticityTractionElement",
448  "AxisymmetricLinearElasticityTractionElement::fill_in_contribution_to_residuals()",
449  OOMPH_EXCEPTION_LOCATION);
450  }
451 
452  //Premultiply the weights and the square-root of the determinant of
453  //the metric tensor
454  double W = w*sqrt(Adet);
455 
456  //Now calculate the load
457  Vector<double> traction(n_dim+1);
458  get_traction(time,
459  ipt,
460  interpolated_x,
461  interpolated_normal,
462  traction);
463 
464  //Loop over the test functions, nodes of the element
465  for(unsigned l=0;l<n_node;l++)
466  {
467  //Loop over the displacement components
468  for(unsigned i=0;i<n_dim+1;i++)
469  {
470  // Equation number
471  local_eqn = this->nodal_local_eqn(l,u_nodal_index[i]);
472  /*IF it's not a boundary condition*/
473  if(local_eqn >= 0)
474  {
475  //Add the loading terms to the residuals
476  residuals[local_eqn] -= traction[i]*psi(l)*interpolated_x[0]*W;
477  }
478  }
479  } //End of loop over shape functions
480  } //End of loop over integration points
481 
482  }

681 template <class ELASTICITY_BULK_ELEMENT, class NAVIER_STOKES_BULK_ELEMENT>
682  void FSIAxisymmetricLinearElasticityTractionElement<
683  ELASTICITY_BULK_ELEMENT, NAVIER_STOKES_BULK_ELEMENT>::
684  fill_in_contribution_to_residuals_axisym_fsi_traction(
685  Vector<double> &residuals)
686  {
687 
688  //Find out how many nodes there are
689  unsigned n_node = nnode();
690 
691 #ifdef PARANOID
692  //Find out how many positional dofs there are
693  unsigned n_position_type = this->nnodal_position_type();
694  if(n_position_type != 1)
695  {
696  throw OomphLibError(
697  "LinearElasticity is not yet implemented for more than one position type.",
698  OOMPH_CURRENT_FUNCTION,
699  OOMPH_EXCEPTION_LOCATION);
700  }
701 #endif
702 
703  //Find out the dimension of the node
704  unsigned n_dim = this->nodal_dimension();
705 
706  //Cache the nodal indices at which the displacement components are stored
707  unsigned u_nodal_index[n_dim+1];
708  for(unsigned i=0;i<n_dim+1;i++)
709  {
710  u_nodal_index[i] = this->U_index_axisym_fsi_traction[i];
711  }
712 
713  //Integer to hold the local equation number
714  int local_eqn=0;
715 
716  //Set up memory for the shape functions
717  Shape psi(n_node);
718  DShape dpsids(n_node,n_dim-1);
719 
720  // Get FSI parameter
721  const double q_value=q();
722 
723  // Storage for traction
724  Vector<double> traction(3);
725 
726  //Set the value of n_intpt
727  unsigned n_intpt = integral_pt()->nweight();
728 
729  //Loop over the integration points
730  for(unsigned ipt=0;ipt<n_intpt;ipt++)
731  {
732  //Get the integral weight
733  double w = integral_pt()->weight(ipt);
734 
735  //Only need to call the local derivatives
736  dshape_local_at_knot(ipt,psi,dpsids);
737 
738  //Calculate the coordinates
739  Vector<double> interpolated_x(n_dim,0.0);
740 
741  //Also calculate the surface tangent vectors
742  DenseMatrix<double> interpolated_A(n_dim-1,n_dim,0.0);
743 
744  //Calculate displacements and derivatives
745  for(unsigned l=0;l<n_node;l++)
746  {
747  //Loop over directions
748  for(unsigned i=0;i<n_dim;i++)
749  {
750  //Calculate the Eulerian coords
751  const double x_local = nodal_position(l,i);
752  interpolated_x[i] += x_local*psi(l);
753 
754  //Loop over LOCAL derivative directions, to calculate the tangent(s)
755  for(unsigned j=0;j<n_dim-1;j++)
756  {
757  interpolated_A(j,i) += x_local*dpsids(l,j);
758  }
759  }
760  }
761 
762  //Now find the local metric tensor from the tangent Vectors
763  DenseMatrix<double> A(n_dim-1);
764  for(unsigned i=0;i<n_dim-1;i++)
765  {
766  for(unsigned j=0;j<n_dim-1;j++)
767  {
768  //Initialise surface metric tensor to zero
769  A(i,j) = 0.0;
770 
771  //Take the dot product
772  for(unsigned k=0;k<n_dim;k++)
773  {
774  A(i,j) += interpolated_A(i,k)*interpolated_A(j,k);
775  }
776  }
777  }
778 
779  //Get the outer unit normal
780  Vector<double> interpolated_normal(n_dim);
781  outer_unit_normal(ipt,interpolated_normal);
782 
783  //Find the determinant of the metric tensor
784  double Adet =0.0;
785  switch(n_dim)
786  {
787  case 2:
788  Adet = A(0,0);
789  break;
790  case 3:
791  Adet = A(0,0)*A(1,1) - A(0,1)*A(1,0);
792  break;
793  default:
794  throw
795  OomphLibError(
796  "Wrong dimension in TimeHarmonicLinElastLoadedByPressureElement",
797  "TimeHarmonicLinElastLoadedByPressureElement::fill_in_contribution_to_residuals()",
798  OOMPH_EXCEPTION_LOCATION);
799  }
800 
801  //Premultiply the weights and the square-root of the determinant of
802  //the metric tensor
803  double W = w*sqrt(Adet);
804 
805  // Get bulk element for traction
806  NAVIER_STOKES_BULK_ELEMENT* ext_el_pt=
807  dynamic_cast<NAVIER_STOKES_BULK_ELEMENT*>
808  (this->external_element_pt(0,ipt));
809  Vector<double> s_ext(this->external_element_local_coord(0,ipt));
810 
811  // Get traction from bulk element
812  ext_el_pt->traction(s_ext,
813  interpolated_normal,
814  traction);
815 
816  //Loop over the test functions, nodes of the element
817  for(unsigned l=0;l<n_node;l++)
818  {
819  //Loop over the displacement components
820  for(unsigned i=0;i<n_dim+1;i++)
821  {
822  // Equation number
823  local_eqn = this->nodal_local_eqn(l,u_nodal_index[i]);
824  /*IF it's not a boundary condition*/
825  if(local_eqn >= 0)
826  {
827  //Add the loading terms (multiplied by fsi scaling factor)
828  //to the residuals
829  residuals[local_eqn] -=
830  q_value*traction[i]*psi(l)*interpolated_x[0]*W;
831  }
832  }
833  } //End of loop over shape functions
834 
835  } //End of loop over integration points
836  }

845 
846 
847 }
