142 //========================================================================
143 /// The OomphLibException constructor takes the error description,
144 /// function name, a location string provided by the
145 /// OOMPH_EXCEPTION_LOCATION and an exception type "WARNING" or "ERROR"
146 /// and combines them into a standard error message that is written to the
147 /// exception stream. The output_width of the message can also be specified.
148 /// Optionally provide a traceback of the function calls.
149 //==========================================================================
150 OomphLibException::OomphLibException(const std::string &error_description,
151  const std::string &function_name,
152  const char *location,
153  const std::string &exception_type,
154  std::ostream &exception_stream,
155  const unsigned &output_width,
156  bool list_trace_back) :
157  std::runtime_error("OomphException")
158 {
159  // By default we shout
160  Suppress_error_message=false;
161 
162  // Store exception stream
163  Exception_stream_pt=&exception_stream;
164 
165  // Create storage for error message
166  Exception_stringstream_pt=new std::stringstream;
167 
168  //Build an exception header string from the information passed
169  //Start with a couple of new lines to space things out
170  std::string exception_header="\n\n";
171 
172  //Now add a dividing line
173  for(unsigned i=0;i<output_width;i++) {exception_header += "=";}
174  exception_header += "\n";
175 
176  //Write the type of exception
177  exception_header += "Oomph-lib ";
178  exception_header += exception_type;
179 
180  //Add the function in which it occurs
181  exception_header += "\n\n at ";
182  exception_header += location;
183  exception_header += "\n\n in ";
184  exception_header += function_name;
185 
186  //Finish with two new lines
187  exception_header +="\n\n";
188 
189  //and a closing line
190  for(unsigned i=0;i<(unsigned)(0.8*output_width);i++)
191  {exception_header += "-";}
192 
193  //Output the error header to the stream
194  (*Exception_stringstream_pt) << exception_header << std::endl;
195 
196  //Report the error
197  (*Exception_stringstream_pt) << std::endl << error_description << std::endl;
198 
199 #ifdef OOMPH_HAS_STACKTRACE
200  // Print the stacktrace
201  if (list_trace_back)
202  {
203  print_stacktrace((*Exception_stringstream_pt));
204  }
205 #endif
206 
207  //Finish off with another set of double lines
208  for(unsigned i=0;i<output_width;i++) {(*Exception_stringstream_pt) << "=";}
209  (*Exception_stringstream_pt) << std::endl << std::endl;
210 
211  // Copy message to stream in terminate helper in case the message
212  // doesn't get caught and/or doesn/t make it to the destructor
213  (*TerminateHelper::Exception_stringstream_pt)<<
214  (*Exception_stringstream_pt).str();
215 
216 }
