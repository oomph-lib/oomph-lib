71 /// \short Calculate the Jacobian of the mapping between local and global
72 /// coordinates at the position s for face elements for two dimensional
73 /// problems
74 /// The jacobian of the 1D face element is computed which is dt/ds_t
75 //=============================================================================
76 template<>
77 double BiharmonicFluxElement<2>::J_eulerian(const Vector<double> &s) const
78 {
79 
80  //Find the number of nodes
81  unsigned n_node = this->nnode();
82 
83  //Set up dummy memory for the shape functions
84  Shape psi(n_node,Nface_nodal_dof);
85  DShape dpsids(n_node,Nface_nodal_dof,1);
86 
87  //Get the shape functions and local derivatives
88  this->dshape_local(s,psi,dpsids);
89 
90  // computed dx_i/ds_t along edge element
91  Vector<double> interpolated_dxds_t(2);
92 
93  //Get the bulk position type corresponding to the slope
94  const unsigned slope_index = this->bulk_position_type(1);
95  for(unsigned l=0;l<n_node;l++)
96  {
97  for (unsigned d = 0; d < 2; d++)
98  {
99  interpolated_dxds_t[d] += this->node_pt(l)->x_gen(0,d) * dpsids(l,0,0)
100  + this->node_pt(l)->x_gen(slope_index,d)*dpsids(l,1,0);
101  }
102  }
103 
104  // dt/ds_t = dx_0/ds_t*t_0 + dx_1/ds_t*t_1
105  //
106  // given : t_0 = dx_0/ds_t / ( (dx_0/ds_t)^2 + (dx_1/ds_t)^2 )^0.5
107  // t_1 = dx_1/ds_t / ( (dx_0/ds_t)^2 + (dx_1/ds_t)^2 )^0.5
108  //
109  // then : dt/ds_t = ( (dx_0/ds_t)^2 + (dx_1/ds_t)^2 )^0.5
110  //
111  // (note : it is gaurantee that dt/ds_t is +ve, therefore do not need PARANOID
112  // check of jacobian)
113  double dtds_t = sqrt(interpolated_dxds_t[0]*interpolated_dxds_t[0] +
114  interpolated_dxds_t[1]*interpolated_dxds_t[1]);
115 
116  //Return the Jacobian
117  return dtds_t;
118 }

123 /// \short Compute the elemental contribution to the residual vector for 2D
124 /// problem type 0 biharmonic flux elements
125 //=============================================================================
126 template<>
127 void BiharmonicFluxElement<2>::
128 fill_in_generic_residual_contribution_biharmonic_flux(Vector<double> &residual)
129 {
130  //Find out how many nodes there are in the face element
131  unsigned n_node = this->nnode();
132 
133  //set up memory for shape functions
134  Shape psi(n_node,Nface_nodal_dof);
135 
136  //setup memory for derivative of shape functions
137  DShape dpsi(n_node,Nface_nodal_dof,1);
138 
139  //Set the value of Nintpt
140  unsigned n_intpt = this->integral_pt()->nweight();
141 
142  // local coordinate position of integration point in face element
143  Vector<double> s(1);
144 
145  // edge sign adjust
146  int edge_sign = //-int(this->s_fixed_value())*int((s_fixed_index-0.5)*2);
147  - this->normal_sign();
148 
149  //Flip for the different normal conventions (TIDY THIS UP)
150  if((this->face_index() == 2) || (this->face_index() == -2))
151  {edge_sign *= -1;}
152 
153  // Get the bulk position type corresponding to the slope
154  const unsigned slope_index = this->bulk_position_type(1);
155 
156  // Ge the position type corresponding to the outer slope
157  const unsigned outer_slope_index = 3 - slope_index;
158 
159 //Loop over the integration points
160  //--------------------------------
161  for(unsigned ipt=0;ipt<n_intpt;ipt++)
162  {
163 
164  //Get the integral weight
165  double w = this->integral_pt()->weight(ipt);
166 
167  // value of local coordinate s at integration point
168  s[0] = this->integral_pt()->knot(ipt,0);
169 
170  // get the (1D) shape fn
171  this->dshape_local(s,psi,dpsi);
172 
173  // compute dx_i/ds_t and dx_i/ds_n at ipt
174  Vector<double> dxds_t(2,0.0);
175  Vector<double> dxds_n(2,0.0);
176  for (unsigned n = 0; n < n_node; n++)
177  {
178  for (unsigned d = 0; d < 2; d++)
179  {
180  for (unsigned k = 0; k < Nface_nodal_dof; k++)
181  {
182  dxds_t[d] += dpsi(n,k,0)*node_pt(n)->x_gen(slope_index*k,d);
183  dxds_n[d] += psi(n,k)*node_pt(n)->x_gen(outer_slope_index+
184  slope_index*k,d);
185  }
186  }
187  }
188 
189  // compute ds_n/dn
190  double ds_ndn = -edge_sign*sqrt(dxds_t[0]*dxds_t[0]+dxds_t[1]*dxds_t[1])/
191  (-dxds_n[0]*dxds_t[1]+dxds_t[0]*dxds_n[1]);
192 
193  // compute ds_t/dn
194  double ds_tdn = edge_sign*(dxds_t[1]*dxds_n[1]+dxds_t[0]*dxds_n[0])/
195  (sqrt(dxds_t[0]*dxds_t[0]+dxds_t[1]*dxds_t[1])*
196  (-dxds_n[0]*dxds_t[1]+dxds_t[0]*dxds_n[1]));
197 
198  // compute interpolated_m at s
199  double interpolated_m = 0.0;
200  Vector<double> m(2);
201  for (unsigned n = 0; n < n_node; n++)
202  {
203  this->node_pt(n)->get_coordinates_on_boundary(Boundary,m);
204  for (unsigned k = 0; k < Nface_nodal_dof; k++)
205  {
206  interpolated_m += psi(n,k)*m[k];
207  }
208  }
209 
210  // get the fluxes
211  double flux0 = 0.0;
212  get_flux0(interpolated_m,flux0);
213  double flux1 = 0.0;
214  get_flux1(interpolated_m,flux1);
215 
216  // get J
217  double J = J_eulerian(s);
218 
219  //Premultiply the weights and the Jacobian
220  double W = w*J;
221 
222  //Now add to the appropriate equations
223 
224 
225  //Loop over the test function nodes
226  for(unsigned n=0;n<n_node;n++)
227  {
228  // loop over the face element position types
229  for (unsigned k = 0;k < Nface_nodal_dof;k++)
230  {
231 
232  // apply contribution for flux0
233 
234  // determine bulk position type
235  unsigned bulk_p_type = slope_index*k;
236 
237  // local equation number
238  int local_eqn = this->nodal_local_eqn(n,bulk_p_type);
239 
240  // if node not pinned
241  if (local_eqn >= 0)
242  {
243  // add contribution to residual
244  residual[local_eqn] += flux1*psi(n,k)*W;
245  }
246 
247  // apply first contribution for flux1
248 
249  // if node not pinned
250  if (local_eqn >= 0)
251  {
252  // add contribution to residual
253  residual[local_eqn] -= flux0*dpsi(n,k,0)*ds_tdn*W;
254  }
255 
256  // apply second contribution for flux1
257 
258  // determine bulk position type
259  bulk_p_type = outer_slope_index + slope_index*k;
260 
261  // local equation number
262  local_eqn = this->nodal_local_eqn(n,bulk_p_type);
263 
264  // if node not pinned
265  if (local_eqn >= 0)
266  {
267  // add contribution to residual
268  residual[local_eqn] -= flux0*psi(n,k)*ds_ndn*W;
269  }
270  }
271  }
272  }
273 }
