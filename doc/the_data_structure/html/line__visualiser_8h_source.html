278 
279  // Copy values

281  j++)
282  {
283  data[i][j]=received_data[line_i][counter_d[line_i]+j];
284  }
285 
286  //Increase counter
287  counter_d[line_i]+=received_size[line_i][counter_s[line_i] ];
288  counter_s[line_i]++;
289  }
290  } // end somebody has found it -- no output at all if nobody
291  // has found the point (e.g. outside mesh)
292  }
293  }
294  // Send data to root
295  else
296  {

300  Comm_pt->mpi_comm());
301 
302  //Send the sizes of fields to the main process
303  if (buff_size==0) size_values.resize(1);
304  MPI_Send(&size_values[0], buff_size, MPI_UNSIGNED, 0, tag,
305  Comm_pt->mpi_comm());
306 
307  //Send the number of data fields to the main process
308  buff_size = local_values.size();
309  MPI_Send(&buff_size, 1, MPI_UNSIGNED, 0, tag,
310  Comm_pt->mpi_comm());
311 
312  //Send the data to the main process
313  if (buff_size==0) local_values.resize(1);
314  MPI_Send(&local_values[0], buff_size, MPI_DOUBLE, 0, tag,
315  Comm_pt->mpi_comm());
316  }
317 
318 #endif // Serial version
319  }
320  }
321  else
322  {
323 
324  // Loop over the points
325  for (unsigned i=0; i<Nplot_points; i++)
326  {
327  // Check if the point was found in the mesh
328  if (Plot_point[i].first != NULL) // success

548  /// Helper function to setup from file
549  void setup_from_file(Mesh* mesh_pt, const std::string file_name,
550  const double &scale)
551  {
552  // Open file use ifstream
553  std::ifstream file_input(file_name.c_str(),std::ios_base::in);
554  if (!file_input)
555  {
556  std::ostringstream error_message;
557  error_message << "Cannot open file " << file_name << "\n";
558  throw OomphLibError(error_message.str(),
559  OOMPH_CURRENT_FUNCTION,
560  OOMPH_EXCEPTION_LOCATION);
561  }
562  if (!file_input.is_open())
563  {
564  std::ostringstream error_message;
565  error_message << "Cannot open file " << file_name << "\n";
566  throw OomphLibError(error_message.str(),
567  OOMPH_CURRENT_FUNCTION,
568  OOMPH_EXCEPTION_LOCATION);
569  }
570 
571  // Declaration of variables
572  std::string line;
573  Vector<Vector<double> > coord_vec_tmp; // Coord array
574 
575  // Loop over the lines of the input file
576  while(getline(file_input,line)/* != 0*/)
577  {
578  // Test if the first char of the line is a number
579  // using ascii enumeration of chars
580  if (isdigit(line[0]))
581  {
582  Vector<double> tmp(3);
583 
584  // Read the 3 first doubles of the line
585  // Return 3 if success and less if error
586  int n=sscanf(line.c_str(),"%lf %lf %lf",&tmp[0],&tmp[1],&tmp[2]);
587 
588  if (n == 3) // success
589  {
590  // Rescaling
591  for (unsigned i=0; i<3; i++)
592  {
593  tmp[i]*=scale;
594  }
595 
596  // Add the new point to the list
597  coord_vec_tmp.push_back(tmp);
598  }
599  else // error
600  {
601  oomph_info << "Line ignored \n";
602  }
603  }
604  }
605 
606  // Call to the helper function
607  setup(mesh_pt,coord_vec_tmp);
608  }

614  {

616  Nplot_points=coord_vec.size();

622 
623  // Dimension

625 
626  // Make space

628

630  MeshAsGeomObject mesh_geom_tmp(mesh_pt);
631

635

637  double tt_start=TimingHelpers::timer();

658  /* << coord_vec[i][1] << "\n"; */
659  }

663  /* << coord_vec[i][0] << " " */
664  /* << coord_vec[i][1] << "\n"; */
665  }
