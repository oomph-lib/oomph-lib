82  template<>
83  void TRaviartThomasDarcyElement<0>::get_q_basis_local(const Vector<double> &s,
84  Shape &q_basis) const
85  {
86  // RT_0 basis functions
87  q_basis(0,0) = Sign_edge[0]*std::sqrt(2)*s[0];
88  q_basis(0,1) = Sign_edge[0]*std::sqrt(2)*s[1];
89 
90  q_basis(1,0) = Sign_edge[1]*(s[0]-1);
91  q_basis(1,1) = Sign_edge[1]*s[1];
92 
93  q_basis(2,0) = Sign_edge[2]*s[0];
94  q_basis(2,1) = Sign_edge[2]*(s[1]-1);
95  }

257 //===========================================================================
258  /// Compute the local form of the q basis at local coordinate s
259 //===========================================================================
260  template<>
261  void TRaviartThomasDarcyElement<1>::get_q_basis_local(const Vector<double> &s,
262  Shape &q_basis) const
263  {
264  // RT_1 basis functions
265 
266  Vector<double> g1_vect=edge_flux_interpolation_point(0,0);
267  Vector<double> g2_vect=edge_flux_interpolation_point(0,1);
268  double g1=g1_vect[0];
269  double g2=g2_vect[0];
270  q_basis(0,0) = Sign_edge[0]*std::sqrt(2.0)*s[0]*(s[1]-g2)/(g1-g2);
271  q_basis(0,1) = Sign_edge[0]*std::sqrt(2.0)*s[1]*(s[1]-g2)/(g1-g2);
272 
273  q_basis(1,0) = Sign_edge[0]*std::sqrt(2.0)*s[0]*(s[1]-g1)/(g2-g1);
274  q_basis(1,1) = Sign_edge[0]*std::sqrt(2.0)*s[1]*(s[1]-g1)/(g2-g1);
275 
276 
277  g1_vect=edge_flux_interpolation_point(1,0);
278  g2_vect=edge_flux_interpolation_point(1,1);
279  g1=g1_vect[0];
280  g2=g2_vect[0];
281  q_basis(2,0) = Sign_edge[1]*(s[0]-1.0)*(s[1]-g1)/(g2-g1);
282  q_basis(2,1) = Sign_edge[1]*s[1]*(s[1]-g1)/(g2-g1);
283 
284  q_basis(3,0) = Sign_edge[1]*(s[0]-1.0)*(s[1]-g2)/(g1-g2);
285  q_basis(3,1) = Sign_edge[1]*s[1]*(s[1]-g2)/(g1-g2);
286 
287 
288  g1_vect=edge_flux_interpolation_point(2,0);
289  g2_vect=edge_flux_interpolation_point(2,1);
290  g1=g1_vect[0];
291  g2=g2_vect[0];
292  q_basis(4,0) = Sign_edge[2]*s[0]*(s[0]-g2)/(g1-g2);
293  q_basis(4,1) = Sign_edge[2]*(s[1]-1.0)*(s[0]-g2)/(g1-g2);
294 
295  q_basis(5,0) = Sign_edge[2]*s[0]*(s[0]-g1)/(g2-g1);
296  q_basis(5,1) = Sign_edge[2]*(s[1]-1.0)*(s[0]-g1)/(g2-g1);
297 
298  q_basis(6,0) = s[1]*s[0];
299  q_basis(6,1) = s[1]*(s[1]-1.0);
300 
301  q_basis(7,0) = s[0]*(s[0]-1.0);
302  q_basis(7,1) = s[0]*s[1];
303  }

307  /// Compute the local form of the q basis and dbasis/ds at local coordinate s
308 //===========================================================================
309  template<>
310  void TRaviartThomasDarcyElement<1>::get_div_q_basis_local(
311  const Vector<double> &s,
312  Shape &div_q_basis_ds) const
313  {
314 
315  Vector<double> g1_vect=edge_flux_interpolation_point(0,0);
316  Vector<double> g2_vect=edge_flux_interpolation_point(0,1);
317  double g1=g1_vect[0];
318  double g2=g2_vect[0];
319  div_q_basis_ds(0) = Sign_edge[0]*std::sqrt(2.0)*(3.0*s[1]-2.0*g2)/(g1-g2);
320  div_q_basis_ds(1) = Sign_edge[0]*std::sqrt(2.0)*(2.0*g1-3.0*s[1])/(g1-g2);
321 
322 
323  g1_vect=edge_flux_interpolation_point(1,0);
324  g2_vect=edge_flux_interpolation_point(1,1);
325  g1=g1_vect[0];
326  g2=g2_vect[0];
327  div_q_basis_ds(2) = Sign_edge[1]*(2.0*g1-3.0*s[1])/(g1-g2);
328  div_q_basis_ds(3) = Sign_edge[1]*(3.0*s[1]-2.0*g2)/(g1-g2);
329 
330 
331 
332  g1_vect=edge_flux_interpolation_point(2,0);
333  g2_vect=edge_flux_interpolation_point(2,1);
334  g1=g1_vect[0];
335  g2=g2_vect[0];
336  div_q_basis_ds(4) = Sign_edge[2]*(3.0*s[0]-2.0*g2)/(g1-g2);
337  div_q_basis_ds(5) = Sign_edge[2]*(2.0*g1-3.0*s[0])/(g1-g2);
338 
339  div_q_basis_ds(6) = 3.0*s[1]-1.0;
340  div_q_basis_ds(7) = 3.0*s[0]-1.0;
341 
342  // Scale the basis by the ratio of the length of the edge to the length of
343  // the corresponding edge on the reference element
344  scale_basis(div_q_basis_ds);
345  }
