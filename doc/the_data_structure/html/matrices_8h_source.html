420  /// Copy constructor: Deep copy!
421  DenseMatrix(const DenseMatrix& source_matrix)
422  {
423  // Set row and column lengths
424  N=source_matrix.nrow();
425  M=source_matrix.ncol();
426  // Assign space for the data
427  Matrixdata = new T[N*M];
428  // Copy the data across from the other matrix
429  for(unsigned long i=0;i<N;i++)
430  {

436  }

438  /// Copy assignment

440  {

443  {
444  // Check row and column length
445  unsigned long n=source_matrix.nrow();
446  unsigned long m=source_matrix.ncol();
447  if ( (N!=n) || (M!=m) )
