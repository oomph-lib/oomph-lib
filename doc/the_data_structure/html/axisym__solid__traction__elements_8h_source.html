45 /// A class for elements that allow the imposition of an applied traction
46 /// in the principle of virtual displacements.
47 /// The geometrical information can be read from the FaceGeometry<ELEMENT>
48 /// class and and thus, we can be generic enough without the need to have
49 /// a separate equations class.
50 //======================================================================
51 template <class ELEMENT>
52 class AxisymmetricSolidTractionElement :
53 public virtual FaceGeometry<ELEMENT>, public virtual FaceElement
54 {
55 private:

59  const Vector<double> &n,

62 protected:

64  /// Return the surface traction force
65  void get_traction(const double& time, const Vector<double> &x,
66  const Vector<double> &n,Vector<double> &result) const
67  {
68  //If the function pointer is zero return zero
69  if(Traction_fct_pt == 0)
70  {
71  //Loop over dimensions and set body forces to zero
72  //It's axisymmetric, so we only have "two" dimensions
73  for(unsigned i=0;i<2;i++) {result[i] = 0.0;}
74  }
75  //Otherwise call the function
76  else
77  {
78  (*Traction_fct_pt)(time,x,n,result);
79  }
80  }

139 /// Return the residuals for the AxisymmetricSolidTractionElements
140 //=======================================================================
141 template<class ELEMENT>
142 void AxisymmetricSolidTractionElement<ELEMENT>::
143 fill_in_contribution_to_residuals(Vector<double> &residuals)
144 {
145  //Find out how many nodes there are
146  unsigned n_node = nnode();
147  //Find out how many positional dofs there are
148  unsigned n_position_type = nnodal_position_type();
149 
150  //Integer to hold the local equation number
151  int local_eqn=0;
152 
153  //Set up memory for the shape functions
154  //The surface is 1D, so we only have one local derivative
155  Shape psi(n_node,n_position_type);
156  DShape dpsids(n_node,n_position_type,1);
157 
158  //Set the value of n_intpt
159  unsigned n_intpt = integral_pt()->nweight();
160 
161  //Loop over the integration points
162  for(unsigned ipt=0;ipt<n_intpt;ipt++)
163  {
164  //Get the integral weight
165  double w = integral_pt()->weight(ipt);
166 
167  //Only need to call the local derivatives
168  dshape_local_at_knot(ipt,psi,dpsids);
169 
170  //Calculate the global position and lagrangian coordinate
171  Vector<double> interpolated_x(2,0.0), interpolated_xi(2,0.0);
172  //Calculate the global and lagrangian derivtives wrt the local coordinates
173  Vector<double> interpolated_dxds(2,0.0), interpolated_dxids(2,0.0);
174 
175  //Calculate displacements and derivatives
176  for(unsigned l=0;l<n_node;l++)
177  {
178  //Loop over positional dofs
179  for(unsigned k=0;k<n_position_type;k++)
180  {
181  //Loop over the number of lagrangian coordinates (2)
182  for(unsigned i=0;i<2;i++)
183  {
184  //Calculate the global position
185  interpolated_x[i] +=
186  nodal_position_gen(l,bulk_position_type(k),i)*psi(l,k);
187  interpolated_xi[i] +=
188  this->lagrangian_position_gen(l,bulk_position_type(k),i)*psi(l,k);
189  //Calculate the derivatives of the global and lagrangian coordinates
190  interpolated_dxds[i] +=
191  nodal_position_gen(l,bulk_position_type(k),i)*dpsids(l,k,0);
192  interpolated_dxids[i] +=
193  this->lagrangian_position_gen(l,bulk_position_type(k),i)
194  *dpsids(l,k,0);
195  }
196  }
197  }
198 
199  //Now calculate the entries of the deformed surface metric tensor
200  //Now find the local deformed metric tensor from the tangent Vectors
201  DenseMatrix<double> A(2);
202  //The off-diagonal terms are Zero
203  A(0,1) = A(1,0) = 0.0;
204  //The diagonal terms are a little complicated
205  A(0,0) =
206  (interpolated_dxds[0] - interpolated_x[1]*interpolated_dxids[1])*
207  (interpolated_dxds[0] - interpolated_x[1]*interpolated_dxids[1]) +
208  (interpolated_dxds[1] + interpolated_x[0]*interpolated_dxids[1])*
209  (interpolated_dxds[1] + interpolated_x[0]*interpolated_dxids[1]);
210 
211 
212  A(1,1) = (interpolated_x[0]*sin(interpolated_xi[1]) +
213  interpolated_x[1]*cos(interpolated_xi[1]))*
214  (interpolated_x[0]*sin(interpolated_xi[1]) +
215  interpolated_x[1]*cos(interpolated_xi[1]));
216 
217  //Premultiply the weights and the square-root of the determinant of
218  //the metric tensor
219  double W = w*sqrt(A(0,0)*A(1,1));
220 
221  //Also find the normal -- just the cross product of the metric tensors
222  //but I want to express it in terms of e_r and e_theta components
223  //N.B. There is an issue at theta = 0,pi, where the normal is e_{r},
224  //but given that I never assemble it, should be OK!
225  //The minus sign is chosen to ensure that the normal is really outward
226  Vector<double> interpolated_normal(2);
227  //Component in the e_{r} direction
228  interpolated_normal[0] = -1.0*
229  (interpolated_x[0]*sin(interpolated_xi[1]) +
230  interpolated_x[1]*cos(interpolated_xi[1]))*
231  (interpolated_dxds[1] + interpolated_x[0]*interpolated_dxids[1]);
232  //Component in the e_{theta} direction
233  interpolated_normal[1] = -1.0*
234  (interpolated_x[0]*sin(interpolated_xi[1]) +
235  interpolated_x[1]*cos(interpolated_xi[1]))*
236  (interpolated_x[1]*interpolated_dxids[1] - interpolated_dxds[0]);
237 
238  //If we're on the north or south face need to flip normal
239  if(s_fixed_value()==-1)
240  {
241  interpolated_normal[0] *= -1.0;
242  interpolated_normal[1] *= -1.0;
243  }
244 
245  //Now adjust and scale the normal
246  double length = 0.0;
247  for(unsigned i=0;i<2;i++)
248  {
249  interpolated_normal[i] *= normal_sign();
250  length += interpolated_normal[i]*interpolated_normal[i];
251  }
252  for(unsigned i=0;i<2;i++)
253  {
254  interpolated_normal[i] /= sqrt(length);
255  }
256 
257  //Now calculate the load
258  Vector<double> traction(2);
259 
260  //Normal is outwards
261  get_traction(time(),interpolated_x,interpolated_normal,traction);
262 
263  //=====LOAD TERMS FROM PRINCIPLE OF VIRTUAL DISPLACEMENTS========
264 
265  //Loop over the test functions, nodes of the element
266  for(unsigned l=0;l<n_node;l++)
267  {
268  //Loop of types of dofs
269  for(unsigned k=0;k<n_position_type;k++)
270  {
271  //Loop over the displacement components
272  for(unsigned i=0;i<2;i++)
273  {
274  local_eqn =
275  this->position_local_eqn(l,bulk_position_type(k),i);
276  /*IF it's not a boundary condition*/
277  if(local_eqn >= 0)
278  {
279  //Add the loading terms to the residuals
280  residuals[local_eqn] -= traction[i]*psi(l,k)*W;
281  }
282  }
283  } //End of if not boundary condition
284  } //End of loop over shape functions
285  } //End of loop over integration points
286 
287 }
288 
289 
290 }
