46 /// RefineableElements are FiniteElements that may be subdivided into
47 /// children to provide a better local approximation to the solution.
48 /// After non-uniform refinement adjacent elements need not necessarily have
49 /// nodes in common. A node that does not have a counterpart in its
50 /// neighbouring element is known as a hanging node and its position and
51 /// any data that it stores must be constrained to ensure inter-element
52 /// continuity.
53 ///
54 /// Generic data and function interfaces associated with refinement
55 /// are defined in this class.
56 ///
57 /// Additional data includes:
58 /// - a pointer to a general Tree object that is used to track the
59 /// refinement history,
60 /// - a refinement level (not necessarily the same as the level in the tree!),
61 /// - a flag indicating whether the element should be refined,
62 /// - a flag indicating whether the element should be de-refined,
63 /// - a global element number for plotting/validation purposes,
64 /// - storage for local equation numbers associated with hanging nodes.
65 ///
66 /// Additional functions perform the following generic tasks:
67 /// - provide access to additional data,
68 /// - setup local equation numbering for data associated with hanging nodes,
69 /// - generic finite-difference calculation of contributions to the
70 /// elemental jacobian from nodal data to include hanging nodes,
71 /// - split of the element into its sons,
72 /// - select and deselect the element for refinement,
73 /// - select and deselect the sons of the element for de-refinement (merging),
74 ///
75 /// In addition, there are a number of interfaces that specify element-specific
76 /// tasks. These should be overloaded in RefineableElements of particular
77 /// geometric types and perform the following tasks:
78 /// - return a pointer to the root and father elements in the tree structure,
79 /// - define the number of sons into which the element is divided,
80 /// - build the element: construct nodes, assign their positions,
81 /// values and any boundary conditions,
82 /// - recreate the element from its sons if they are merged,
83 /// - deactivate the element, perform any operations that are
84 /// required when the element is still in the tree, but no longer active
85 /// - set the number and provide access to the values interpolated
86 /// by the nodes,
87 /// - setup the hanging nodes
88 ///
89 /// In mixed element different sets of nodes are used to interpolate different
90 /// unknowns. Interfaces are provided for functions that can be used to find
91 /// the position of the nodes that interpolate the different unknowns. These
92 /// functions are used to setup hanging node information automatically in
93 /// particular elements, e.g. Taylor Hood Navier--Stokes.
94 /// The default implementation assumes that the elements are isoparameteric.
95 ///

234  /// \short Split the element into the number of sons to be
235  /// constructed and return a
236  /// vector of pointers to the sons. Elements are allocated, but they are
237  /// not given any properties. The refinement level of the sons is one
238  /// higher than that of the father elemern.
239  template<class ELEMENT>
240  void split(Vector<ELEMENT*>& son_pt) const
241  {
242  // Increase refinement level
243  int son_refine_level=Refine_level+1;
244 
245  //How many sons are to be constructed
246  unsigned n_sons = required_nsons();
247  //Resize the son pointer
248  son_pt.resize(n_sons);
249 
250  //Loop over the sons and construct
251  for(unsigned i=0;i<n_sons;i++)
252  {
253  son_pt[i]=new ELEMENT;
254  //Set the refinement level of the newly constructed son.
255  son_pt[i]->set_refinement_level(son_refine_level);
256  }
257  }

320  /// \short Unbuild the element, i.e. mark the nodes that were created
321  /// during its creation for possible deletion
322  virtual void unbuild()
323  {
324  //Get pointer to father element
325  RefineableElement* father_pt = father_element_pt();
326  //If there is no father, nothing to do
327  if(father_pt==0) {return;}
328 
329  //Loop over all the nodes
330  unsigned n_node = this->nnode();
331  for(unsigned n=0;n<n_node;n++)
332  {
333  //If any node in this element is in the father, it can't be deleted
334  if(father_pt->get_node_number(this->node_pt(n)) >= 0)
335  {
336  node_pt(n)->set_non_obsolete();
337  }
338  }
339  }

482  /// Return a pointer to the "father" element at the specified refinement level
483  void get_father_at_refinement_level(unsigned& refinement_level,
484  RefineableElement*& father_at_reflevel_pt)
485  {
486  // Get the father in the tree (it shouldn't try to get a null Tree...)
487  Tree* father_pt = Tree_pt->father_pt();
488  // Get the refineable element associated with this father
489  RefineableElement* father_el_pt=dynamic_cast<RefineableElement*>
490  (father_pt->object_pt());
491  // Get the refinement level
492  unsigned level=father_el_pt->refinement_level();
493  // If the level matches the required one then return, if not call again
494  if (level==refinement_level)
495  {
496  father_at_reflevel_pt=father_el_pt;
497  }
498  else
499  {
500  // Recursive call
501  father_el_pt->get_father_at_refinement_level(refinement_level,
502  father_at_reflevel_pt);
503  }
504  }
