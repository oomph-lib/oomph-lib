261  /// \short Get 'flux' for Z2 error recovery: Upper triangular entries
262  /// in strain rate tensor.
263  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)
264  {
265 #ifdef PARANOID
266  unsigned num_entries = 6;
267  if (flux.size() < num_entries)
268  {
269  std::ostringstream error_message;
270  error_message << "The flux vector has the wrong number of entries, "
271  << flux.size() << ", whereas it should be at least "
272  << num_entries << std::endl;
273  throw OomphLibError(error_message.str(),
274  OOMPH_CURRENT_FUNCTION,
275  OOMPH_EXCEPTION_LOCATION);
276  }
277 #endif
278 
279  // Get strain rate matrix
280  DenseMatrix<double> strainrate(3);
281  this->strain_rate(s, strainrate);
282 
283  // Pack into flux Vector
284  unsigned icount = 0;
285 
286  // Start with diagonal terms
287  for (unsigned i = 0; i < 3; i++)
288  {
289  flux[icount] = strainrate(i, i);
290  icount++;
291  }
292 
293  // Off diagonals row by row
294  for (unsigned i = 0; i < 3; i++)
295  {
296  for (unsigned j = i + 1; j < 3; j++)
297  {
298  flux[icount] = strainrate(i, j);
299  icount++;
300  }
301  }
302  }

312  /// \short Create a list of pairs for all unknowns in this element,
313  /// so that the first entry in each pair contains the global equation
314  /// number of the unknown, while the second one contains the number
315  /// of the "DOF type" that this unknown is associated with.
316  /// (Function can obviously only be called if the equation numbering
317  /// scheme has been set up.) Velocity=0; Pressure=1
318  void get_dof_numbers_for_unknowns(
319  std::list<std::pair<unsigned long, unsigned>>& dof_lookup_list) const
320  {
321  // number of nodes
322  unsigned n_node = this->nnode();
323 
324  // number of pressure values
325  unsigned n_press = this->npres_axi_nst();
326 
327  // temporary pair (used to store dof lookup prior to being added to list)
328  std::pair<unsigned, unsigned> dof_lookup;
329 
330  // pressure dof number
331  unsigned pressure_dof_number = 3;
332 
333  // loop over the pressure values
334  for (unsigned n = 0; n < n_press; n++)
335  {
336  // determine local eqn number
337  int local_eqn_number = this->p_local_eqn(n);
338 
339  // ignore pinned values - far away degrees of freedom resulting
340  // from hanging nodes can be ignored since these are be dealt
341  // with by the element containing their master nodes
342  if (local_eqn_number >= 0)
343  {
344  // store dof lookup in temporary pair: First entry in pair
345  // is global equation number; second entry is dof type
346  dof_lookup.first = this->eqn_number(local_eqn_number);
347  dof_lookup.second = pressure_dof_number;
348 
349  // add to list
350  dof_lookup_list.push_front(dof_lookup);
351  }
352  }
353 
354  // loop over the nodes
355  for (unsigned n = 0; n < n_node; n++)
356  {
357  // find the number of values at this node
358  unsigned nv = this->node_pt(n)->nvalue();
359 
360  // loop over these values
361  for (unsigned v = 0; v < nv; v++)
362  {
363  // determine local eqn number
364  int local_eqn_number = this->nodal_local_eqn(n, v);
365 
366  // ignore pinned values
367  if (local_eqn_number >= 0)
368  {
369  // store dof lookup in temporary pair: First entry in pair
370  // is global equation number; second entry is dof type
371  dof_lookup.first = this->eqn_number(local_eqn_number);
372  dof_lookup.second = v;
373 
374  // add to list
375  dof_lookup_list.push_front(dof_lookup);
376  }
377  }
378  }
379  }
380  };

491  Shape& ppsi,
492  DShape& dppsidx,
493  Shape& ptest,
494  DShape& dptestdx) const
495  {

497  ppsi[0] = 1.0;

499  ppsi[2] = s[1];
500

502  dppsidx(1, 0) = 1.0;
503  dppsidx(2, 0) = 0.0;
504 
505  dppsidx(0, 1) = 0.0;
506  dppsidx(1, 1) = 0.0;
507  dppsidx(2, 1) = 1.0;

509 
510  // Get the values of the shape functions and their local derivatives
511  Shape psi(7);
512  DShape dpsi(7, 2);
513  dshape_local(s, psi, dpsi);
514 
515  // Allocate memory for the inverse 2x2 jacobian
516  DenseMatrix<double> inverse_jacobian(2);
517 
518  // Now calculate the inverse jacobian
519  const double det = local_to_eulerian_mapping(dpsi, inverse_jacobian);
520

524

796  /// in strain rate tensor.
797  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)

802  {
