277  /// \short Add algebraic update information for node: What's the
278  /// ID of the mesh update function (typically used within the mesh)
279  /// Which Mesh implements the update operation? Also,
280  /// pass the vector of geometric objects and
281  /// the vectors of reference values that are
282  /// needed for the update operation. Negative values for ID are only
283  /// allowed when called from node constructor, as indicated
284  /// by the final argument which defaults to false.
285  void add_node_update_info(const int& id,
286  AlgebraicMesh* mesh_pt,
287  const Vector<GeomObject*>& geom_object_pt,
288  const Vector<double>& ref_value,
289  const bool& called_from_constructor=false)
290  {
291  // Sanity check
292  if (id<0)
293  {
294  if (!called_from_constructor)
295  {
296  std::ostringstream error_message;
297  error_message
298  << "\nNegative ID, " << id
299  << ", only allowed if called from constructor and\n"
300  << "indicated as such by optional boolean flag."
301  << std::endl;
302  throw OomphLibError(error_message.str(),
303  OOMPH_CURRENT_FUNCTION,
304  OOMPH_EXCEPTION_LOCATION);
305  }
306  }
307 
308  // If there's just one entry -- check if it's the default dummy one
309  if (Mesh_pt.size()==1)
310  {
311  if (Mesh_pt.begin()->second==Dummy_mesh_pt)
312  {
313  if (Default_it_mesh_pt->second==Dummy_mesh_pt)
314  {
315  kill_node_update_info(Dummy_node_update_fct_id);
316  }
317  }
318  }
319 
320  // Now insert the actual info
321  Mesh_pt.insert(std::make_pair(id,mesh_pt));
322  Geom_object_pt.insert(std::make_pair(id,geom_object_pt));
323  Ref_value.insert(std::make_pair(id,ref_value));
324 
325  // Always use the "first" update fct as default -- can be overwritten
326  // outside (usually only done for self test)
327  set_default_node_update(Mesh_pt.begin()->first);
328  }
