146  //============================================================================
147  /// The contents of the vector are redistributed to match the new
148  /// distribution. In a non-MPI build this method works, but does nothing.
149  /// \b NOTE 1: The current distribution and the new distribution must have
150  /// the same number of global rows.
151  /// \b NOTE 2: The current distribution and the new distribution must have
152  /// the same Communicator.
153  //============================================================================
154  void DoubleVector::redistribute(
155  const LinearAlgebraDistribution* const& dist_pt)
156  {
157 #ifdef OOMPH_HAS_MPI
158 #ifdef PARANOID
159  if (!Internal_values)
160  {
161  // if this vector does not own the double* values then it cannot be
162  // distributed.
163  // note: this is not stictly necessary - would just need to be careful
164  // with delete[] below.
165  std::ostringstream error_message;
166  error_message << "This vector does not own its data (i.e. it has been "
167  << "passed in via set_external_values() and therefore "
168  << "cannot be redistributed";
169  throw OomphLibError(
170  error_message.str(), OOMPH_CURRENT_FUNCTION, OOMPH_EXCEPTION_LOCATION);
171  }
172  // paranoid check that the nrows for both distributions is the
173  // same
174  if (dist_pt->nrow() != this->nrow())
175  {
176  std::ostringstream error_message;
177  error_message << "The number of global rows in the new distribution ("
178  << dist_pt->nrow() << ") is not equal to the number"
179  << " of global rows in the current distribution ("
180  << this->nrow() << ").\n";
181  throw OomphLibError(
182  error_message.str(), OOMPH_CURRENT_FUNCTION, OOMPH_EXCEPTION_LOCATION);
183  }
184  // paranoid check that the current distribution and the new distribution
185  // have the same Communicator
186  OomphCommunicator temp_comm(*dist_pt->communicator_pt());
187  if (!(temp_comm == *this->distribution_pt()->communicator_pt()))
188  {
189  std::ostringstream error_message;
190  error_message << "The new distribution and the current distribution must "
191  << "have the same communicator.";
192  throw OomphLibError(
193  error_message.str(), OOMPH_CURRENT_FUNCTION, OOMPH_EXCEPTION_LOCATION);
194  }
195 #endif
196 
197  // check the distributions are not the same
198  if (!((*this->distribution_pt()) == *dist_pt))
199  {
200  // get the rank and the number of processors
201  int my_rank = this->distribution_pt()->communicator_pt()->my_rank();
202  int nproc = this->distribution_pt()->communicator_pt()->nproc();
203 
204  // if both vectors are distributed
205  if (this->distributed() && dist_pt->distributed())
206  {
207  // new nrow_local and first_row data
208  Vector<unsigned> new_first_row_data(nproc);
209  Vector<unsigned> new_nrow_local_data(nproc);
210  Vector<unsigned> current_first_row_data(nproc);
211  Vector<unsigned> current_nrow_local_data(nproc);
212  for (int i = 0; i < nproc; i++)
213  {
214  new_first_row_data[i] = dist_pt->first_row(i);
215  new_nrow_local_data[i] = dist_pt->nrow_local(i);
216  current_first_row_data[i] = this->first_row(i);
217  current_nrow_local_data[i] = this->nrow_local(i);
218  }
219 
220  // compute which local rows are expected to be received from each
221  // processor / sent to each processor
222  Vector<unsigned> new_first_row_for_proc(nproc);
223  Vector<unsigned> new_nrow_local_for_proc(nproc);
224  Vector<unsigned> new_first_row_from_proc(nproc);
225  Vector<unsigned> new_nrow_local_from_proc(nproc);
226 
227  // for every processor compute first_row and nrow_local that will
228  // will sent and received by this processor
229  for (int p = 0; p < nproc; p++)
230  {
231  // start with data to be sent
232  if ((new_first_row_data[p] < (current_first_row_data[my_rank] +
233  current_nrow_local_data[my_rank])) &&
234  (current_first_row_data[my_rank] <
235  (new_first_row_data[p] + new_nrow_local_data[p])))
236  {
237  new_first_row_for_proc[p] =
238  std::max(current_first_row_data[my_rank], new_first_row_data[p]);
239  new_nrow_local_for_proc[p] =
240  std::min((current_first_row_data[my_rank] +
241  current_nrow_local_data[my_rank]),
242  (new_first_row_data[p] + new_nrow_local_data[p])) -
243  new_first_row_for_proc[p];
244  }
245 
246  // and data to be received
247  if ((new_first_row_data[my_rank] <
248  (current_first_row_data[p] + current_nrow_local_data[p])) &&
249  (current_first_row_data[p] <
250  (new_first_row_data[my_rank] + new_nrow_local_data[my_rank])))
251  {
252  new_first_row_from_proc[p] =
253  std::max(current_first_row_data[p], new_first_row_data[my_rank]);
254  new_nrow_local_from_proc[p] =
255  std::min(
256  (current_first_row_data[p] + current_nrow_local_data[p]),
257  (new_first_row_data[my_rank] + new_nrow_local_data[my_rank])) -
258  new_first_row_from_proc[p];
259  }
260  }
261 
262  // temporary storage for the new data
263  double* temp_data = new double[new_nrow_local_data[my_rank]];
264 
265  // "send to self" or copy Data that does not need to be sent else where
266  // to temp_data
267  if (new_nrow_local_for_proc[my_rank] != 0)
268  {
269  unsigned j =
270  new_first_row_for_proc[my_rank] - current_first_row_data[my_rank];
271  unsigned k =
272  new_first_row_for_proc[my_rank] - new_first_row_data[my_rank];
273  for (unsigned i = 0; i < new_nrow_local_for_proc[my_rank]; i++)
274  {
275  temp_data[k + i] = Values_pt[j + i];
276  }
277  }
278 
279  // send and receive circularly
280  for (int p = 1; p < nproc; p++)
281  {
282  // next processor to send to
283  unsigned dest_p = (my_rank + p) % nproc;
284 
285  // next processor to receive from
286  unsigned source_p = (nproc + my_rank - p) % nproc;
287 
288  // send and receive the value
289  MPI_Status status;
290  MPI_Sendrecv(Values_pt + new_first_row_for_proc[dest_p] -
291  current_first_row_data[my_rank],
292  new_nrow_local_for_proc[dest_p],
293  MPI_DOUBLE,
294  dest_p,
295  1,
296  temp_data + new_first_row_from_proc[source_p] -
297  new_first_row_data[my_rank],
298  new_nrow_local_from_proc[source_p],
299  MPI_DOUBLE,
300  source_p,
301  1,
302  this->distribution_pt()->communicator_pt()->mpi_comm(),
303  &status);
304  }
305 
306  // copy from temp data to Values_pt
307  delete[] Values_pt;
308  unsigned nrow_local = dist_pt->nrow_local();
309  Values_pt = new double[nrow_local];
310  for (unsigned i = 0; i < nrow_local; i++)
311  {
312  Values_pt[i] = temp_data[i];
313  }
314  delete[] temp_data;
315  }
316 
317  // if this vector is distributed but the new distributed is global
318  else if (this->distributed() && !dist_pt->distributed())
319  {
320  // copy existing Values_pt to temp_data
321  unsigned nrow_local = this->nrow_local();
322  double* temp_data = new double[nrow_local];
323  for (unsigned i = 0; i < nrow_local; i++)
324  {
325  temp_data[i] = Values_pt[i];
326  }
327 
328  // clear and resize Values_pt
329  delete[] Values_pt;
330  Values_pt = new double[this->nrow()];
331 
332  // create a int vector of first rows
333  int* dist_first_row = new int[nproc];
334  int* dist_nrow_local = new int[nproc];
335  for (int p = 0; p < nproc; p++)
336  {
337  dist_first_row[p] = this->first_row(p);
338  dist_nrow_local[p] = this->nrow_local(p);
339  }
340 
341  // gather the local vectors from all processors on all processors
342  int my_nrow_local(this->nrow_local());
343  MPI_Allgatherv(temp_data,
344  my_nrow_local,
345  MPI_DOUBLE,
346  Values_pt,
347  dist_nrow_local,
348  dist_first_row,
349  MPI_DOUBLE,
350  this->distribution_pt()->communicator_pt()->mpi_comm());
351 
352  // update the distribution
353  this->build_distribution(dist_pt);
354 
355  // delete the temp_data
356  delete[] temp_data;
357 
358  // clean up
359  delete[] dist_first_row;
360  delete[] dist_nrow_local;
361  }
362 
363  // if this vector is not distrubted but the target vector is
364  else if (!this->distributed() && dist_pt->distributed())
365  {
366  // cache the new nrow_local
367  unsigned nrow_local = dist_pt->nrow_local();
368 
369  // and first_row
370  unsigned first_row = dist_pt->first_row();
371 
372  // temp storage for the new data
373  double* temp_data = new double[nrow_local];
374 
375  // copy the data
376  for (unsigned i = 0; i < nrow_local; i++)
377  {
378  temp_data[i] = Values_pt[first_row + i];
379  }
380 
381  // copy to Values_pt
382  delete[] Values_pt;
383  Values_pt = temp_data;
384 
385  // update the distribution
386  this->build_distribution(dist_pt);
387  }
388 
389  // copy the Distribution
390  this->build_distribution(dist_pt);
391  }
392 #endif
393  }

591  //============================================================================
592  /// returns the maximum coefficient
593  //============================================================================
594  double DoubleVector::max() const
595  {
596  // the number of local rows
597  unsigned nrow = this->nrow_local();
598 
599  // get the local maximum
600  double max = 0.0;
601  for (unsigned i = 0; i < nrow; i++)
602  {
603  if (std::fabs(Values_pt[i]) > std::fabs(max))
604  {
605  max = std::fabs(Values_pt[i]);
606  }
607  }
608 
609  // now return the maximum
610 #ifdef OOMPH_HAS_MPI
611  // if this vector is not distributed then the local maximum is the global
612  // maximum
613  if (!this->distributed())
614  {
615  return max;
616  }
617  // else if the vector is distributed but only on a single processor
618  // then the local maximum is the global maximum
619  else if (this->distribution_pt()->communicator_pt()->nproc() == 1)
620  {
621  return max;
622  }
623  // otherwise use MPI_Allreduce to find the global maximum
624  else
625  {
626  double local_max = max;
627  MPI_Allreduce(&local_max,
628  &max,
629  1,
630  MPI_DOUBLE,
631  MPI_MAX,
632  this->distribution_pt()->communicator_pt()->mpi_comm());
633  return max;
634  }
635 #else
636  return max;
637 #endif
638  }

641  /// output the contents of the vector
642  //============================================================================
643  void DoubleVector::output(std::ostream& outfile,
644  const int& output_precision) const
645  {
646  // temp pointer to values
647  double* temp;
648 
649  // number of global row
650  unsigned nrow = this->nrow();
651 
652 #ifdef OOMPH_HAS_MPI
653 
654  // number of local rows
655  int nrow_local = this->nrow_local();
656 
657  // gather from all processors
658  if (this->distributed() &&
659  this->distribution_pt()->communicator_pt()->nproc() > 1)
660  {
661  // number of processors
662  int nproc = this->distribution_pt()->communicator_pt()->nproc();
663 
664  // number of gobal row
665  unsigned nrow = this->nrow();
666 
667  // get the vector of first_row s and nrow_local s
668  int* dist_first_row = new int[nproc];
669  int* dist_nrow_local = new int[nproc];
670  for (int p = 0; p < nproc; p++)
671  {
672  dist_first_row[p] = this->first_row(p);
673  dist_nrow_local[p] = this->nrow_local(p);
674  }
675 
676  // gather
677  temp = new double[nrow];
678  MPI_Allgatherv(Values_pt,
679  nrow_local,
680  MPI_DOUBLE,
681  temp,

958  //===========================================================================
959  /// \short Concatenate DoubleVectors.
960  /// Takes a Vector of DoubleVectors. If the out vector is built, we will not
961  /// build a new distribution. Otherwise we build a uniform distribution.
962  ///
963  /// The rows of the out vector is seen "as it is" in the in vectors.
964  /// For example, if we have DoubleVectors with distributions A and B,
965  /// distributed across two processors (p0 and p1),
966  ///
967  /// A: [a0] (on p0) B: [b0] (on p0)
968  /// [a1] (on p1) [b1] (on P1),
969  ///
970  /// then the out_vector is
971  ///
972  /// [a0 (on p0)
973  /// a1] (on p0)
974  /// [b0] (on p1)
975  /// b1] (on p1),
976  ///
977  /// Communication is required between processors. The sum of the global
978  /// number of rows in the in vectors must equal to the global number of rows
979  /// in the out vector. This condition must be met if one is to supply an out
980  /// vector with a distribution, otherwise we can let the function generate
981  /// the out vector distribution itself.
982  //===========================================================================
983  void concatenate(const Vector<DoubleVector*>& in_vector_pt,
984  DoubleVector& out_vector)
985  {
986  // How many in vectors to concatenate?
987  unsigned nvectors = in_vector_pt.size();
988 
989  // PARANIOD checks which involves the in vectors only
990 #ifdef PARANOID
991  // Check that there is at least one vector.
992  if (nvectors == 0)
993  {
994  std::ostringstream error_message;
995  error_message << "There is no vector to concatenate...\n"
996  << "Perhaps you forgot to fill in_vector_pt?\n";
997  throw OomphLibError(error_message.str(),
998  OOMPH_CURRENT_FUNCTION,
999  OOMPH_EXCEPTION_LOCATION);
1000  }
1001 
1002  // Does this vector need concatenating?
1003  if (nvectors == 1)
1004  {
1005  std::ostringstream warning_message;
1006  warning_message << "There is only one vector to concatenate...\n"
1007  << "This does not require concatenating...\n";
1008  OomphLibWarning(warning_message.str(),
1009  OOMPH_CURRENT_FUNCTION,
1010  OOMPH_EXCEPTION_LOCATION);
1011  }
1012 
1013  // Check that all the DoubleVectors in in_vector_pt are built
1014  for (unsigned vec_i = 0; vec_i < nvectors; vec_i++)
1015  {
1016  if (!in_vector_pt[vec_i]->built())
1017  {
1018  std::ostringstream error_message;
1019  error_message << "The vector in position " << vec_i
1020  << " is not built.\n"
1021  << "I cannot concatenate an unbuilt vector.\n";
1022  throw OomphLibError(error_message.str(),
1023  OOMPH_CURRENT_FUNCTION,
1024  OOMPH_EXCEPTION_LOCATION);
1025  }
1026  }
1027 #endif
1028 
1029  // The communicator pointer for the first in vector.
1030  const OomphCommunicator* const comm_pt =
1031  in_vector_pt[0]->distribution_pt()->communicator_pt();
1032 
1033  // Check if the first in vector is distributed.
1034  bool distributed = in_vector_pt[0]->distributed();
1035 
1036  // If the out vector is not built, build it with a uniform distribution.
1037  if (!out_vector.built())
1038  {
1039  // Nrow for the out vector is the sum of the nrow of the in vectors.
1040  unsigned tmp_nrow = 0;
1041  for (unsigned vec_i = 0; vec_i < nvectors; vec_i++)
1042  {
1043  tmp_nrow += in_vector_pt[vec_i]->nrow();
1044  }
1045 
1046  // Build the out vector with uniform distribution.
1047  out_vector.build(
1048  LinearAlgebraDistribution(comm_pt, tmp_nrow, distributed), 0.0);
1049  }
1050  else
1051  {
1052 #ifdef PARANOID
1053  // Check that the sum of nrow of in vectors match the nrow in the out
1054  // vectors.
1055  unsigned in_nrow = 0;
1056  for (unsigned vec_i = 0; vec_i < nvectors; vec_i++)
1057  {
1058  in_nrow += in_vector_pt[vec_i]->nrow();
1059  }
1060 
1061  if (in_nrow != out_vector.nrow())
1062  {
1063  std::ostringstream error_message;
1064  error_message << "The sum of nrow of the in vectors does not match\n"
1065  << "the nrow of the out vector.\n";
1066  throw OomphLibError(error_message.str(),
1067  OOMPH_CURRENT_FUNCTION,
1068  OOMPH_EXCEPTION_LOCATION);
1069  }
1070 #endif
1071  }
1072 
1073 #ifdef PARANOID
1074  // Check that all communicators of the vectors to concatenate are the same
1075  // by comparing all communicators against the out vector.
1076  const OomphCommunicator out_comm =
1077  *(out_vector.distribution_pt()->communicator_pt());
1078 
1079  for (unsigned vec_i = 0; vec_i < nvectors; vec_i++)
1080  {
1081  // Get the Communicator for the current vector.
1082  const OomphCommunicator in_comm =
1083  *(in_vector_pt[vec_i]->distribution_pt()->communicator_pt());
1084 
1085  if (out_comm != in_comm)
1086  {
1087  std::ostringstream error_message;
1088  error_message << "The vector in position " << vec_i << " has a\n"
1089  << "different communicator from the out vector.\n";
1090  throw OomphLibError(error_message.str(),
1091  OOMPH_CURRENT_FUNCTION,
1092  OOMPH_EXCEPTION_LOCATION);
1093  }
1094  }
1095 
1096  // Check that the distributed boolean is the same for all vectors.
1097  if (out_comm.nproc() != 1)
1098  {
1099  const bool out_distributed = out_vector.distributed();
1100  for (unsigned vec_i = 0; vec_i < nvectors; vec_i++)
1101  {
1102  if (out_distributed != in_vector_pt[vec_i]->distributed())
1103  {
1104  std::ostringstream error_message;
1105  error_message << "The vector in position " << vec_i << " has a\n"
1106  << "different distributed boolean from "
1107  << "the out vector.\n";
1108  throw OomphLibError(error_message.str(),
1109  OOMPH_CURRENT_FUNCTION,
1110  OOMPH_EXCEPTION_LOCATION);
1111  }
1112  }
1113  }
1114 #endif
1115 
1116 
1117  // Now we do the concatenation.
1118  if ((comm_pt->nproc() == 1) || !distributed)
1119  {
1120  // Serial version of the code.
1121  // This is trivial, we simply loop through the in vectors and
1122  // fill in the out vector.
1123 
1124  // Out vector index.
1125  unsigned out_i = 0;
1126 
1127  // Out vector values.
1128  double* out_value_pt = out_vector.values_pt();
1129 
1130  // Loop through the in vectors.
1131  for (unsigned vec_i = 0; vec_i < nvectors; vec_i++)
1132  {
1133  // Nrow of current in vector.
1134  unsigned in_nrow = in_vector_pt[vec_i]->nrow();
1135 
1136  // In vector values.
1137  double* in_value_pt = in_vector_pt[vec_i]->values_pt();
1138 
1139  // Loop through the entries of this in vector.
1140  for (unsigned i = 0; i < in_nrow; i++)
1141  {
1142  out_value_pt[out_i++] = in_value_pt[i];
1143  }
1144  }
1145  }
1146  // Otherwise we are dealing with a distributed vector.
1147  else
1148  {
1149 #ifdef OOMPH_HAS_MPI
1150  // Get the number of processors
1151  unsigned nproc = comm_pt->nproc();
1152 
1153  // My rank
1154  unsigned my_rank = comm_pt->my_rank();
1155 
1156  // Storage for the data (per processor) to send
1157  Vector<Vector<double>> values_to_send(nproc);
1158 
1159  // The sum of the nrow for the in vectors (so far). This is used as an
1160  // offset to calculate the global equation number in the out vector
1161  unsigned long sum_of_vec_nrow = 0;
1162 
1163  // Loop over the in vectors and work out:
1164  // out_p: the rank the of receiving processor
1165  // out_local_eqn: the local equation number of the receiving processor
1166  //
1167  // Then put the value and out_local_eqn at out_p in values_to_send
1168 
1169  LinearAlgebraDistribution* out_distribution_pt =
1170  out_vector.distribution_pt();
1171  for (unsigned in_vec_i = 0; in_vec_i < nvectors; in_vec_i++)
1172  {
1173  // Loop through the local equations
1174  unsigned in_vec_nrow_local = in_vector_pt[in_vec_i]->nrow_local();
1175  unsigned in_vec_first_row = in_vector_pt[in_vec_i]->first_row();
1176 
1177  for (unsigned in_row_i = 0; in_row_i < in_vec_nrow_local; in_row_i++)
1178  {
1179  // Calculate the global equation number for this in_row_i
1180  unsigned out_global_eqn =
1181  in_row_i + in_vec_first_row + sum_of_vec_nrow;
1182 
1183  // Get the processor that this global row belongs to.
1184  // The rank_of_global_row(...) function loops through all the
1185  // processors and does two unsigned comparisons. Since we have to do
1186  // this for every row, it may be better to store a list mapping for
1187  // very large number of processors.
1188  unsigned out_p =
1189  out_distribution_pt->rank_of_global_row(out_global_eqn);
1190  // unsigned out_p = out_distribution_pt
1191  // ->rank_of_global_row_map(out_global_eqn);
1192 
1193  // Knowing out_p enables us to work out the out_first_row and
1194  // out_local_eqn.
1195  unsigned out_first_row = out_distribution_pt->first_row(out_p);
1196  unsigned out_local_eqn = out_global_eqn - out_first_row;
1197 
1198  // Now push back the out_local_eqn and the value
1199  values_to_send[out_p].push_back(out_local_eqn);
1200  values_to_send[out_p].push_back(
1201  (*in_vector_pt[in_vec_i])[in_row_i]);
1202  }
1203 
1204  // Update the offset.
1205  sum_of_vec_nrow += in_vector_pt[in_vec_i]->nrow();
1206  }
1207 
1208  // Prepare to send the data!
1209 
1210  // Storage for the number of data to be sent to each processor.
1211  Vector<int> send_n(nproc, 0);
1212 
1213  // Storage for all the values to be send to each processor.
1214  Vector<double> send_values_data;
1215 
1216  // Storage location within send_values_data
1217  Vector<int> send_displacement(nproc, 0);
1218 
1219  // Get the total amount of data which needs to be sent, so we can
1220  // reserve space for it.
1221  unsigned total_ndata = 0;
1222  for (unsigned rank = 0; rank < nproc; rank++)
1223  {
1224  if (rank != my_rank)
1225  {
1226  total_ndata += values_to_send[rank].size();
1227  }
1228  }
1229 
1230  // Now we don't have to re-allocate data/memory when push_back is
1231  // called. Nb. Using push_back without reserving memory may cause
1232  // multiple re-allocation behind the scenes, this is expensive.
1233  send_values_data.reserve(total_ndata);
1234 
1235  // Loop over all the processors to "flat pack" the data for sending.
1236  for (unsigned rank = 0; rank < nproc; rank++)
1237  {
1238  // Set the offset for the current processor
1239  send_displacement[rank] = send_values_data.size();
1240 
1241  // Don't bother to do anything if
1242  // the processor in the loop is the current processor.
1243  if (rank != my_rank)
1244  {
1245  // Put the values into the send data vector.
1246  unsigned n_data = values_to_send[rank].size();
1247  for (unsigned j = 0; j < n_data; j++)
1248  {
1249  send_values_data.push_back(values_to_send[rank][j]);
1250  } // Loop over the data
1251  } // if rank != my_rank
1252 
1253  // Find the number of data to be added to the vector.
1254  send_n[rank] = send_values_data.size() - send_displacement[rank];
1255  } // Loop over processors
1256 
1257  // Storage for the number of data to be received from each processor.
1258  Vector<int> receive_n(nproc, 0);
1259  MPI_Alltoall(&send_n[0],
1260  1,
1261  MPI_INT,
1262  &receive_n[0],
1263  1,
1264  MPI_INT,
1265  comm_pt->mpi_comm());
1266 
1267  // Prepare the data to be received
1268  // by working out the displacement from the received data.
1269  Vector<int> receive_displacement(nproc, 0);
1270  int receive_data_count = 0;
1271  for (unsigned rank = 0; rank < nproc; rank++)
1272  {
1273  receive_displacement[rank] = receive_data_count;
1274  receive_data_count += receive_n[rank];
1275  }
1276 
1277  // Now resize the receive buffer for all data from all processors.
1278  // Make sure that it has size of at least one.
1279  if (receive_data_count == 0)
1280  {
1281  receive_data_count++;
1282  }
1283  Vector<double> receive_values_data(receive_data_count);
1284 
1285  // Make sure that the send buffer has size at least one
1286  // so that we don't get a segmentation fault.
1287  if (send_values_data.size() == 0)
1288  {
1289  send_values_data.resize(1);
1290  }
1291 
1292  // Now send the data between all processors
1293  MPI_Alltoallv(&send_values_data[0],
1294  &send_n[0],
1295  &send_displacement[0],
1296  MPI_DOUBLE,
1297  &receive_values_data[0],
1298  &receive_n[0],
1299  &receive_displacement[0],
1300  MPI_DOUBLE,
1301  comm_pt->mpi_comm());
1302 
1303  // Data from all other processors are stored in:
1304  // receive_values_data
1305  // Data already on this processor is stored in:
1306  // values_to_send[my_rank]
1307 
1308  // Loop through the data on this processor.
1309  unsigned location_i = 0;
1310  unsigned my_values_to_send_size = values_to_send[my_rank].size();
1311  while (location_i < my_values_to_send_size)
1312  {
1313  out_vector[unsigned(values_to_send[my_rank][location_i])] =
1314  values_to_send[my_rank][location_i + 1];
1315 
1316  location_i += 2;
1317  }
1318 
1319  // Before we loop through the data on other processors, we need to check
1320  // if any data has been received.
1321  bool data_has_been_received = false;
1322  unsigned send_rank = 0;
1323  while (send_rank < nproc)
1324  {
1325  if (receive_n[send_rank] > 0)
1326  {
1327  data_has_been_received = true;
1328  break;
1329  }
1330  send_rank++;
1331  }
1332 
1333  location_i = 0;
1334  if (data_has_been_received)
1335  {
1336  unsigned receive_values_data_size = receive_values_data.size();
1337  while (location_i < receive_values_data_size)
1338  {
1339  out_vector[unsigned(receive_values_data[location_i])] =
1340  receive_values_data[location_i + 1];
1341  location_i += 2;
1342  }
1343  }
1344 #else
1345  {
1346  std::ostringstream error_message;
1347  error_message << "I don't know what to do with distributed vectors\n"
1348  << "without MPI... :(";
1349  throw OomphLibError(error_message.str(),
1350  OOMPH_CURRENT_FUNCTION,
1351  OOMPH_EXCEPTION_LOCATION);
1352  }
1353 #endif
1354  }
