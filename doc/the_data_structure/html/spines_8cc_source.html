83 /// Update function to update all nodes of mesh.
84 /// [Doesn't make sense to use this mesh with SolidElements anyway,
85 /// so we buffer the case if update_all_solid_nodes (which defaults
86 /// to false) is set to true.]
87 //============================================================
88 void SpineMesh::node_update(const bool& update_all_solid_nodes)
89 {
90 #ifdef PARANOID
91  if (update_all_solid_nodes)
92  {
93  std::string error_message =
94  "Doesn't make sense to use an SpineMesh with\n";
95  error_message +=
96  "SolidElements so specifying update_all_solid_nodes=true\n";
97  error_message += "doesn't make sense either\n";
98 
99  throw OomphLibError(error_message,
100  OOMPH_CURRENT_FUNCTION,
101  OOMPH_EXCEPTION_LOCATION);
102  }
103 #endif
104 
105  //Loop over all the nodes
106  unsigned long Node_pt_range = Node_pt.size();
107  for(unsigned long l=0;l<Node_pt_range;l++)
108  {
109 #ifdef PARANOID
110  if(!dynamic_cast<SpineNode*>(Node_pt[l]))
111  {
112  std::ostringstream error_stream;
113  error_stream
114  << "Error: Node " << l << "is a " << typeid(Node_pt[l]).name()
115  << ", not a SpineNode" << std::endl;
116  throw OomphLibError(error_stream.str(),
117  OOMPH_CURRENT_FUNCTION,
118  OOMPH_EXCEPTION_LOCATION);
119  }
120 #endif
121 
122  //Need to cast to spine node to get to update function
123  dynamic_cast<SpineNode*>(Node_pt[l])->node_update();
124  }
125 }

128 /// Assign (global) equation numbers to spines, nodes and elements
129 //====================================================================
130 unsigned long SpineMesh::assign_global_spine_eqn_numbers
131 (Vector<double *> &Dof_pt)
132 {
133  //Find the current number of dofs
134  unsigned long equation_number = Dof_pt.size();
135 
136  // Loop over spines and set global equation numbers for the spine heights
137  // (they are the only Data items whose global eqn numbers are assigned
138  // here)
139  unsigned long Spine_pt_range = Spine_pt.size();
140  for(unsigned long i=0;i<Spine_pt_range;i++)
141  {
142  Spine_pt[i]->spine_height_pt()->assign_eqn_numbers(equation_number,Dof_pt);
143  }
144 
145  //Return the total number of equations
146  return(equation_number);
147 }

150 /// \short Function to describe the dofs of the Spine. The ostream
151 /// specifies the output stream to which the description
152 /// is written; the string stores the currently
153 /// assembled output that is ultimately written to the
154 /// output stream by Data::describe_dofs(...); it is typically
155 /// built up incrementally as we descend through the
156 /// call hierarchy of this function when called from
