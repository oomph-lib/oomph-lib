40 /// Excute the update function: Update the current (and if
41 /// update_all_time_levels_for_new_node==true also the previous)
42 /// nodal position. Also update the current nodal values if
43 /// an auxiliary update function is defined.
44 /// Note: Updating of previous positions is only required (and should
45 /// only be performed for) newly created MacroElementNodeUpdateNodes
46 /// i.e. when this function is called from...
47 /// If a node is hanging, its position is determined via its hanging
48 /// node constraints after updating the position of its master nodes.
49 //========================================================================
50 void MacroElementNodeUpdateNode::node_update(const bool&
51  update_all_time_levels_for_new_node)
52 {
53 
54  // Number of time levels that need to be updated
55  unsigned ntime;
56  if (update_all_time_levels_for_new_node)
57  {
58  //Present value plus the previous values
59  ntime = 1 + Position_time_stepper_pt->nprev_values();
60  }
61  else
62  {
63  //Just the present value
64  ntime=1;
65  }
66 
67  // Is it a hanging node?
68  if (is_hanging())
69  {
70 
71  // Loop over all master nodes and update their position
72  // That's all we need to update the position of hanging nodes!
73  // (Recall that for hanging nodes Node::x(...) is not
74  // guaranteed to be kept up-to-date; the (constrained!) nodal
75  // position of hanging nodes must be determined via
76  // Node::position() which determines the position
77  // via the hanging node constraints from the position of
78  // the master nodes)
79  unsigned nmaster=hanging_pt()->nmaster();
80  for (unsigned imaster=0;imaster<nmaster;imaster++)
81  {
82  dynamic_cast<MacroElementNodeUpdateNode*>(hanging_pt()->
83  master_node_pt(imaster))
84  ->node_update(update_all_time_levels_for_new_node);
85  }
86  }
87  // Node isn't hanging --> update it directly
88  else
89  {
90  // If no update element is defined, keep the nodal positions where
91  // they were (i.e. don't do anything), else update
92  if (Node_update_element_pt!=0)
93  {
94  // Vector of local coordinates
95  unsigned n_dim=ndim();
96  Vector<double> x_new(n_dim);
97 
98  // Loop over time levels
99  for (unsigned t=0;t<ntime;t++)
100  {
101  //Update via macro element representation
102  Node_update_element_pt->get_x(t,S_in_node_update_element,x_new);
103  for (unsigned i=0;i<n_dim;i++)
104  {
105  x(t,i)=x_new[i];
106  }
107  }
108  }
109  }
110 
111  // Perform auxiliary update of function values? Node passes itself
112  // to this function so its position etc. is available to the auxiliary
113  // node update function.
114  if (Aux_node_update_fct_pt!=0)
115  {
116  Aux_node_update_fct_pt(this);
117  }
118 
119 }

121 }
