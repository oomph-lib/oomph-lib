82  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)
83 {
84 #ifdef PARANOID
85  unsigned num_entries=DIM+((DIM*DIM)-DIM)/2;
86  if (flux.size()!=num_entries)
87  {
88  std::ostringstream error_message;

99  DenseMatrix<double> strain(DIM);
100  this->get_strain(s,strain);

103  unsigned icount=0;

489 /// 2D rebuild from sons: reconstruct the solid pressure
490 //===================================================================
491 template<>
492 inline void RefineableQPVDElementWithPressure<2>::
493 rebuild_from_sons(Mesh* &mesh_pt)
494 {
495  using namespace QuadTreeNames;
496 
497  //Storage for the solid pressure of each son
498  double centre_solid_p[4]={0.0,0.0,0.0,0.0};
499 
500  // Loop over the sons and assign the central solid pressure
501  for (unsigned ison=0;ison<4;ison++)
502  {
503  // Add the sons midnode pressure
504  centre_solid_p[ison] =
505  dynamic_cast<RefineableQPVDElementWithPressure<2>*>
506  (quadtree_pt()->son_pt(ison)->object_pt())->solid_p(0);
507  }
508 
509  // Use the average for the central solid pressure
510  double p_value = 0.25*(centre_solid_p[0] + centre_solid_p[1] +
511  centre_solid_p[2] + centre_solid_p[3]);
512  //Actually set the pressure
513  set_solid_p(0,p_value);
514 
515 
516  //Slope in s_0 direction
517  //----------------------
518 
519  // Use average of the 2 FD approximations based on the
520  // elements central pressure values
521  // [Other options: Take average of the four
522  // pressure derivatives]
523  double slope1=centre_solid_p[SE] - centre_solid_p[SW];
524  double slope2=centre_solid_p[NE] - centre_solid_p[NW];
525 
526 
527  // Use the average value
528  p_value = 0.5*(slope1 + slope2);
529  set_solid_p(1,p_value);
530 
531  //Slope in s_1 direction
532  //----------------------
533 
534  // Use average of the 2 FD approximations based on the
535  // elements central pressure values
536  // [Other options: Take average of the four
537  // pressure derivatives]
538 
539  slope1= centre_solid_p[NE] - centre_solid_p[SE];
540  slope2= centre_solid_p[NW] - centre_solid_p[SW];
541 
542  // Use the average
543  p_value = 0.5*(slope1 + slope2);
544  set_solid_p(2,p_value);
545 }

569

574  s_father[1]=-0.5;
575  }
576  // South east son
577  else if (son_type==SE)
578  {
579  s_father[0]= 0.5;
580  s_father[1]=-0.5;
581  }
582  // North east son
583  else if (son_type==NE)
584  {
585  s_father[0]=0.5;
586  s_father[1]=0.5;
587  }
588 
589  // North west son
590  else if (son_type==NW)
591  {
592  s_father[0]=-0.5;
593  s_father[1]= 0.5;
594  }
595

602  set_solid_p(0,press);

606  set_solid_p(2,0.5*cast_father_element_pt->solid_p(2));

660  double slope4 = centre_solid_p[RUB] - centre_solid_p[LUB];
