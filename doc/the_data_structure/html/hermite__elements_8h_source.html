85 /// General QHermiteElement class. Local coordinates are not assumed
86 /// to be aligned with the global coordinates so the Jacobian
87 /// of the mapping between local and global coordinates is
88 /// a full matrix. For cases where the coordinates are aligned,
89 /// you should use the derived class, DiagQHermiteElement, which
90 /// uses a simplified mapping that makes the evaluation of
91 /// derivatives of the shape functions much cheaper.
92 //=======================================================================

137  ///Check whether the local coordinate are valid or not
138  bool local_coord_is_valid(const Vector<double> &s)
139  {
140  unsigned ncoord = dim();
141  for(unsigned i=0;i<ncoord;i++)
142  {
143  // We're outside
144  if((s[i] - s_max() > 0.0) ||

174  /// \short Function to compute the geometric shape functions and
175  /// also first and second derivatives wrt local coordinates at
176  /// local coordinate s.
177  /// Numbering:
178  /// \b 1D:
179  /// d2psids(i,0) = \f$ d^2 \psi_j / d s^2 \f$
180  /// \b 2D:
181  /// d2psids(i,0) = \f$ \partial^2 \psi_j / \partial s_0^2 \f$
182  /// d2psids(i,1) = \f$ \partial^2 \psi_j / \partial s_1^2 \f$
183  /// d2psids(i,2) = \f$ \partial^2 \psi_j / \partial s_0 \partial s_1 \f$
184  /// \b 3D:
185  /// d2psids(i,0) = \f$ \partial^2 \psi_j / \partial s_0^2 \f$
186  /// d2psids(i,1) = \f$ \partial^2 \psi_j / \partial s_1^2 \f$
187  /// d2psids(i,2) = \f$ \partial^2 \psi_j / \partial s_2^2 \f$
188  /// d2psids(i,3) = \f$ \partial^2 \psi_j / \partial s_0 \partial s_1 \f$
189  /// d2psids(i,4) = \f$ \partial^2 \psi_j / \partial s_0 \partial s_2 \f$
190  /// d2psids(i,5) = \f$ \partial^2 \psi_j / \partial s_1 \partial s_2 \f$

225  /// Get local coordinates of node j in the element; vector sets its own size
226  void local_coordinate_of_node(const unsigned& j, Vector<double>& s) const
227  {
228  s.resize(DIM);
229  Vector<unsigned> j_sub(DIM);
230  unsigned j_copy = j;
231  unsigned NNODE_1D=2;
232  const double S_min = this->s_min();
233  const double S_range = this->s_max() - S_min;
234  for(unsigned i=0;i<DIM;i++)
235  {
236  j_sub[i] = j_copy%NNODE_1D;
237  j_copy = (j_copy - j_sub[i])/NNODE_1D;
238  s[i]= S_min + double(j_sub[i])/(double)(NNODE_1D-1)*S_range;
239  }
240  }

242  /// Get local fraction of node j in the element; vector sets its own size
243  void local_fraction_of_node(const unsigned& j, Vector<double>& s_fraction)
244  {
245  s_fraction.resize(DIM);
246  Vector<unsigned> j_sub(DIM);
247  unsigned j_copy = j;
248  unsigned NNODE_1D=2;
249  for(unsigned i=0;i<DIM;i++)
250  {
251  j_sub[i] = j_copy%NNODE_1D;
252  j_copy = (j_copy - j_sub[i])/NNODE_1D;
253  s_fraction[i]= j_sub[i];
254  }
255  }

328 template<>
329 inline void QHermiteElement<1>::get_s_plot(
330  const unsigned& i,
331  const unsigned& nplot,
332  Vector<double>& s,
333  const bool& use_equally_spaced_interior_sample_points) const
334  {
335  if (nplot>1)
336  {
337  s[0]=-1.0+2.0*double(i)/double(nplot-1);
338  if (use_equally_spaced_interior_sample_points)
339  {
340  double range=2.0;
341  double dx_new=range/double(nplot);
342  double range_new=double(nplot-1)*dx_new;
343  s[0]=-1.0+0.5*dx_new+range_new*(1.0+s[0])/range;
344  }
345  }
346  else
347  {
348  s[0]=0.0;
349  }
350  }
