45 /// \short A class for all elements that solve the Advection
46 /// Diffusion equations in conservative form using isoparametric elements.
47 /// \f[
48 /// \frac{\partial}{\partial x_{i}}\left(

100  /// \short Return the index at which the unknown value

109 /// \short du/dt at local node n.
110  /// Uses suitably interpolated value for hanging nodes.
111  double du_dt_cons_adv_diff(const unsigned &n) const
112  {
113  // Get the data's timestepper
114  TimeStepper* time_stepper_pt= this->node_pt(n)->time_stepper_pt();
115 
116  //Initialise dudt
117  double dudt=0.0;
118  //Loop over the timesteps, if there is a non Steady timestepper
119  if (!time_stepper_pt->is_steady())
120  {
121  //Find the index at which the variable is stored
122  const unsigned u_nodal_index = u_index_cons_adv_diff();
123 
124  // Number of timsteps (past & present)
125  const unsigned n_time = time_stepper_pt->ntstorage();
126 
127  for(unsigned t=0;t<n_time;t++)
128  {
129  dudt += time_stepper_pt->weight(1,t)*nodal_value(t,n,u_nodal_index);
130  }
131  }
132  return dudt;
133  }

333  /// \short Get diffusivity tensor at (Eulerian) position
334  /// x and/or local coordinate s.
335  /// This function is
336  /// virtual to allow overloading in multi-physics problems where
337  /// the wind function might be determined by
338  /// another system of equations
339  inline virtual void get_diff_cons_adv_diff(const unsigned& ipt,
340  const Vector<double> &s,
341  const Vector<double>& x,
342  DenseMatrix<double>& D) const
343  {
344  //If no wind function has been set, return identity
345  if(Diff_fct_pt==0)
346  {
347  for(unsigned i=0;i<DIM;i++)
348  {
349  for(unsigned j=0;j<DIM;j++)
350  {
351  if(i==j) {D(i,j) = 1.0;} else {D(i,j) = 0.0;}
352  }
353  }
354  }
355  else
356  {
357  // Get diffusivity tensor
358  (*Diff_fct_pt)(x,D);
359  }
360  }

363  /// Get flux: \f$\mbox{flux}[i] = \mbox{d}u / \mbox{d}x_i \f$
364  void get_flux(const Vector<double>& s, Vector<double>& flux) const
365  {
366  //Find out how many nodes there are in the element
367  unsigned n_node = nnode();
368 
369  //Get the nodal index at which the unknown is stored
370  unsigned u_nodal_index = u_index_cons_adv_diff();
371 
372  //Set up memory for the shape and test functions
373  Shape psi(n_node);
374  DShape dpsidx(n_node,DIM);
375 
376  //Call the derivatives of the shape and test functions
377  dshape_eulerian(s,psi,dpsidx);
378 
379  //Initialise to zero
380  for(unsigned j=0;j<DIM;j++) {flux[j] = 0.0;}
381 
382  // Loop over nodes
383  for(unsigned l=0;l<n_node;l++)
384  {
385  //Loop over derivative directions
386  for(unsigned j=0;j<DIM;j++)
387  {
388  flux[j] += nodal_value(l,u_nodal_index)*dpsidx(l,j);
389  }
390  }
391  }

393  /// Get flux: \f$\mbox{flux}[i] = \mbox{d}u / \mbox{d}x_i \f$
394  void get_total_flux(const Vector<double>& s,
395  Vector<double>& total_flux) const
396  {
397  //Find out how many nodes there are in the element
398  const unsigned n_node = nnode();
399 
400  //Get the nodal index at which the unknown is stored
401  const unsigned u_nodal_index = u_index_cons_adv_diff();
402 
403  //Set up memory for the shape and test functions
404  Shape psi(n_node);
405  DShape dpsidx(n_node,DIM);
406 
407  //Call the derivatives of the shape and test functions
408  dshape_eulerian(s,psi,dpsidx);
409 
410  //Storage for the Eulerian position
411  Vector<double> interpolated_x(DIM,0.0);
412  //Storage for the concentration
413  double interpolated_u = 0.0;
414  //Storage for the derivatives of the concentration
415  Vector<double> interpolated_dudx(DIM,0.0);
416 
417  // Loop over nodes
418  for(unsigned l=0;l<n_node;l++)
419  {
420  //Get the value at the node
421  const double u_value = this->nodal_value(l,u_nodal_index);
422  interpolated_u += u_value*psi(l);
423  // Loop over directions
424  for(unsigned j=0;j<DIM;j++)
425  {
426  interpolated_x[j] += this->nodal_position(l,j)*psi(l);
427  interpolated_dudx[j] += u_value*dpsidx(l,j);
428  }
429  }
430 
431  //Dummy integration point
432  unsigned ipt=0;
433 
434  //Get the conserved wind (non-divergence free)
435  Vector<double> conserved_wind(DIM);
436  get_conserved_wind_cons_adv_diff(ipt,s,interpolated_x,conserved_wind);
437 
438  //Get diffusivity tensor
439  DenseMatrix<double> D(DIM,DIM);
440  get_diff_cons_adv_diff(ipt,s,interpolated_x,D);
441 
442  //Calculate the total flux made up of the diffusive flux
443  //and the conserved wind
444  for(unsigned i=0;i<DIM;i++)
445  {
446  total_flux[i] = 0.0;
447  for(unsigned j=0;j<DIM;j++)
448  {
449  total_flux[i] += D(i,j)*interpolated_dudx[j];
450  }
451  total_flux[i] -= conserved_wind[i]*interpolated_u;
452  }
453  }

490  /// Return FE representation of function value u(s) at local coordinate s
491  inline double interpolated_u_cons_adv_diff(const Vector<double> &s) const
492  {
493  //Find number of nodes
494  unsigned n_node = nnode();
495 
496  //Get the nodal index at which the unknown is stored
497  unsigned u_nodal_index = u_index_cons_adv_diff();
