41  {
42  // Create an output stream

44

47  << "representation of the solution. As such, it\n"

53  OOMPH_CURRENT_FUNCTION,

57  // Storage for the eight corners
58  Vector<double> corner_LDB(3,0.0);
59  Vector<double> corner_RDB(3,0.0);

70  // First coordinate of the bottom-left coordinates of a surface
71  zeta[0]=-1.0;
72 
73  // Second coordinate of the bottom-left coordinates of a surface
74  zeta[1]=-1.0;
75 
76  // Calculate the space-time coordinates of the LDB corner
77  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
78  OcTreeNames::B,zeta,corner_LDB);
79 
80  // Calculate the space-time coordinates of the LUB corner
81  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
82  OcTreeNames::U,zeta,corner_LUB);
83 
84  // Calculate the space-time coordinates of the LDF corner
85  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
86  OcTreeNames::F,zeta,corner_LDF);
87 
88  // Calculate the space-time coordinates of the RDB corner
89  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
90  OcTreeNames::R,zeta,corner_RDB);
91 
92  // First coordinate of the the top-right coordinates of a surface
93  zeta[0]=1.0;
94 
95  // Second coordinate of the the top-right coordinates of a surface
96  zeta[1]=1.0;
97 
98  // Calculate the space-time coordinates of the RUB corner
99  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
100  OcTreeNames::B,zeta,corner_RUB);
101 
102  // Calculate the space-time coordinates of the RDF corner
103  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
104  OcTreeNames::D,zeta,corner_RDF);
105 
106  // Calculate the space-time coordinates of the LUF corner
107  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
108  OcTreeNames::L,zeta,corner_LUF);
109 
110  // Calculate the space-time coordinates of the RUF corner
111  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
112  OcTreeNames::R,zeta,corner_RUF);
113 
114  // Get the position of the 4 corners of the center slice
115  Vector<double> corner_LD(3,0.0);
116  Vector<double> corner_RD(3,0.0);
117  Vector<double> corner_LU(3,0.0);
118  Vector<double> corner_RU(3,0.0);
119 
120  // Set the coordinates of the point on a surface
121  zeta[0]=-1.0;
122  zeta[1]=s[2];
123 
124  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
125  OcTreeNames::D,zeta,corner_LD);
126  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
127  OcTreeNames::U,zeta,corner_LU);
128  zeta[0]=1.0;
129  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
130  OcTreeNames::D,zeta,corner_RD);
131  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
132  OcTreeNames::U,zeta,corner_RU);
133 
134  // Get position on the B,F faces;
135  Vector<double> face_B(3);
136  Vector<double> face_F(3);
137 
138  zeta[0]=s[0];
139  zeta[1]=s[1];
140  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
141  OcTreeNames::B,zeta,face_B);
142  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
143  OcTreeNames::F,zeta,face_F);
144 
145 
146  // Get position on the edges of the middle slice
147  Vector<double> edge_mid_L(3);
148  Vector<double> edge_mid_R(3);
149  Vector<double> edge_mid_D(3);
150  Vector<double> edge_mid_U(3);
151  zeta[0]=s[0];
152  zeta[1]=s[2];
153  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
154  OcTreeNames::U,zeta,edge_mid_U);
155 
156  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
157  OcTreeNames::D,zeta,edge_mid_D);
158  zeta[0]=s[1];
159  zeta[1]=s[2];
160  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
161  OcTreeNames::L,zeta,edge_mid_L);
162 
163  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
164  OcTreeNames::R,zeta,edge_mid_R);
165 
166  // Get position on the edges of the back slice
167  Vector<double> edge_back_L(3);
168  Vector<double> edge_back_R(3);
169  Vector<double> edge_back_D(3);
170  Vector<double> edge_back_U(3);
171  zeta[0]=s[0];
172  zeta[1]=-1.0;
173  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
174  OcTreeNames::U,zeta,edge_back_U);
175 
176  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
177  OcTreeNames::D,zeta,edge_back_D);
178  zeta[0]=s[1];
179  zeta[1]=-1.0;

181  OcTreeNames::L,zeta,edge_back_L);
182 
183  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
184  OcTreeNames::R,zeta,edge_back_R);
185 
186  // Get position on the edges of the front slice
187  Vector<double> edge_front_L(3);
188  Vector<double> edge_front_R(3);
189  Vector<double> edge_front_D(3);
190  Vector<double> edge_front_U(3);
191  zeta[0]=s[0];
192  zeta[1]=1.0;
193  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
194  OcTreeNames::U,zeta,edge_front_U);
195 
196  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
197  OcTreeNames::D,zeta,edge_front_D);
198  zeta[0]=s[1];
199  zeta[1]=1.0;
200  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
201  OcTreeNames::L,zeta,edge_front_L);
202 
203  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
204  OcTreeNames::R,zeta,edge_front_R);
205 
206  // The number of dimensions (=space+time)
207  unsigned n_dim=3;
208 
209  // Loop over the coordinate directions
210  for (unsigned i=0; i<n_dim; i++)
211  {
212  //-----------------------------
213  // Position on the middle slice
214  //-----------------------------
215  double slice_mid;
216 
217  // The points on the up and down edges of the middle "rectangular slice"
218  double point_up=0.0;
219  double point_down=0.0;
220 
221  // Calculate the point on the upper edge
222  point_up=corner_LU[i]+(corner_RU[i]-corner_LU[i])*0.5*(s[0]+1.0);
223 
224  // Calculate the point on the lower edge
225  point_down=corner_LD[i]+(corner_RD[i]-corner_LD[i])*0.5*(s[0]+1.0);
226 
227  // Position in the rectangular middle slice
228  slice_mid=point_down+0.5*(1.0+s[1])*(point_up-point_down);
229 
230  // Get the differences to the edges of the middle slice
231  double diff_L,diff_R,diff_D,diff_U;
232  diff_L=edge_mid_L[i]-slice_mid;
233  diff_R=edge_mid_R[i]-slice_mid;
234  diff_D=edge_mid_D[i]-slice_mid;
235  diff_U=edge_mid_U[i]-slice_mid;
236 
237  // Map it to get the position in the middle slice
238  slice_mid+=(diff_L*(1.0-0.5*(s[0]+1.0))+
239  diff_R*0.5*(s[0]+1.0)+
240  diff_D*(1.0-0.5*(s[1]+1.0))+
241  diff_U*0.5*(s[1]+1.0));
242 
243  //---------------------------
244  // Position on the back slice
245  //---------------------------
246  double slice_back;
247 
248  // Calculate the point on the upper edge of the back "rectangular slice"
249  point_up=corner_LUB[i]+(corner_RUB[i]-corner_LUB[i])*0.5*(s[0]+1.0);
250 
251  // Calculate the point on the lower edge of the back "rectangular slice"
252  point_down=corner_LDB[i]+(corner_RDB[i]-corner_LDB[i])*0.5*(s[0]+1.0);
253 
254  // Position in the rectangular back slice
255  slice_back=point_down+0.5*(1.0+s[1])*(point_up-point_down);
256 
257  // Get the differences to the edges of the middle slice
258  diff_L=edge_back_L[i]-slice_back;
259  diff_R=edge_back_R[i]-slice_back;
260  diff_D=edge_back_D[i]-slice_back;
261  diff_U=edge_back_U[i]-slice_back;
262 
263  // Map it to get the position in the back slice
264  slice_back+=(diff_L*(1.0-0.5*(s[0]+1.0))+
265  diff_R*0.5*(s[0]+1.0)+
266  diff_D*(1.0-0.5*(s[1]+1.0))+
267  diff_U*0.5*(s[1]+1.0));
268 
269  //----------------------------
270  // Position on the front slice
271  //----------------------------
272  double slice_front;
273 
274  // Calculate the point on the upper edge of the back "rectangular slice"
275  point_up=corner_LUF[i]+(corner_RUF[i]-corner_LUF[i])*0.5*(s[0]+1.0);
276 
277  // Calculate the point on the lower edge of the back "rectangular slice"
278  point_down=corner_LDF[i]+(corner_RDF[i]-corner_LDF[i])*0.5*(s[0]+1.0);
279 
280  // Position in the rectangular back slice
281  slice_front=point_down+0.5*(1.0+s[1])*(point_up-point_down);
282 
283  // Get the differences to the edges of the middle slice
284  diff_L=edge_front_L[i]-slice_front;
285  diff_R=edge_front_R[i]-slice_front;
286  diff_D=edge_front_D[i]-slice_front;
287  diff_U=edge_front_U[i]-slice_front;
288 
289  // Map it to get the position in the back slice
290  slice_front+=(diff_L*(1.0-0.5*(s[0]+1.0))+
291  diff_R*0.5*(s[0]+1.0)+
292  diff_D*(1.0-0.5*(s[1]+1.0))+
293  diff_U*0.5*(s[1]+1.0));
294 
295  //-------------------------------------------------------------------------
296  // Get difference between the back and front slices and the actual boundary
297  //-------------------------------------------------------------------------
298  double diff_back=face_B[i]-slice_back;
299  double diff_front=face_F[i]-slice_front;
300 
301  //----------
302  // Final map
303  //----------
304  r[i]=slice_mid+0.5*(1.0+s[2])*diff_front+0.5*(1.0-s[2])*diff_back;
305  }
306  } // End of macro_map

311  void QExtrudedMacroElement<3>::
312  output_macro_element_boundaries(std::ostream& outfile,const unsigned& nplot)
313  {
314  // Use the OcTree enumeration for corners, edges and faces
315  using namespace OcTreeNames;
316 
317  // Storage for the local coordinates (of a point on a face)
318  Vector<double> s(2);
319 
320  // Storage for the global coordinates
321  Vector<double> x(3);
322 
323  // Loop over the faces
324  for (unsigned idirect=L; idirect<=F; idirect++)
325  {
326  // Output the header
327  outfile << "ZONE I=" << nplot << ", J=" << nplot << std::endl;
328 
329  // Loop over the plot points in the second surface direction
330  for (unsigned i=0; i<nplot; i++)
331  {
332  // Calculate the second local coordinate associated with this plot point
333  s[1]=-1.0+2.0*double(i)/double(nplot-1);
334 
335  // Loop over the plot points in the first surface direction
336  for (unsigned j=0; j<nplot; j++)
337  {
338  // Calculate the first local coordinate associated with this plot point
339  s[0]=-1.0+2.0*double(j)/double(nplot-1);
340 
341  // To make the extrusion machinery consistent with the Domain machinery
342  // a time level argument has to be provided. For our purposes we set
343  // this to zero to ensure that the appropriate output is provided.
344  unsigned t=0;
345 
346  // Get the global coordinates associated with these local coordinates
347  Extruded_domain_pt->macro_element_boundary(t,Macro_element_number,
348  idirect,s,x);
349 
350  // Output the global (space-time) coordinates
351  outfile << x[0] << " " << x[1] << " " << x[2] << std::endl;
352  }
353  } // for (unsigned i=0;i<nplot;i++)
354  } // for (unsigned idirect=L;idirect<=F;idirect++)
355  } // End of output_macro_element_boundaries
