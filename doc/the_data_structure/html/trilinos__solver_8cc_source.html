41 /// Function which uses problem_pt's get_jacobian(...) function to
42 /// generate a linear system which is then solved. This function deletes
43 /// any existing internal data and then generates a new AztecOO solver.
44 //=============================================================================
45  void TrilinosAztecOOSolver::solve(Problem* const &problem_pt,
46  DoubleVector &solution)
47  {
48 
49 
50 // MemoryUsage::doc_memory_usage("start of TrilinosAztecOOSolver::solve");
51 // MemoryUsage::insert_comment_to_continous_top(
52 // "start of TrilinosAztecOOSolver::solve");
53 
54  // clean up from previous solve
55  clean_up_memory();
56 
57  // if we were previously using a problem based solve then we should delete
58  // the oomphlib matrix as it was created during the last solve
59  if (Using_problem_based_solve)
60  {
61  delete Oomph_matrix_pt;
62  Oomph_matrix_pt = NULL;
63  }
64 
65  // this is a problem based solve
66  Using_problem_based_solve = true;
67 
68  // store the problem_pt
69  Problem_pt = problem_pt;
70 
71  //Get oomph-lib Jacobian matrix and residual vector
72 
73  // record the start time
74  double start_t = TimingHelpers::timer();
75 
76 // MemoryUsage::doc_memory_usage("start of get_jacobian()");
77 // MemoryUsage::insert_comment_to_continous_top("start of get_jacobian()");
78 
79  // create the residual
80  DoubleVector residual;
81 
82  // create the jacobian
83  CRDoubleMatrix* cr_matrix_pt = new CRDoubleMatrix;
84  Oomph_matrix_pt = cr_matrix_pt;
85  problem_pt->get_jacobian(residual,*cr_matrix_pt);
86  this->build_distribution(residual.distribution_pt());
87 
88  // record the end time and compute the matrix setup time
89  double end_t = TimingHelpers::timer();
90  Jacobian_setup_time = end_t-start_t;
91  if (this->Doc_time)
92  {
93  oomph_info << "Time to generate Jacobian [sec] : "
94  << Jacobian_setup_time << std::endl;
95  }
96 
97 
98 // MemoryUsage::doc_memory_usage("after get_jacobian() in trilinos solver");
99 // MemoryUsage::insert_comment_to_continous_top(
100 // "after get_jacobian() in trilinos solver");
101 
102  // store the distribution of the solution vector
103  if (!solution.built())
104  {
105  solution.build(this->distribution_pt(),0.0);
106  }
107  LinearAlgebraDistribution solution_dist(solution.distribution_pt());
108 
109  // redistribute the distribution
110  solution.redistribute(this->distribution_pt());
111 
112 // MemoryUsage::doc_memory_usage("before trilinos solve");
113 // MemoryUsage::insert_comment_to_continous_top("before trilinos solve ");

116  solve(Oomph_matrix_pt, residual, solution);

135 /// Function to solve the linear system defined by matrix_pt and rhs.
136 /// \b NOTE 1. The matrix has to be of type CRDoubleMatrix or
137 /// DistributedCRDoubleMatrix.
138 /// \b NOTE 2. This function will delete any existing internal data and
139 /// generate a new AztecOO solver.
140 //=============================================================================
141 void TrilinosAztecOOSolver::solve(DoubleMatrixBase* const& matrix_pt,
142  const DoubleVector &rhs,
143  DoubleVector &result)
144 {
145 
146  // start the timer
147  double start_t = TimingHelpers::timer();
148 
149 #ifdef PARANOID
150  // check that the matrix is square
151  if (matrix_pt->nrow() != matrix_pt->ncol())
152  {
153  std::ostringstream error_message_stream;
154  error_message_stream
155  << "The matrix at matrix_pt must be square.";
156  throw OomphLibError(error_message_stream.str(),
157  OOMPH_CURRENT_FUNCTION,
158  OOMPH_EXCEPTION_LOCATION);
159  }
160  // check that the matrix and the rhs vector have the same nrow()
161  if (matrix_pt->nrow() != rhs.nrow())
162  {
163  std::ostringstream error_message_stream;
164  error_message_stream
165  << "The matrix and the rhs vector must have the same number of rows.";
166  throw OomphLibError(error_message_stream.str(),
167  OOMPH_CURRENT_FUNCTION,
168  OOMPH_EXCEPTION_LOCATION);
169  }
170 
171  // if the matrix is distributable then it too should have the same
172  // communicator as the rhs vector and should not be distributed
173  CRDoubleMatrix* cr_matrix_pt = dynamic_cast<CRDoubleMatrix*>(matrix_pt);
174  if (cr_matrix_pt != 0)
175  {
176  OomphCommunicator temp_comm(*rhs.distribution_pt()->communicator_pt());
177  if (!(temp_comm == *cr_matrix_pt->distribution_pt()->communicator_pt()))
178  {
179  std::ostringstream error_message_stream;
180  error_message_stream
181  << "The matrix matrix_pt must have the same communicator as the vectors"
182  << " rhs and result must have the same communicator";
183  throw OomphLibError(error_message_stream.str(),
184  OOMPH_CURRENT_FUNCTION,
185  OOMPH_EXCEPTION_LOCATION);
186  }
187  }
188  else
189  {
190  throw OomphLibError("Matrix must be of type CRDoubleMatrix",
191  OOMPH_CURRENT_FUNCTION,
192  OOMPH_EXCEPTION_LOCATION);
193  }
194 
195  // if the result vector is setup then check it is not distributed and has
196  // the same communicator as the rhs vector
197  if (result.built())
198  {
199  if (!(*result.distribution_pt() == *rhs.distribution_pt()))
200  {
201  std::ostringstream error_message_stream;
202  error_message_stream
203  << "The result vector distribution has been setup; it must have the "
204  << "same distribution as the rhs vector.";
205  throw OomphLibError(error_message_stream.str(),
206  OOMPH_CURRENT_FUNCTION,
207  OOMPH_EXCEPTION_LOCATION);
208  }
209  }
210 #endif
211 
212  // build the result if not built
213  if (!result.built())
214  {
215  result.build(rhs.distribution_pt(),0.0);
216  }
217 
218  if (Use_iterative_solver_as_preconditioner)
219  {
220  // Only call the setup method if this is the first time we call the
221  // solve method
222  if (First_time_solve_when_used_as_preconditioner)
223  {
224  // setup the solver
225  solver_setup(matrix_pt);
226  // Do not call the setup again
227  First_time_solve_when_used_as_preconditioner=false;
228  // Enable resolve since we are not going to build the solver, the
229  // matrix and the wrapper to the preconditioner again
230  Enable_resolve=true;
231  }
232  }
233  else
234  {
235  // setup the solver
236  solver_setup(matrix_pt);
237  }
238 
239  // create Epetra version of r
240  Epetra_Vector* epetra_r_pt = TrilinosEpetraHelpers::
241  create_distributed_epetra_vector(rhs);
242 
243  // create an empty Epetra vector for z
244  Epetra_Vector* epetra_z_pt = TrilinosEpetraHelpers::
245  create_distributed_epetra_vector(result);
246 
247  double start_t_trilinos = TimingHelpers::timer();
248 
249  // solve the system
250  solve_using_AztecOO(epetra_r_pt,epetra_z_pt);
251 
252  double end_t_trilinos = TimingHelpers::timer();
253  if (this->Doc_time)
254  {
255  oomph_info << "Time for trilinos solve itself : "
256  << end_t_trilinos-start_t_trilinos
257  << "s" << std::endl;
258  }
259  // Copy result to z
260  TrilinosEpetraHelpers::copy_to_oomphlib_vector(epetra_z_pt,result);
261 
262  // clean up memory
263  delete epetra_r_pt;
264  delete epetra_z_pt;
265 
266  // delete solver data if required
267  if (!Enable_resolve)
268  {
269  clean_up_memory();
270  }
271 
272  // stop timers and compute solve time
273  double end_t = TimingHelpers::timer();
274  Linear_solver_solution_time = end_t-start_t;
275 
276  // output timings and info
277  if (this->Doc_time)
278  {
279  oomph_info << "Time for complete trilinos solve : "
280  << Linear_solver_solution_time
281  << "s" << std::endl;
282  }
283 }

287 /// Helper function for setting up the solver. Converts the oomph-lib
288 /// matrices to Epetra matrices, sets up the preconditioner, creates the
289 /// Trilinos Aztec00 solver and passes in the matrix, preconditioner and
290 /// parameters.
291 //=============================================================================
292 void TrilinosAztecOOSolver::solver_setup(DoubleMatrixBase* const& matrix_pt)
293 {
294 
295  // clean up the memory
296  // - delete all except Oomph_matrix_pt, which may have been set in the
297  // problem based solve
298  clean_up_memory();
299 
300  // cast to CRDoubleMatrix
301  // note cast check performed in matrix based solve(...) method
302  CRDoubleMatrix* cast_matrix_pt = dynamic_cast<CRDoubleMatrix*>(matrix_pt);
303 
304  // store the distribution
305  // distribution of preconditioner is same as matrix
306  this->build_distribution(cast_matrix_pt->distribution_pt());
307 
308  // create the new solver
309  AztecOO_solver_pt = new AztecOO();
310 
311  // if the preconditioner is an oomph-lib preconditioner then we set it up
312  TrilinosPreconditionerBase* trilinos_prec_pt =
313  dynamic_cast<TrilinosPreconditionerBase* >(Preconditioner_pt);
314  if (trilinos_prec_pt == 0)
315  {
316  if (Setup_preconditioner_before_solve)
317  {
318  // setup the preconditioner
319  // start of prec setup
320  double prec_setup_start_t = TimingHelpers::timer();
321  Preconditioner_pt->setup(matrix_pt);
322  // start of prec setup
323  double prec_setup_finish_t = TimingHelpers::timer();
324  if (Doc_time)
325  {
326  double t_prec_setup = prec_setup_finish_t - prec_setup_start_t;
327  oomph_info << "Time for preconditioner setup [sec]: "
328  << t_prec_setup << std::endl;
329  }
330 #ifdef PARANOID
331  if (*Preconditioner_pt->distribution_pt() != *this->distribution_pt())
332  {
333  std::ostringstream error_message;
334  error_message << "The oomph-lib preconditioner and the solver must "
335  << "have the same distribution";
336  throw OomphLibError(error_message.str(),
337  OOMPH_CURRENT_FUNCTION,
338  OOMPH_EXCEPTION_LOCATION);
339  }
340 #endif
341  }
342 
343  // wrap the oomphlib preconditioner in the Epetra_Operator derived
344  // OoomphLibPreconditionerEpetraOperator to allow it to be passed to the
345  // trilinos preconditioner
346  Epetra_preconditioner_pt =
347  new OomphLibPreconditionerEpetraOperator(Preconditioner_pt);
348  }
349 
350  // create the matrix
351  double start_t_matrix = TimingHelpers::timer();
352  if (Use_aztecoo_workaround_for_epetra_matrix_setup)
353  {
354  Epetra_matrix_pt = TrilinosEpetraHelpers
355  ::create_distributed_epetra_matrix_for_aztecoo(cast_matrix_pt);
356  }
357  else
358  {
359  Epetra_matrix_pt = TrilinosEpetraHelpers
360  ::create_distributed_epetra_matrix(cast_matrix_pt,
361  cast_matrix_pt->distribution_pt());
362  }
363 
364  // record the end time and compute the matrix setup time
365  double end_t_matrix = TimingHelpers::timer();
366  if (trilinos_prec_pt == 0)
367  {
368  if (Using_problem_based_solve)
369  {
370  dynamic_cast<CRDoubleMatrix*>(Oomph_matrix_pt)->clear();
371  delete Oomph_matrix_pt;
372  Oomph_matrix_pt = NULL;
373  }
374 
375  // delete Oomph-lib matrix if requested
376  else if (Delete_matrix)
377  {
378  dynamic_cast<CRDoubleMatrix*>(matrix_pt)->clear();
379  }
380  }
381 
382  // output times
383  if (Doc_time)
384  {
385  oomph_info << "Time to generate Trilinos matrix : "
386  << double(end_t_matrix-start_t_matrix)
387  << "s" << std::endl;
388  }
389 
390  // set the matrix
391  AztecOO_solver_pt->SetUserMatrix(Epetra_matrix_pt);
392 
393  //set the preconditioner
394  if (trilinos_prec_pt == 0)
395  {
396  AztecOO_solver_pt->
397  SetPrecOperator(Epetra_preconditioner_pt);
398  }
399 
400 #ifdef PARANOID
401  // paranoid check the preconditioner exists
402  if (Preconditioner_pt == 0)
403  {
404  std::ostringstream error_message;
405  error_message << "Preconditioner_pt == 0. (Remember default "
406  << "preconditioner is IdentityPreconditioner)";
407  throw OomphLibError(error_message.str(),
408  OOMPH_CURRENT_FUNCTION,
409  OOMPH_EXCEPTION_LOCATION);
410  }
411 #endif
412 
413  // if the preconditioner is a trilinos preconditioner then setup the
414  // preconditioner
415  if (trilinos_prec_pt != 0)
416  {
417  // only setup the preconditioner if required
418  if (Setup_preconditioner_before_solve)
419  {
420  // start of prec setup
421  double prec_setup_start_t = TimingHelpers::timer();
422 
423  // setup the preconditioner
424  trilinos_prec_pt->set_matrix_pt(matrix_pt);
425  trilinos_prec_pt->setup(Epetra_matrix_pt);
426 
427 
428  // set the preconditioner
429  AztecOO_solver_pt->
430  SetPrecOperator(trilinos_prec_pt->epetra_operator_pt());
431 
432  // start of prec setup
433  double prec_setup_finish_t = TimingHelpers::timer();
434  if (Doc_time)
435  {
436  double t_prec_setup = prec_setup_finish_t - prec_setup_start_t;
437  oomph_info << "Time for preconditioner setup [sec]: "
438  << t_prec_setup << std::endl;
439  }
440  }
441 
442  // delete the oomph-matrix if required
443  if (Using_problem_based_solve)
444  {
445  dynamic_cast<CRDoubleMatrix*>(Oomph_matrix_pt)->clear();
446  delete Oomph_matrix_pt;
447  Oomph_matrix_pt = NULL;
448  }
449 
450  // delete Oomph-lib matrix if requested
451  else if (Delete_matrix)
452  {
453  dynamic_cast<CRDoubleMatrix*>(matrix_pt)->clear();
454  }
455  }
456 
457  // set solver options
458  if (Doc_time)
459  {
460  AztecOO_solver_pt->SetAztecOption(AZ_output, AZ_warnings);
461  }
462  else
463  {
464  AztecOO_solver_pt->SetAztecOption(AZ_output, AZ_none);
465  }
466  AztecOO_solver_pt->SetAztecOption(AZ_kspace, Max_iter);
467 
468  // set solver type
469  switch (Solver_type)
470  {
471  case CG:
472  AztecOO_solver_pt->SetAztecOption(AZ_solver, AZ_cg);
473  break;
474 
475  case GMRES:
476  AztecOO_solver_pt->SetAztecOption(AZ_solver, AZ_gmres);
477  break;
478 
479  case BiCGStab:
480  AztecOO_solver_pt->SetAztecOption(AZ_solver, AZ_bicgstab);
481  break;
482 
483  default:
484  std::ostringstream error_message;
485  error_message << "Solver_type set to incorrect value. "
486  << "Acceptable values are "
487  << CG << ", " << GMRES << " and " << BiCGStab
488  << ". Current value is " << Solver_type << ".";
489  throw OomphLibError(error_message.str(),
490  OOMPH_CURRENT_FUNCTION,
491  OOMPH_EXCEPTION_LOCATION);
492  }
493 }

496 /// \short Function to resolve a linear system using the existing solver
497 /// data, allowing a solve with a new right hand side vector. This
498 /// function must be used after a call to solve(...) with
499 /// enable_resolve set to true.
500 //=============================================================================
501 void TrilinosAztecOOSolver::resolve(const DoubleVector &rhs,
502  DoubleVector &solution)
503 {
504  // start the timer
505  double start_t = TimingHelpers::timer();
506 
507 #ifdef PARANOID
508  if (Epetra_matrix_pt->NumGlobalRows() != static_cast<int>(rhs.nrow()))
509  {
510  std::ostringstream error_message;
511  error_message << "The rhs vector and the matrix must have the same number "
512  << "of rows.\n"
513  << "The rhs vector has " << rhs.nrow() << " rows.\n"
514  << "The matrix has " << Epetra_matrix_pt->NumGlobalRows()
515  << " rows.\n";
516  throw OomphLibError(error_message.str(),
517  OOMPH_CURRENT_FUNCTION,
518  OOMPH_EXCEPTION_LOCATION);
519  }
520 
521 // if the result vector is setup then check it is not distributed and has
522  // the same communicator as the rhs vector
523  if (solution.built())
524  {
525  if (!(*solution.distribution_pt() == *rhs.distribution_pt()))
526  {
527  std::ostringstream error_message_stream;
528  error_message_stream
529  << "The result vector distribution has been setup; it must have the "
530  << "same distribution as the rhs vector.";
531  throw OomphLibError(error_message_stream.str(),
532  OOMPH_CURRENT_FUNCTION,
533  OOMPH_EXCEPTION_LOCATION);
534  }
535  }
536 #endif
537 
538  // build the result if not built
539  if (!solution.built())
540  {
541  solution.build(rhs.distribution_pt(),0.0);
542  }
543 
544 
545  // create Epetra version of r
546  Epetra_Vector* epetra_r_pt = TrilinosEpetraHelpers::
547  create_distributed_epetra_vector(rhs);
548 
549  // create an empty Epetra vector for z
550  Epetra_Vector* epetra_z_pt = TrilinosEpetraHelpers::
551  create_distributed_epetra_vector(solution);
552 
553  // solve the system
554  solve_using_AztecOO(epetra_r_pt,epetra_z_pt);
555 
556  // Copy result to z
557  if (!solution.distribution_built())
558  {
559  solution.build(rhs.distribution_pt(),0.0);
560  }
561  TrilinosEpetraHelpers::copy_to_oomphlib_vector(epetra_z_pt,solution);
562 
563  // clean up memory
564  delete epetra_r_pt;
565  delete epetra_z_pt;
566 
567  double end_t = TimingHelpers::timer();
568  Linear_solver_solution_time = end_t-start_t;
569 
570  // output timings and info
571  if (this->Doc_time)
572  {
573  oomph_info << "Time for resolve : "
574  << Linear_solver_solution_time
575  << "s" << std::endl;
576  }
577 }

581 /// Helper function performs the actual solve once the AztecOO
582 /// solver is set up (i.e. solver_setup() is called)
583 //=============================================================================
584 void TrilinosAztecOOSolver::solve_using_AztecOO(Epetra_Vector* &rhs_pt,
585  Epetra_Vector* &soln_pt)
586 {
587 #ifdef PARANOID
588  // check the matrix and rhs are of consistent sizes
589  if (AztecOO_solver_pt == 0 )
590  {
591  std::ostringstream error_message;
592  error_message << "Solver must be called with solve(...) "
593  << "before resolve(...) to set it up.\n";
594  throw OomphLibError(error_message.str(),
595  OOMPH_CURRENT_FUNCTION,
596  OOMPH_EXCEPTION_LOCATION);
597  }
598 #endif
599 
600  // set the vectors
601  AztecOO_solver_pt->SetLHS(soln_pt);
602  AztecOO_solver_pt->SetRHS(rhs_pt);
603 
604  // perform solve
605  AztecOO_solver_pt->Iterate(Max_iter, Tolerance);
606 
607 
608  // output iterations and final norm
609  Iterations = AztecOO_solver_pt->NumIters();
610  if (Doc_time)
611  {
612  double norm = AztecOO_solver_pt->TrueResidual();
613  oomph_info << "Linear solver iterations : "
614  << Iterations << std::endl;
615  oomph_info << "Final Relative Residual Norm: " << norm << std::endl;
616  }
617 }
