81  /// \short Get 'flux' for Z2 error recovery: Upper triangular entries
82  /// in strain rate tensor.
83  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)
84  {
85 #ifdef PARANOID
86  unsigned num_entries=2*(DIM+((DIM*DIM)-DIM)/2);
87  if (flux.size()!=num_entries)
88  {
89  std::ostringstream error_message;
90  error_message << "The flux vector has the wrong number of entries, "
91  << flux.size() << ", whereas it should be "
92  << num_entries << std::endl;
93  throw OomphLibError(
94  error_message.str(),
95  "RefineableLinearisedNavierStokesEquations::get_Z2_flux()",
96  OOMPH_EXCEPTION_LOCATION);
97  }
98 #endif
99 
100  // Get strain rate matrix
101  DenseMatrix<double> real_strainrate(DIM);
102  this->strain_rate(s,real_strainrate,0);
103  DenseMatrix<double> imag_strainrate(DIM);
104  this->strain_rate(s,imag_strainrate,1);
105 
106 
107  // Pack into flux Vector
108  unsigned icount=0;
109 
110  // Start with diagonal terms
111  for(unsigned i=0;i<DIM;i++)
112  {
113  flux[icount]=real_strainrate(i,i);
114  icount++;
115  }
116 
117  // Off diagonals row by row
118  for(unsigned i=0;i<DIM;i++)
119  {
120  for(unsigned j=i+1;j<DIM;j++)
121  {
122  flux[icount]=real_strainrate(i,j);
123  icount++;
124  }
125  }
126 
127  // Start with diagonal terms
128  for(unsigned i=0;i<DIM;i++)
129  {
130  flux[icount]=imag_strainrate(i,i);
131  icount++;
132  }
133 
134  // Off diagonals row by row
135  for(unsigned i=0;i<DIM;i++)
136  {
137  for(unsigned j=i+1;j<DIM;j++)
138  {
139  flux[icount]=imag_strainrate(i,j);
140  icount++;
141  }
142  }
143  }

275  /// Rebuild from sons: Reconstruct pressure from the (merged) sons
276  void rebuild_from_sons(Mesh* &mesh_pt)
277  {
278  using namespace QuadTreeNames;
279 
280  // Central pressure value:
281  // -----------------------
282 
283  // Use average of the sons central pressure values
284  // Other options: Take average of the four (discontinuous)
285  // pressure values at the father's midpoint]
286 
287  Vector<double> av_press(2,0.0);
288 
289  // Loop over the sons
290  for(unsigned ison=0;ison<4;ison++)
291  {
292  // Loop over the two pressure components
293  for(unsigned i=0;i<2;i++)
294  {
295  // Add the sons midnode pressure
296  av_press[i] += quadtree_pt()->son_pt(ison)->object_pt()->
297  internal_data_pt(P_linearised_nst_internal_index[i])->value(0);
298  }
299  }
300 
301  // Use the average
302  for(unsigned i=0;i<2;i++)
303  {
304  internal_data_pt(P_linearised_nst_internal_index[i])
305  ->set_value(0,0.25*av_press[i]);
306  }
307 
308  Vector<double> slope1(2,0.0), slope2(2,0.0);
309 
310  // Loop over pressure components
311  for(unsigned i=0;i<2;i++)
312  {
313  // Slope in s_0 direction
314  // ----------------------
315 
316  // Use average of the 2 FD approximations based on the
317  // elements central pressure values
318  // [Other options: Take average of the four
319  // pressure derivatives]
320 
321  slope1[i] = quadtree_pt()->son_pt(SE)->object_pt()->
322  internal_data_pt(P_linearised_nst_internal_index[i])->value(0) -
323  quadtree_pt()->son_pt(SW)->object_pt()->
324  internal_data_pt(P_linearised_nst_internal_index[i])->value(0);
325 
326  slope2[i] = quadtree_pt()->son_pt(NE)->object_pt()->
327  internal_data_pt(P_linearised_nst_internal_index[i])->value(0) -
328  quadtree_pt()->son_pt(NW)->object_pt()->
329  internal_data_pt(P_linearised_nst_internal_index[i])->value(0);
330 
331  // Use the average
332  internal_data_pt(P_linearised_nst_internal_index[i])->
333  set_value(1,0.5*(slope1[i]+slope2[i]));
334 
335  // Slope in s_1 direction
336  // ----------------------
337 
338  // Use average of the 2 FD approximations based on the
339  // elements central pressure values
340  // [Other options: Take average of the four
341  // pressure derivatives]
342 
343  slope1[i] = quadtree_pt()->son_pt(NE)->object_pt()->
344  internal_data_pt(P_linearised_nst_internal_index[i])->value(0) -
345  quadtree_pt()->son_pt(SE)->object_pt()->
346  internal_data_pt(P_linearised_nst_internal_index[i])->value(0);
347 
348  slope2[i] = quadtree_pt()->son_pt(NW)->object_pt()->
349  internal_data_pt(P_linearised_nst_internal_index[i])->value(0) -
350  quadtree_pt()->son_pt(SW)->object_pt()->
351  internal_data_pt(P_linearised_nst_internal_index[i])->value(0);
352 
353  // Use the average
354  internal_data_pt(P_linearised_nst_internal_index[i])->
355  set_value(2,0.5*(slope1[i]+slope2[i]));
356  } // End of loop over pressure components
357  }

373  /// \short Get the function value u in Vector.
374  /// Note: Given the generality of the interface (this function
375  /// is usually called from black-box documentation or interpolation
376  /// routines), the values Vector sets its own size in here.
377  void get_interpolated_values(const Vector<double>&s,
378  Vector<double>& values)
379  {
380  // Determine size of values Vector: U^C, U^S, W^C, W^S, V^C, V^S
381  const unsigned n_values = 4*DIM;
382 
383  // Set size of values Vector and initialise to zero
384  values.resize(n_values,0.0);
385 
386  // Calculate velocities: values[0],...
387  for(unsigned i=0;i<n_values;i++)
388  {
389  values[i] = interpolated_u_linearised_nst(s,i);
390  }
391  }
392 
393  /// \short Get all function values [U^C,U^S,...,P^S] at previous timestep t
394  /// (t=0: present; t>0: previous timestep).
395  /// \n
396  /// Note: Given the generality of the interface (this function is
397  /// usually called from black-box documentation or interpolation
398  /// routines), the values Vector sets its own size in here.
399  /// \n
400  /// Note: No pressure history is kept, so pressure is always
401  /// the current value.
402  void get_interpolated_values(const unsigned& t, const Vector<double>&s,
403  Vector<double>& values)
404  {
405  // Set size of Vector: U^C, U^S, W^C, W^S, V^C, V^S
406  values.resize(4*DIM);
407 
408  // Initialise to zero
409  for(unsigned i=0;i<4*DIM;i++) { values[i]=0.0; }
410 
411  // Determine number of nodes in element
412  const unsigned n_node = nnode();
413 
414  // Shape functions
415  Shape psif(n_node);
416  shape(s,psif);
417 
418  // Calculate velocities: values[0],...
419  for(unsigned i=0;i<(4*DIM);i++)
420  {
421  // Get the local index at which the i-th velocity is stored
422  const unsigned u_local_index = u_index_linearised_nst(i);
423  for(unsigned l=0;l<n_node;l++)
424  {
425  values[i] += nodal_value(t,l,u_local_index)*psif[l];
426  }
427  }
428  }

434  /// Further build for Crouzeix_Raviart interpolates the internal
435  /// pressure dofs from father element: Make sure pressure values and
436  /// dp/ds agree between fathers and sons at the midpoints of the son
437  /// elements.
438  void further_build()
439  {
440  // Call the generic further build
441  RefineableLinearisedNavierStokesEquations::further_build();
442 
443  using namespace QuadTreeNames;
444 
445  // What type of son am I? Ask my quadtree representation...
446  int son_type=quadtree_pt()->son_type();
447 
448  // Pointer to my father (in element impersonation)
449  RefineableElement* father_el_pt=
450  quadtree_pt()->father_pt()->object_pt();
451 
452  Vector<double> s_father(2);
453 
454  // Son midpoint is located at the following coordinates in father element:
455 
456  // South west son
457  if(son_type==SW)
458  {
459  s_father[0]=-0.5;
460  s_father[1]=-0.5;
461  }
462  // South east son
463  else if(son_type==SE)
464  {
465  s_father[0]= 0.5;
466  s_father[1]=-0.5;
467  }
468  // North east son
469  else if(son_type==NE)
470  {
471  s_father[0]=0.5;
472  s_father[1]=0.5;
473  }
474 
475  // North west son
476  else if(son_type==NW)
477  {
478  s_father[0]=-0.5;
479  s_father[1]= 0.5;
480  }
481 
482  // Pressure values in father element
483  // ---------------------------------
484 
485  // Find pointer to father element
486  RefineableLinearisedQCrouzeixRaviartElement*
487  cast_father_el_pt=
488  dynamic_cast<RefineableLinearisedQCrouzeixRaviartElement*>
489  (father_el_pt);
490 
491  // Set up storage for pressure in father element
492  Vector<double> press(2,0.0);
493 
494  // Loop over pressure components
495  for(unsigned i=0;i<2;i++)
496  {
497  // Get pressure from father element
498  press[i]
499  = cast_father_el_pt->interpolated_p_linearised_nst(s_father,i);
500 
501  // Pressure value gets copied straight into internal dof:
502  internal_data_pt(P_linearised_nst_internal_index[i])->
503  set_value(0,press[i]);
504 
505  // The slopes get copied from father
506  internal_data_pt(P_linearised_nst_internal_index[i])->
507  set_value(1,0.5*cast_father_el_pt
508  ->internal_data_pt(P_linearised_nst_internal_index[i])
509  ->value(1));
510 
511  internal_data_pt(P_linearised_nst_internal_index[i])->
512  set_value(2,0.5*cast_father_el_pt
513  ->internal_data_pt(P_linearised_nst_internal_index[i])
514  ->value(2));
515  } // End of loop over pressure components
516  }

607 
608  // Get nodal indeces of the two pressure components
609  Vector<int> p_index(2);
610  for(unsigned i=0;i<2;i++)
611  {
612  p_index[i] = this->p_index_linearised_nst(i);
613  }

782  /// \short The velocity nodes are the same as the geometric nodes.
783  /// The pressure nodes must be calculated by using the same methods
784  /// as the geometric nodes, but by recalling that there are only two
785  /// pressure nodes per edge.
786  Node* get_interpolating_node_at_local_coordinate(const Vector<double> &s,
787  const int &n_value)
788  {
789  // If we are calculating pressure nodes
790  if(n_value==static_cast<int>(2*DIM)
791  || n_value==static_cast<int>((2*DIM)+1))
792  {
793  // Storage for the index of the pressure node
794  unsigned total_index = 0;
795  // The number of nodes along each 1d edge is 2.
796  const unsigned NNODE_1D = 2;
797  // Storage for the index along each boundary
798  // Note that it's only a 2D spatial element
799  Vector<int> index(2);
800  // Loop over the coordinates
801  for(unsigned i=0;i<2;i++)
802  {
803  // If we are at the lower limit, the index is zero
804  if(s[i]==-1.0) { index[i] = 0; }
805 
806  // If we are at the upper limit, the index is the number of nodes
807  // minus 1
808  else if(s[i] == 1.0) { index[i] = NNODE_1D-1; }
809 
810  // Otherwise, we have to calculate the index in general
811  else
812  {
813  // For uniformly spaced nodes the 0th node number would be
814  double float_index = 0.5*(1.0 + s[i])*(NNODE_1D-1);
815  index[i] = int(float_index);
816  //What is the excess. This should be safe because the
817  //taking the integer part rounds down
818  double excess = float_index - index[i];
819  //If the excess is bigger than our tolerance there is no node,
820  //return null
821  if((excess > FiniteElement::Node_location_tolerance) && (
822  (1.0 - excess) > FiniteElement::Node_location_tolerance))
823  {
824  return 0;
825  }
826  }
827  ///Construct the general pressure index from the components.
828  total_index += index[i]*
829  static_cast<unsigned>(pow(static_cast<float>(NNODE_1D),
830  static_cast<int>(i)));
831  }
832  // If we've got here we have a node, so let's return a pointer to it
833  return this->node_pt(this->Pconv[total_index]);
834  }
835  // Otherwise velocity nodes are the same as pressure nodes
836  else
837  {
838  return this->get_node_at_local_coordinate(s);
839  }
840  }
