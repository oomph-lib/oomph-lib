127  /// \short Get 'flux' for Z2 error recovery: Upper triangular entries
128  /// in strain rate tensor.
129  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)
130  {
131 #ifdef PARANOID
132  unsigned num_entries=3;
133  if (flux.size()!=num_entries)
134  {
135  std::ostringstream error_message;
136  error_message << "The flux vector has the wrong number of entries, "
137  << flux.size() << ", whereas it should be "
138  << num_entries << std::endl;
139  throw OomphLibError(error_message.str(),
140  OOMPH_CURRENT_FUNCTION,
141  OOMPH_EXCEPTION_LOCATION);
142  }
143 #endif
144 
145  // Get strain rate matrix
146  DenseMatrix<double> strainrate(2);
147  //this->strain_rate(s,strainrate);
148  this->strain_rate_by_r(s,strainrate);
149 
150  // Pack into flux Vector
151  unsigned icount=0;
152 
153  // Start with diagonal terms
154  for(unsigned i=0;i<2;i++)
155  {
156  flux[icount]=strainrate(i,i);
157  icount++;
158  }
159 
160  //Off diagonals row by row
161  for(unsigned i=0;i<2;i++)
162  {
163  for(unsigned j=i+1;j<2;j++)
164  {
165  flux[icount]=strainrate(i,j);
166  icount++;
167  }
168  }
169  }

309 /// \short Get the function value u in Vector.
310 /// Note: Given the generality of the interface (this function
311 /// is usually called from black-box documentation or interpolation routines),
312 /// the values Vector sets its own size in here.
313  void get_interpolated_values(const unsigned& t,
314  const Vector<double>&s,
315  Vector<double>& values)
316  {
317 #ifdef PARANOID
318  //Find out the number of timesteps (present & previous)
319  // (the present element might not have been initialised yet but
320  // its root must know about the time integrator)
321  RefineableElement* root_el_pt = this->tree_pt()->root_pt()->object_pt();
322 
323  unsigned N_prev_time = root_el_pt->node_pt(0)->
324  time_stepper_pt()->nprev_values();
325 
326  if (t>N_prev_time)
327  {
328  std::ostringstream error_message;
329  error_message
330  << "The value of t in get_interpolated_values(...), "
331  << t << std::endl
332  << "is greater than the number of previous stored timesteps";
333 
334  throw OomphLibError(
335  error_message.str(),
336  OOMPH_CURRENT_FUNCTION,
337  OOMPH_EXCEPTION_LOCATION);
338  }
339 #endif
340 
341  // Set size of Vector: u,v,p
342  values.resize(2+1);
343 
344  // Initialise
345  for(unsigned i=0;i<2+1;i++) {values[i]=0.0;}
346 
347  //Find out how many nodes there are
348  unsigned n_node = this->nnode();
349 
350  // Shape functions
351  Shape psif(n_node);
352  this->shape(s,psif);
353 
354  //Calculate velocities: values[0],...
355  for(unsigned i=0;i<2;i++)
356  {
357  //Get the index at which the i-th velocity is stored
358  unsigned u_nodal_index = this->u_index_pnst(i);
359  for(unsigned l=0;l<n_node;l++)
360  {
361  values[i] += this->nodal_value(t,l,u_nodal_index)*psif[l];
362  }
363  }
364 
365  //Calculate pressure: values[DIM]
366  //(no history is carried in the pressure)
367  values[2] = this->interpolated_p_pnst(s);
368  }

410  /// \short The velocity nodes are the same as the geometric nodes. The
411  /// pressure nodes must be calculated by using the same methods as
412  /// the geometric nodes, but by recalling that there are only two pressure
413  /// nodes per edge.
414  Node* get_interpolating_node_at_local_coordinate(const Vector<double> &s,
415  const int &value_id)
416  {
417  //If we are calculating pressure nodes
418  if(value_id==2)
419  {
420  //Storage for the index of the pressure node
421  unsigned total_index=0;
422  //The number of nodes along each 1d edge is 2.
423  unsigned NNODE_1D = 2;
424  //Storage for the index along each boundary
425  Vector<int> index(2);
426  //Loop over the coordinates
427  for(unsigned i=0;i<2;i++)
428  {
429  //If we are at the lower limit, the index is zero
430  if(s[i]==-1.0)
431  {
432  index[i] = 0;
433  }
434  //If we are at the upper limit, the index is the number of nodes minus 1
435  else if(s[i] == 1.0)
436  {
437  index[i] = NNODE_1D-1;
438  }
439  //Otherwise, we have to calculate the index in general
440  else
441  {
442  //For uniformly spaced nodes the 0th node number would be
443  double float_index = 0.5*(1.0 + s[i])*(NNODE_1D-1);
444  index[i] = int(float_index);
445  //What is the excess. This should be safe because the
446  //taking the integer part rounds down
447  double excess = float_index - index[i];
448  //If the excess is bigger than our tolerance there is no node,
449  //return null
450  if((excess > FiniteElement::Node_location_tolerance) && (
451  (1.0 - excess) > FiniteElement::Node_location_tolerance))
452  {
453  return 0;
454  }
455  }
456  ///Construct the general pressure index from the components.
457  total_index += index[i]*
458  static_cast<unsigned>(pow(static_cast<float>(NNODE_1D),
459  static_cast<int>(i)));
460  }
461  //If we've got here we have a node, so let's return a pointer to it
462  return this->pressure_node_pt(total_index);
463  }
464  //Otherwise velocity nodes are the same as pressure nodes
465  else
466  {
467  return this->get_node_at_local_coordinate(s);
468  }
469  }

500  /// \short Add to the set \c paired_load_data pairs containing
501  /// - the pointer to a Data object
502  /// and
503  /// - the index of the value in that Data object
504  /// .
505  /// for all values (pressures, velocities) that affect the
506  /// load computed in the \c get_load(...) function.
507  /// (Overloads non-refineable version and takes hanging nodes
508  /// into account)
509  void insert_load_data(std::set<std::pair<Data*,unsigned> > &paired_load_data)
510  {
511  //Get the nodal indices at which the velocities are stored
512  unsigned u_index[2];
513  for(unsigned i=0;i<2;i++) {u_index[i] = this->u_index_pnst(i);}
514 
515  //Loop over the nodes
516  unsigned n_node = this->nnode();
517  for(unsigned n=0;n<n_node;n++)
518  {
519  // Pointer to current node
520  Node* nod_pt=this->node_pt(n);
521 
522  // Check if it's hanging:
523  if (nod_pt->is_hanging())
524  {
525  // It's hanging -- get number of master nodes
526  unsigned nmaster=nod_pt->hanging_pt()->nmaster();
527 
528  // Loop over masters
529  for (unsigned j=0;j<nmaster;j++)
530  {
531  Node* master_nod_pt=nod_pt->hanging_pt()->master_node_pt(j);
532 
533  //Loop over the velocity components and add pointer to their data
534  //and indices to the vectors
535  for(unsigned i=0;i<2;i++)
536  {
537  paired_load_data.insert(std::make_pair(master_nod_pt,u_index[i]));
538  }
539  }
540  }
541  //Not hanging
542  else
543  {
544  //Loop over the velocity components and add pointer to their data
545  //and indices to the vectors
546  for(unsigned i=0;i<2;i++)
547  {
548  paired_load_data.insert(std::make_pair(this->node_pt(n),u_index[i]));
549  }
550  }
551  }
552 
553  //Get the nodal index at which the pressure is stored
554  int p_index = this->p_nodal_index_pnst();
555 
556  //Loop over the pressure data
557  unsigned n_pres= this->npres_pnst();
558  for(unsigned l=0;l<n_pres;l++)
559  {
560  //Get the pointer to the nodal pressure
561  Node* pres_node_pt = this->pressure_node_pt(l);
562  // Check if the pressure dof is hanging
563  if(pres_node_pt->is_hanging(p_index))
564  {
565  //Get the pointer to the hang info object
566  // (pressure is stored as p_index--th nodal dof).
567  HangInfo* hang_info_pt = pres_node_pt->hanging_pt(p_index);
568 
569  // Get number of pressure master nodes (pressure is stored
570  unsigned nmaster = hang_info_pt->nmaster();
571 
572  // Loop over pressure master nodes
573  for(unsigned m=0;m<nmaster;m++)
574  {
575  //The p_index-th entry in each nodal data is the pressure, which
576  //affects the traction
577  paired_load_data.insert(
578  std::make_pair(hang_info_pt->master_node_pt(m),p_index));
579  }
580  }
581  // It's not hanging
582  else
583  {
584  //The p_index-th entry in each nodal data is the pressure, which
585  //affects the traction
586  paired_load_data.insert(std::make_pair(pres_node_pt,p_index));
587  }
588  }
589 
590  }

834 inline void RefineablePolarCrouzeixRaviartElement::
835 rebuild_from_sons(Mesh* &mesh_pt)
836 {
837  using namespace QuadTreeNames;
838 
839  //Central pressure value:
840  //-----------------------
841 
842  // Use average of the sons central pressure values
843  // Other options: Take average of the four (discontinuous)
844  // pressure values at the father's midpoint]
845 
846  double av_press=0.0;
847 
848  // Loop over the sons
849  for (unsigned ison=0;ison<4;ison++)
850  {
851  // Add the sons midnode pressure
852  // Note that we can assume that the pressure is stored at the same
853  // location because these are EXACTLY the same type of elements
854  av_press += quadtree_pt()->son_pt(ison)->object_pt()->
855  internal_data_pt(this->P_pnst_internal_index)->value(0);
856  }
857 
858  // Use the average
859  internal_data_pt(this->P_pnst_internal_index)->set_value(0,0.25*av_press);
860 
861 
862  //Slope in s_0 direction
863  //----------------------
864 
865  // Use average of the 2 FD approximations based on the
866  // elements central pressure values
867  // [Other options: Take average of the four
868  // pressure derivatives]
869 
870  double slope1=
871  quadtree_pt()->son_pt(SE)->object_pt()->
872  internal_data_pt(this->P_pnst_internal_index)->value(0) -
873  quadtree_pt()->son_pt(SW)->object_pt()->
874  internal_data_pt(this->P_pnst_internal_index)->value(0);
875 
876  double slope2 =
877  quadtree_pt()->son_pt(NE)->object_pt()->
878  internal_data_pt(this->P_pnst_internal_index)->value(0) -
879  quadtree_pt()->son_pt(NW)->object_pt()->
880  internal_data_pt(this->P_pnst_internal_index)->value(0);
881 
882 
883  // Use the average
884  internal_data_pt(this->P_pnst_internal_index)->
885  set_value(1,0.5*(slope1+slope2));
886 
887 
888 
889  //Slope in s_1 direction
890  //----------------------
891 
892  // Use average of the 2 FD approximations based on the
893  // elements central pressure values
894  // [Other options: Take average of the four
895  // pressure derivatives]
896 
897  slope1 =
898  quadtree_pt()->son_pt(NE)->object_pt()
899  ->internal_data_pt(this->P_pnst_internal_index)->value(0) -
900  quadtree_pt()->son_pt(SE)->object_pt()
901  ->internal_data_pt(this->P_pnst_internal_index)->value(0);
902 
903  slope2=
904  quadtree_pt()->son_pt(NW)->object_pt()
905  ->internal_data_pt(this->P_pnst_internal_index)->value(0) -
906  quadtree_pt()->son_pt(SW)->object_pt()
907  ->internal_data_pt(this->P_pnst_internal_index)->value(0);
908 
909 
910  // Use the average
911  internal_data_pt(this->P_pnst_internal_index)->
912  set_value(2,0.5*(slope1+slope2));
913 }
