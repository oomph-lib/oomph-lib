39  /// Strouhal number St = a/(UT) for application of no slip condition.
40  /// Initialised to 1.0.
41  double Strouhal_for_no_slip=1.0;
42 
43  /// Apply no-slip condition for N.St. on a moving wall node
44  /// u = St dR/dt, where the Strouhal number St = a/(UT) is defined by
45  /// FSI_functions::Strouhal_for_no_slip and is initialised to 1.0.
46  /// Note: This requires the x,y,[z] velocity components to be stored
47  /// in nodal values 0,1,[2]. This is the default for all currently
48  /// existing Navier-Stokes elements. If you use any others,
49  /// use this function at your own risk.
50  void apply_no_slip_on_moving_wall(Node* node_pt)
51  {
52  // Find out spatial dimension of node
53  unsigned ndim=node_pt->ndim();
54  // Assign velocity
55  for (unsigned i=0;i<ndim;i++)
56  {
57  node_pt->set_value(i,Strouhal_for_no_slip*(node_pt->dposition_dt(i)));
58  }
59  }
60 
61 }

148  //==================================================================
149  /// Get the contribution to the load vector provided by
150  /// the adjacent fluid element: Pass number of integration point
151  /// in solid element,
152  /// and the unit normal vector (pointing into the fluid on the "front"
153  /// of the FSIWallElement) and return the load vector.
154  /// Note that the load is non-dimensionalised on the wall-stress scale,
155  /// i.e. it is obtained by computing the traction (on the fluid stress-scale)
156  /// from the adjacent fluid element and then multiplying it by
157  /// the stress-scale-ratio \f$ Q. \f$.
158  /// If element is loaded on both sides, the fluid load computed
159  /// from the reverse element is \b subtracted, because it's
160  /// computed with the same normal vector.
161  //==================================================================
162  void FSIWallElement::fluid_load_vector(const unsigned& intpt,
163  const Vector<double>& N,
164  Vector<double>& load)
165  {
166 
167  // Size of load vector
168  unsigned n_load=load.size();
169 
170  // Initialise
171  for (unsigned i=0;i<n_load;i++) load[i]=0.0;
172 
173  // Create vector for fluid load
174  Vector<double> fluid_load(n_load);
175 
176  // Loop over front and back if required: Get number of fluid-loaded faces
177  unsigned n_loaded_face=2;
178  if (Only_front_is_loaded_by_fluid) n_loaded_face=1;
179 
180  for (unsigned face=0;face<n_loaded_face;face++)
181  {
182  //Get the local coordinate in the fluid element (copy
183  // operation for Vector)
184  Vector<double> s_adjacent(external_element_local_coord(face,intpt));
185 
186  //Call the load function for adjacent element
187  FSIFluidElement* el_f_pt=dynamic_cast<FSIFluidElement*>
188  (external_element_pt(face,intpt));
189  if (el_f_pt!=0)
190  {
191  el_f_pt->get_load(s_adjacent,N,fluid_load);
192  }
193  else
194  {
195 #ifdef PARANOID
196  if (!Dont_warn_about_missing_adjacent_fluid_elements)
197  {
198  std::ostringstream warning_stream;
199  warning_stream
200  << "Info: No adjacent element set in FSIWallElement.\n\n"
201  << "Note: you can disable this message by setting \n "
202  << "FSIWallElement::Dont_warn_about_missing_adjacent_fluid_elements"
203  << "\n to true or recompiling without PARANOID.\n";
204  OomphLibWarning(warning_stream.str(),
205  "FSIWallElement::fluid_load_vector()",
206  OOMPH_EXCEPTION_LOCATION);
207  }
208 #endif
209  }
210 
211  // Adjust sign of fluid traction. If normal on the front
212  // points into the fluid, the normal at the back points away
213  // from it.
214  double sign=1.0;
215  if (face==1) sign=-1.0;
216 
217  //The load is scaled by the stress-scale ratio Q
218  for(unsigned i=0;i<n_load;i++)
219  {
220  load[i] += fluid_load[i]*sign*q();
221  }
222 
223  } // end of loop over faces
224 
225  }

229  /// Update the nodal positions in all fluid elements that affect
230  /// the traction on this FSIWallElement
231  //==================================================================
232  void FSIWallElement::node_update_adjacent_fluid_elements()
233  {
234  // Don't update elements repeatedly
235  std::map<FSIFluidElement*,bool> done;
236 
237  // Number of integration points
238  unsigned n_intpt=integral_pt()->nweight();
239 
240  // Loop over front and back if required: Get number of fluid-loaded faces
241  unsigned n_loaded_face=2;
242  if (Only_front_is_loaded_by_fluid) n_loaded_face=1;
243  for (unsigned face=0;face<n_loaded_face;face++)
244  {
245 
246  // Loop over all integration points in wall element
247  for (unsigned iint=0;iint<n_intpt;iint++)
248  {
249  // Get fluid element that affects this integration point
250  FSIFluidElement* el_f_pt=dynamic_cast<FSIFluidElement*>
251  (external_element_pt(face,iint));
252 
253  // Is there an adjacent fluid element?
254  if (el_f_pt!=0)
255  {
256  // Have we updated its positions yet?
257  if (!done[el_f_pt])
258  {
259  // Update nodal positions
260  el_f_pt->node_update();
261  done[el_f_pt]=true;
262  }
263  }
264  }
265  }
266  }

277 /// Overload the function that must return all field data involved
278 /// in the interactions from the external (fluid) element. It allows
279 /// the velocity degrees of freedom to be ignored if we want to
280 /// ignore the shear stresses when computing the Jacobian.
281 //=======================================================================
282  void FSIWallElement::identify_all_field_data_for_external_interaction(
283  Vector<std::set<FiniteElement*> > const &external_elements_pt,
284  std::set<std::pair<Data*,unsigned> > &paired_interaction_data)
285  {
286  //Loop over each inteaction
287  const unsigned n_interaction = this->ninteraction();
288  for(unsigned i=0;i<n_interaction;i++)
289  {
290  //Loop over each element in the set
291  for(std::set<FiniteElement*>::const_iterator it=
292  external_elements_pt[i].begin();
293  it != external_elements_pt[i].end(); it++)
294  {
295  //Cast the element the specific fluid element
296  FSIFluidElement *external_fluid_el_pt =
297  dynamic_cast<FSIFluidElement*>(*it);
298 
299  //Only add pressure load if so desired
300  if (Ignore_shear_stress_in_jacobian)
301  {
302  // Add the "pressure" load data
303  external_fluid_el_pt->identify_pressure_data(paired_interaction_data);
304  }
305  else
306  {
307  // Add the "direct" load data (usually velocities and pressures)
308  // to the set
309  external_fluid_el_pt->identify_load_data(paired_interaction_data);
310  }
311  }
312  } //End of loop over interactions
313  }
