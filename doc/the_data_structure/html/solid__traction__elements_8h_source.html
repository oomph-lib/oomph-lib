312 template<class ELEMENT>
313 void SolidTractionElement<ELEMENT>::
314 fill_in_contribution_to_residuals_solid_traction(Vector<double> &residuals)
315 {
316  //Find out how many nodes there are
317  unsigned n_node = nnode();
318 
319  //Find out how many positional dofs there are
320  unsigned n_position_type = this->nnodal_position_type();
321 
322  //Find out the dimension of the node
323  unsigned n_dim = this->nodal_dimension();
324 
325  //Integer to hold the local equation number
326  int local_eqn=0;
327 
328  //Set up memory for the shape functions
329  //Note that in this case, the number of lagrangian coordinates is always
330  //equal to the dimension of the nodes
331  Shape psi(n_node,n_position_type);
332  DShape dpsids(n_node,n_position_type,n_dim-1);
333 
334  //Set the value of n_intpt
335  unsigned n_intpt = integral_pt()->nweight();
336 
337  //Loop over the integration points
338  for(unsigned ipt=0;ipt<n_intpt;ipt++)
339  {
340  //Get the integral weight
341  double w = integral_pt()->weight(ipt);
342 
343  //Only need to call the local derivatives
344  dshape_local_at_knot(ipt,psi,dpsids);
345 
346  //Calculate the Eulerian and Lagrangian coordinates
347  Vector<double> interpolated_x(n_dim,0.0);
348  Vector<double> interpolated_xi(n_dim,0.0);
349 
350  //Also calculate the surface Vectors (derivatives wrt local coordinates)
351  DenseMatrix<double> interpolated_A(n_dim-1,n_dim,0.0);
352 
353  //Calculate displacements and derivatives
354  for(unsigned l=0;l<n_node;l++)
355  {
356  //Loop over positional dofs
357  for(unsigned k=0;k<n_position_type;k++)
358  {
359  //Loop over displacement components (deformed position)
360  for(unsigned i=0;i<n_dim;i++)
361  {
362  //Calculate the Eulerian and Lagrangian positions
363  interpolated_x[i] +=
364  nodal_position_gen(l,bulk_position_type(k),i)*psi(l,k);
365 
366  interpolated_xi[i] +=
367  this->lagrangian_position_gen(l,bulk_position_type(k),i)*psi(l,k);
368 
369  //Loop over LOCAL derivative directions, to calculate the tangent(s)
370  for(unsigned j=0;j<n_dim-1;j++)
371  {
372  interpolated_A(j,i) +=
373  nodal_position_gen(l,bulk_position_type(k),i)*dpsids(l,k,j);
374  }
375  }
376  }
377  }
378 
379  //Now find the local deformed metric tensor from the tangent Vectors
380  DenseMatrix<double> A(n_dim-1);
381  for(unsigned i=0;i<n_dim-1;i++)
382  {
383  for(unsigned j=0;j<n_dim-1;j++)
384  {
385  //Initialise surface metric tensor to zero
386  A(i,j) = 0.0;
387  //Take the dot product
388  for(unsigned k=0;k<n_dim;k++)
389  {
390  A(i,j) += interpolated_A(i,k)*interpolated_A(j,k);
391  }
392  }
393  }
394 
395  //Get the outer unit normal
396  Vector<double> interpolated_normal(n_dim);
397  outer_unit_normal(ipt,interpolated_normal);
398 
399  //Find the determinant of the metric tensor
400  double Adet =0.0;
401  switch(n_dim)
402  {
403  case 2:
404  Adet = A(0,0);
405  break;
406  case 3:
407  Adet = A(0,0)*A(1,1) - A(0,1)*A(1,0);
408  break;
409  default:
410  throw
411  OomphLibError("Wrong dimension in SolidTractionElement",
412  "SolidTractionElement::fill_in_contribution_to_residuals()",
413  OOMPH_EXCEPTION_LOCATION);
414  }
415 
416  //Premultiply the weights and the square-root of the determinant of
417  //the metric tensor
418  double W = w*sqrt(Adet);
419 
420  //Now calculate the load
421  Vector<double> traction(n_dim);
422  get_traction(ipt,
423  interpolated_xi,
424  interpolated_x,
425  interpolated_normal,
426  traction);
427 
428  //=====LOAD TERMS FROM PRINCIPLE OF VIRTUAL DISPLACEMENTS========
429 
430  //Loop over the test functions, nodes of the element
431  for(unsigned l=0;l<n_node;l++)
432  {
433  //Loop of types of dofs
434  for(unsigned k=0;k<n_position_type;k++)
435  {
436  //Loop over the displacement components
437  for(unsigned i=0;i<n_dim;i++)
438  {
439  local_eqn = this->position_local_eqn(l,bulk_position_type(k),i);
440  /*IF it's not a boundary condition*/
441  if(local_eqn >= 0)
442  {
443  //Add the loading terms to the residuals
444  residuals[local_eqn] -= traction[i]*psi(l,k)*W;
445  }
446  }
447  } //End of if not boundary condition
448  } //End of loop over shape functions
449  } //End of loop over integration points
450 
451 }
