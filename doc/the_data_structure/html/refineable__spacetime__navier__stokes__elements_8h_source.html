433  /// \short Get 'flux' for Z2 error recovery: Upper triangular entries
434  /// in strain rate tensor.
435  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)
436  {
437 #ifdef PARANOID
438  // Calculate the number of entries there should be
439  unsigned num_entries=(DIM+(DIM*(DIM-1))/2)+DIM;
440 
441  // Check if the flux vector has the correct size
442  if (flux.size()<num_entries)
443  {
444  std::ostringstream error_message;
445  error_message << "The flux vector has the wrong number of entries, "
446  << flux.size() << ", whereas it should be at least "
447  << num_entries << std::endl;
448  throw OomphLibError(error_message.str(),
449  OOMPH_CURRENT_FUNCTION,
450  OOMPH_EXCEPTION_LOCATION);
451  }
452 #endif
453 
454  // Allocate space for the strain-rate
455  DenseMatrix<double> strainrate(DIM,DIM,0.0);
456 
457  // Get strain rate matrix
458  this->strain_rate(s,strainrate);
459 
460  // Pack into flux Vector
461  unsigned icount=0;
462 
463  // Loop over the diagonal entries
464  for (unsigned i=0;i<DIM;i++)
465  {
466  // Add the next strain rate entry to the flux vector
467  flux[icount]=strainrate(i,i);
468 
469  // Increment the counter
470  icount++;
471  }
472 
473  // Loop over the rows of the matrix
474  for (unsigned i=0;i<DIM;i++)
475  {
476  // Loop over the lower triangular columns
477  for (unsigned j=i+1;j<DIM;j++)
478  {
479  // Add the next strain rate entry to the flux vector
480  flux[icount]=strainrate(i,j);
481 
482  // Increment the counter
483  icount++;
484  }
485  } // for (unsigned i=0;i<DIM;i++)
486 
487  // The number of nodes in the element
488  unsigned n_node=this->nnode();
489 
490  // Set up memory for the shape and test functions
491  Shape psif(n_node);
492 
493  // Set up memory for the derivatives of the shape and test functions
494  DShape dpsifdx(n_node,DIM+1);
495 
496  // Call the derivatives of the shape and test functions
497  dshape_eulerian(s,psif,dpsifdx);
498 
499  // Loop over velocity components
500  for (unsigned j=0;j<DIM;j++)
501  {
502  // Find the index at which the variable is stored
503  unsigned u_nodal_index=this->u_index_nst(j);
504 
505  // Loop over nodes
506  for (unsigned l=0;l<n_node;l++)
507  {
508  // Add the time-derivative contribution from the l-th node
509  flux[icount]+=this->nodal_value(l,u_nodal_index)*dpsifdx(l,DIM);
510  }
511 
512  // Increment the counter
513  icount++;
514  } // for (unsigned j=0;j<DIM;j++)
515  } // End of get_Z2_flux
