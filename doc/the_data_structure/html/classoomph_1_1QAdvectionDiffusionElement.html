template<unsigned DIM, unsigned NNODE_1D>
class oomph::QAdvectionDiffusionElement< DIM, NNODE_1D >

//////////////////////////////////////////////////////////////////////// ////////////////////////////////////////////////////////////////////////

QAdvectionDiffusionElement elements are linear/quadrilateral/brick-shaped Advection Diffusion elements with isoparametric interpolation for the function.
