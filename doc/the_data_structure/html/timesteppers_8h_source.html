53  /// \short Class to keep track of discrete/continous time. It is essential
54  /// to have a single Time object when using multiple time-stepping schemes;
55  /// e.g., in fluid-structure interaction problems, it is common to use
56  /// different schemes for the fluid and solid domains.
57  /// Storage is allocated for the current value
58  /// of the (continuous) time and a limited history of previous timesteps.
59  /// The number of previous timesteps must be equal to the number required
60  /// by the "highest order" scheme.
61  //====================================================================

154  /// \short Return the value of the continuous time at the t-th previous
155  /// time level (t=0: current; t>0 previous).
156  double time(const unsigned& t = 0) const
157  {
158 #ifdef PARANOID
159  if (t > ndt())
160  {
161  using namespace StringConversion;
162  std::string error_msg = "Timestep " + to_string(t) + " out of range!";
163  throw OomphLibError(
164  error_msg, OOMPH_CURRENT_FUNCTION, OOMPH_EXCEPTION_LOCATION);
165  }
166 #endif
167  // Load the current value of the time
168  double time_local = Continuous_time;
169  // Loop over the t previous timesteps and subtract each dt
170  for (unsigned i = 0; i < t; i++)
171  {
172  time_local -= Dt[i];
173  }
174  return time_local;
175  }

177  /// \short Update all stored values of dt by shifting each value along
178  /// the array. This function must be called before starting to solve at a
179  /// new time level.
180  void shift_dt()
181  {
182  unsigned n_dt = Dt.size();
183  // Return straight away if there are no stored timesteps
184  if (n_dt == 0)
185  {
186  return;
187  }
188  // Start from the end of the array and shift every entry back by one
189  for (unsigned i = (n_dt - 1); i > 0; i--)
190  {
191  Dt[i] = Dt[i - 1];
192  }
193  }
194  };

203  /// \short Base class for time-stepping schemes.
204  /// Timestepper provides an approximation of the temporal derivatives of
205  /// Data such that the i-th derivative of the j-th value in Data is
206  /// represented as
207  /// \code
208  /// unsigned n_value=data_pt->nvalue();
209  /// Vector<double> time_derivative(n_value);
210  /// for (unsigned j=0;j<n_value;j++)
211  /// {
212  /// time_derivative[j]=0.0;
213  /// for (unsigned t=0;t<ntstorage();t++)
214  /// {
215  /// time_derivative[j] += data_pt->value(t,j)*weight(i,t);
216  /// }
217  /// }
218  /// \endcode
219  /// where the time index \c t is such that
220  /// - \c t \c = \c 0 refers to the current value of the Data
221  /// - \c t \c > \c 0 refers to the `history' values, i.e. the doubles that
222  /// are stored in Data to represent the value's time derivatives. For BDF
223  /// schemes these doubles are the values at previous times;
224  /// for other timestepping schemes they can represent quantities
225  /// such as previous first and second derivatives, etc.
226  ///
227  /// TimeSteppers can evaluate all derivatives up to their \c order().
228  ///
229  /// The first \c nprev_values() `history values' represent the
230  /// values at the previous timesteps. For BDF schemes,
231  /// \c nprev_values() \c = \c ntstorage()-1 , i.e. all `history values'
232  /// represent previous values. For \c Newmark<NSTEPS>, only the first
233  /// NSTEPS `history values' represent previous values (NSTEPS=1
234  /// gives the normal Newmark scheme).
235  //====================================================================
