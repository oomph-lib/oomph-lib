230  /// Get flux: \f$ \mbox{flux}[i] = \nabla u = \mbox{d}u / \mbox{d}x_i \f$
231  void get_flux(const Vector<double>& s, Vector<double>& flux) const
232  {
233  //Find out how many nodes there are in the element
234  unsigned n_node = nnode();
235 
236  //Get the nodal index at which the unknown is stored
237  unsigned u_nodal_index = u_index_axisym_adv_diff();
238 
239  //Set up memory for the shape and test functions
240  Shape psi(n_node);
241  DShape dpsidx(n_node,2);
242 
243  //Call the derivatives of the shape and test functions
244  dshape_eulerian(s,psi,dpsidx);
245 
246  //Initialise to zero
247  for(unsigned j=0;j<2;j++)
248  {
249  flux[j] = 0.0;
250  }
251 
252  //Loop over nodes
253  for(unsigned l=0;l<n_node;l++)
254  {
255  //Loop over derivative directions
256  for(unsigned j=0;j<2;j++)

258  flux[j] += nodal_value(l,u_nodal_index)*dpsidx(l,j);
259  }
260  }

289  /// Return FE representation of function value u(s) at local coordinate s
290  inline double interpolated_u_adv_diff(const Vector<double> &s) const
291  {
292  //Find number of nodes
293  unsigned n_node = nnode();
294 
295  //Get the nodal index at which the unknown is stored
296  unsigned u_nodal_index = u_index_axisym_adv_diff();
297 
298  //Local shape function
299  Shape psi(n_node);
300 
301  //Find values of shape function
302  shape(s,psi);
303 
304  //Initialise value of u
305  double interpolated_u = 0.0;
306 
307  //Loop over the local nodes and sum
308  for(unsigned l=0;l<n_node;l++)
309  {
310  interpolated_u += nodal_value(l,u_nodal_index)*psi[l];
311  }
312 
313  return(interpolated_u);
314  }

317  ///\short Return derivative of u at point s with respect to all data
318  ///that can affect its value.
319  ///In addition, return the global equation numbers corresponding to the
320  ///data. This is virtual so that it can be overloaded in the
321  ///refineable version
322  virtual void dinterpolated_u_adv_diff_ddata(const Vector<double> &s,
323  Vector<double> &du_ddata,
324  Vector<unsigned> &global_eqn_number)
325  {
326  //Find number of nodes
327  const unsigned n_node = nnode();
328 
329  //Get the nodal index at which the unknown is stored
330  const unsigned u_nodal_index = u_index_axisym_adv_diff();
331 
332  //Local shape function
333  Shape psi(n_node);
334 
335  //Find values of shape function
336  shape(s,psi);
337 
338  //Find the number of dofs associated with interpolated u
339  unsigned n_u_dof = 0;
340  for(unsigned l=0;l<n_node;l++)
341  {
342  int global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
343  //If it's positive add to the count
344  if (global_eqn >=0)
345  {
346  ++n_u_dof;
347  }
348  }
349 
350  //Now resize the storage schemes
351  du_ddata.resize(n_u_dof,0.0);
352  global_eqn_number.resize(n_u_dof,0);
353 
354  //Loop over the nodes again and set the derivatives
355  unsigned count = 0;
356  for(unsigned l=0;l<n_node;l++)
357  {
358  //Get the global equation number
359  int global_eqn=this->node_pt(l)->eqn_number(u_nodal_index);
360  //If it's positive
361  if (global_eqn >= 0)
362  {
363  //Set the global equation number
364  global_eqn_number[count] = global_eqn;
365  //Set the derivative with respect to the unknown
366  du_ddata[count] = psi[l];
367  //Increase the counter
368  ++count;
369  }
370  }
371  }

535 /// \short Define the shape functions and test functions and derivatives
536 /// w.r.t. global coordinates and return Jacobian of mapping.
537 ///
538 /// Galerkin: Test functions = shape functions
539 //======================================================================
540 
541 template<unsigned NNODE_1D>
542 double QSteadyAxisymAdvectionDiffusionElement<NNODE_1D>::
543 dshape_and_dtest_eulerian_adv_diff(const Vector<double> &s,
544  Shape &psi,
545  DShape &dpsidx,
546  Shape &test,
547  DShape &dtestdx) const
548 {
549  //Call the geometrical shape functions and derivatives
550  double J = this->dshape_eulerian(s,psi,dpsidx);
551 
552  //Loop over the test functions and derivatives and set them equal to the
553  //shape functions
554  for(unsigned i=0;i<NNODE_1D;i++)
555  {
556  test[i] = psi[i];
557  for(unsigned j=0;j<2;j++)
558  {
559  dtestdx(i,j) = dpsidx(i,j);
560  }
561  }
562 
563  //Return the jacobian
564  return J;
565 }

576 template<unsigned NNODE_1D>

927 //===========================================================================
928 /// \short Compute the element's residual vector and the (zero) Jacobian
929 /// matrix for the Robin boundary condition:
930 /// \f[
931 /// \Delta u \cdot \mathbf{n} + \alpha (\mathbf{x}) = \beta (\mathbf{x})
932 /// \f]
933 //===========================================================================
934 template<class ELEMENT>
935 void SteadyAxisymAdvectionDiffusionFluxElement<ELEMENT>::
936 fill_in_generic_residual_contribution_adv_diff_flux(Vector<double> &residuals,
937  DenseMatrix<double> &jacobian,
938  unsigned flag)
939 {
940  //Find out how many nodes there are
941  const unsigned n_node = nnode();
942 
943  // Locally cache the index at which the variable is stored
944  const unsigned u_index_axisym_adv_diff = U_index_adv_diff;
945 
946  //Set up memory for the shape and test functions
947  Shape psif(n_node), testf(n_node);
948 
949  //Set the value of n_intpt
950  const unsigned n_intpt = integral_pt()->nweight();
951 
952  //Set the Vector to hold local coordinates
953  Vector<double> s(1);
954 
955  //Integers used to store the local equation number and local unknown
956  //indices for the residuals and jacobians
957  int local_eqn=0, local_unknown=0;
958 
959  //Loop over the integration points
960  //--------------------------------
961  for(unsigned ipt=0;ipt<n_intpt;ipt++)
962  {
963 
964  //Assign values of s
965  for(unsigned i=0;i<1;i++)
966  {
967  s[i] = integral_pt()->knot(ipt,i);
968  }
969 
970  //Get the integral weight
971  double w = integral_pt()->weight(ipt);
972 
973  //Find the shape and test functions and return the Jacobian
974  //of the mapping
975  double J = shape_and_test(s,psif,testf);
976 
977  //Premultiply the weights and the Jacobian
978  double W = w*J;
979 
980  //Calculate local values of the solution and its derivatives
981  //Allocate
982  double interpolated_u=0.0;
983  Vector<double> interpolated_x(2,0.0);
984 
985  //Calculate position
986  for(unsigned l=0;l<n_node;l++)
987  {
988  //Get the value at the node
989  double u_value = raw_nodal_value(l,u_index_axisym_adv_diff);
990  interpolated_u += u_value*psif(l);
991  //Loop over coordinate direction
992  for(unsigned i=0;i<2;i++)
993  {
994  interpolated_x[i] += nodal_position(l,i)*psif(l);
995  }
996  }
997 
998  //Get the imposed beta (beta=flux when alpha=0.0)
999  double beta;
1000  get_beta(interpolated_x,beta);
1001 
1002  //Get the imposed alpha
1003  double alpha;
1004  get_alpha(interpolated_x,alpha);
1005 
1006  //r is the first position component
1007  double r = interpolated_x[0];
1008 
1009  //Now add to the appropriate equations
1010 
1011  //Loop over the test functions
1012  for(unsigned l=0;l<n_node;l++)
1013  {
1014  //Set the local equation number
1015  local_eqn = nodal_local_eqn(l,u_index_axisym_adv_diff);
1016  /*IF it's not a boundary condition*/
1017  if(local_eqn >= 0)
1018  {
1019  //Add the prescribed beta terms
1020  residuals[local_eqn] -= r*(beta - alpha*interpolated_u)*testf(l)*W;
1021 
1022  //Calculate the Jacobian
1023  //----------------------
1024  if ( (flag) && (alpha!=0.0) )
1025  {
1026  //Loop over the velocity shape functions again
1027  for(unsigned l2=0;l2<n_node;l2++)
1028  {
1029  //Set the number of the unknown
1030  local_unknown = nodal_local_eqn(l2,u_index_axisym_adv_diff);
1031 
1032  //If at a non-zero degree of freedom add in the entry
1033  if(local_unknown >= 0)
1034  {
1035  jacobian(local_eqn,local_unknown) +=
1036  r*alpha*psif[l2]*testf[l]*W;
1037  }
1038  }
1039  }
1040 
1041  }
1042  } //end loop over test functions
1043 
1044  } //end loop over integration points
1045 
1046 }//end fill_in_generic_residual_contribution_adv_diff_flux
1047
