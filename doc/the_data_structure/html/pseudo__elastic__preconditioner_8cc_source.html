115  // Setup method for the PseudoElasticPreconditioner.
116  //=============================================================================
117  void PseudoElasticPreconditioner::setup()
118  {
119  // clean
120  this->clean_up_memory();
121 
122 #ifdef PARANOID
123  // paranoid check that meshes have been set
124  if (Elastic_mesh_pt==0)
125  {
126  std::ostringstream error_message;
127  error_message << "The elastic mesh must be set.";
128  throw OomphLibError(error_message.str(),
129  OOMPH_CURRENT_FUNCTION,
130  OOMPH_EXCEPTION_LOCATION);
131 
132  }
133  if (Lagrange_multiplier_mesh_pt==0)
134  {
135  std::ostringstream error_message;
136  error_message << "The Lagrange multiplier mesh must be set.";
137  throw OomphLibError(error_message.str(),
138  OOMPH_CURRENT_FUNCTION,
139  OOMPH_EXCEPTION_LOCATION);
140  }
141 #endif
142 
143  // set the meshes
144  this->set_nmesh(2);
145  this->set_mesh(0,Elastic_mesh_pt);
146  this->set_mesh(1,Lagrange_multiplier_mesh_pt);
147 
148  // Figure out number of dof types
149  unsigned n_solid_dof_types = 0;
150  unsigned n_dof_types = 0;
151 
152  if (this->is_master_block_preconditioner())
153  {
154  // get the number of solid dof types from the first element
155  n_solid_dof_types = this->ndof_types_in_mesh(0);
156 
157  // get the total number of dof types
158  n_dof_types = n_solid_dof_types
159  + this->ndof_types_in_mesh(1);
160  }
161  else
162  {
163  n_dof_types = this->ndof_types();
164  n_solid_dof_types = n_dof_types - (n_dof_types/3);
165  }
166 #ifdef PARANOID
167  if (n_dof_types%3 != 0)
168  {
169  std::ostringstream error_message;
170  error_message << "This preconditioner requires DIM*3 types of DOF";
171  throw OomphLibError(error_message.str(),
172  OOMPH_CURRENT_FUNCTION,
173  OOMPH_EXCEPTION_LOCATION);
174  }
175 #endif
176 
177  // determine the dimension
178  Dim = n_dof_types/3;
179 
180 #ifdef PARANOID
181  // Recast Jacobian matrix to CRDoubleMatrix
182  CRDoubleMatrix* cr_matrix_pt = dynamic_cast<CRDoubleMatrix*>(matrix_pt());
183 
184  if (cr_matrix_pt==0)
185  {
186  std::ostringstream error_message;
187  error_message << "FSIPreconditioner only works with"
188  << " CRDoubleMatrix matrices" << std::endl;
189  throw OomphLibError(error_message.str(),
190  OOMPH_CURRENT_FUNCTION,
191  OOMPH_EXCEPTION_LOCATION);
192  }
193 #endif
194 
195  // Setup the dof_list scheme for block_setup.
196  // The natural DOF types order is (in 3D):
197  // 0 1 2 3 4 5 6 7 8 <- natural DOF type index
198  // x y z xc yc zc lx ly lz <- DOf type
199  //
200  // We need to group the directional displacements together:
201  // 0 1 2 3 4 5 6 7 8 <- block index
202  // x xc y yc xz zc lx ly lz <- DOF type
203  //
204  // The mapping required is:
205  // 0 2 4 1 3 5 6 7 8
206  //
207  // In general we want:
208  //
209  // dof_to_block_map[DOF type] = block type
210  Vector<unsigned> dof_list_for_block_setup(n_dof_types);
211  for (unsigned dim_i = 0; dim_i < Dim; dim_i++)
212  {
213  // bulk solid dof types
214  dof_list_for_block_setup[dim_i] = 2*dim_i;
215 
216  // constrained solid dof types
217  dof_list_for_block_setup[dim_i + Dim] = 2*dim_i + 1;
218 
219  // lagr dof types
220  dof_list_for_block_setup[dim_i + 2*Dim] = 2*Dim + dim_i;
221  }
222 
223  // Setup the block ordering. We have the following:
224  // Block types [0, 2*Dim) are the solid blocks.
225  // Block types [2*Dim, 3*Dim) are the Lagrange multiplier blocks.
226  //
227  // For d = 0, 1, 2,
228  // Bulk solid doftypes: 2*d
229  // Constrained solid doftypes: 2*d+1
230  // Lgr doftyoes: 2*Dim + d
231  this->block_setup(dof_list_for_block_setup);
232 
233  // Create dof_list for subsidiary preconditioner. This will be used later
234  // in turn_into_subsidiary_block_preconditioner(...)
235  Vector<unsigned> dof_list_for_subsidiary_prec(n_solid_dof_types);
236 
237  // The dof types are currently in the order (in 3D):
238  // 0 1 2 3 4 5 6 7 8 <- DOF index
239  // x y z xc yc zc lx ly lz <- DOF type
240  //
241  // We need to group the directional displacements together:
242  // x xc y yc xz zc
243  //
244  // The mapping required is:
245  // 0 3 1 4 2 5
246  //
247  // In general we want:
248  // dof_list[subsidiary DOF type] = master DOF type
249  for (unsigned dim_i = 0; dim_i < Dim; dim_i++)
250  {
251  // bulk solid dof
252  dof_list_for_subsidiary_prec[2*dim_i] = dim_i;
253 
254  // constrained solid dof
255  dof_list_for_subsidiary_prec[2*dim_i+1] = dim_i + Dim;
256  }
257 
258  // Get the solid blocks
259  DenseMatrix<CRDoubleMatrix*> solid_matrix_pt(n_solid_dof_types,
260  n_solid_dof_types,0);
261 
262  for (unsigned row_i = 0; row_i < n_solid_dof_types; row_i++)
263  {
264  for (unsigned col_i = 0; col_i < n_solid_dof_types; col_i++)
265  {
266  solid_matrix_pt(row_i,col_i) = new CRDoubleMatrix;
267  this->get_block(row_i,col_i,*solid_matrix_pt(row_i,col_i));
268  }
269  }
270 
271  // compute the scaling (if required)
272  if (Use_inf_norm_of_s_scaling)
273  {
274  Scaling = CRDoubleMatrixHelpers::inf_norm(solid_matrix_pt);
275  }
276  else
277  {
278  Scaling = 1.0;
279  }
280 
281  // Add the scaled identify matrix to the constrained solid blocks.
282  for(unsigned d = 0; d < Dim; d++)
283  {
284  // Where is the constrained block located?
285  unsigned block_i = 2*d+1;
286 
287  // Data from the constrained block.
288  double* s_values = solid_matrix_pt(block_i,block_i)->value();
289  int* s_column_index = solid_matrix_pt(block_i,block_i)->column_index();
290  int* s_row_start = solid_matrix_pt(block_i,block_i)->row_start();
291  int s_nrow_local = solid_matrix_pt(block_i,block_i)->nrow_local();
292  int s_first_row = solid_matrix_pt(block_i,block_i)->first_row();
293 
294  // Loop through the diagonal entries and add the scaling.
295  for (int i = 0; i < s_nrow_local; i++)
296  {
297  bool found = false;
298  for (int j = s_row_start[i];
299  j < s_row_start[i+1] && !found; j++)
300  {
301  if (s_column_index[j] == i + s_first_row)
302  {
303  s_values[j] += Scaling;
304  found = true;
305  } // if
306  } // for row start
307 
308  // Check if the diagonal entry is found.
309  if(!found)
310  {
311  std::ostringstream error_message;
312  error_message << "The diagonal entry for the constained block("
313  << block_i<<","<<block_i<<")\n"
314  << "on local row " << i << " does not exist."
315  << std::endl;
316  throw OomphLibError(error_message.str(),
317  OOMPH_CURRENT_FUNCTION,
318  OOMPH_EXCEPTION_LOCATION);
319  }
320  } // for nrow_local
321  } // for Dim
322 
323 
324  // setup the solid subsidiary preconditioner ////////////////////////////////
325  // this preconditioner uses the full S matrix
326  if (E_preconditioner_type == Exact_block_preconditioner)
327  {
328  ExactBlockPreconditioner<CRDoubleMatrix>* s_prec_pt =
329  new ExactBlockPreconditioner<CRDoubleMatrix>;
330 
331  // Add mesh (not actually used since this only acts as a subsidiary
332  // preconditioner...
333  //s_prec_pt->add_mesh(Elastic_mesh_pt);
334 
335  Vector<Vector<unsigned> > doftype_to_doftype_map(n_solid_dof_types,
336  Vector<unsigned>(1,0));
337 
338  for (unsigned i = 0; i < n_solid_dof_types; i++)
339  {
340  doftype_to_doftype_map[i][0] = i;
341  }
342 
343  s_prec_pt->turn_into_subsidiary_block_preconditioner(
344  this,dof_list_for_subsidiary_prec,doftype_to_doftype_map);
345 
346  if (Elastic_subsidiary_preconditioner_function_pt != 0)
347  {
348  s_prec_pt->
349  set_subsidiary_preconditioner_function
350  (Elastic_subsidiary_preconditioner_function_pt);
351  }
352 
353  // Set the replacement blocks.
354  for (unsigned d = 0; d < Dim; d++)
355  {
356  // The dof-block located in the block-ordering.
357  unsigned block_i = 2*d+1;
358 
359  // The dof-block located in the dof-ordering.
360  unsigned dof_block_i = Dim + d;
361  this->set_replacement_dof_block(dof_block_i,dof_block_i,
362  solid_matrix_pt(block_i,block_i));
363  }
364 
365  s_prec_pt->Preconditioner::setup(matrix_pt());
366  Elastic_preconditioner_pt = s_prec_pt;
367  }
368  // otherwise it is a block based preconditioner
369  else
370  {
371  GeneralPurposeBlockPreconditioner<CRDoubleMatrix>* s_prec_pt = 0;
372 
373  // set the block preconditioning method
374  switch (E_preconditioner_type)
375  {
376  case Block_diagonal_preconditioner:
377  {
378  s_prec_pt = new BlockDiagonalPreconditioner<CRDoubleMatrix>;
379  }
380  break;
381  case Block_upper_triangular_preconditioner:
382  {
383  BlockTriangularPreconditioner<CRDoubleMatrix>* block_triangular_prec_pt
384  = new BlockTriangularPreconditioner<CRDoubleMatrix>;
385  block_triangular_prec_pt->upper_triangular();
386 
387  s_prec_pt = block_triangular_prec_pt;
388  }
389  break;
390  case Block_lower_triangular_preconditioner:
391  {
392  BlockTriangularPreconditioner<CRDoubleMatrix>* block_triangular_prec_pt
393  = new BlockTriangularPreconditioner<CRDoubleMatrix>;
394  block_triangular_prec_pt->lower_triangular();
395 
396  s_prec_pt = block_triangular_prec_pt;
397  }
398  break;
399  default:
400  {
401  std::ostringstream error_msg;
402  error_msg << "There is no such block based preconditioner.\n"
403  << "Candidates are:\n"
404  << "PseudoElasticPreconditioner::Block_diagonal_preconditioner\n"
405  << "PseudoElasticPreconditioner::Block_upper_triangular_preconditioner\n"
406  << "PseudoElasticPreconditioner::Block_lower_triangular_preconditioner\n";
407  throw OomphLibError(error_msg.str(),
408  OOMPH_CURRENT_FUNCTION,
409  OOMPH_EXCEPTION_LOCATION);
410  }
411  break;
412  }
413 
414 
415  // Add mesh (not actually used since this only acts as a subsidiary
416  // preconditioner...
417  //s_prec_pt->add_mesh(Elastic_mesh_pt);
418 
419  // The block to block map
420  Vector<Vector<unsigned> > doftype_to_doftype_map(
421  Dim,Vector<unsigned>(2,0));
422  Vector<unsigned> s_prec_dof_to_block_map(Dim,0);
423 
424  unsigned tmp_index = 0;
425  for (unsigned d = 0; d < Dim; d++)
426  {
427  s_prec_dof_to_block_map[d] = d;
428  doftype_to_doftype_map[d][0] = tmp_index++;
429  doftype_to_doftype_map[d][1] = tmp_index++;
430  }
431 
432  s_prec_pt->turn_into_subsidiary_block_preconditioner
433  (this,dof_list_for_subsidiary_prec,doftype_to_doftype_map);
434 
435  if(Elastic_subsidiary_preconditioner_function_pt != 0)
436  {
437  s_prec_pt->set_subsidiary_preconditioner_function
438  (Elastic_subsidiary_preconditioner_function_pt);
439  }
440 
441  // Set the replacement blocks.
442  for (unsigned d = 0; d < Dim; d++)
443  {
444  // The dof-block located in the block-ordering.
445  unsigned block_i = 2*d+1;
446 
447  // Then dof-block located in the dof-ordering.
448  unsigned dof_block_i = Dim + d;
449  this->set_replacement_dof_block(dof_block_i,dof_block_i,
450  solid_matrix_pt(block_i,block_i));
451  }
452 
453  s_prec_pt->set_dof_to_block_map(s_prec_dof_to_block_map);
454  s_prec_pt->Preconditioner::setup(matrix_pt());
455 
456  Elastic_preconditioner_pt = s_prec_pt;
457  }
458 
459  // No longer require the solid blocks
460  for (unsigned row_i = 0; row_i < n_solid_dof_types; row_i++)
461  {
462  for (unsigned col_i = 0; col_i < n_solid_dof_types; col_i++)
463  {
464  delete solid_matrix_pt(row_i,col_i); solid_matrix_pt(row_i,col_i) = 0;
465  }
466  }
467 
468  // next setup the lagrange multiplier preconditioners
469  Lagrange_multiplier_preconditioner_pt.resize(Dim);
470  for (unsigned d = 0; d < Dim; d++)
471  {
472  CRDoubleMatrix* b_pt = new CRDoubleMatrix;
473  this->get_block(2*Dim+d,2*d+1,*b_pt);
474 
475  // if a non default preconditioner is specified create
476  // the preconditioners
477  if (Lagrange_multiplier_subsidiary_preconditioner_function_pt != 0)
478  {
479  Lagrange_multiplier_preconditioner_pt[d] =
480  (*Lagrange_multiplier_subsidiary_preconditioner_function_pt)();
481  }
482  // else use default superlu preconditioner
483  else
484  {
485  Lagrange_multiplier_preconditioner_pt[d] = new SuperLUPreconditioner;
486  }
487 
488  // and setup
489  Lagrange_multiplier_preconditioner_pt[d]->setup(b_pt);
490  delete b_pt; b_pt = 0;
491  }
492 
493  }

505  //=============================================================================
506  /// \short Apply the lagrange multiplier subsidiary preconditioner.
507  //=============================================================================
508  void PseudoElasticPreconditioner::
509  lagrange_multiplier_preconditioner_solve(const DoubleVector& r,
510  DoubleVector& z)
511  {
512  // apply the lagrange multiplier preconditioner
513  for (unsigned d = 0; d < Dim; d++)
514  {
515  DoubleVector x;
516  this->get_block_vector(Dim*2+d,r,x);
517  DoubleVector y;
518  Lagrange_multiplier_preconditioner_pt[d]->preconditioner_solve(x,y);
519  Lagrange_multiplier_preconditioner_pt[d]->preconditioner_solve(y,x);
520  unsigned nrow_local = x.nrow_local();
521  double* x_pt = x.values_pt();
522  for (unsigned i = 0; i < nrow_local; i++)
523  {
524  x_pt[i] = x_pt[i] * Scaling;
525  }
526  this->return_block_vector(Dim*2+d,x,z);
527  }
