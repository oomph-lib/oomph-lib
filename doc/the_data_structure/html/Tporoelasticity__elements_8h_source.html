266  /// Returns the nth gauss point along an edge: if sign_edge(edge)==1, returns
267  /// regular gauss point; if sign_edge(edge)==-1, returns 1-(gauss point)
268  double edge_gauss_point(const unsigned &edge,
269  const unsigned &n) const
270  {
271 #ifdef RANGE_CHECKING
272  if(edge >= 3)
273  {
274  std::ostringstream error_message;
275  error_message << "Range Error: edge " << edge
276  << " is not in the range (0,2)";
277  throw OomphLibError(error_message.str(),
278  OOMPH_CURRENT_FUNCTION,
279  OOMPH_EXCEPTION_LOCATION);
280  }
281  if(n >= nedge_gauss_point())
282  {

285  << " is not in the range (0,"
286  << nedge_gauss_point()-1 <<")";
287  throw OomphLibError(error_message.str(),
288  OOMPH_CURRENT_FUNCTION,

295  /// Returns the global coordinates of the nth gauss point along an edge
296  void edge_gauss_point_global(const unsigned &edge,
297  const unsigned &n,
298  Vector<double> &x) const
299  {
300 #ifdef RANGE_CHECKING
301  if(edge >= 3)
302  {
303  std::ostringstream error_message;
304  error_message << "Range Error: edge " << edge
305  << " is not in the range (0,2)";
306  throw OomphLibError(error_message.str(),
307  OOMPH_CURRENT_FUNCTION,
308  OOMPH_EXCEPTION_LOCATION);
309  }
310  if(n >= nedge_gauss_point())
311  {
312  std::ostringstream error_message;
313  error_message << "Range Error: n " << n
314  << " is not in the range (0,"
315  << nedge_gauss_point()-1 <<")";
316  throw OomphLibError(error_message.str(),
317  OOMPH_CURRENT_FUNCTION,
318  OOMPH_EXCEPTION_LOCATION);
319  }
320 #endif
321 
322  // Get the location of the n-th gauss point along the edge in terms of the
323  // distance along the edge itself
324  double gauss_point=edge_gauss_point(edge,n);
325 
326  // Convert the edge number to the number of the mid-edge node along that
327  // edge
328  unsigned node_number=Q_edge_conv[edge];
329 
330  // Storage for the local coords of the gauss point
331  Vector<double> s_gauss(2,0);
332 
333  // The edge basis functions are defined in a clockwise manner, so we have
334  // to effectively "flip" the coordinates along edges 0 and 1 to match this
335  switch(node_number)
336  {
337  case 3:
338  s_gauss[0]=1-gauss_point;
339  s_gauss[1]=gauss_point;
340  break;
341  case 4:
342  s_gauss[0]=0;
343  s_gauss[1]=1-gauss_point;
344  break;
345  case 5:
346  s_gauss[0]=gauss_point;
347  s_gauss[1]=0;
348  break;
349  }
350 
351  // Calculate the global coordinates from the local ones
352  interpolated_x(s_gauss,x);
353  }

435  /// Scale the edge basis to allow arbitrary edge mappings
436  void scale_basis(Shape& basis) const
437  {
438  // Storage for the lengths of the edges of the element
439  Vector<double> length(3,0.0);
440 
441  // Temporary storage for the vertex positions
442  double x0,y0,x1,y1;
443 
444  // loop over the edges of the element and calculate their lengths (in x-y
445  // space)
446  for(unsigned i=0;i<3;i++)
447  {
448  x0=this->node_pt(i)->x(0);
449  y0=this->node_pt(i)->x(1);
450  x1=this->node_pt((i+1)%3)->x(0);
451  y1=this->node_pt((i+1)%3)->x(1);
452 
453  length[i]=std::sqrt(std::pow(y1-y0,2)+std::pow(x1-x0,2));
454  }
455 
456  // lengths of the sides of the reference element (in the same order as the
457  // basis functions)
458  const double ref_length[3]={std::sqrt(2.0),1,1};
459 
460  // get the number of basis functions associated with the edges
461  unsigned n_q_basis_edge=nq_basis_edge();
462 
463  // rescale the edge basis functions to allow arbitrary edge mappings from
464  // element to ref. element
465  const unsigned n_index2 = basis.nindex2();
466  for(unsigned i=0;i<n_index2;i++)
467  {
468  for(unsigned l=0;l<n_q_basis_edge;l++)
469  {
470  basis(l,i)*=(length[l/(ORDER+1)]/ref_length[l/(ORDER+1)]);
471  }
472  }
473  }
