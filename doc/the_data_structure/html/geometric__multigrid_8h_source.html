152  /// Clean up anything that needs to be cleaned up
153  void clean_up_memory()
154  {
155  // We only need to destroy data if the solver has been set up and
156  // the data hasn't already been cleared
157  if (Has_been_setup)
158  {
159  // Loop over all of the levels in the hierarchy
160  for (unsigned i=0; i<Nlevel-1; i++)
161  {
162  // Delete the pre-smoother associated with this level
163  delete Pre_smoothers_storage_pt[i];
164 
165  // Make it a null pointer

175  delete Mg_matrices_storage_pt[i];

177  // Make it a null pointer

181  // Loop over all but the coarsest of the levels in the hierarchy
182  for (unsigned i=0; i<Nlevel-1; i++)
183  {

185  delete Interpolation_matrices_storage_pt[i];

188  Interpolation_matrices_storage_pt[i]=0;

514  /// \short Virtual function in the base class that needs to be implemented
515  /// later but for now just leave it empty
516  void solve(Problem* const& problem_pt, DoubleVector& result)
517  {
518  // Dynamically cast problem_pt of type Problem to a MGProblem pointer
519  MGProblem* mg_problem_pt=dynamic_cast<MGProblem*>(problem_pt);
520 
521 #ifdef PARANOID
522  // PARANOID check - If the dynamic_cast produces a null pointer the
523  // input was not a MGProblem
524  if (0==mg_problem_pt)
525  {
526  throw OomphLibError("Input problem must be of type MGProblem.",
527  OOMPH_CURRENT_FUNCTION,
528  OOMPH_EXCEPTION_LOCATION);
529  }
530  // PARANOID check - If a node in the input problem has more than
531  // one value we cannot deal with it so arbitarily check the first one
532  if (problem_pt->mesh_pt()->node_pt(0)->nvalue()!=1)
533  {
534  // How many dofs are there in the first node
535  unsigned n_value=problem_pt->mesh_pt()->node_pt(0)->nvalue();
536 
537  // Make the error message
538  std::ostringstream error_message_stream;
539  error_message_stream << "Cannot currently deal with more than 1 dof"
540  << " per node. This problem has " << n_value
541  << " dofs per node."
542  << std::endl;
543 
544  // Throw the error message
545  throw OomphLibError(error_message_stream.str(),
546  OOMPH_CURRENT_FUNCTION,
547  OOMPH_EXCEPTION_LOCATION);
548  }
549 #endif
550 
551  // Assign the new MGProblem pointer to Mg_problem_pt
552  Mg_problem_pt=mg_problem_pt;
553 
554  // Set up all of the required MG structures
555  full_setup();
556 
557  // Run the MG method and assign the solution to result
558  mg_solve(result);
559 
560  // Only output if the V-cycle output isn't suppressed
561  if (!Suppress_v_cycle_output)
562  {
563  // Notify user that the hierarchy of levels is complete
564  oomph_info << "\n================="
565  << "Multigrid Solve Complete"
566  << "=================\n"
567  << std::endl;
568  }
569 
570  // If the user did not request all output be suppressed
571  if (!Suppress_all_output)
572  {
573  // If the user requested all V-cycle output be suppressed
574  if (Suppress_v_cycle_output)
575  {
576  // Create an extra line spacing
577  oomph_info << std::endl;
578  }
579 
580  // Output number of V-cycles taken to solve
581  if (Has_been_solved)
582  {
583  oomph_info << "Total number of V-cycles required for solve: "
584  << V_cycle_counter << std::endl;
585  }
586  else
587  {
588  oomph_info << "Total number of V-cycles used: "
589  << V_cycle_counter << std::endl;
590  }
591  } // if (!Suppress_all_output)
592 
593  // Only enable and assign the stream pointer again if we originally
594  // suppressed everything otherwise it won't be set yet
595  if (Suppress_all_output)
596  {
597  // Now enable the stream pointer again
598  oomph_info.stream_pt()=Stream_pt;
599  }
600  } // End of solve

975  /// \short Set up the MG hierarchy
976  //===================================================================
977  // Function to set up the hierachy of levels. Creates a vector of
978  // pointers to each MG level
979  template<unsigned DIM>
980  void MGSolver<DIM>::setup_mg_hierarchy()
981  {
982  // Initialise the timer start variable
983  double t_m_start=0.0;
984 
985  // Notify the user if it is allowed
986  if (!Suppress_all_output)
987  {
988  // Notify user of progress
989  oomph_info << "\n===============Creating Multigrid Hierarchy==============="
990  << std::endl;
991 
992  // Start clock
993  t_m_start=TimingHelpers::timer();
994  }
995 
996  // Create a bool to indicate whether or not we could create an unrefined
997  // copy. This bool will be assigned the value FALSE when the current copy
998  // is the last level of the multigrid hierarchy
999  bool managed_to_create_unrefined_copy=true;
1000 
1001  // Now keep making copies and try to make an unrefined copy of
1002  // the mesh
1003  unsigned level=0;
1004 
1005  // Set up all of the levels by making a completely unrefined copy
1006  // of the problem using the function make_new_problem
1007  while (managed_to_create_unrefined_copy)
1008  {
1009  // Make a new object of the same type as the derived problem
1010  MGProblem* new_problem_pt=Mg_problem_pt->make_new_problem();
1011 
1012  // Do anything that needs to be done before we can refine the mesh
1013  new_problem_pt->actions_before_adapt();
1014 
1015  // To create the next level in the hierarchy we need to create a mesh
1016  // which matches the refinement of the current problem and then unrefine
1017  // the mesh. This can alternatively be done using the function
1018  // refine_base_mesh_as_in_reference_mesh_minus_one which takes a
1019  // reference mesh to do precisely the above with
1020  managed_to_create_unrefined_copy=
1021  new_problem_pt->mg_bulk_mesh_pt()->
1022  refine_base_mesh_as_in_reference_mesh_minus_one(
1023  Mg_hierarchy[level]->mg_bulk_mesh_pt());
1024 
1025  // If we were able to unrefine the problem on the current level
1026  // then add the unrefined problem to a vector of the levels
1027  if (managed_to_create_unrefined_copy)
1028  {
1029  // Another level has been created so increment the level counter
1030  level++;
1031 
1032  // If the documentation of everything has not been suppressed
1033  // then tell the user we managed to create another level
1034  if (!Suppress_all_output)
1035  {
1036  // Notify user that unrefinement was successful
1037  oomph_info << "\nSuccess! Level " << level
1038  << " has been created." << std::endl;
1039  }
1040 
1041  // Do anything that needs to be done after refinement
1042  new_problem_pt->actions_after_adapt();
1043 
1044  // Do the equation numbering for the new problem
1045  oomph_info << "\n - Number of equations: "
1046  << new_problem_pt->assign_eqn_numbers() << std::endl;
1047 
1048  // Add the new problem pointer onto the vector of MG levels
1049  // and increment the value of level by 1
1050  Mg_hierarchy.push_back(new_problem_pt);
1051  }
1052  // If we weren't able to create an unrefined copy
1053  else
1054  {
1055  // Delete the new problem
1056  delete new_problem_pt;
1057 
1058  // Make it a null pointer
1059  new_problem_pt=0;
1060 
1061  // Assign the number of levels to Nlevel
1062  Nlevel=Mg_hierarchy.size();
1063 
1064  // If we're allowed to document then tell the user we've reached
1065  // the coarsest level of the hierarchy
1066  if (!Suppress_all_output)
1067  {
1068  // Notify the user
1069  oomph_info << "\n Reached the coarsest level! "
1070  << "Number of levels: " << Nlevel << std::endl;
1071  }
1072  } // if (managed_to_unrefine)
1073  } // while (managed_to_unrefine)
1074 
1075  // Given that we know the number of levels in the hierarchy we can resize the
1076  // vectors which will store all the information required for our solver:
1077  // Resize the vector storing all of the system matrices
1078  Mg_matrices_storage_pt.resize(Nlevel,0);
1079 
1080  // Resize the vector storing all of the solution vectors (X_mg)
1081  X_mg_vectors_storage.resize(Nlevel);
1082 
1083  // Resize the vector storing all of the RHS vectors (Rhs_mg)
1084  Rhs_mg_vectors_storage.resize(Nlevel);
1085 
1086  // Resize the vector storing all of the residual vectors
1087  Residual_mg_vectors_storage.resize(Nlevel);
1088 
1089  // Allocate space for the pre-smoother storage vector (remember, we do
1090  // not need a smoother on the coarsest level we use a direct solve there)
1091  Pre_smoothers_storage_pt.resize(Nlevel-1,0);
1092 
1093  // Allocate space for the post-smoother storage vector (remember, we do
1094  // not need a smoother on the coarsest level we use a direct solve there)
1095  Post_smoothers_storage_pt.resize(Nlevel-1,0);
1096 
1097  // Resize the vector storing all of the interpolation matrices
1098  Interpolation_matrices_storage_pt.resize(Nlevel-1,0);
1099 
1100  // Resize the vector storing all of the restriction matrices
1101  Restriction_matrices_storage_pt.resize(Nlevel-1,0);
1102 
1103  if (!Suppress_all_output)
1104  {
1105  // Stop clock
1106  double t_m_end=TimingHelpers::timer();
1107  double total_setup_time=double(t_m_end-t_m_start);
1108  oomph_info << "\nCPU time for creation of hierarchy of MG problems [sec]: "
1109  << total_setup_time << std::endl;
1110 
1111  // Notify user that the hierarchy of levels is complete
1112  oomph_info << "\n===============Hierarchy Creation Complete================"
1113  << "\n" << std::endl;

1181  /// \short Set up the MG hierarchy structures
1182  //===================================================================
1183  // Function to set up the hierachy of levels. Creates a vector of
1184  // pointers to each MG level
1185  template<unsigned DIM>
1186  void MGSolver<DIM>::setup_mg_structures()
1187  {
1188  // Initialise the timer start variable
1189  double t_m_start=0.0;
1190 
1191  // Start the clock (if we're allowed to time things)
1192  if (!Suppress_all_output)
1193  {
1194  // Start the clock
1195  t_m_start=TimingHelpers::timer();
1196  }
1197 
1198  // Allocate space for the system matrix on each level
1199  for (unsigned i=0; i<Nlevel; i++)
1200  {
1201  // Dynamically allocate a new CRDoubleMatrix
1202  Mg_matrices_storage_pt[i]=new CRDoubleMatrix;
1203  }
1204 
1205  // Loop over each level and extract the system matrix, solution vector
1206  // right-hand side vector and residual vector (to store the value of r=b-Ax)
1207  for (unsigned i=0; i<Nlevel; i++)
1208  {
1209  // If we're allowed to output
1210  if (!Suppress_all_output)
1211  {
1212  // Output the level we're working on
1213  oomph_info << "Setting up MG structures on level: " << i
1214  << "\n" << std::endl;
1215  }
1216 
1217  // Resize the solution and RHS vector
1218  unsigned n_dof=Mg_hierarchy[i]->ndof();
1219  LinearAlgebraDistribution* dist_pt=
1220  new LinearAlgebraDistribution(Mg_hierarchy[i]->communicator_pt(),
1221  n_dof,false);
1222 
1223 #ifdef PARANOID
1224 #ifdef OOMPH_HAS_MPI
1225  // Set up the warning messages
1226  std::string warning_message="Setup of distribution has not been ";
1227  warning_message+="tested with MPI.";
1228 
1229  // If we're not running the code in serial
1230  if (dist_pt->communicator_pt()->nproc()>1)
1231  {
1232  // Throw a warning
1233  OomphLibWarning(warning_message,
1234  OOMPH_CURRENT_FUNCTION,
1235  OOMPH_EXCEPTION_LOCATION);
1236  }
1237 #endif
1238 #endif
1239 
1240  // Build the approximate solution
1241  X_mg_vectors_storage[i].clear();
1242  X_mg_vectors_storage[i].build(dist_pt);
1243 
1244  // Build the point source function
1245  Rhs_mg_vectors_storage[i].clear();
1246  Rhs_mg_vectors_storage[i].build(dist_pt);
1247 
1248  // Build the residual vector (r=b-Ax)
1249  Residual_mg_vectors_storage[i].clear();
1250  Residual_mg_vectors_storage[i].build(dist_pt);
1251 
1252  // Delete the distribution pointer
1253  delete dist_pt;
1254 
1255  // Make it a null pointer
1256  dist_pt=0;
1257 
1258  // Build the matrix distribution
1259  Mg_matrices_storage_pt[i]->clear();
1260  Mg_matrices_storage_pt[i]->distribution_pt()->
1261  build(Mg_hierarchy[i]->communicator_pt(),n_dof,false);
1262 
1263  // Compute system matrix on the current level. On the finest level of the
1264  // hierarchy the system matrix and RHS vector is given by the Jacobian and
1265  // vector of residuals which define the original problem which the Galerkin
1266  // approximation to the system matrix is used on the subsequent levels so
1267  // that the correct contributions are taken from each dof on the level above
1268  // (that is to say, it should match the contribution taken from the solution
1269  // vector and RHS vector on the level above)
1270  if (i==0)
1271  {
1272  // Initialise the timer start variable
1273  double t_jac_start=0.0;
1274 
1275  // If we're allowed to output things
1276  if (!Suppress_all_output)
1277  {
1278  // Start timer for Jacobian setup
1279  t_jac_start=TimingHelpers::timer();
1280  }
1281 
1282  // The system matrix on the finest level is the Jacobian and the RHS
1283  // vector is given by the residual vector which accompanies the Jacobian
1284  Mg_hierarchy[0]->get_jacobian(Rhs_mg_vectors_storage[0],
1285  *Mg_matrices_storage_pt[0]);
1286 
1287  if (!Suppress_all_output)
1288  {
1289  // Document the time taken
1290  double t_jac_end=TimingHelpers::timer();

1292  oomph_info << " - Time for setup of Jacobian [sec]: "
1293  << jacobian_setup_time << "\n" << std::endl;
1294  }
1295  }
1296  else
1297  {
1298  // Initialise the timer start variable
1299  double t_gal_start=0.0;
1300 
1301  // If we're allowed
1302  if (!Suppress_all_output)
1303  {
1304  // Start timer for Galerkin matrix calculation
1305  t_gal_start=TimingHelpers::timer();
1306  }
1307 
1308  // The system matrix on the coarser levels must be formed using the
1309  // Galerkin approximation which we do by calculating the product
1310  // A^2h = I^2h_h * A^h * I^h_2h, i.e. the coarser version of the
1311  // finer grid system matrix is formed by multiplying by the (fine grid)
1312  // restriction matrix from the left and the (fine grid) interpolation
1313  // matrix from the left
1314 
1315  // First we need to calculate A^h * I^h_2h which we store as A^2h
1316  Mg_matrices_storage_pt[i-1]->
1317  multiply(*Interpolation_matrices_storage_pt[i-1],
