43  /// Base class for upgraded disk-like GeomObject (i.e. 2D surface in 3D space)
44  /// with specification of boundaries. The GeomObject's position(...)
45  /// function computes the 3D (Eulerian) position vector r as a function of the
46  /// 2D intrinsic (Lagrangian) coordinates, zeta, without reference to
47  /// any boundaries. This class specifies the boundaries by specifying
48  /// a mapping from a 1D intrinsic boundary coordinate, zeta_bound,
49  /// to the 2D intrinsic (Lagrangian) coordinates, zeta.
50  ///
51  /// The class is made functional by provision (in the derived class!) of:
52  /// - a pointer to a GeomObject<1,2> that parametrises the 2D intrinisic
53  /// coordinates zeta along boundary b in terms of the 1D boundary
54  /// coordinate, zeta_bound,
55  /// - the initial value of the 1D boundary coordinate zeta_bound on boundary
56  /// b,
57  /// - the final value of the 1D boundary coordinate zeta_bound on boundary b
58  /// .
59  /// for each of these boundaries. All three containers for these
60  /// must be resized (consistently) and populated in the derived class.
61  /// Number of boundaries is inferred from their size.
62  ///
63  /// Class also provides broken virtual function to specify boundary triads,
64  /// and various output functions.
65  //===========================================================================

254  /// \short Specify intrinsic coordinates of a point within a specified
255  /// region -- region ID, r, should be positive.
256  void add_region_coordinates(const unsigned& r,
257  Vector<double>& zeta_in_region)
258  {
259  // Verify if not using the default region number (zero)
260  if (r == 0)
261  {
262  std::ostringstream error_message;
263  error_message
264  << "Please use another region id different from zero.\n"
265  << "It is internally used as the default region number.\n";
266  throw OomphLibError(error_message.str(),
267  OOMPH_CURRENT_FUNCTION,
268  OOMPH_EXCEPTION_LOCATION);
269  }
270  // Need two coordinates
271  unsigned n = zeta_in_region.size();
272  if (n != 2)
273  {
274  std::ostringstream error_message;
275  error_message << "Vector specifying zeta coordinates of point in\n"
276  << "region has be length 2; yours has length " << n
277  << std::endl;
278  throw OomphLibError(error_message.str(),
279  OOMPH_CURRENT_FUNCTION,
280  OOMPH_EXCEPTION_LOCATION);
281  }
282 
283  // First check if the region with the specified id does not already exist
284  std::map<unsigned, Vector<double>>::iterator it;
285  it = Zeta_in_region.find(r);
286 
287  // If it is already a region defined with that id throw an error
288  if (it != Zeta_in_region.end())
289  {
290  std::ostringstream error_message;
291  error_message << "The region id (" << r << ") that you are using for"
292  << "defining\n"
293  << "your region is already in use. Use another\n"
294  << "region id and verify that you are not re-using\n"
295  << " previously defined regions ids.\n"
296  << std::endl;
297  OomphLibWarning(error_message.str(),
298  OOMPH_CURRENT_FUNCTION,
299  OOMPH_EXCEPTION_LOCATION);
300  }
301 
302  // If it does not exist then create the map
303  Zeta_in_region[r] = zeta_in_region;
304  }
305

337  /// \short Warped disk in 3d: zeta[0]=x; zeta[1]=y (so it doesn't have
338  /// coordinate singularities), with specification of two boundaries (b=0,1)
339  /// that turn the whole thing into a circular disk.
340  //=========================================================================
341  class WarpedCircularDisk : public DiskLikeGeomObjectWithBoundaries
342  {
343  public:
344  /// Constructor. Pass amplitude and azimuthal wavenumber of
345  /// warping as arguments. Can specify vertical offset as final, optional
346  /// argument.
347  WarpedCircularDisk(const double& epsilon,
348  const unsigned& n,
349  const double& z_offset = 0.0)
350  : Epsilon(epsilon), N(n), Z_offset(z_offset)
351  {
352  // How many boundaries do we have?
353  unsigned nb = 2;
354  Boundary_parametrising_geom_object_pt.resize(nb);
355  Zeta_boundary_start.resize(nb, 0.0);
356  Zeta_boundary_end.resize(nb, 0.0);
357 
358  // GeomObject<1,2> representing the first boundary
359  Boundary_parametrising_geom_object_pt[0] = new Ellipse(1.0, 1.0);
360  Zeta_boundary_start[0] = 0.0;
361  Zeta_boundary_end[0] = MathematicalConstants::Pi;
362 
363  // GeomObject<1,2> representing the second boundary
364  Boundary_parametrising_geom_object_pt[1] = new Ellipse(1.0, 1.0);
365  Zeta_boundary_start[1] = MathematicalConstants::Pi;
366  Zeta_boundary_end[1] = 2.0 * MathematicalConstants::Pi;
367  }

407  /// \short Position Vector at Lagrangian coordinate zeta
408  void position(const Vector<double>& zeta, Vector<double>& r) const
409  {
410  // Position Vector
411  r[0] = zeta[0];
412  r[1] = zeta[1];
413  double radius = sqrt(r[0] * r[0] + r[1] * r[1]);
414  double phi = atan2(r[1], r[0]);
415  r[2] = Z_offset + w(radius, phi);
416  }

419  /// \short Parametrised position on object: r(zeta). Evaluated at
420  /// previous timestep. t=0: current time; t>0: previous
421  /// timestep. Object is steady so calls time-independent version
422  void position(const unsigned& t,
423  const Vector<double>& zeta,
424  Vector<double>& r) const

430  /// Boundary triad on boundary b at boundary coordinate zeta_bound
431  void boundary_triad(const unsigned& b,
432  const double& zeta_bound,
433  Vector<double>& r,
434  Vector<double>& tangent,
435  Vector<double>& normal,
436  Vector<double>& binormal)
437  {
438  double phi = zeta_bound;
439 
440  // Position Vector
441  r[0] = cos(phi);
442  r[1] = sin(phi);
443  r[2] = Z_offset + w(1.0, phi);
444 
445  Vector<double> dr_dr(3);
446  dr_dr[0] = cos(phi);
447  dr_dr[1] = sin(phi);
448  dr_dr[2] = dwdr(1.0, phi);
449  double inv_norm = 1.0 / sqrt(dr_dr[0] * dr_dr[0] + dr_dr[1] * dr_dr[1] +
450  dr_dr[2] * dr_dr[2]);
451 
452  normal[0] = dr_dr[0] * inv_norm;
453  normal[1] = dr_dr[1] * inv_norm;
454  normal[2] = dr_dr[2] * inv_norm;
455 
456  Vector<double> dr_dphi(3);
457  dr_dphi[0] = -sin(phi);
458  dr_dphi[1] = cos(phi);
459  dr_dphi[2] = dwdphi(1.0, phi);
460 
461  inv_norm = 1.0 / sqrt(dr_dphi[0] * dr_dphi[0] + dr_dphi[1] * dr_dphi[1] +
462  dr_dphi[2] * dr_dphi[2]);
463 
464  tangent[0] = dr_dphi[0] * inv_norm;
465  tangent[1] = dr_dphi[1] * inv_norm;
466  tangent[2] = dr_dphi[2] * inv_norm;
467 
468  binormal[0] = tangent[1] * normal[2] - tangent[2] * normal[1];
469  binormal[1] = tangent[2] * normal[0] - tangent[0] * normal[2];
470  binormal[2] = tangent[0] * normal[1] - tangent[1] * normal[0];
471  }
472 
473  private:
474  /// Vertical deflection
475  double w(const double& r, const double& phi) const
476  {
477  return Epsilon * cos(double(N) * phi) * pow(r, 2);
478  }
479 
480  /// Deriv of vertical deflection w.r.t. radius
481  double dwdr(const double& r, const double& phi) const
482  {
483  return Epsilon * cos(double(N) * phi) * 2.0 * r;
484  }
485 
486  /// Deriv of vertical deflection w.r.t. angle
487  double dwdphi(const double& r, const double& phi) const
488  {
489  return -Epsilon * double(N) * sin(double(N) * phi) * pow(r, 2);
490  }
491

494 
495  /// Wavenumber of non-axisymmetric deformation
496  unsigned N;
497

508  /// \short Warped disk in 3d: zeta[0]=x; zeta[1]=y (so it doesn't have
509  /// coordinate singularities), with specification of two boundaries (b=0,1)
510  /// that turn the whole thing into a circular disk. In addition
511  /// has two internal boundaries (b=2,3), a distance h_annulus from
512  /// the outer edge. Annual (outer) region is region 1.
513  //=========================================================================
514  class WarpedCircularDiskWithAnnularInternalBoundary
515  : public virtual WarpedCircularDisk

517  public:

519  /// warping as arguments. Can specify vertical offset as final, optional
520  /// argument.

522  const double& epsilon,
523  const unsigned& n,
524  const double& z_offset = 0.0)
525  : WarpedCircularDisk(epsilon, n, z_offset), H_annulus(h_annulus)
526  {

528  Boundary_parametrising_geom_object_pt.resize(4);
529  Zeta_boundary_start.resize(4);
530  Zeta_boundary_end.resize(4);
531 
532  // Radius of the internal annular boundary
533  double r_annulus = 1.0 - h_annulus;
534 
535  // GeomObject<1,2> representing the third boundary
536  Boundary_parametrising_geom_object_pt[2] =
537  new Ellipse(r_annulus, r_annulus);
538  Zeta_boundary_start[2] = 0.0;
539  Zeta_boundary_end[2] = MathematicalConstants::Pi;
540 
541  // GeomObject<1,2> representing the fourth boundary
542  Boundary_parametrising_geom_object_pt[3] =
543  new Ellipse(r_annulus, r_annulus);
544  Zeta_boundary_start[3] = MathematicalConstants::Pi;
545  Zeta_boundary_end[3] = 2.0 * MathematicalConstants::Pi;
