47 /// \short A class for all elements that solve the
48 /// Advection Diffusion equations in a cylindrical polar coordinate system
49 /// using isoparametric elements.

115 /// \short du/dt at local node n.
116  /// Uses suitably interpolated value for hanging nodes.
117  double du_dt_axi_adv_diff(const unsigned &n) const
118  {
119  // Get the data's timestepper
120  TimeStepper* time_stepper_pt= this->node_pt(n)->time_stepper_pt();
121 
122  //Initialise dudt
123  double dudt=0.0;
124  //Loop over the timesteps, if there is a non Steady timestepper
125  if (!time_stepper_pt->is_steady())
126  {
127  //Find the index at which the variable is stored
128  const unsigned u_nodal_index = u_index_axi_adv_diff();
129 
130  // Number of timsteps (past & present)
131  const unsigned n_time = time_stepper_pt->ntstorage();
132 
133  for(unsigned t=0;t<n_time;t++)
134  {
135  dudt += time_stepper_pt->weight(1,t)*nodal_value(t,n,u_nodal_index);
136  }
137  }
138  return dudt;
139  }

369  /// \short Get flux: [du/dr,du/dz]
370  void get_flux(const Vector<double>& s, Vector<double>& flux) const
371  {
372  //Find out how many nodes there are in the element
373  const unsigned n_node = nnode();
374 
375  //Get the nodal index at which the unknown is stored
376  const unsigned u_nodal_index = u_index_axi_adv_diff();
377 
378  //Set up memory for the shape and test functions
379  Shape psi(n_node);
380  DShape dpsidx(n_node,2);
381 
382  //Call the derivatives of the shape and test functions
383  dshape_eulerian(s,psi,dpsidx);
384 
385  //Initialise to zero
386  for(unsigned j=0;j<2;j++) {flux[j] = 0.0;}
387 
388  //Loop over nodes
389  for(unsigned l=0;l<n_node;l++)
390  {
391  const double u_value = this->nodal_value(l,u_nodal_index);
392  //Add in the derivative directions
393  flux[0] += u_value*dpsidx(l,0);
394  flux[1] += u_value*dpsidx(l,1);
395  }
396  }

425  /// Return FE representation of function value u(s) at local coordinate s
426  inline double interpolated_u_axi_adv_diff(const Vector<double> &s) const
427  {
428  //Find number of nodes
429  const unsigned n_node = nnode();
430 
431  //Get the nodal index at which the unknown is stored
432  const unsigned u_nodal_index = u_index_axi_adv_diff();
433 
434  //Local shape function
435  Shape psi(n_node);
436 
437  //Find values of shape function
438  shape(s,psi);
439 
440  //Initialise value of u
441  double interpolated_u = 0.0;
442 
443  //Loop over the local nodes and sum
444  for(unsigned l=0;l<n_node;l++)
445  {
446  interpolated_u += nodal_value(l,u_nodal_index)*psi[l];
447  }
448 
449  return(interpolated_u);
450  }

453  ///\short Return derivative of u at point s with respect to all data
454  ///that can affect its value.
455  ///In addition, return the global equation numbers corresponding to the
456  ///data. This is virtual so that it can be overloaded in the
457  ///refineable version
458  virtual void dinterpolated_u_axi_adv_diff_ddata(
459  const Vector<double> &s,
460  Vector<double> &du_ddata,
461  Vector<unsigned> &global_eqn_number)
462  {
463  //Find number of nodes
464  const unsigned n_node = nnode();
465 
466  //Get the nodal index at which the unknown is stored
467  const unsigned u_nodal_index = u_index_axi_adv_diff();
468 
469  //Local shape function
470  Shape psi(n_node);
471 
472  //Find values of shape function
473  shape(s,psi);
474 
475  //Find the number of dofs associated with interpolated u
476  unsigned n_u_dof = 0;
477  for(unsigned l=0;l<n_node;l++)
478  {
479  int global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
480  //If it's positive add to the count
481  if (global_eqn >=0)
482  {
483  ++n_u_dof;
484  }
485  }
486 
487  //Now resize the storage schemes
488  du_ddata.resize(n_u_dof,0.0);
489  global_eqn_number.resize(n_u_dof,0);
490 
491  //Loop over the nodes again and set the derivatives
492  unsigned count = 0;
493  for(unsigned l=0;l<n_node;l++)
494  {
495  //Get the global equation number
496  int global_eqn=this->node_pt(l)->eqn_number(u_nodal_index);
497  //If it's positive
498  if (global_eqn >= 0)
499  {
500  //Set the global equation number
501  global_eqn_number[count] = global_eqn;
502  //Set the derivative with respect to the unknown
503  du_ddata[count] = psi[l];
504  //Increase the counter
505  ++count;
506  }
507  }
508  }
