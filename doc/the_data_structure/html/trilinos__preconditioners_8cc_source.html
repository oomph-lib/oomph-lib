48 /// \short Function to set up a preconditioner for the linear system
49 /// defined by matrix_pt. This function must be called before using
50 /// preconditioner_solve.
51 /// \b NOTE 1. matrix_pt must point to an object of class CRDoubleMatrix or
52 /// DistributedCRDoubleMatrix
53 /// This method should be called by oomph-lib solvers and preconditioners
54 //=============================================================================
55  void TrilinosPreconditionerBase::setup()
56 {
57  //clean up the memory
58  clean_up_memory();
59 
60 #ifdef PARANOID
61  // check the matrix is square
62  if ( matrix_pt()->nrow() != matrix_pt()->ncol() )
63  {
64  std::ostringstream error_message;
65  error_message << "Preconditioners require a square matrix. "
66  << "Matrix is " << matrix_pt()->nrow()
67  << " by " << matrix_pt()->ncol() << std::endl;
68  throw OomphLibError(error_message.str(),
69  OOMPH_CURRENT_FUNCTION,
70  OOMPH_EXCEPTION_LOCATION);
71  }
72 #endif
73 
74 
75  // get a pointer to the cr double matrix
76  CRDoubleMatrix* cr_matrix_pt = dynamic_cast<CRDoubleMatrix*>(matrix_pt());
77 
78 #ifdef PARANOID
79  if (cr_matrix_pt == 0)
80  {
81  throw OomphLibError("Matrix must be of type CRDoubleMatrix",
82  OOMPH_CURRENT_FUNCTION,
83  OOMPH_EXCEPTION_LOCATION);
84  }
85 #endif
86 
87  // build the distribution of this preconditioner
88  this->build_distribution(cr_matrix_pt->distribution_pt());
89 
90  // create the matrices
91  Epetra_matrix_pt = TrilinosEpetraHelpers::create_distributed_epetra_matrix
92  (cr_matrix_pt,this->distribution_pt());
93 
94  // set up preconditioner
95  setup_trilinos_preconditioner(Epetra_matrix_pt);
96 }

125 //=============================================================================
126 /// \short preconditioner_solve - applies the preconditioner to the vector r
127 /// taking distributed oomph-lib vectors (DistributedVector<double>) as
128 /// arguments.
129 //=============================================================================
130 void TrilinosPreconditionerBase::
131 preconditioner_solve(const DoubleVector &r, DoubleVector &z)
132 {
133 
134  // Get ready to do cumulative timings
135 double t_start = TimingHelpers::timer();
136 
137 #ifdef PARANOID
138  // check preconditioner data exists
139  if (Epetra_preconditioner_pt == 0)
140  {
141  std::ostringstream error_message;
142  error_message << "preconditioner_solve requires that solver data has "
143  << "been set up" << std::endl;
144  throw OomphLibError(error_message.str(),
145  OOMPH_CURRENT_FUNCTION,
146  OOMPH_EXCEPTION_LOCATION);
147  }
148 
149  if (*this->distribution_pt() != r.distribution_pt())
150  {
151  std::ostringstream error_message;
152  error_message << "The rhs vector and the matrix must have the same "
153  << "distribution.\n";
154  throw OomphLibError(error_message.str(),
155  OOMPH_CURRENT_FUNCTION,
156  OOMPH_EXCEPTION_LOCATION);
157  }
158 #endif
159 
160  // create Epetra version of r
161  Epetra_Vector* epetra_r_pt =
162  TrilinosEpetraHelpers::create_distributed_epetra_vector(r);
163 
164  // create an empty Epetra vector for z
165  Epetra_Vector* epetra_z_pt =
166  TrilinosEpetraHelpers::create_distributed_epetra_vector
167  (this->distribution_pt());
168 
169  // Apply preconditioner
170  Epetra_preconditioner_pt->ApplyInverse(*epetra_r_pt,*epetra_z_pt);
171 
172  // Copy result to z
173  z.build(this->distribution_pt(),0.0);
174  TrilinosEpetraHelpers::copy_to_oomphlib_vector(epetra_z_pt,z);
175 
176  // clean up memory
177  delete epetra_r_pt;
178  delete epetra_z_pt;
179 
180 
181  // Add to cumulative solve time
182  double t_end = TimingHelpers::timer();
183  Cumulative_preconditioner_solve_time+=(t_end-t_start);
184 
185 }

252 setup_trilinos_preconditioner(Epetra_CrsMatrix* epetra_matrix_pt)
253 {
254  // set up a parameter list
255  Teuchos::ParameterList ifpack_parameters;
256  ifpack_parameters.set("fact: level-of-fill", ILU_fill_level);
257  ifpack_parameters.set("fact: ilut level-of-fill", ILUT_fill_level);
258  ifpack_parameters.set("fact: absolute threshold", Absolute_threshold);
259  ifpack_parameters.set("fact: relative threshold", Relative_threshold);
260 
261  // create the preconditioner
262  Ifpack ifpack_factory;
263  Ifpack_Preconditioner* tmp_pt= ifpack_factory.Create(Preconditioner_type,
264  epetra_matrix_pt,
265  Overlap);
266 
267  tmp_pt->SetParameters(ifpack_parameters);
268  tmp_pt->Initialize();
269  tmp_pt->Compute();
270 
271  // Now store the pointer to the newly created Ifpack_Preconditioner
272  Epetra_preconditioner_pt=tmp_pt;
