158  /// \short Return this element's contribution to the total volume enclosed
159  /// by collection of these elements
160  double contribution_to_enclosed_volume()
161  {
162  // Initialise
163  double vol=0.0;
164 
165  //Find out how many nodes there are
166  const unsigned n_node = this->nnode();
167 
168  //Set up memeory for the shape functions
169  Shape psi(n_node);
170  DShape dpsids(n_node,1);
171 
172  //Set the value of n_intpt
173  const unsigned n_intpt = this->integral_pt()->nweight();
174 
175  //Storage for the local coordinate
176  Vector<double> s(1);
177 
178  //Loop over the integration points
179  for(unsigned ipt=0;ipt<n_intpt;ipt++)
180  {
181  //Get the local coordinate at the integration point
182  s[0] = this->integral_pt()->knot(ipt,0);
183 
184  //Get the integral weight
185  double W = this->integral_pt()->weight(ipt);
186 
187  //Call the derivatives of the shape function at the knot point
188  this->dshape_local_at_knot(ipt,psi,dpsids);
189 
190  // Get position and tangent vector
191  Vector<double> interpolated_t1(2,0.0);
192  Vector<double> interpolated_x(2,0.0);
193  for(unsigned l=0;l<n_node;l++)
194  {
195  //Loop over directional components
196  for(unsigned i=0;i<2;i++)
197  {
198  interpolated_x[i] += this->nodal_position(l,i)*psi(l);
199  interpolated_t1[i] += this->nodal_position(l,i)*dpsids(l,0);
200  }
201  }
202 
203  //Calculate the length of the tangent Vector
204  double tlength = interpolated_t1[0]*interpolated_t1[0] +
205  interpolated_t1[1]*interpolated_t1[1];
206 
207  //Set the Jacobian of the line element
208  double J = sqrt(tlength)*interpolated_x[0];
209 
210  //Now calculate the normal Vector
211  Vector<double> interpolated_n(2);
212  this->outer_unit_normal(ipt,interpolated_n);
213 
214  // Assemble dot product
215  double dot = 0.0;
216  for(unsigned k=0;k<2;k++)
217  {
218  dot += interpolated_x[k]*interpolated_n[k];
219  }
220 
221  // Add to volume with sign chosen so that...
222 
223  // Factor of 1/3 comes from div trick
224  vol += 2.0*MathematicalConstants::Pi*dot*W*J/3.0;
225 
226  }
227 
228  return vol;
229  }

246  //Get the value of Nintpt

255  Shape psi(n_node);
256  DShape dpsids(n_node,Dim-1);
257 
258  // Cache the Strouhal number
259  const double local_st=st();
260 
261  // Cache the slip rate coefficient
262  const double local_inverse_slip_rate_coeff=inverse_slip_rate_coefficient();
263 
264  //Loop over the integration points
265  for(unsigned ipt=0;ipt<n_intpt;ipt++)
266  {
267  //Assign values of s

269  {
270  s[i] = integral_pt()->knot(ipt,i);
271  }
272

274  Vector<double> interpolated_normal(Dim);
275  outer_unit_normal(ipt,interpolated_normal);
276 
277  // Calculate the unit tangent vector
278  Vector<double> interpolated_tangent(Dim);
279  interpolated_tangent[0]=-interpolated_normal[1];
280  interpolated_tangent[1]= interpolated_normal[0];
281 
282  // Get solid velocity and porous flux from adjacent solid
283  POROELASTICITY_BULK_ELEMENT* ext_el_pt=
284  dynamic_cast<POROELASTICITY_BULK_ELEMENT*>(
285  external_element_pt(0,ipt));
286  Vector<double> s_ext(external_element_local_coord(0,ipt));
287  Vector<double> du_dt(3);
288  Vector<double> q(2);
289  ext_el_pt->interpolated_du_dt(s_ext,du_dt);
290  ext_el_pt->interpolated_q(s_ext,q);
291  x_bulk[0]=ext_el_pt->interpolated_x(s_ext,0);
292  x_bulk[1]=ext_el_pt->interpolated_x(s_ext,1);
293 
294  // Get own coordinates:
295  Vector<double> x(Dim);
296  this->interpolated_x(s,x);
297 
298 #ifdef PARANOID
299  if (!AxisymmetricPoroelasticityTractionElementHelper::Allow_gap_in_FSI)
300  {
301 
302  double error=sqrt((x[0]-x_bulk[0])*(x[0]-x_bulk[0])+
303  (x[1]-x_bulk[1])*(x[1]-x_bulk[1]));
304  double tol=1.0e-10;
305  if (error>tol)
306  {
307  std::stringstream junk;
308  junk
309  << "Gap between external and face element coordinate\n"
310  << "is suspiciously large: "
311  << error << "\nBulk/external at: "
312  << x_bulk[0] << " " << x_bulk[1] << "\n"
313  << "Face at: " << x[0] << " " << x[1] << "\n";
314  OomphLibWarning(junk.str(),
315  OOMPH_CURRENT_FUNCTION,
316  OOMPH_EXCEPTION_LOCATION);
317  }
318  }
319 #endif
320 
321  // Get permeability from the bulk poroelasticity element
322  const double permeability=ext_el_pt->permeability();
323  const double local_permeability_ratio=ext_el_pt->permeability_ratio();
324 
325  // Local coordinate in bulk element
326  Vector<double> s_bulk(Dim);
327  s_bulk=local_coordinate_in_bulk(s);
328 
329  // Get the fluid traction from the NSt bulk element
330  Vector<double> traction_nst(3);
331  dynamic_cast<FLUID_BULK_ELEMENT*>(bulk_element_pt())->traction(
332  s_bulk,interpolated_normal,traction_nst);
333 
334  // Get fluid velocity from bulk element
335  Vector<double> fluid_veloc(Dim+1,0.0);
336  dynamic_cast<FLUID_BULK_ELEMENT*>(bulk_element_pt())->
337  interpolated_u_axi_nst(s_bulk,fluid_veloc);
338 
339  // Get fluid pressure from bulk element
340  double p_fluid=dynamic_cast<FLUID_BULK_ELEMENT*>(bulk_element_pt())->
341  interpolated_p_axi_nst(s_bulk);
342 
343  // Calculate the normal components
344  double scaled_normal_wall_veloc=0.0;
345  double scaled_normal_poro_veloc=0.0;
346  double scaled_tangential_wall_veloc=0.0;
347  double scaled_tangential_poro_veloc=0.0;
348  double normal_nst_veloc=0.0;
349  for(unsigned i=0;i<Dim;i++)
350  {
351  scaled_normal_wall_veloc+=
352  local_st*du_dt[i]*interpolated_normal[i];
353 
354  scaled_normal_poro_veloc+=
355  local_st*permeability*q[i]*interpolated_normal[i];
356 
357  scaled_tangential_wall_veloc+=
358  local_st*du_dt[i]*interpolated_tangent[i];
359 
360  scaled_tangential_poro_veloc+=
361  -traction_nst[i]*sqrt(local_permeability_ratio)*
362  local_inverse_slip_rate_coeff*interpolated_tangent[i];
363 
364  normal_nst_veloc+=fluid_veloc[i]*interpolated_normal[i];
365  }
366 
367  // Calculate the combined poroelasticity "velocity" (RHS of BJS BC).
368  double total_poro_normal_component=
369  scaled_normal_wall_veloc+scaled_normal_poro_veloc;
370  double total_poro_tangential_component=
371  scaled_tangential_wall_veloc+scaled_tangential_poro_veloc;
372  Vector<double> poro_veloc(2,0.0);
373  for(unsigned i=0;i<Dim;i++)
374  {
375  poro_veloc[i]+=
376  total_poro_normal_component*interpolated_normal[i]+
377  total_poro_tangential_component*interpolated_tangent[i];
378  }
379 
380 
381  //Call the derivatives of the shape function at the knot point
382  this->dshape_local_at_knot(ipt,psi,dpsids);
383 
384  // Get tangent vector
385  Vector<double> interpolated_t1(2,0.0);
386  for(unsigned l=0;l<n_node;l++)
387  {
388  //Loop over directional components
389  for(unsigned i=0;i<2;i++)
390  {
391  interpolated_t1[i] += this->nodal_position(l,i)*dpsids(l,0);
392  }
393  }
394 
395  //Set the Jacobian of the line element
396  double J = sqrt(1.0+
397  (interpolated_t1[0]*interpolated_t1[0])/
398  (interpolated_t1[1]*interpolated_t1[1]))*x[0];
399 
400 
401  // Default geometry; evaluate everything in deformed (Nst) config.
402  double lagrangian_eulerian_translation_factor=1.0;
403 
404  // Get pointer to associated face element to get geometric information
405  // from (if set up)
406  FSILinearisedAxisymPoroelasticTractionElement<POROELASTICITY_BULK_ELEMENT,
407  FLUID_BULK_ELEMENT>* ext_face_el_pt=
408  dynamic_cast<FSILinearisedAxisymPoroelasticTractionElement<POROELASTICITY_BULK_ELEMENT,
409  FLUID_BULK_ELEMENT>*>(external_element_pt(1,ipt));
410 
411  // Update geometry
412  if (ext_face_el_pt!=0)
413  {
414  Vector<double> s_ext_face(external_element_local_coord(1,ipt));
415 
416  // Get correction factor for geometry
417  lagrangian_eulerian_translation_factor=
418  ext_face_el_pt->lagrangian_eulerian_translation_factor(s_ext_face);
419  }

423  // Output

425  << x_bulk[0] << " " // column 1

427  << fluid_veloc[0] << " " // column 3
