306  /// \short Write the triangulateio data to disk as a poly file,
307  /// mainly used for debugging
308  void write_triangulateio_to_polyfile(TriangulateIO &triangle_io,
309  std::ostream &poly_file)
310  {
311  //Up the precision dramatiacally
312  poly_file.precision(20);
313 
314  //Output the number of points and their attributes
315  //Store the number of attributes
316  const int n_attr = triangle_io.numberofpointattributes;
317  poly_file << triangle_io.numberofpoints << " " << 2 << " "
318  << n_attr << " " ;
319  //Determine whether there are point markers
320  bool point_markers=true;
321  if(triangle_io.pointmarkerlist==NULL) {point_markers=false;}
322  //Indicate this in the file
323  poly_file << point_markers << "\n";
324 
325  //Now output the point data
326  poly_file << "#Points\n";
327  for(int n=0;n<triangle_io.numberofpoints;++n)
328  {
329  //Output the point number and x and y coordinates
330  poly_file << n+1 << " "
331  << triangle_io.pointlist[2*n] << " "
332  << triangle_io.pointlist[2*n+1] << " ";
333  //Output any attributes
334  for(int i=0;i<n_attr;++i)
335  {
336  poly_file << triangle_io.pointattributelist[n_attr*n+i] << " ";
337  }
338  //Output the boundary marker
339  if(point_markers)
340  {
341  poly_file << triangle_io.pointmarkerlist[n] << " ";
342  }
343  poly_file << "\n";
344  }
345 
346  //Now move onto the segments
347  poly_file << "#Lines\n";
348  poly_file << triangle_io.numberofsegments << " ";
349  //Determine whether there are segment markers
350  bool seg_markers=true;
351  if(triangle_io.segmentmarkerlist==NULL) {seg_markers=false;}
352  //Output this info in the file
353  poly_file << seg_markers << "\n";
354 
355  //Now output the segment data
356  for(int n=0;n<triangle_io.numberofsegments;++n)
357  {
358  poly_file << n+1 << " "
359  << triangle_io.segmentlist[2*n] << " "
360  << triangle_io.segmentlist[2*n+1] << " ";
361  //If there is a boundary marker output
362  if(seg_markers)
363  {
364  poly_file << triangle_io.segmentmarkerlist[n] << " ";
365  }
366  poly_file << "\n";
367  }
368 
369  //Now output the number of holes
370  poly_file << "#No holes\n";
371  poly_file << triangle_io.numberofholes << "\n";
372  //Output the hole data
373  for(int h=0;h<triangle_io.numberofholes;++h)
374  {
375  poly_file << h+1 << " "
376  << triangle_io.holelist[2*h] << " "
377  << triangle_io.holelist[2*h+1] << "\n";
378  }
379 
380  //Now output the number of regions
381  poly_file << "#Assignment of attributes to regions\n";
382  poly_file << triangle_io.numberofregions << "\n";
383 
384  //Loop over the regions
385  for(int r=0;r<triangle_io.numberofregions;++r)
386  {
387  poly_file << r+1 << " ";
388  for(unsigned i=0;i<4;i++)
389  {
390  poly_file << triangle_io.regionlist[4*r+i] << " ";
391  }
392  poly_file << "\n";
393  }
394  }

398  /// Create a triangulateio data file from ele node and poly
399  /// files. This is used if the mesh is generated by using Triangle externally.
400  /// The triangulateio structure is required to dump the mesh topology for
401  /// restarts.
402  void create_triangulateio_from_polyfiles(
403  const std::string& node_file_name,
404  const std::string& element_file_name,
405  const std::string& poly_file_name,

407  bool &use_attributes)

413  std::ifstream element_file(element_file_name.c_str(),std::ios_base::in);
414 
415  // Check that the file actually opened correctly
416  if(!element_file.is_open())
417  {
418  std::string error_msg("Failed to open element file: ");
419  error_msg += "\"" + element_file_name + "\".";

427  static_cast<unsigned>(triangle_io.numberoftriangles);
428 
429  //Read in the number of nodes per element
430  element_file >> triangle_io.numberofcorners;
431  const unsigned n_local_node =
432  static_cast<unsigned>(triangle_io.numberofcorners);
433 
434  //Read in the element attributes
435  element_file >> triangle_io.numberoftriangleattributes;
436  const unsigned n_attributes =
437  static_cast<unsigned>(triangle_io.numberoftriangleattributes);
438 
439  //Allocate storage in the data structure
440  triangle_io.trianglelist =
441  (int *) malloc(triangle_io.numberoftriangles *
442  triangle_io.numberofcorners * sizeof(int));
443 
444  if(n_attributes > 0)
445  {
446  triangle_io.triangleattributelist =
447  (double *) malloc(triangle_io.numberoftriangles *
448  triangle_io.numberoftriangleattributes
449  * sizeof(double));
450  }
451 
452  //Dummy storage
453  int dummy_element_number;
454 
455  //Initialise counter
456  unsigned counter=0;
457  unsigned counter2=0;
458 
459  // Read global node numbers for all elements
460  for(unsigned e=0;e<n_element;e++)
461  {
462  element_file >> dummy_element_number;
463  for(unsigned j=0;j<n_local_node;j++)
464  {
465  element_file >> triangle_io.trianglelist[counter];
466  ++counter;
467  }
468  for(unsigned j=0;j<n_attributes;j++)
469  {
470  element_file >> triangle_io.triangleattributelist[counter2];

472  }
473  }

475  element_file.close();

478  // -----------------

492  unsigned n_node =
493  static_cast<unsigned>(triangle_io.numberofpoints);
494

496  unsigned dimension;

498 
499 #ifdef PARANOID

509  node_file >> triangle_io.numberofpointattributes;
510  unsigned n_point_attributes =

541  for(unsigned j=0;j<n_point_attributes;++j)
