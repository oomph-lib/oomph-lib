62 /// A class for all isoparametric elements that solve the
63 /// UnsteadyHeat equations.
64 /// \f[
65 /// \frac{\partial^2 u}{\partial x_i^2}=\frac{\partial u}{\partial t}+f(t,x_j)
66 /// \f]
67 /// This contains the generic maths. Shape functions, geometric
68 /// mapping etc. must get implemented in derived class.
69 /// Note that this class assumes an isoparametric formulation, i.e. that
70 /// the scalar unknown is interpolated using the same shape funcitons
71 /// as the position.
72 //=============================================================

123  /// \short du/dt at local node n.
124  /// Uses suitably interpolated value for hanging nodes.
125  double du_dt_ust_heat(const unsigned &n) const
126  {
127  // Get the data's timestepper
128  TimeStepper* time_stepper_pt= this->node_pt(n)->time_stepper_pt();
129 
130  //Initialise dudt
131  double dudt=0.0;
132 
133  //Loop over the timesteps, if there is a non Steady timestepper
134  if (!time_stepper_pt->is_steady())
135  {
136  //Find the index at which the variable is stored
137  const unsigned u_nodal_index = u_index_ust_heat();
138 
139  // Number of timsteps (past & present)
140  const unsigned n_time = time_stepper_pt->ntstorage();
141 
142  //Add the contributions to the time derivative
143  for(unsigned t=0;t<n_time;t++)
144  {
145  dudt += time_stepper_pt->weight(1,t)*nodal_value(t,n,u_nodal_index);
146  }
147  }
148  return dudt;
149  }

262  /// Get flux: flux[i] = du/dx_i
263  void get_flux(const Vector<double>& s, Vector<double>& flux) const
264  {
265  //Find out how many nodes there are in the element
266  unsigned n_node = nnode();
267 
268  //Find the index at which the variable is stored
269  unsigned u_nodal_index = u_index_ust_heat();
270 
271  //Set up memory for the shape and test functions
272  Shape psi(n_node);
273  DShape dpsidx(n_node,DIM);
274 
275  //Call the derivatives of the shape and test functions
276  dshape_eulerian(s,psi,dpsidx);
277 
278  //Initialise to zero
279  for(unsigned j=0;j<DIM;j++) {flux[j] = 0.0;}
280 
281  // Loop over nodes
282  for(unsigned l=0;l<n_node;l++)
283  {
284  //Loop over derivative directions
285  for(unsigned j=0;j<DIM;j++)
286  {
287  flux[j] += nodal_value(l,u_nodal_index)*dpsidx(l,j);
288  }
289  }
290  }

570 //======================================================================
571 /// Define the shape functions and test functions and derivatives
572 /// w.r.t. global coordinates and return Jacobian of mapping.
573 ///
574 /// Galerkin: Test functions = shape functions
575 //======================================================================
576 template<unsigned DIM, unsigned NNODE_1D>
577 double QUnsteadyHeatElement<DIM,NNODE_1D>::
578  dshape_and_dtest_eulerian_ust_heat(const Vector<double> &s,
579  Shape &psi,
580  DShape &dpsidx,
581  Shape &test,
582  DShape &dtestdx) const
583 {
584  //Call the geometrical shape functions and derivatives
585  double J = this->dshape_eulerian(s,psi,dpsidx);
586 
587  //Loop over the test functions and derivatives and set them equal to the
588  //shape functions
589  for(unsigned i=0;i<NNODE_1D;i++)
590  {
591  test[i] = psi[i];
592  for(unsigned j=0;j<DIM;j++)
593  {
594  dtestdx(i,j) = dpsidx(i,j);
595  }
596  }
597 
598  //Return the jacobian
599  return J;
600 }

603 //======================================================================
604 /// Define the shape functions and test functions and derivatives
605 /// w.r.t. global coordinates and return Jacobian of mapping.
606 ///
607 /// Galerkin: Test functions = shape functions
608 //======================================================================
609 template<unsigned DIM,unsigned NNODE_1D>
610 double QUnsteadyHeatElement<DIM,NNODE_1D>::
611  dshape_and_dtest_eulerian_at_knot_ust_heat(
612  const unsigned &ipt,
613  Shape &psi,
614  DShape &dpsidx,
615  Shape &test,
616  DShape &dtestdx) const
617 {
618  //Call the geometrical shape functions and derivatives
619  double J = this->dshape_eulerian_at_knot(ipt,psi,dpsidx);
620 
621  //Set the test functions equal to the shape functions
622  //(sets internal pointers)
623  test = psi;
624  dtestdx = dpsidx;
625 
626  //Return the jacobian
627  return J;
628 }
