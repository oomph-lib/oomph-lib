43 /// Work out the position derivative taking into account the movement
44 /// relative to the original centre of mass
45 //======================================================================
46  void ImmersedRigidBodyElement::dposition_dt(const Vector<double> &zeta,
47  const unsigned &j,
48  Vector<double> &drdt)
49  {
50  //Switch on time level
51  switch(j)
52  {
53  //Current time, just return the position
54  case 0:
55  return position(zeta,drdt);
56  break;
57 
58  //Time derivative
59  case 1:
60  {
61  //Get the initial position of the underlying geometric object
62  Vector<double> initial_x(2);
63  Geom_object_pt->position(zeta,initial_x);
64  //Scale relative to the centre of mass
65  double X = initial_x[0] - Initial_centre_of_mass[0];
66  double Y = initial_x[1] - Initial_centre_of_mass[1];
67 
68  //Now calculate the original angle and radius
69  double phi_orig=atan2(Y,X);
70  double r_orig=sqrt(X*X+Y*Y);
71 
72  // Get first time derivatives of all displacement data
73  Vector<double> veloc(3);
74  // Get the velocity of the data
75  this->Centre_displacement_data_pt->time_stepper_pt()
76  ->time_derivative(1,this->Centre_displacement_data_pt,
77  veloc);
78 
79  //Now use the chain rule to specify the boundary velocities
80  drdt[0] = veloc[0];
81  drdt[1] = veloc[1];
82 
83  if(Include_geometric_rotation)
84  {
85  drdt[0] += - r_orig*sin(
86  phi_orig + this->Centre_displacement_data_pt->value(2))*veloc[2];
87  drdt[1] += r_orig*cos(
88  phi_orig + this->Centre_displacement_data_pt->value(2))*veloc[2];
89  }
90  }
91  //Done
92  return;
93  break;
94 
95  default:
96  std::ostringstream warning_stream;
97  warning_stream
98  << "Using default (static) assignment " << j
99  << "-th time derivative in GeomObject::dposition_dt(...) is zero\n"
100  << "Overload for your specific geometric object if this is not \n"
101  << "appropriate. \n";
102  OomphLibWarning(warning_stream.str(),
103  "GeomObject::dposition_dt()",
104  OOMPH_EXCEPTION_LOCATION);
105 
106  unsigned n=drdt.size();
107  for (unsigned i=0;i<n;i++) {drdt[i]=0.0;}
108  break;
109  }
110  }

147  void ImmersedRigidBodyElement::get_force_and_torque(const double& time,
148  Vector<double>& force,
149  double& torque)
150  {
151  // Get external force
152  if (External_force_fct_pt==0)
153  {
154  force[0]=0.0;
155  force[1]=0.0;
156  }
157  else
158  {
159  External_force_fct_pt(time,force);
160  }
161 
162  // Get external torque
163  if (External_torque_fct_pt==0)
164  {
165  torque=0.0;
166  }
167  else
168  {
169  External_torque_fct_pt(time,torque);
170  }
171 
172  // Add drag from any (fluid) mesh attached to surface of body
173  Vector<double> element_drag_force(2);
174  Vector<double> element_drag_torque(1);
175  if (Drag_mesh_pt==0)
176  {
177  return;
178  }
179  else
180  {
181  unsigned nel=Drag_mesh_pt->nelement();
182 
183  for (unsigned e=0;e<nel;e++)
184  {
185  dynamic_cast<ElementWithDragFunction*>(Drag_mesh_pt->element_pt(e))->
186  get_drag_and_torque(element_drag_force,
187  element_drag_torque);
188  force[0]+=element_drag_force[0];
189  force[1]+=element_drag_force[1];
190  torque+=element_drag_torque[0];
191  }
192  }
193  }

206  this->flush_external_data();

212  //Allocate storage for all load data in the mesh
213  std::set<std::pair<Data*,unsigned> > bulk_load_data_pt;
214 
215  //Loop over all elements in the drag mesh
216  const unsigned n_element = drag_mesh_pt->nelement();
217  for(unsigned e=0;e<n_element;++e)
218  {
219  //Cast the bulk element associated with each FaceElement to
220  //an FSIFluidElement
221  FSIFluidElement* bulk_elem_pt =
222  dynamic_cast<FSIFluidElement*>(
223  dynamic_cast<FaceElement*>(drag_mesh_pt->element_pt(e))
224  ->bulk_element_pt());
225  //Check that the cast worked
226  if(bulk_elem_pt==0)
227  {
228  throw OomphLibError("Drag mesh must consist of FSIFluidElements\n",
229  OOMPH_CURRENT_FUNCTION,
230  OOMPH_EXCEPTION_LOCATION);
231  }
232 
233  //Add the geometric and load data from the bulk element to the
234  //set allocated above
235  bulk_elem_pt->identify_geometric_data(bulk_geometric_data_pt);
236  bulk_elem_pt->identify_load_data(bulk_load_data_pt);
237  }
238 
239  //Need to add all these data as external data to this (Rigid Body) object
240  for(std::set<Data*>::iterator it = bulk_geometric_data_pt.begin();
241  it!=bulk_geometric_data_pt.end();++it)
242  {
243  this->add_external_data(*it);
244  }
245 
246  //Now do the same but make custom data for the load data
247  for(std::set<std::pair<Data*,unsigned> >::iterator it =
248  bulk_load_data_pt.begin();
249  it!=bulk_load_data_pt.end();++it)
250  {
251  Data* temp_data_pt = new HijackedData(it->second,it->first);
252  List_of_external_hijacked_data.push_back(
253  this->add_external_data(temp_data_pt));
254  }

294 /// Calculate the contributions to the residuals and the jacobian
295 //======================================================================
296  void ImmersedRigidBodyElement::get_residuals_rigid_body_generic(
297  Vector<double> &residuals,
298  DenseMatrix<double> &jacobian,
299  const bool& flag)
300  {
301  // Get timestepper and time
302  TimeStepper* timestepper_pt=
303  this->Centre_displacement_data_pt->time_stepper_pt();
304  double time=timestepper_pt->time();
305 
306  // Get second time derivatives of all displacement data
307  Vector<double> accel(3);
308  timestepper_pt->time_derivative(2,this->Centre_displacement_data_pt,
309  accel);
310 
311  // Get force and torque
312  Vector<double> external_force(2);
313  double external_torque;
314  get_force_and_torque(time,external_force,external_torque);
315 
316  //Get the timescale ratio product of Reynolds number
317  //and Strouhal number squared
318  const double Lambda_sq =
319  this->re()*this->st()*this->st()*this->density_ratio();
320 
321  //Get the effective ReInvFr which must be multiplied by the
322  //density ratio to compute the gravitational load on the rigid body
323  const double scaled_re_inv_fr =
324  this->re_invfr()*this->density_ratio();
325  //Get the gravitational load
326  Vector<double> G = g();
327 
328  // Newton's law
329  int local_eqn=0;
330  local_eqn = this->centre_displacement_local_eqn(0);
331  if(local_eqn >= 0)
332  {
333  residuals[local_eqn]=
334  Lambda_sq*Mass*accel[0]- external_force[0] - Mass*scaled_re_inv_fr*G[0];
335 
336  // Get Jacobian too?
337  if (flag)
338  {
339  jacobian(local_eqn,local_eqn)=
340  Lambda_sq*Mass*timestepper_pt->weight(2,0);
341  }
342  }
343 
344  local_eqn = this->centre_displacement_local_eqn(1);
345  if(local_eqn >= 0)
346  {
347  residuals[local_eqn]=Lambda_sq*Mass*accel[1] -
348  external_force[1] - Mass*scaled_re_inv_fr*G[1];
349  // Get Jacobian too?
350  if (flag)
351  {
352  jacobian(local_eqn,local_eqn)=
353  Lambda_sq*Mass*timestepper_pt->weight(2,0);
354  }
355  }
356 
357  local_eqn = this->centre_displacement_local_eqn(2);
358  if(local_eqn >= 0)
359  {
360  residuals[local_eqn]=Lambda_sq*Moment_of_inertia*accel[2]-external_torque;
361  // Get Jacobian too?
362  if (flag)
363  {
364  jacobian(local_eqn,local_eqn)=
365  Lambda_sq*Moment_of_inertia*timestepper_pt->weight(2,0);
366  }
367  }
368  }

373 /// and a vector of pointers to TriangleMeshPolyLines
374 /// that define the boundary segments of the polygon.
375 /// Each TriangleMeshPolyLine has its own boundary ID and can contain
376 /// multiple (straight-line) segments. The optional final argument
377 /// is a pointer to a Data object whose three values represent
378 /// the two displacements of and the rotation angle about the polygon's
379 /// centre of mass.
380 //=======================================================================
381  ImmersedRigidBodyTriangleMeshPolygon::
382  ImmersedRigidBodyTriangleMeshPolygon(
383  const Vector<double>& hole_center,
384  const Vector<TriangleMeshCurveSection*>&
385  boundary_polyline_pt,
386  TimeStepper* const &time_stepper_pt,
387  Data* const &centre_displacement_data_pt) :
388  TriangleMeshCurve(boundary_polyline_pt),
389  TriangleMeshClosedCurve(boundary_polyline_pt, hole_center),
390  TriangleMeshPolygon(boundary_polyline_pt, hole_center),
391  ImmersedRigidBodyElement(time_stepper_pt,centre_displacement_data_pt)
392  {
393  //The underlying geometric object can be used to update the configuration
394  //internally before a remesh
395  this->Can_update_configuration = true;
396 
397  // Original rotation angle is zero
398  Initial_Phi=0.0;
399 
400  // Compute coordinates of centre of gravity etc
401  Vector<double> r_left(2);
402  Vector<double> r_right(2);
403  Mass=0.0;
404  Initial_centre_of_mass[0]=0.0;
405  Initial_centre_of_mass[1]=0.0;
406  double inertia_x=0.0;
407  double inertia_y=0.0;
408 
409  // Loop over polylines
410  unsigned nboundary=boundary_polyline_pt.size();
411  for (unsigned i=0;i<nboundary;i++)
412  {
413  // Loop over the segments to get the vertex coordinates
414  unsigned nseg=boundary_polyline_pt[i]->nsegment();
415  for(unsigned j=0;j<nseg;j++)
416  {
417  // Get the vertex coordinates
418  r_left =this->polyline_pt(i)->vertex_coordinate(j);
419  r_right=this->polyline_pt(i)->vertex_coordinate(j+1);
420 
421  // Mass (area)
422  Mass+=0.5*(r_left[0]*r_right[1]-r_right[0]*r_left[1]);
423 
424  // Centroid
425  Initial_centre_of_mass[0]+=(r_left[0]+r_right[0])*
426  (r_left[0]*r_right[1]-r_right[0]*r_left[1]);
427  Initial_centre_of_mass[1]+=(r_left[1]+r_right[1])*
428  (r_left[0]*r_right[1]-r_right[0]*r_left[1]);
429  }
430  if (nboundary==1)
431  {
432  // Get the vertex coordinates
433  r_left =this->polyline_pt(0)->vertex_coordinate(nseg);
434  r_right=this->polyline_pt(0)->vertex_coordinate(0);
435 
436  // Mass (area)
437  Mass+=0.5*(r_left[0]*r_right[1]-r_right[0]*r_left[1]);
438 
439  // Centroid
440  Initial_centre_of_mass[0]+=(r_left[0]+r_right[0])*
441  (r_left[0]*r_right[1]-r_right[0]*r_left[1]);
442  Initial_centre_of_mass[1]+=(r_left[1]+r_right[1])*
443  (r_left[0]*r_right[1]-r_right[0]*r_left[1]);
444  }
445  }
446 
447  // Normalise
448  Initial_centre_of_mass[0]/=(6.0*Mass);
449  Initial_centre_of_mass[1]/=(6.0*Mass);
450 
451  // Another loop over polylines for moment of inertia
452  for (unsigned i=0;i<nboundary;i++)
453  {
454  // Loop over the segments to get the vertex coordinates
455  unsigned nseg=boundary_polyline_pt[i]->nsegment();
456  for(unsigned j=0;j<nseg;j++)
457  {
458  // Get the vertex coordinates
459  r_left =this->polyline_pt(i)->vertex_coordinate(j);
460  r_right=this->polyline_pt(i)->vertex_coordinate(j+1);
461 
462  // Get moment about centroid
463  r_left[0]-=Initial_centre_of_mass[0];
464  r_left[1]-=Initial_centre_of_mass[1];
465  r_right[0]-=Initial_centre_of_mass[0];
466  r_right[1]-=Initial_centre_of_mass[1];
467 
468  // Moment of inertia
469  inertia_x+=1.0/12.0*(r_left[1]*r_left[1]+
470  r_left[1]*r_right[1]+
471  r_right[1]*r_right[1])*
472  (r_left[0]*r_right[1]-r_right[0]*r_left[1]);
473 
474  inertia_y+=1.0/12.0*(r_left[0]*r_left[0]+
475  r_left[0]*r_right[0]+
476  r_right[0]*r_right[0])*
477  (r_left[0]*r_right[1]-r_right[0]*r_left[1]);
478  }
479 
480  if (nboundary==1)
481  {
482  // Get the vertex coordinates
483  r_left =this->polyline_pt(0)->vertex_coordinate(nseg);
484  r_right=this->polyline_pt(0)->vertex_coordinate(0);
485
