45 /// A class for all isoparametric elements that solve the
46 /// LinearWave equations.
47 /// \f[
48 /// \frac{\partial^2 u}{\partial x_i^2}=
49 /// \frac{\partial^2 u}{\partial t^2}+f(t,x_j)
50 /// \f]
51 /// This contains the generic maths. Shape functions, geometric
52 /// mapping etc. must get implemented in derived class.
53 //=============================================================
54 template<unsigned DIM>
55 class LinearWaveEquations : public virtual FiniteElement
56 {
57 public:
58 /// Function pointer to source function fct(x,f(x)) --
59 /// x is a Vector!
60 typedef void (*LinearWaveSourceFctPt)(const double& time,
61 const Vector<double>& x,

73 /// Return the index at which the unknown value
74 /// is stored. The default value, 0, is appropriate for single-physics
75 /// problems, when there is only one variable, the value that satisfies the
76 /// linear wave equation.
77 /// In derived multi-physics elements, this function should be overloaded
78 /// to reflect the chosen storage scheme. Note that these equations require
79 /// that the unknown is always stored at the same index at each node.
80 virtual inline unsigned u_index_lin_wave() const
81 {
82 return 0;
83 }

85 /// du/dt at local node n.
86 /// Uses suitably interpolated value for hanging nodes.
87 double du_dt_lin_wave(const unsigned& n) const
88 {
89 // Get the data's timestepper
90 TimeStepper* time_stepper_pt = node_pt(n)->time_stepper_pt();
91
92 // Initialise d^2u/dt^2
93 double dudt = 0.0;
94 // Loop over the timesteps, if there is a non steady timestepper
95 if (!time_stepper_pt->is_steady())
96 {
97 // Find the index at which the linear wave unknown is stored
98 const unsigned u_nodal_index = u_index_lin_wave();
99
100 // Number of timsteps (past & present)
101 const unsigned n_time = time_stepper_pt->ntstorage();
102
103 // Add the contributions to the derivative
104 for (unsigned t = 0; t < n_time; t++)
105 {
106 dudt +=
107 time_stepper_pt->weight(1, t) * nodal_value(t, n, u_nodal_index);
108 }
109 }
110 return dudt;
111 }
112
113 /// d^2u/dt^2 at local node n.
114 /// Uses suitably interpolated value for hanging nodes.
115 double d2u_dt2_lin_wave(const unsigned& n) const
116 {
117 // Get the data's timestepper
118 TimeStepper* time_stepper_pt = node_pt(n)->time_stepper_pt();
119
120 // Initialise d^2u/dt^2
121 double ddudt = 0.0;
122 // Loop over the timesteps, if there is a non steady timestepper
123 if (!time_stepper_pt->is_steady())
124 {
125 // Find the index at which the linear wave unknown is stored
126 const unsigned u_nodal_index = u_index_lin_wave();
127
128 // Number of timsteps (past & present)
129 const unsigned n_time = time_stepper_pt->ntstorage();
130
131 // Add the contributions to the derivative
132 for (unsigned t = 0; t < n_time; t++)
133 {
134 ddudt +=
135 time_stepper_pt->weight(2, t) * nodal_value(t, n, u_nodal_index);
136 }
137 }
138 return ddudt;
139 }

203 /// Get source term at continous time t and (Eulerian) position x.
204 /// Virtual so it can be overloaded in derived multiphysics elements.
205 inline void get_source_lin_wave(const double& t,
206 const unsigned& ipt,

208 double& source) const
209 {
210 // If no source function has been set, return zero
211 if (Source_fct_pt == 0)

223 /// Get flux: flux[i] = du/dx_i
224 void get_flux(const Vector<double>& s, Vector<double>& flux) const
225 {
226 // Find out how many nodes there are in the element
227 unsigned n_node = nnode();
228
229 // Find the index at which the linear wave unknown is stored
230 unsigned u_nodal_index = u_index_lin_wave();
231
232 // Set up memory for the shape and test functions
233 Shape psi(n_node);
234 DShape dpsidx(n_node, DIM);
235
236 // Call the derivatives of the shape and test functions
237 dshape_eulerian(s, psi, dpsidx);
238
239 // Initialise to zero
240 for (unsigned j = 0; j < DIM; j++)
241 {
242 flux[j] = 0.0;
243 }
244
245 // Loop over nodes
246 for (unsigned l = 0; l < n_node; l++)
247 {
248 // Loop over derivative directions
249 for (unsigned j = 0; j < DIM; j++)
250 {
251 flux[j] += nodal_value(l, u_nodal_index) * dpsidx(l, j);
252 }
253 }
254 }

276 /// Return FE representation of function value u(s) at local coordinate s
277 inline double interpolated_u_lin_wave(const Vector<double>& s) const
278 {
279 // Find number of nodes
280 unsigned n_node = nnode();
281
282 // Find the index at which the linear wave unknown is stored
283 unsigned u_nodal_index = u_index_lin_wave();
284
285 // Local shape function
286 Shape psi(n_node);
287
288 // Find values of shape function
289 shape(s, psi);
290
291 // Initialise value of u
292 double interpolated_u = 0.0;
293
294 // Loop over the local nodes and sum
295 for (unsigned l = 0; l < n_node; l++)
296 {
297 interpolated_u += nodal_value(l, u_nodal_index) * psi[l];
298 }
299
300 return (interpolated_u);
301 }

506 //======================================================================
507 /// Define the shape functions and test functions and derivatives
508 /// w.r.t. global coordinates and return Jacobian of mapping.
509 ///
510 /// Galerkin: Test functions = shape functions
511 //======================================================================
512 template<unsigned DIM, unsigned NNODE_1D>
513 double QLinearWaveElement<DIM, NNODE_1D>::dshape_and_dtest_eulerian_lin_wave(
514 const Vector<double>& s,
515 Shape& psi,
516 DShape& dpsidx,
517 Shape& test,
518 DShape& dtestdx) const
519 {
520 // Call the geometrical shape functions and derivatives
521 double J = this->dshape_eulerian(s, psi, dpsidx);
522
523 // Loop over the test functions and derivatives and set them equal to the
524 // shape functions
525 for (unsigned i = 0; i < NNODE_1D; i++)
526 {
527 test[i] = psi[i];
528 for (unsigned j = 0; j < DIM; j++)
529 {
530 dtestdx(i, j) = dpsidx(i, j);
531 }
532 }
533
534 // Return the jacobian
535 return J;
536 }

538 //======================================================================
539 /// Define the shape functions and test functions and derivatives
540 /// w.r.t. global coordinates and return Jacobian of mapping.
541 ///
542 /// Galerkin: Test functions = shape functions
543 //======================================================================
544 template<unsigned DIM, unsigned NNODE_1D>
545 double QLinearWaveElement<DIM, NNODE_1D>::
546 dshape_and_dtest_eulerian_at_knot_lin_wave(const unsigned& ipt,
547 Shape& psi,
548 DShape& dpsidx,
549 Shape& test,
550 DShape& dtestdx) const
551 {
552 // Call the geometrical shape functions and derivatives
553 double J = this->dshape_eulerian_at_knot(ipt, psi, dpsidx);
554
555 // Set the test functions equal to the shape functions
556 //(sets internal pointers)
557 test = psi;
558 dtestdx = dpsidx;
559
560 // Return the jacobian
561 return J;
562 }
