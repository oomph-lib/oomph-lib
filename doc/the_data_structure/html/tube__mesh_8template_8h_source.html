46 /// \short 3D tube mesh class.

52 /// line parametrised by \f$ \xi_0 = \xi_0^{lo} \f$.
53 /// - Boundary 1: The outer wall, represetned by \f$\xi_2 = 1\f$.
54 /// - Boundary 2: The out flow, represented by \f$\xi_0 = \xi_0^{hi}\f$.

116 /// The domain is specified by the GeomObject that identifies

124 ///

134 /// \short Constructor for adaptive deformable quarter tube mesh class.
135 /// Pass pointer to geometric object that
136 /// specifies the volume, start and end coordinates for the centreline
137 /// on the geometric object. Values of theta at which dividing lines
138 /// are to be placed, fractions of the radius for the central box
139 /// at the dividing lines, the number of layers

143  const Vector<double> &centreline_limits,
144  const Vector<double> &theta_positions,
145  const Vector<double> &radius_box,
146  const unsigned& nlayer,

148  &Mesh::Default_TimeStepper):
149  TubeMesh<ELEMENT>(wall_pt,centreline_limits,theta_positions,
150  radius_box,

152  {
153  // Loop over all elements and set macro element pointer
154  for (unsigned ielem=0;ielem<TubeMesh<ELEMENT>::nelement();ielem++)
155  {
156  dynamic_cast<RefineableQElement<3>*>(
157  TubeMesh<ELEMENT>::element_pt(ielem))->
158  set_macro_elem_pt(this->Domain_pt->macro_element_pt(ielem));
159  }
160 
161 
162  // Setup Octree forest: Turn elements into individual octrees
163  // and plant in forest
164  Vector<TreeRoot*> trees_pt;
165  for (unsigned iel=0;iel<TubeMesh<ELEMENT>::nelement();iel++)
166  {
167  FiniteElement* el_pt=TubeMesh<ELEMENT>::finite_element_pt(iel);
168  ELEMENT* ref_el_pt=dynamic_cast<ELEMENT*>(el_pt);
169  OcTreeRoot* octree_root_pt=new OcTreeRoot(ref_el_pt);
170  trees_pt.push_back(octree_root_pt);
171  }
172  this->Forest_pt = new OcTreeForest(trees_pt);
173 
174 #ifdef PARANOID
175  // Run self test
176  unsigned success_flag=
177  dynamic_cast<OcTreeForest*>(this->Forest_pt)->self_test();
178  if (success_flag==0)
179  {
180  oomph_info << "Successfully built octree forest " << std::endl;
181  }
182  else
183  {
184  throw OomphLibError(

186  OOMPH_CURRENT_FUNCTION,
187  OOMPH_EXCEPTION_LOCATION);
188  }
189 #endif
190 
191  }
