97  /// \short Setup the preconditioner: store the matrix pointer and the
98  /// communicator pointer then call preconditioner specific setup()
99  /// function.
100  void setup(DoubleMatrixBase* matrix_pt)
101  {
102  // Store matrix pointer
103  set_matrix_pt(matrix_pt);
104 
105  // Extract and store communicator pointer
106  DistributableLinearAlgebraObject* dist_obj_pt =
107  dynamic_cast<DistributableLinearAlgebraObject*>(matrix_pt);
108  if (dist_obj_pt != 0)
109  {
110  set_comm_pt(dist_obj_pt->distribution_pt()->communicator_pt());
111  }
112  else
113  {
114  set_comm_pt(0);
115  }

117  double setup_time_start = TimingHelpers::timer();
118  setup();

120  Setup_time = setup_time_finish - setup_time_start;

278  /// setup method - just sets the distribution
279  virtual void setup()
280  {
281  // first attempt to cast to DistributableLinearAlgebraObject
282  DistributableLinearAlgebraObject* dist_matrix_pt =
283  dynamic_cast<DistributableLinearAlgebraObject*>(matrix_pt());
284 
285  // if it is a distributable matrix
286  if (dist_matrix_pt != 0)
287  {
288  // store the distribution
289  this->build_distribution(dist_matrix_pt->distribution_pt());
290  }
291  // else it is not a distributable matrix
292  else
293  {
294  // # of rows in the matrix
295  unsigned n_row = matrix_pt()->nrow();
296 
297  LinearAlgebraDistribution dist(comm_pt(), n_row, false);
298  this->build_distribution(dist);
299  }
300  }

302  /// \short Apply the preconditioner. This method should apply the
303  /// preconditioner operator to the vector r and return the vector z.
304  virtual void preconditioner_solve(const DoubleVector& r, DoubleVector& z)
305  {
306 #ifdef PARANOID
307  if (*r.distribution_pt() != *this->distribution_pt())
308  {
309  std::ostringstream error_message_stream;
310  error_message_stream
311  << "The r vector must have the same distribution as the "
312  "preconditioner. "
313  << "(this is the same as the matrix passed to setup())";
314  throw OomphLibError(error_message_stream.str(),
315  OOMPH_CURRENT_FUNCTION,
316  OOMPH_EXCEPTION_LOCATION);
317  }
318  if (z.built())
319  {
320  if (*z.distribution_pt() != *this->distribution_pt())
321  {
322  std::ostringstream error_message_stream;
323  error_message_stream
324  << "The z vector distribution has been setup; it must have the "
325  << "same distribution as the r vector (and preconditioner).";
326  throw OomphLibError(error_message_stream.str(),
327  OOMPH_CURRENT_FUNCTION,
328  OOMPH_EXCEPTION_LOCATION);
329  }
330  }
331 #endif
332 
333  // apply
334  z = r;
335  }
336

338  /// \short Apply the preconditioner. This method should apply the
339  /// preconditioner operator to the vector r and return the vector z.
340  void preconditioner_solve_transpose(const DoubleVector& r, DoubleVector& z)
341  {
342  // Applying the preconditioner to the transposed system is exactly the
343  // same as applying the preconditioner to the original system
344  preconditioner_solve(r, z);
345  }
