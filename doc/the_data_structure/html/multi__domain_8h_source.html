220  /// \short Set up the two-way multi-domain interactions for the
221  /// problem pointed to by \c problem_pt.
222  /// Use this for cases where first_mesh_pt and second_mesh_pt
223  /// occupy the same physical space and are populated by
224  /// ELEMENT_0 and ELEMENT_1 respectively, and are combined to solve
225  /// a single problem. The elements in two meshes interact both ways
226  /// the elements in each mesh act as "external elements" for the
227  /// elements in the "other" mesh. The interaction indices allow the
228  /// specification of which interaction we're setting up in the two
229  /// meshes. They default to zero, which is appropriate if there's
230  /// only a single interaction.

238  /// \short Function to set up the one-way multi-domain interaction for
239  /// problems where the meshes pointed to by \c mesh_pt and \c external_mesh_pt
240  /// occupy the same physical space, and the elements in \c external_mesh_pt
241  /// act as "external elements" for the \c ElementWithExternalElements
242  /// in \c mesh_pt (but not vice versa):
243  /// - \c mesh_pt points to the mesh of ElemenWithExternalElements for which
244  /// the interaction is set up.
245  /// - \c external_mesh_pt points to the mesh that contains the elements
246  /// of type EXT_ELEMENT that act as "external elements" for the
247  /// \c ElementWithExternalElements in \ mesh_pt.
248  /// - The interaction_index parameter defaults to zero and must be otherwise
249  /// set by the user if there is more than one mesh that provides sources
250  /// for the Mesh pointed to by mesh_pt.

253  Mesh* const &mesh_pt,

257  /// \short Function to set up the one-way multi-domain interaction for
258  /// FSI-like problems.
259  /// - \c mesh_pt points to the mesh of \c ElemenWithExternalElements for which
260  /// the interaction is set up. In an FSI example, this mesh would contain
261  /// the \c FSIWallElements (either beam/shell elements or the
262  /// \c FSISolidTractionElements that apply the traction to
263  /// a "bulk" solid mesh that is loaded by the fluid.)
264  /// - \c external_mesh_pt points to the mesh that contains the elements
265  /// of type EXT_ELEMENT that provide the "source" for the
266  /// \c ElementWithExternalElements. In an FSI example, this
267  /// mesh would contain the "bulk" fluid elements.
268  /// - \c external_face_mesh_pt points to the mesh of \c FaceElements
269  /// attached to the \c external_mesh_pt. The mesh pointed to by
270  /// \c external_face_mesh_pt has the same dimension as \c mesh_pt.
271  /// The elements contained in \c external_face_mesh_pt are of type
272  /// FACE_ELEMENT_GEOM_OBJECT. In an FSI example, these elements
273  /// are usually the \c FaceElementAsGeomObjects (templated by the
274  /// type of the "bulk" fluid elements to which they are attached)
275  /// that define the FSI boundary of the fluid domain.
276  /// - The interaction_index parameter defaults to zero and must otherwise be
277  /// set by the user if there is more than one mesh that provides "external
278  /// elements" for the Mesh pointed to by mesh_pt (e.g. in the case
279  /// when a beam or shell structure is loaded by fluid from both sides.)
280  template<class EXT_ELEMENT, class FACE_ELEMENT_GEOM_OBJECT>
281  void setup_multi_domain_interaction(Problem* problem_pt,
282  Mesh* const &mesh_pt,
283  Mesh* const &external_mesh_pt,
284  Mesh* const &external_face_mesh_pt,
285  const unsigned& interaction_index=0);
