51 /// \short Describes the distribution of a distributable linear algebra type
52 /// object. Typically this is a container (such as a DoubleVector) or an
53 /// operator (e.g Preconditioner or LinearSolver).
54 /// This object is used in both serial and parallel implementations. In the
55 /// serial context (no MPI) this just contains an integer indicating
56 /// the number of rows.
57 /// In parallel either each processor holds a subset of the set of global rows.
58 /// (each processor contains only a single continuous block of rows -
59 /// parametised with variables denoting the first row and the number of local
60 /// rows) or, all rows are be duplicated across all processors.
61 /// In parallel this object also contains an OomphCommunicator object which
62 /// primarily contains the MPI_Comm communicator associated with this object.
63 //=============================================================================

368  /// \short return the local index corresponding to the global index
369  unsigned global_to_local_row_map(const unsigned& global_i) const
370  {
371 #ifdef PARANOID
372  if (global_i >= Nrow)
373  {
374  throw OomphLibError
375  ("Requested global row outside the number of global rows",
376  OOMPH_CURRENT_FUNCTION,
377  OOMPH_EXCEPTION_LOCATION);
378  }
379 #endif
380  int local_i = static_cast<int>(global_i);
381  int p = 0;
382  while ((int)(local_i - (int)nrow_local(p)) >= 0)
383  {
384  local_i -= (int)nrow_local(p);
385  p++;
386  }
387  return (unsigned)local_i;
388  }

395  {

552  /// \short Takes a vector of LinearAlgebraDistribution objects and
553  /// concatenates them such that the nrow_local of the out_distribution
554  /// is the sum of the nrow_local of all the in_distributions and the number
555  /// of global rows of the out_distribution is the sum of the number of global
556  /// rows of all the in_distributions.
557  /// This results in a permutation of the rows in the out_distribution.
558  /// Think of this in terms of DoubleVectors, if we have DoubleVectors with
559  /// distributions A and B, distributed across two processors (p0 and p1),
560  /// A: [a0] (on p0) B: [b0] (on p0)
561  /// [a1] (on p1) [b1] (on P1),
562  ///
563  /// then the out_distribution is
564  /// [a0 (on p0)
565  /// b0] (on p0)
566  /// [a1 (on p1)
567  /// b1] (on p1),
568  ///
569  /// as opposed to
570  /// [a0 (on p0)
571  /// a1] (on p0)
572  /// [b0 (on p1)
573  /// b1] (on p1).
574  ///
575  /// Note (1): The out_distribution may not be uniformly distributed even
576  /// if the in_distributions are uniform distributions.
577  /// Try this out with two distributions of global rows 3 and 5, uniformly
578  /// distributed across two processors. Compare this against a distribution
579  /// of global row 8 distributed across two processors.
580  ///
581  /// Note (2): There is no equivalent function which takes a Vector of
582  /// LinearAlgebraDistribution objects (as opposed to pointers), there should
583  /// not be one since we do not want to invoke the assignment operator when
584  /// creating the Vector of LinearAlgebraDistribution objects.
585  void concatenate(const Vector<LinearAlgebraDistribution*> &in_distribution_pt,
586  LinearAlgebraDistribution &out_distribution);
587  }
