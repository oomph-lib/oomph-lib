53  /// Compute element residual vector and/or element Jacobian matrix
54  ///
55  /// flag=0: compute only residual vector
56  /// flag=1: compute both
57  ///
58  /// Pure version without hanging nodes
59  //======================================================================
60  template <unsigned SPATIAL_DIM>
61  void SpaceTimeUnsteadyHeatEquations<SPATIAL_DIM>::
62  fill_in_generic_residual_contribution_ust_heat(Vector<double>& residuals,
63  DenseMatrix<double>& jacobian,
64  const unsigned& flag)
65  {
66  // Find out how many nodes there are
67  unsigned n_node=nnode();
68 
69  // Find the index at which the variable is stored
70  unsigned u_nodal_index=u_index_ust_heat();
71 
72  // Set up memory for the shape functions
73  Shape psi(n_node);
74 
75  // Set up memory for the test functions
76  Shape test(n_node);
77 
78  // Allocate space for the derivatives of the shape functions
79  DShape dpsidx(n_node,SPATIAL_DIM+1);
80 
81  // Allocate space for the derivatives of the test functions
82  DShape dtestdx(n_node,SPATIAL_DIM+1);
83 
84  // Set the value of n_intpt
85  unsigned n_intpt=integral_pt()->nweight();
86 
87  // Storage for the local coordinates
88  Vector<double> s(SPATIAL_DIM+1,0.0);
89 
90  // Get the Alpha parameter
91  double alpha_local=alpha();
92 
93  // Get the Beta parameter
94  double beta_local=beta();
95 
96  // Integer to hold the local equation
97  int local_eqn=0;
98 
99  // Integer to hold the local unknowns
100  int local_unknown=0;
101 
102  // Loop over the integration points
103  for (unsigned ipt=0; ipt<n_intpt; ipt++)
104  {
105  // Assign values of s
106  for (unsigned i=0; i<SPATIAL_DIM+1; i++)
107  {
108  // Calculate the i-th local coordinate
109  s[i]=integral_pt()->knot(ipt,i);
110  }
111 
112  // Get the integral weight
113  double w=integral_pt()->weight(ipt);
114 
115  // Call the derivatives of the shape and test functions
116  double J=dshape_and_dtest_eulerian_at_knot_ust_heat(ipt,psi,dpsidx,
117  test,dtestdx);
118 
119  // Premultiply the weights and the Jacobian
120  double W=w*J;
121 
122  // Storage for the interpolated time value
123  double interpolated_t=0.0;
124 
125  // Storage for the interpolated solution value
126  double interpolated_u=0.0;
127 
128  // Storage for the interpolated time-derivative of the solution
129  double interpolated_dudt=0.0;
130 
131  // Storage for the spatial coordinates
132  Vector<double> interpolated_x(SPATIAL_DIM,0.0);
133 
134  // Storage for the spatial derivatives of the solution
135  Vector<double> interpolated_dudx(SPATIAL_DIM,0.0);
136 
137  // Storage for the mesh velocity
138  Vector<double> mesh_velocity(SPATIAL_DIM,0.0);
139 
140  //-------------------------------------------------
141  // Calculate derivatives and source function value:
142  //-------------------------------------------------
143  // Loop over the nodes
144  for (unsigned l=0; l<n_node; l++)
145  {
146  // Get the nodal value at the l-th node
147  double u_value=raw_nodal_value(l,u_nodal_index);
148 
149  // Update the interpolated time value
150  interpolated_t+=raw_nodal_position(l,SPATIAL_DIM)*psi(l);
151 
152  // Loop over the coordinate directions (both spatial AND time)
153  for (unsigned j=0; j<SPATIAL_DIM; j++)
154  {
155  // Update the interpolated x value
156  interpolated_x[j]+=raw_nodal_position(l,j)*psi(l);
157 
158  // Update the interpolated du/dx_j value
159  interpolated_dudx[j]+=u_value*dpsidx(l,j);
160  }
161 
162  // Update the interpolated u value
163  interpolated_u+=u_value*psi(l);
164 
165  // Update the interpolated du/dt value
166  interpolated_dudt+=u_value*dpsidx(l,SPATIAL_DIM);
167  } // for (unsigned l=0;l<n_node;l++)
168 
169  // Initialise the source term value
170  double source=0.0;
171 
172  // Get the interpolated source term value
173  get_source_ust_heat(interpolated_t,ipt,interpolated_x,source);
174 
175  //---------------------------------
176  // Assemble residuals and Jacobian:
177  //---------------------------------
178  // Loop over the nodes (or equivalently the test functions)
179  for (unsigned l=0; l<n_node; l++)
180  {
181  // Get the local equation number
182  local_eqn=nodal_local_eqn(l,u_nodal_index);
183 
184  // If it's not a boundary condition
185  if (local_eqn>=0)
186  {
187  // Add source term and time derivative
188  residuals[local_eqn]+=(source+alpha_local*interpolated_dudt)*test(l)*W;
189 
190  // Loop over the coordinate directions
191  for (unsigned k=0; k<SPATIAL_DIM; k++)
192  {
193  // Add in the contribution from the Laplace operator
194  residuals[local_eqn]+=beta_local*interpolated_dudx[k]*dtestdx(l,k)*W;
195  }
196 
197  //------------------------
198  // Calculate the Jacobian:
199  //------------------------
200  // If we also need to construct the Jacobian
201  if (flag)
202  {
203  // Loop over the velocity shape functions again
204  for (unsigned l2=0; l2<n_node; l2++)
205  {
206  // Get the local equation number
207  local_unknown=nodal_local_eqn(l2,u_nodal_index);
208 
209  // If we're at a non-zero degree of freedom add in the entry
210  if (local_unknown>=0)
211  {
212  // Add in the time derivative contribution
213  jacobian(local_eqn,local_unknown)+=(alpha_local*test(l)*
214  dpsidx(l2,SPATIAL_DIM)*W);
215 
216  // Laplace operator
217  for (unsigned i=0; i<SPATIAL_DIM; i++)
218  {
219  // Add the test function contribution to the Jacobian
220  jacobian(local_eqn,local_unknown)+=(beta_local*dpsidx(l2,i)*
221  dtestdx(l,i)*W);
222  }
223  } // if (local_unknown>=0)
224  } // for (unsigned l2=0;l2<n_node;l2++)
225  } // if (flag)
226  } // if (local_eqn>=0)
227  } // for (unsigned l=0;l<n_node;l++)
228  } // for (unsigned ipt=0;ipt<n_intpt;ipt++)
229  } // End of fill_in_generic_residual_contribution_ust_heat

235  template <unsigned SPATIAL_DIM>
236  void SpaceTimeUnsteadyHeatEquations<SPATIAL_DIM>::compute_norm(double& norm)
237  {
238  // Initialise
239  norm=0.0;
240 
241  // Vector of local coordinates
242  Vector<double> s(SPATIAL_DIM+1,0.0);
243 
244  // Vector for coordinates
245  Vector<double> x(SPATIAL_DIM+1,0.0);
246 
247  // Find out how many nodes there are in the element
248  unsigned n_node=nnode();
249 
250  // Allocate memory for the shape and test functions
251  Shape psi(n_node);
252 
253  // Set the value of n_intpt
254  unsigned n_intpt=integral_pt()->nweight();
255 
256  // Loop over the integration points
257  for (unsigned ipt=0; ipt<n_intpt; ipt++)
258  {
259  // Assign values of s
260  for (unsigned i=0; i<SPATIAL_DIM+1; i++)
261  {
262  // Get the i-th local coordinate at the ipt-th integration point
263  s[i]=integral_pt()->knot(ipt,i);
264  }
265 
266  // Get the integral weight
267  double w=integral_pt()->weight(ipt);
268 
269  // Get Jacobian of mapping
270  double J=J_eulerian(s);
271 
272  // Pre-multiply the weights and the Jacobian
273  double W=w*J;
274 
275  // Get FE function value
276  double u=interpolated_u_ust_heat(s);
277 
278  // Update the norm value
279  norm+=u*u*W;
280  } // for (unsigned ipt=0;ipt<n_intpt;ipt++)
281  } // End of compute_norm
