36  /// \short Get all the boundary information of an element using the
37  /// input (quad_mesh_pt) mesh. If the element lies on a boundary then
38  /// the user will be given the corresponding boundary index and the
39  /// index of the face of quad_el_pt attached to the boundary. If the
40  /// element does NOT lie on any boundaries, this function simply
41  /// returns a vector of size zero.
42  //========================================================================
43  template<class ELEMENT>
44  Vector<std::pair<unsigned,int> > ExtrudedCubeMeshFromQuadMesh<ELEMENT>::
45  get_element_boundary_information(QuadMeshBase* quad_mesh_pt,
46  FiniteElement* quad_el_pt)
47  {
48  // Allocate space for the boundary info (initialised to empty)
49  Vector<std::pair<unsigned,int> > boundary_info;
50 
51  // Find the number of boundaries in the input mesh
52  unsigned n_boundary=quad_mesh_pt->nboundary();
53 
54  // Loop over the boundaries of the mesh
55  for (unsigned b=0; b<n_boundary; b++)
56  {
57  // How many elements are there on the b-th boundary?
58  unsigned n_boundary_element=quad_mesh_pt->nboundary_element(b);
59 
60  // Loop over the elements on the b-th boundary
61  for (unsigned e=0; e<n_boundary_element; e++)
62  {
63  // Is the e-th element on the b-th boundary the input element?
64  if (quad_el_pt==quad_mesh_pt->boundary_element_pt(b,e))
65  {
66  // Create a pair to hold the boundary index and element face index
67  std::pair<unsigned,int> boundary_and_face_index;
68 
69  // Set the first entry (boundary index)
70  boundary_and_face_index.first=b;
71 
72  // Set the second entry (face index)
73  boundary_and_face_index.second=
74  quad_mesh_pt->face_index_at_boundary(b,e);
75 
76  // Add the boundary index to the boundary_info vector
77  boundary_info.push_back(boundary_and_face_index);
78  }
79  } // for (unsigned e=0;e<n_boundary_element;e++)
80  } // for (unsigned b=0;b<n_boundary;b++)
81 
82  // Return the result
83  return boundary_info;
84  } // End of get_element_boundary_information

87  /// Generic mesh construction. This function contains all the details of
88  /// the mesh generation process, including all the tedious loops, counting
89  /// spacing and boundary functions.
90  /// NOTE: The boundary number of the extruded mesh will follow the same
91  /// numbering as the input quad mesh. The newly created "front" and "back"
92  /// face of the 3D mesh will added after those boundaries. For example,
93  /// if the input mesh has 4 boundaries; b_0, b_1, b_2 & b_3 then the
94  /// extruded mesh will have 6 boundaries; eb_0, eb_1, eb_2, eb_3, eb_4 &
95  /// eb_5 where eb_4 is the "front" face and eb5 is the "back" face. The
96  /// boundaries eb_i here satisfy eb_i = (b_i) x [Zmin,Zmax].
97  //========================================================================

180  std::ostringstream error_message;

183  error_message << "Extrude2DQuadMeshTo3DCubeMesh needs the number of "
184  << "nodes (in the 2D element) in one direction\n to match "
185  << "the number of nodes in one direction in the 3D element!";

189  OOMPH_CURRENT_FUNCTION,

195

232  // already created an edge (and thus the nodes on that edge). If the
233  // nodes have already been created then all we have to do is copy them
234  // over in reverse order
235  std::map<std::pair<Edge,double>,Vector<Node*> > edge_to_nodes_map;
236 
237  // As several edges can attach to a single (corner) node we can
238  // accidentally create duplicate nodes (i.e. nodes that lie at
239  // the same spatial position) because the edges aren't covered in
240  // the required order (something we shouldn't rely on anyway!). To
241  // get around this we store the corner nodes of each quad element
242  // with its corresponding z-value. This will in turn return the
243  // corresponding node in the extruded mesh
244  std::map<std::pair<Node*,double>,Node*> quad_corner_node_to_cube_node_map;
245 
246  // Loop over all the elements in one slice
247  for (unsigned e=0; e<n_element_in_quad_mesh; e++)
248  {
249  // Get a pointer to the e-th element in the input mesh
250  FiniteElement* quad_el_pt=quad_mesh_pt->finite_element_pt(e);
251 
252  // Get the index of the cube element in the extruded mesh
253  unsigned element_index=i*n_element_in_quad_mesh+e;
254 
255  // Create the e-th element in the i-th slice of the extruded mesh
256  ELEMENT* cube_el_pt=new ELEMENT;
257 
258  // Put it in global storage!
259  Element_pt[element_index]=cube_el_pt;
260 
261  // Index of the local node number (in the quad element)
262  unsigned quad_local_node_number=0;
263 
264  // Index of the local node number (in the cube element)
265  unsigned cube_local_node_number=0;
266 
267  // Cache the boundary information associated with this element
268  Vector<std::pair<unsigned,int> > el_boundary_information=
269  boundary_information[e];
270 
271  // Loop over the nodes in the third local coordinate direction
272  for (unsigned n_2=0; n_2<n_node_1d; n_2++)
273  {
274  // Calculate the z-value at the n_2-th (z-)node in the i-th element slice
275  double z_value=z_spacing_function(i,n_2);
276 
277  //---------------------------------------------------------------
278  // At a given node slice, there are three options:
279  // (1) If i>0 and n_2=0 then we already created the nodes
280  // (in the current slice) when we created the nodes in
281  // the previous slice;
282  // (2) If (1) doesn't hold we may have still have created
283  // certain nodes along a face/edge of an element that
284  // we previously created (in this node slice);
285  // (3) No nodes in this node slice have been created yet so
286  // set them all up!
287  // NOTE: if (1) occurs, we can skip to the next node slice but
288  // we can't if (2) occurs (or (3), obviously!).
289  //---------------------------------------------------------------
290  //---------------------------------------------------------------
291  // Case (1): If we're past the first element slice and we're on
292  // the first node slice (in the element) then copy the nodes from
293  // the appropriate element.
294  //---------------------------------------------------------------
295  if ((i>0)&&(n_2==0))
296  {
297  // Get the index of the cube element (to copy) in the extruded mesh
298  unsigned copy_element_index=(i-1)*n_element_in_quad_mesh+e;
299 
300  // Get a pointer to the element in the previous slice
301  // containing the node we want
302  FiniteElement* copy_cube_el_pt=finite_element_pt(copy_element_index);
303 
304  // Storage for the local node number in the element that we're copying
305  unsigned copy_cube_local_node_number=0;
306 
307  // Loop over the nodes in the second local coordinate direction
308  for (unsigned n_1=0; n_1<n_node_1d; n_1++)
309  {
310  // Loop over the nodes in the first local coordinate direction
311  for (unsigned n_0=0; n_0<n_node_1d; n_0++)
312  {
313  // Calculate the local node number (in the extruded mesh)
314  cube_local_node_number=n_0+(n_node_1d*n_1);
315 
316  // Calculate the local node number (in the input mesh)
317  quad_local_node_number=n_0+(n_node_1d*n_1);
318 
319  // Calculate the local node number (in the element we're
320  // copying it from)
321  copy_cube_local_node_number=cube_local_node_number+
322  n_node_1d*n_node_1d*(n_node_1d-1);
323 
324  // Copy the node over
325  cube_el_pt->node_pt(cube_local_node_number)=
326  copy_cube_el_pt->node_pt(copy_cube_local_node_number);
327  } // for (unsigned n_0=0;n_0<n_node_1d;n_0++)
328  } // for (unsigned n_1=0;n_1<n_node_1d;n_1++)
329  }
330  //---------------------------------------------------------------
331  // Case (2) & (3): Loop over the edges in the current node slice
332  // and check if any of them have already been set up (and thus
333  // stored in edge_to_nodes_map). If they have then copy them over
334  // and construct all the other nodes. If they haven't then simply
335  // construct all of the nodes.
336  // NOTE: if an edge has been created already, we can't make the
337  // nodes first, assign the coordinates THEN check as it would
338  // be a wasteful operation. Instead, we get the nodes describing
339  // an edge from the input mesh (quad_mesh_pt) and pair it with
340  // the current z-value.
341  //---------------------------------------------------------------
342  else
343  {
344  // Create a vector of bools to tell us if we've set the nodes up
345  // in the current node slice
346  std::vector<bool> has_node_been_setup(n_node_1d*n_node_1d,false);
347 
348  // Number of edges in a slice (N/E/S/W)
349  unsigned n_edge=4;
350 
351  // Storage for the edges
352  Vector<std::pair<Edge,double> > edges_and_z_value;
353 
354  // Vector to tell us if we are copying any edges (to make sure
355  // we don't store the edge again later!)
356  std::vector<bool> has_edge_been_done(n_edge,false);
357 
358  // What is the last edge? Convert this to an int so the compiler doesn't
359  // complain about comparisons between unsigned and signed integers...
360  int last_edge=int(QuadTreeNames::N+n_edge);
361 
362  // Loop over the edges
363  for (int i_edge=QuadTreeNames::N; i_edge<last_edge; i_edge++)
364  {
365  // Pointer for the first node associated with this edge
366  Node* node1_pt=0;
367 
368  // Pointer for the second node associated with this edge
369  Node* node2_pt=0;
370 
371  // Find the corner nodes associated with this edge
372  switch (i_edge)
373  {
374  case QuadTreeNames::N:
375  // The first node
376  node1_pt=quad_el_pt->node_pt((n_node_1d*n_node_1d)-1);
377 
378  // The second node
379  node2_pt=quad_el_pt->node_pt(n_node_1d*(n_node_1d-1));
380 
381  // Break
382  break;
383  case QuadTreeNames::E:
384  // The first node
385  node1_pt=quad_el_pt->node_pt(n_node_1d-1);
386 
387  // The second node
388  node2_pt=quad_el_pt->node_pt((n_node_1d*n_node_1d)-1);
389 
390  // Break
391  break;
392  case QuadTreeNames::S:
393  // The first node
394  node1_pt=quad_el_pt->node_pt(0);
395 
396  // The second node
397  node2_pt=quad_el_pt->node_pt(n_node_1d-1);
398 
399  // Break
400  break;
401  case QuadTreeNames::W:
402  // The first node
403  node1_pt=quad_el_pt->node_pt(n_node_1d*(n_node_1d-1));
404 
405  // The second node
406  node2_pt=quad_el_pt->node_pt(0);
407 
408  // Break
409  break;
410  default:
411  // Create an ostringstream object to create an error message
412  std::ostringstream error_message_stream;
413 
414  // Create the error message
415  error_message_stream << "Input is " << i_edge
416  << " but it can only\n"
417  << "be either N/E/S/W, i.e. "
418  << QuadTreeNames::N << ","
419  << QuadTreeNames::E << ","
420  << QuadTreeNames::S << " or "
421  << QuadTreeNames::W << std::endl;

424  throw OomphLibError(error_message_stream.str(),
425  OOMPH_CURRENT_FUNCTION,
426  OOMPH_EXCEPTION_LOCATION);
427  }

430  Edge edge(node1_pt,node2_pt);
431 
432  // Pair it up with the z-value at the current node slice
433  std::pair<Edge,double> edge_and_z_value_pair(edge,z_value);
434 
435  // Store the edge (with its corresponding z-value)
436  edges_and_z_value.push_back(edge_and_z_value_pair);

441

443  if (it!=edge_to_nodes_map.end())
444  {
445  // Get the vector of node pointers
446  Vector<Node*> edge_nodes_pt=it->second;
447 
448 #ifdef PARANOID

457 #endif

460  switch (i_edge)
461  {
462  case QuadTreeNames::N:
463  // Indicate that the Northern edge has been done
464  has_edge_been_done[0]=true;
465 
466  // Break
467  break;
468  case QuadTreeNames::E:
469  // Indicate that the Eastern edge has been done
470  has_edge_been_done[1]=true;
471 
472  // Break
473  break;
474  case QuadTreeNames::S:
475  // Indicate that the Southern edge has been done
476  has_edge_been_done[2]=true;
477 
478  // Break
479  break;

481  // Indicate that the Western edge has been done
482  has_edge_been_done[3]=true;
483 
484  // Break
485  break;

488  throw OomphLibError("Incorrect edge type!",
489  OOMPH_CURRENT_FUNCTION,
490  OOMPH_EXCEPTION_LOCATION);
491  }
492

494  // Assign the nodes appropriately depending on which edge has
495  // already been created.
496  // NOTE: the nodes will have been created in anti-clockwise
497  // order which means we will have to assign them in clockwise
498  // order (from the perspective of the current element).
499  //---------------------------------------------------------------
500  // Loop over the nodes
501  for (unsigned i_node=0; i_node<n_node_1d; i_node++)
502  {
503  // Calculate the value of quad_local_node_number depending on
504  // which edge the node lies on
505  switch (i_edge)
506  {
507  case QuadTreeNames::N:
508  // Calculate this Node's quad local node number
509  quad_local_node_number=((n_node_1d*n_node_1d)-1)-i_node;
510 
511  // Break
512  break;
513  case QuadTreeNames::E:
514  // Calculate this Node's quad local node number
515  quad_local_node_number=(n_node_1d-1)+(i_node*n_node_1d);
516 
517  // Break
518  break;
519  case QuadTreeNames::S:
520  // Calculate this Node's quad local node number
521  quad_local_node_number=i_node;
522 
523  // Break
524  break;
525  case QuadTreeNames::W:
526  // Calculate this Node's quad local node number
527  quad_local_node_number=
528  (n_node_1d*(n_node_1d-1))-(i_node*n_node_1d);
529 
530  // Break
531  break;
532  }
533 
534  // Calculate this Node's cube local node number
535  cube_local_node_number=quad_local_node_number+
536  (n_node_1d*n_node_1d*n_2);
537 
538  // Assign the i_node-th node
539  cube_el_pt->node_pt(cube_local_node_number)=
540  edge_nodes_pt[(n_node_1d-1)-i_node];
541 
542  // Indicate that the appropriate node has been set up
543  has_node_been_setup[quad_local_node_number]=true;
544  } // for (unsigned i_node=0;i_node<n_node_1d;i_node++)
545  } // if (it!=edge_to_nodes_map.end())
546  } // for (unsigned i_edge=0;i<n_edge;i_edge++)

548  // The number of corners
549  unsigned n_corner=4;
550 
551  // Loop over the corner nodes
552  for (unsigned i_corner=0; i_corner<n_corner; i_corner++)
553  {
554  // Calculate the value of quad_local_node_number depending on
555  // which corner the node lies on
556  switch (i_corner)
557  {
558  case 0:
559  // Bottom-left corner
560  quad_local_node_number=0;
561 
562  // Break
563  break;
564  case 1:
565  // Bottom-right corner
566  quad_local_node_number=n_node_1d-1;
567 
568  // Break
569  break;
570  case 2:
571  // Top-left corner
572  quad_local_node_number=n_node_1d*(n_node_1d-1);
573 
574  // Break
575  break;
576  case 3:
577  // Top-right corner
578  quad_local_node_number=(n_node_1d*n_node_1d)-1;
579 
580  // Break
581  break;
582  }
583 
584  // If the node hasn't already been copied from some other edge
585  if (!has_node_been_setup[quad_local_node_number])
586  {
587  // Get a pointer to the first corner node (in the quad mesh)
588  Node* quad_corner_node_pt=
589  quad_el_pt->node_pt(quad_local_node_number);
590

592  std::pair<Node*,double> quad_corner_node_and_z_value(
593  quad_corner_node_pt,z_value);

890  // Break

898  case 3:
899  // Western edge

929  // which corner the node lies on

935

937  break;
938  case 1:

941 
942  // Break
943  break;
944  case 2:
945  // Top-left corner
946  quad_local_node_number=n_node_1d*(n_node_1d-1);
947 
948  // Break
949  break;
950  case 3:
951  // Top-right corner
952  quad_local_node_number=(n_node_1d*n_node_1d)-1;
953 
954  // Break
955  break;
956  }
957
