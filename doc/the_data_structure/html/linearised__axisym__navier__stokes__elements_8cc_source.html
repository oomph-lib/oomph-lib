242  /// Get strain-rate tensor: \f$ e_{ij} \f$ where
243  /// \f$ i,j = r,z,\theta \f$ (in that order). We evaluate this tensor
244  /// at a value of theta such that the product of theta and the azimuthal
245  /// mode number (k) gives \f$ \pi/4 \f$. Therefore
246  /// \f$ \cos(k \theta) = \sin(k \theta) = 1/\sqrt{2} \f$.
247  //=======================================================================
248  void LinearisedAxisymmetricNavierStokesEquations::
249  strain_rate(const Vector<double>& s, DenseMatrix<double>& strainrate) const
250  {
251 #ifdef PARANOID
252  if ((strainrate.ncol()!=3)||(strainrate.nrow()!=3))
253  {
254  std::ostringstream error_message;
255  error_message << "The strain rate has incorrect dimensions "
256  << strainrate.ncol() << " x "
257  << strainrate.nrow() << " Not 3" << std::endl;
258 
259  throw OomphLibError(
260  error_message.str(),
261  "LinearisedAxisymmetricNavierStokeEquations::strain_rate()",
262  OOMPH_EXCEPTION_LOCATION);
263  }
264 #endif
265 
266  // Determine number of nodes in element
267  const unsigned n_node = nnode();
268 
269  // Set up memory for the shape and test functions
270  Shape psi(n_node);
271  DShape dpsidx(n_node,2);
272 
273  // Call the derivatives of the shape functions
274  dshape_eulerian(s,psi,dpsidx);
275 
276  // Radius
277  double interpolated_r = 0.0;
278 
279  // Velocity components and their derivatives
280  double UC = 0.0, US = 0.0;
281  double dUCdr = 0.0, dUSdr = 0.0;
282  double dUCdz = 0.0, dUSdz = 0.0;
283  double WC = 0.0, WS = 0.0;
284  double dWCdr = 0.0, dWSdr = 0.0;
285  double dWCdz = 0.0, dWSdz = 0.0;
286  double VC = 0.0, VS = 0.0;
287  double dVCdr = 0.0, dVSdr = 0.0;
288  double dVCdz = 0.0, dVSdz = 0.0;
289 
290  // Get the local storage for the indices
291  unsigned u_nodal_index[6];
292  for(unsigned i=0;i<6;++i)
293  {
294  u_nodal_index[i] = u_index_linearised_axi_nst(i);
295  }
296 
297  // Loop over nodes to assemble velocities and their derivatives
298  // w.r.t. r and z (x_0 and x_1)
299  for(unsigned l=0;l<n_node;l++)
300  {
301  interpolated_r += nodal_position(l,0)*psi[l];
302 
303  UC += nodal_value(l,u_nodal_index[0])*psi[l];
304  US += nodal_value(l,u_nodal_index[1])*psi[l];
305  WC += nodal_value(l,u_nodal_index[2])*psi[l];
306  WS += nodal_value(l,u_nodal_index[3])*psi[l];
307  VC += nodal_value(l,u_nodal_index[4])*psi[l];
308  VS += nodal_value(l,u_nodal_index[4])*psi[l];
309 
310  dUCdr += nodal_value(l,u_nodal_index[0])*dpsidx(l,0);
311  dUSdr += nodal_value(l,u_nodal_index[1])*dpsidx(l,0);
312  dWCdr += nodal_value(l,u_nodal_index[2])*dpsidx(l,0);
313  dWSdr += nodal_value(l,u_nodal_index[3])*dpsidx(l,0);
314  dVCdr += nodal_value(l,u_nodal_index[4])*dpsidx(l,0);
315  dVSdr += nodal_value(l,u_nodal_index[5])*dpsidx(l,0);
316 
317  dUCdz += nodal_value(l,u_nodal_index[0])*dpsidx(l,1);
318  dUSdz += nodal_value(l,u_nodal_index[1])*dpsidx(l,1);
319  dWCdz += nodal_value(l,u_nodal_index[2])*dpsidx(l,1);
320  dWSdz += nodal_value(l,u_nodal_index[3])*dpsidx(l,1);
321  dVCdz += nodal_value(l,u_nodal_index[4])*dpsidx(l,1);
322  dVSdz += nodal_value(l,u_nodal_index[5])*dpsidx(l,1);
323  }
324 
325  // Cache azimuthal mode number
326  const int k = this->azimuthal_mode_number();
327 
328  // We wish to evaluate the strain-rate tensor at a value of theta
329  // such that k*theta = pi/4 radians. That way we pick up equal
330  // contributions from the real and imaginary parts of the velocities.
331  // sin(pi/4) = cos(pi/4) = 1/sqrt(2)
332  const double cosktheta = 1.0/sqrt(2);
333  const double sinktheta = cosktheta;
334 
335  // Assemble velocities and their derivatives w.r.t. r, z and theta
336  // from real and imaginary parts
337  const double ur = UC*cosktheta + US*sinktheta;
338  const double utheta = VC*cosktheta + VS*sinktheta;
339 
340  const double durdr = dUCdr*cosktheta + dUSdr*sinktheta;
341  const double durdz = dUCdz*cosktheta + dUSdz*sinktheta;
342  const double durdtheta = k*US*cosktheta - k*UC*sinktheta;
343 
344  const double duzdr = dWCdr*cosktheta + dWSdr*sinktheta;
345  const double duzdz = dWCdz*cosktheta + dWSdz*sinktheta;
346  const double duzdtheta = k*WS*cosktheta - k*WC*sinktheta;
347 
348  const double duthetadr = dVCdr*cosktheta + dVSdr*sinktheta;
349  const double duthetadz = dVCdz*cosktheta + dVSdz*sinktheta;
350  const double duthetadtheta = k*VS*cosktheta - k*VC*sinktheta;
351 
352  // Assign strain rates without negative powers of the radius
353  // and zero those with:
354  strainrate(0,0)=durdr;
355  strainrate(0,1)=0.5*(durdz+duzdr);
356  strainrate(1,0)=strainrate(0,1);
357  strainrate(0,2)=0.5*duthetadr;
358  strainrate(2,0)=strainrate(0,2);
359  strainrate(1,1)=duzdz;
360  strainrate(1,2)=0.5*duthetadz;
361  strainrate(2,1)=strainrate(1,2);
362  strainrate(2,2)=0.0;
363 
364  // Overwrite the strain rates with negative powers of the radius
365  // unless we're at the origin
366  if (std::abs(interpolated_r)>1.0e-16)
367  {
368  double inverse_radius=1.0/interpolated_r;
369  strainrate(0,2)=0.5*(duthetadr + inverse_radius*(durdtheta - utheta));
370  strainrate(2,0)=strainrate(0,2);
371  strainrate(2,2)=inverse_radius*(ur + duthetadtheta);
372  strainrate(1,2)=0.5*(duthetadz + inverse_radius*duzdtheta);
373  strainrate(2,1)=strainrate(1,2);
374  }
375 
376  } // End of strain_rate

381  /// Compute the residuals for the linearised axisymmetric Navier--Stokes
382  /// equations; flag=1(or 0): do (or don't) compute the Jacobian as well.
383  //=======================================================================
384  void LinearisedAxisymmetricNavierStokesEquations::
385  fill_in_generic_residual_contribution_linearised_axi_nst(
386  Vector<double> &residuals,
387  DenseMatrix<double> &jacobian,
388  DenseMatrix<double> &mass_matrix,
389  unsigned flag)
390  {
391  // Get the time from the first node in the element
392  const double time = this->node_pt(0)->time_stepper_pt()->time();
393 
394  // Determine number of nodes in the element
395  const unsigned n_node = nnode();
396 
397  // Determine how many pressure values there are associated with
398  // a single pressure component
399  const unsigned n_pres = npres_linearised_axi_nst();
400 
401  // Get the nodal indices at which the velocity is stored
402  unsigned u_nodal_index[6];
403  for(unsigned i=0;i<6;++i)
404  {
405  u_nodal_index[i] = u_index_linearised_axi_nst(i);
406  }
407 
408  // Set up memory for the fluid shape and test functions
409  // Note that there are two spatial dimensions, r and z, in this problem
410  Shape psif(n_node), testf(n_node);
411  DShape dpsifdx(n_node,2), dtestfdx(n_node,2);
412 
413  // Set up memory for the pressure shape and test functions
414  Shape psip(n_pres), testp(n_pres);
415 
416  // Determine number of integration points
417  const unsigned n_intpt = integral_pt()->nweight();
418 
419  // Set up memory for the vector to hold local coordinates (two dimensions)
420  Vector<double> s(2);
421 
422  // Get physical variables from the element
423  // (Reynolds number must be multiplied by the density ratio)
424  const double scaled_re = re()*density_ratio();
425  const double scaled_re_st = re_st()*density_ratio();
426  const double visc_ratio = viscosity_ratio();
427  const int k = azimuthal_mode_number();
428 
429  // Integers used to store the local equation and unknown numbers
430  int local_eqn = 0, local_unknown = 0;
431 
432  // Loop over the integration points
433  for(unsigned ipt=0;ipt<n_intpt;ipt++)
434  {
435  // Assign values of the local coordinates s
436  for(unsigned i=0;i<2;i++) { s[i] = integral_pt()->knot(ipt,i); }
437 
438  // Get the integral weight
439  const double w = integral_pt()->weight(ipt);
440 
441  // Calculate the fluid shape and test functions, and their derivatives
442  // w.r.t. the global coordinates
443  const double J =
444  dshape_and_dtest_eulerian_at_knot_linearised_axi_nst(ipt,psif,dpsifdx,
445  testf,dtestfdx);
446 
447  // Calculate the pressure shape and test functions
448  pshape_linearised_axi_nst(s,psip,testp);
449 
450  // Premultiply the weights and the Jacobian of the mapping between
451  // local and global coordinates
452  const double W = w*J;
453 
454  // Allocate storage for the position and the derivative of the
455  // mesh positions w.r.t. time
456  Vector<double> interpolated_x(2,0.0);
457  Vector<double> mesh_velocity(2,0.0);
458 
459  // Allocate storage for the velocity components (six of these)
460  // and their derivatives w.r.t. time
461  Vector<double> interpolated_u(6,0.0);
462  Vector<double> dudt(6,0.0);
463 
464  // Allocate storage for the pressure components (two of these)
465  Vector<double> interpolated_p(2,0.0);
466 
467  // Allocate storage for the derivatives of the velocity components
468  // w.r.t. global coordinates (r and z)
469  DenseMatrix<double> interpolated_dudx(6,2,0.0);
470 
471  // Calculate pressure at the integration point
472  // -------------------------------------------
473 
474  // Loop over pressure degrees of freedom (associated with a single
475  // pressure component) in the element
476  for(unsigned l=0;l<n_pres;l++)
477  {
478  // Cache the shape function
479  const double psip_ = psip(l);
480 
481  // Loop over the two pressure components
482  for(unsigned i=0;i<2;i++)
483  {
484  // Get the value
485  const double p_value = this->p_linearised_axi_nst(l,i);
486 
487  // Add contribution
488  interpolated_p[i] += p_value*psip_;
489  }
490  } // End of loop over the pressure degrees of freedom in the element
491 
492  // Calculate velocities and their derivatives at the integration point
493  // -------------------------------------------------------------------
494 
495  // Loop over the element's nodes
496  for(unsigned l=0;l<n_node;l++)
497  {
498  // Cache the shape function
499  const double psif_ = psif(l);
500 
501  // Loop over the two coordinate directions
502  for(unsigned i=0;i<2;i++)
503  {
504  interpolated_x[i] += this->raw_nodal_position(l,i)*psif_;
505  }
506 
507  // Loop over the six velocity components
508  for(unsigned i=0;i<6;i++)
509  {
510  // Get the value
511  const double u_value = this->raw_nodal_value(l,u_nodal_index[i]);
512 
513  // Add contribution
514  interpolated_u[i] += u_value*psif_;
515 
516  // Add contribution to dudt
517  dudt[i] += du_dt_linearised_axi_nst(l,i)*psif_;
518 
519  // Loop over two coordinate directions (for derivatives)
520  for(unsigned j=0;j<2;j++)
521  {
522  interpolated_dudx(i,j) += u_value*dpsifdx(l,j);
523  }
524  }
525  } // End of loop over the element's nodes
526 
527  // Get the mesh velocity if ALE is enabled
528  if(!ALE_is_disabled)
529  {
530  // Loop over the element's nodes
531  for(unsigned l=0;l<n_node;l++)
532  {
533  // Loop over the two coordinate directions
534  for(unsigned i=0;i<2;i++)
535  {
536  mesh_velocity[i] += this->raw_dnodal_position_dt(l,i)*psif(l);
537  }
538  }
539  }
