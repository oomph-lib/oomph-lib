52  /// \short A policy class that serves to establish the
53  /// common interfaces for elements that contain moving nodes. This
54  /// class provides storage for the geometric data that affect the
55  /// update of all the nodes of the element, i.e. USUALLY
56  /// all data that are using during a call
57  /// to the Element's node_update() function. In some cases
58  /// (e.g. FluidInterfaceEdge elements), node_update() is overloaded to
59  /// perform an update of the bulk element, in which case the additional
60  /// bulk geometric data become external data of the element and the
61  /// function GeneralisedElement::update_in_external_fd(i) is overloaded
62  /// to also perform the bulk node update.
63  /// The storage is populated
64  /// during the assignment of the equation numbers via the
65  /// complete_setup_of_dependencies() function and then local equations
66  /// numbers are assigned to these data, accessible via
67  /// geometric_data_local_eqn(n,i). Finally, a function is provided
68  /// that calculates the terms in the jacobian matrix by due to these
69  /// geometric data by finite differences.
70  //=======================================================================

131  /// \short Return the local equation number corresponding to the i-th
132  /// value at the n-th geometric data object.
133  inline int geometric_data_local_eqn(const unsigned &n, const unsigned &i)
134  {
135 #ifdef RANGE_CHECKING
136  unsigned n_data = Geom_data_pt.size();
137  if(n >= n_data)
138  {
139  std::ostringstream error_message;
140  error_message << "Range Error: Data number " << n
141  << " is not in the range (0,"
142  << n_data-1 << ")";
143  throw OomphLibError(error_message.str(),
144  OOMPH_CURRENT_FUNCTION,
145  OOMPH_EXCEPTION_LOCATION);
146  }
147  else
148  {
149  unsigned n_value = Geom_data_pt[n]->nvalue();
150  if(i >= n_value)
151  {
152  std::ostringstream error_message;
153  error_message << "Range Error: value " << i << " at data " << n
154  << " is not in the range (0,"
155  << n_value -1 << ")";
156  throw OomphLibError(error_message.str(),
157  OOMPH_CURRENT_FUNCTION,
158  OOMPH_EXCEPTION_LOCATION);
159  }
160  }
161 #endif
162 #ifdef PARANOID
163  //Check that the equations have been allocated
164  if(Geometric_data_local_eqn==0)
165  {
166  throw OomphLibError(
167  "Geometric data local equation numbers have not been allocated",
168  OOMPH_CURRENT_FUNCTION,
169  OOMPH_EXCEPTION_LOCATION);
170  }
171 #endif
172  return Geometric_data_local_eqn[n][i];
