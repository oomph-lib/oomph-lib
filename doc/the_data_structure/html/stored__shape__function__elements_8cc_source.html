58 /// Delete stored shape functions
59 //========================================================================
60 void StorableShapeElementBase::delete_shape_local_stored()
61 {
62  //Can this element delete the stored data?
63  if((Can_delete_shape_local_stored) && (Shape_stored_pt))
64  {
65  //Find the number of entries in the shape function storage vector
66  //N.B. This *should* be the same for all three vectors
67  unsigned n_intpt = Shape_stored_pt->size();
68  //Loop over the entries of each vector, in reverse and delete them
69  for(unsigned ipt=n_intpt;ipt>0;ipt--)
70  {
71  delete (*Shape_stored_pt)[ipt-1];
72  (*Shape_stored_pt)[ipt-1] = 0;
73  }
74  //Delete the vector itself
75  delete Shape_stored_pt;
76  }
77  //Reset the pointer to zero {Must do this even if copied}
78  Shape_stored_pt = 0;
79 }

218 /// Set the spatial integration scheme and also calculate the values of the
219 /// shape functions and their derivatives w.r.t. the local coordinates,
220 /// placing the values into storage so that they may be re-used,
221 /// without recalculation
222 //=======================================================================
223 void StorableShapeElementBase::
224 set_integration_scheme(Integral* const &integral_pt)
225 {
226  //Assign the integration scheme
227  FiniteElement::set_integration_scheme(integral_pt);
228 
229  //If we are storing the shape functions and first and second derivatives
230  if(D2Shape_local_stored_pt!=0)
231  {
232  pre_compute_d2shape_local_at_knots();
233  }
234  //If we are storing the shape functions and first derivatives
235  else if(DShape_local_stored_pt!=0)
236  {
237  pre_compute_dshape_local_at_knots();
238  }
239  //If we are storing the shape functions
240  else if(Shape_stored_pt!=0)
241  {
242  pre_compute_shape_at_knots();
243  }
244 
245  //If we are storing Eulerian first and second derivatives, recompute them
246  if(D2Shape_eulerian_stored_pt!=0)
247  {
248  pre_compute_d2shape_eulerian_at_knots();
249  }
250  //If we are storing Eulerian first derivatives, recompute them
251  else if(DShape_eulerian_stored_pt!=0)
252  {
253  pre_compute_dshape_eulerian_at_knots();
254  }
255  //If we are storing the Jacobian of the mapping from local to Eulerian
256  //coordinates, recompute it
257  else if(Jacobian_eulerian_stored_pt!=0)
258  {
259  pre_compute_J_eulerian_at_knots();
260  }

265 /// Calculate the shape functions at the integration points and store in
266 /// internal storage of the element
267 //========================================================================
268 void StorableShapeElementBase::pre_compute_shape_at_knots()
269 {
270  // Find the number of nodes in the element
271  unsigned n_node = nnode();
272 #ifdef PARANOID
273  if(n_node == 0)
274  {
275  std::string error_message =
276  "FiniteElement::Node_pt must be resized to a value greater than\n";
277  error_message +=
278  "zero before calling pre_compute_shape_at_knots()";
279 
280  throw OomphLibError(error_message,
281  OOMPH_CURRENT_FUNCTION,
282  OOMPH_EXCEPTION_LOCATION);
283  }
284 #endif
285  //Find number of interpolated position dofs
286  unsigned n_position_type = nnodal_position_type();
287 
288  //In case we have an exisiting integration scheme, wipe the stored data
289  delete_shape_local_stored();
290  //Element is now in charge of deleting its own stored data again
291  Can_delete_shape_local_stored = true;
292 
293  //Allocate internal storage for the shape functions
294  Shape_stored_pt = new Vector<Shape*>;
295 
296  // Storage for the shape functions and their local derivatives
297  Shape psi(n_node,n_position_type);
298 
299  //Loop over the integration points
300  unsigned n_intpt = integral_pt()->nweight();
301  for(unsigned ipt=0;ipt<n_intpt;ipt++)
302  {
303  //Get the shape functions
304  FiniteElement::shape_at_knot(ipt,psi);
305 
306  //Set up local storage for the shape functions and derivatives
307  Shape *psi_pt = new Shape(n_node,n_position_type);
308 
309  //Copy the values of the shape functions and their local derivatives
310  //into the element storage
311  for(unsigned n=0;n<n_node;n++)
312  {
313  for(unsigned k=0;k<n_position_type;k++)
314  {
315  (*psi_pt)(n,k) = psi(n,k);
316  }
317  }
318 
319  //Add the pointers to the shape functions to the internal storage
320  Shape_stored_pt->push_back(psi_pt);
321  } //End of loop over integration points
322 }

329 void StorableShapeElementBase::pre_compute_dshape_local_at_knots()
330 {
331  // Find the number of nodes in the element
332  unsigned n_node = nnode();
333 #ifdef PARANOID

335  {
336  std::string error_message =

344  OOMPH_EXCEPTION_LOCATION);
345  }
346 #endif
347  //Find number of interpolated position dofs
348  unsigned n_position_type = nnodal_position_type();
349  //Find spatial dimension of the element
350  unsigned Dim = dim();
351 
352  //In case we have an exisiting integration scheme, wipe the stored data
353  delete_shape_local_stored();
354  delete_dshape_local_stored();
355  //Element is now in charge of deleting its own stored data again
356  Can_delete_shape_local_stored = true;
357

359  Shape_stored_pt = new Vector<Shape*>;
360  DShape_local_stored_pt = new Vector<DShape*>;
361

363  Shape psi(n_node,n_position_type);
364  DShape dpsids(n_node,n_position_type,Dim);
365 
366  //Loop over the integration points
367  unsigned n_intpt = integral_pt()->nweight();
368  for(unsigned ipt=0;ipt<n_intpt;ipt++)

371  FiniteElement::dshape_local_at_knot(ipt,psi,dpsids);
372 
373  //Set up local storage for the shape functions and derivatives
374  Shape *psi_pt = new Shape(n_node,n_position_type);
375  DShape *dpsids_pt = new DShape(n_node,n_position_type,Dim);
376 
377  //Copy the values of the shape functions and their local derivatives
378  //into the element storage

382  {
383  (*psi_pt)(n,k) = psi(n,k);
384 
385  for(unsigned i=0;i<Dim;i++)
386  {(*dpsids_pt)(n,k,i) = dpsids(n,k,i);}

389

402 void StorableShapeElementBase::pre_compute_d2shape_local_at_knots()
403 {
404  // Find the number of nodes in the element
405  unsigned n_node = nnode();
406 #ifdef PARANOID

408  {

421  unsigned n_position_type = nnodal_position_type();
422  //Find spatial dimension of the element
423  unsigned Dim = dim();
424 
425  //Find the number of second derivatives required
426  //N.B. We are assuming that the mixed derivatives are symmetric here
427  unsigned n_deriv=0;
428  switch(Dim)
429  {
430  case 1: n_deriv = 1; break;
431  case 2: n_deriv = 3; break;
432  case 3: n_deriv = 6; break;

434  }

436  //In case we have an exisiting integration scheme, wipe the stored data
437  delete_shape_local_stored();

444  Shape_stored_pt = new Vector<Shape*>;
445  DShape_local_stored_pt = new Vector<DShape*>;
446  D2Shape_local_stored_pt = new Vector<DShape*>;
447 
448  // Storage for the shape functions and their local derivatives
449  Shape psi(n_node,n_position_type);
450  DShape dpsids(n_node,n_position_type,Dim);
451  DShape d2psids(n_node,n_position_type,n_deriv);
452 
453  //Loop over the integration points
454  unsigned n_intpt = integral_pt()->nweight();
455  for(unsigned ipt=0;ipt<n_intpt;ipt++)
456  {
457  //Get the shape functions and local derivatives at the integration point
458  FiniteElement::d2shape_local_at_knot(ipt,psi,dpsids,d2psids);
459 
460  //Set up local storage for the shape functions and derivatives
461  Shape *psi_pt = new Shape(n_node,n_position_type);
462  DShape *dpsids_pt = new DShape(n_node,n_position_type,Dim);
463  DShape *d2psids_pt = new DShape(n_node,n_position_type,n_deriv);
464 
465  //Copy the values of the shape functions and their local derivatives
466  //into the element storage
467  for(unsigned n=0;n<n_node;n++)
468  {
469  for(unsigned k=0;k<n_position_type;k++)
470  {
471  (*psi_pt)(n,k) = psi(n,k);
472 
473  for(unsigned i=0;i<Dim;i++)
474  {(*dpsids_pt)(n,k,i) = dpsids(n,k,i);}
475 
476  for(unsigned i=0;i<n_deriv;i++)
477  {(*d2psids_pt)(n,k,i) = d2psids(n,k,i);}

481  //Add the pointers to the shape functions and derivatives to the internal

486  } //End of loop over integration points

513 //========================================================================
514 /// Calculate the values of the derivatives of the shape functions at the
515 /// integration points and store in the internal storage of the element
516 //========================================================================
517 void StorableShapeElementBase::pre_compute_dshape_eulerian_at_knots()
518 {
519  //Pre-compute the basic shape functions
520  pre_compute_shape_at_knots();
521 
522  //Find the number of nodes
523  unsigned n_node = nnode();
524  //Get the number of position types and the dimension from the element
525  unsigned n_position_type = this->nnodal_position_type();
526  unsigned n_dim = this->nodal_dimension();
527 
528  //Delete the exisiting stored objects
529  delete_J_eulerian_stored();
530  delete_dshape_eulerian_stored();
531  //Now we're in change of deletion again
532  Can_delete_dshape_eulerian_stored=true;
533 
534  //Allocate storage for the stored shape function derivatives
535  DShape_eulerian_stored_pt = new Vector<DShape*>;
536  Jacobian_eulerian_stored_pt = new Vector<double>;
537 
538  //Assign local variables for the shape function and derivatives
539  Shape psi(n_node,n_position_type);
540  DShape dpsidx(n_node,n_position_type,n_dim);
541 
542  //Loop over the integration points
543  unsigned n_intpt = integral_pt()->nweight();
544  for(unsigned ipt=0;ipt<n_intpt;ipt++)
545  {
546  //Get the values of the shape function and derivatives at the
547  //integration point and add to the value of the Jacobian to the
548  //internally stored vector
549  Jacobian_eulerian_stored_pt->
550  push_back(FiniteElement::dshape_eulerian_at_knot(ipt,psi,dpsidx));
551 
552  //Set up local storage for the shape function derivatives
553  DShape* dpsidx_pt = new DShape(n_node,n_position_type,n_dim);
554 
555  //Now copy the values over
556  for(unsigned l=0;l<n_node;l++)
557  {
558  for(unsigned k=0;k<n_position_type;k++)
559  {
560  //First derivatives
561  for(unsigned i=0;i<n_dim;i++) {(*dpsidx_pt)(l,k,i) = dpsidx(l,k,i);}
562  }
563  }
564 
565  //Add the pointer to the vector of stored DShape objects
566  DShape_eulerian_stored_pt->push_back(dpsidx_pt);
567  } //End of loop over integration points
568 }

570 //========================================================================
571 /// Calculate the values of the first and second derivatives of the shape
572 /// functions at the integration points and store in the internal storage
573 /// of the element
574 //========================================================================
575 void StorableShapeElementBase::pre_compute_d2shape_eulerian_at_knots()
576 {

578  pre_compute_shape_at_knots();
579 
580  //Find the number of nodes
581  unsigned n_node = nnode();
582  //Get the number of position types and the dimension from element
583  unsigned n_position_type = this->nnodal_position_type();
584  unsigned n_dim = this->nodal_dimension();
585 
586  //Find the number of second derivatives required
587  //N.B. We are assuming that the mixed derivatives are symmetric here
588  unsigned n_deriv=0;
589  switch(n_dim)
590  {
591  case 1: n_deriv = 1; break;
592  case 2: n_deriv = 3; break;
593  case 3: n_deriv = 6; break;
594  default: oomph_info << "Really more than 3 dimensions?" << std::endl; break;
595  }
596 
597  //Delete the existing objects, if there are any
598  delete_J_eulerian_stored();
599  delete_dshape_eulerian_stored();
600  delete_d2shape_eulerian_stored();
601  //Now we're in change of deletion again
602  Can_delete_dshape_eulerian_stored=true;
603 
604  //Allocate storage for the stored shape function derivatives
605  DShape_eulerian_stored_pt = new Vector<DShape*>;
606  D2Shape_eulerian_stored_pt = new Vector<DShape*>;
607  Jacobian_eulerian_stored_pt = new Vector<double>;
608 
609  //Assign local variables for the shape function and derivatives
610  Shape psi(n_node,n_position_type);
611  DShape dpsidx(n_node,n_position_type,n_dim);
612  DShape d2psidx(n_node,n_position_type,n_deriv);
613 
614  //Loop over the integration points
615  unsigned n_intpt = integral_pt()->nweight();
616  for(unsigned ipt=0;ipt<n_intpt;ipt++)
617  {
618  //Get the values of the shape function and derivatives at the
619  //integration point and assign the value of the Jacobian to the
620  //internally stored vector
621  Jacobian_eulerian_stored_pt->
622  push_back(FiniteElement::d2shape_eulerian_at_knot(ipt,psi,dpsidx,d2psidx));
623
