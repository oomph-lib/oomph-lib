308  /// \short Output boundary coordinates on boundary b -- template argument
309  /// specifies the bulk element type (needed to create FaceElement
310  /// of appropriate type on mesh boundary).
311  template<class BULK_ELEMENT>
312  void doc_boundary_coordinates(const unsigned& b, std::ofstream& the_file)
313  {
314  if (nelement()==0) return;
315  if (!Boundary_coordinate_exists[b])
316  {
317  oomph_info << "No boundary coordinates were set up for boundary "
318  << b << std::endl;
319  return;
320  }
321 
322  // Get spatial dimension
323  unsigned dim=finite_element_pt(0)->node_pt(0)->ndim();
324 
325  // Loop over all elements on boundaries
326  unsigned nel=this->nboundary_element(b);
327 
328  // Loop over the bulk elements adjacent to boundary b
329  for(unsigned e=0;e<nel;e++)
330  {
331  // Get pointer to the bulk element that is adjacent to boundary b
332  FiniteElement* bulk_elem_pt = this->boundary_element_pt(b,e);
333 
334  //Find the index of the face of element e along boundary b
335  int face_index = this->face_index_at_boundary(b,e);
336 
337  // Create new face element
338  DummyFaceElement<BULK_ELEMENT>* el_pt=
339  new DummyFaceElement<BULK_ELEMENT>(bulk_elem_pt,face_index);
340 
341  // Specify boundary id in bulk mesh (needed to extract
342  // boundary coordinate)
343  el_pt->set_boundary_number_in_bulk_mesh(b);
344 
345  // Doc boundary coordinate
346  Vector<double> s(dim-1);
347  Vector<double> zeta(dim-1);
348  Vector<double> x(dim);
349  unsigned n_plot=5;
350  the_file << el_pt->tecplot_zone_string(n_plot);
351 
352  // Loop over plot points
353  unsigned num_plot_points=el_pt->nplot_points(n_plot);
354  for (unsigned iplot=0;iplot<num_plot_points;iplot++)
355  {
356  // Get local coordinates of plot point
357  el_pt->get_s_plot(iplot,n_plot,s);
358  el_pt->interpolated_zeta(s,zeta);
359  el_pt->interpolated_x(s,x);
360  for (unsigned i=0;i<dim;i++)
361  {
362  the_file << x[i] << " ";
363  }
364  for (unsigned i=0;i<(dim-1);i++)
365  {
366  the_file << zeta[i] << " ";
367  }
368 
369  the_file << std::endl;
370  }
371  el_pt->write_tecplot_zone_footer(the_file,n_plot);
372 
373  // Cleanup
374  delete el_pt;
375  }
