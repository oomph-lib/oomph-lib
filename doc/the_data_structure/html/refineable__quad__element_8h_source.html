47  /// Refineable version of QElement<2,NNODE_1D>.
48  ///
49  /// Refinement is performed by quadtree procedures. When the element is
50  /// subdivided, the geometry of its sons is established by calls
51  /// to their father's
52  /// \code get_x(...) \endcode
53  /// function which refers to
54  /// - the father element's geometric FE mapping (this
55  /// is the default)
56  /// .
57  /// or
58  /// - to a MacroElement 's MacroElement::macro_map (if the pointer
59  /// to the macro element is non-NULL)
60  ///
61  /// The class provides a generic RefineableQElement<2>::build() function
62  /// which deals with generic
63  /// isoparametric QElements in which all values are associated with
64  /// nodes. The RefineableQElement<2>::further_build() function provides
65  /// an interface for any element-specific non-generic build operations.
66  ///
67  //=======================================================================

287  /// \short Build the element, i.e. give it nodal positions, apply BCs, etc.
288  /// Incl. documention into new_nodes_file
289  // NOTE: FOR SOME REASON THIS NEEDS TO LIVE IN *.H TO WORK ON INTEL
290  void build(Mesh*& mesh_pt,
291  Vector<Node*>& new_node_pt,
292  bool& was_already_built,
293  std::ofstream& new_nodes_file)
294  {
295  using namespace QuadTreeNames;
296 
297  // Call the standard (non-elastic) build function
298  RefineableQElement<2>::build(
299  mesh_pt, new_node_pt, was_already_built, new_nodes_file);
300 
301  // Are we done?
302  if (was_already_built) return;
303 
304  // Now need to loop over the nodes again and set solid variables
305 
306  // What type of son am I? Ask my quadtree representation...
307  int son_type = Tree_pt->son_type();
308 
309  // Which element (!) is my father? (We must have a father
310  // since was_already_built is false...)
311  RefineableSolidQElement<2>* father_el_pt =
312  dynamic_cast<RefineableSolidQElement<2>*>(
313  Tree_pt->father_pt()->object_pt());
314 
315 
316 #ifdef PARANOID
317  // Currently we can't handle the case of generalised coordinates
318  // since we haven't established how they should be interpolated
319  // Buffer this case:
320  if (static_cast<SolidNode*>(father_el_pt->node_pt(0))
321  ->nlagrangian_type() != 1)
322  {
323  throw OomphLibError(
324  "We can't handle generalised nodal positions (yet).\n",
325  OOMPH_CURRENT_FUNCTION,
326  OOMPH_EXCEPTION_LOCATION);
327  }
328 #endif
329 
330  Vector<double> s_lo(2);
331  Vector<double> s_hi(2);
332  Vector<double> s(2);
333  Vector<double> xi(2);
334  Vector<double> xi_fe(2);
335  Vector<double> x(2);
336  Vector<double> x_fe(2);
337 
338  // Setup vertex coordinates in father element:
339  //--------------------------------------------
340  switch (son_type)
341  {
342  case SW:
343  s_lo[0] = -1.0;
344  s_hi[0] = 0.0;
345  s_lo[1] = -1.0;
346  s_hi[1] = 0.0;
347  break;
348 
349  case SE:
350  s_lo[0] = 0.0;
351  s_hi[0] = 1.0;
352  s_lo[1] = -1.0;
353  s_hi[1] = 0.0;
354  break;
355 
356  case NE:
357  s_lo[0] = 0.0;
358  s_hi[0] = 1.0;
359  s_lo[1] = 0.0;
360  s_hi[1] = 1.0;
361  break;
362 
363  case NW:
364  s_lo[0] = -1.0;
365  s_hi[0] = 0.0;
366  s_lo[1] = 0.0;
367  s_hi[1] = 1.0;
368  break;
369  }
370 
371  // Pass the undeformed macro element onto the son
372  // hierher why can I read this?
373  if (father_el_pt->Undeformed_macro_elem_pt != 0)
374  {
375  Undeformed_macro_elem_pt = father_el_pt->Undeformed_macro_elem_pt;
376  for (unsigned i = 0; i < 2; i++)
377  {
378  s_macro_ll(i) =
379  father_el_pt->s_macro_ll(i) +
380  0.5 * (s_lo[i] + 1.0) *
381  (father_el_pt->s_macro_ur(i) - father_el_pt->s_macro_ll(i));
382  s_macro_ur(i) =
383  father_el_pt->s_macro_ll(i) +
384  0.5 * (s_hi[i] + 1.0) *
385  (father_el_pt->s_macro_ur(i) - father_el_pt->s_macro_ll(i));
386  }
387  }
388 
389  // Local node number
390  unsigned n = 0;
391 
392  // Find number of 1D nodes in element
393  unsigned n_p = nnode_1d();
394 
395  // Loop over nodes in element
396  for (unsigned i0 = 0; i0 < n_p; i0++)
397  {
398  // Local coordinate in father element
399  s[0] = s_lo[0] + (s_hi[0] - s_lo[0]) * double(i0) / double(n_p - 1);
400 
401  for (unsigned i1 = 0; i1 < n_p; i1++)
402  {
403  // Local coordinate in father element
404  s[1] = s_lo[1] + (s_hi[1] - s_lo[1]) * double(i1) / double(n_p - 1);
405 
406  // Local node number
407  n = i0 + n_p * i1;
408 
409  // Get position from father element -- this uses the macro
410  // element representation(s) if appropriate. If the node
411  // turns out to be a hanging node later on, then
412  // its position gets adjusted in line with its
413  // hanging node interpolation.
414  father_el_pt->get_x_and_xi(s, x_fe, x, xi_fe, xi);
415 
416  // Cast the node to an Solid node
417  SolidNode* elastic_node_pt = static_cast<SolidNode*>(node_pt(n));
418 
419  for (unsigned i = 0; i < 2; i++)
420  {
421  // x_fe is the FE representation -- this is all we can
422  // work with in a solid mechanics problem. If you wish
423  // to reposition nodes on curvilinear boundaries of
424  // a domain to their exact positions on those boundaries
425  // you'll have to do this yourself! [Note: We used to
426  // use the macro-element-based representation
427  // to assign the position of pinned nodes but this is not always
428  // correct since pinning doesn't mean "pin in place" or
429  // "pin to the curvilinear boundary". For instance, we could impose
430  // the boundary displacement manually.
431  // x_fe is the FE representation
432  elastic_node_pt->x(i) = x_fe[i];
433 
434  // Lagrangian coordinates can come from undeformed macro element
435 
436  if (Use_undeformed_macro_element_for_new_lagrangian_coords)
437  {
438  elastic_node_pt->xi(i) = xi[i];
439  }
440  else
441  {
442  elastic_node_pt->xi(i) = xi_fe[i];
443  }
444  }
445 
446 
447  // Are there any history values to be dealt with?
448  TimeStepper* time_stepper_pt =
449  father_el_pt->node_pt(0)->time_stepper_pt();
450 
451  // Number of history values (incl. present)
452  unsigned ntstorage = time_stepper_pt->ntstorage();
453  if (ntstorage != 1)
454  {
455  // Loop over # of history values (excluding present which has been
456  // done above)
457  for (unsigned t = 1; t < ntstorage; t++)
458  {
459  // History values can (and in the case of Newmark timestepping,
460  // the scheme most likely to be used for Solid computations, do)
461  // include non-positional values, e.g. velocities and
462  // accelerations.
463 
464  // Set previous positions of the new node
465  for (unsigned i = 0; i < 2; i++)
466  {
467  elastic_node_pt->x(t, i) =
468  father_el_pt->interpolated_x(t, s, i);
469  }
470  }
471  }
472 
473  } // End of vertical loop over nodes in element
474 
475  } // End of horizontal loop over nodes in element
476  }
477  };
