96  void face_local_coordinate_of_flux_interpolation_point(const unsigned &edge,

98  Vector<double> &s)
99  const
100  {
101  // Get the location of the n-th flux interpolation point along
102  // the edge in terms of the distance along the edge itself
103  Vector<double> flux_interpolation_point=
104  edge_flux_interpolation_point(edge,n);
105 
106  // Convert the edge number to the number of the mid-edge node along that
107  // edge

110  // The edge basis functions are defined in a clockwise manner, so we have
111  // to effectively "flip" some coordinates
112  switch(node_number)
113  {
114  case 3:

117  case 4:

254  /// \short Compute the global coordinates of the nth flux interpolation
255  /// point along an edge
256  void edge_flux_interpolation_point_global(const unsigned &edge,
257  const unsigned &n,
258  Vector<double> &x) const
259  {
260  // Get the location of the n-th flux interpolation point along
261  // the edge in terms of the distance along the edge itself
262  Vector<double> flux_interpolation_point=
263  edge_flux_interpolation_point(edge,n);
264 
265  // Convert the edge number to the number of the mid-edge node along that
266  // edge
267  unsigned node_number=Q_edge_conv[edge];
268 
269  // Storage for the local coords of the flux_interpolation point
270  Vector<double> s_flux_interpolation(2,0.0);
271 
272  // The edge basis functions are defined in a clockwise manner, so we have
273  // to effectively "flip" the coordinates along edges 0 and 1 to match this
274  switch(node_number)
275  {
276  case 3:
277  s_flux_interpolation[0]=1.0-flux_interpolation_point[0];
278  s_flux_interpolation[1]=flux_interpolation_point[0];
279  break;
280  case 4:
281  s_flux_interpolation[0]=0.0;
282  s_flux_interpolation[1]=1.0-flux_interpolation_point[0];
283  break;
284  case 5:
285  s_flux_interpolation[0]=flux_interpolation_point[0];
286  s_flux_interpolation[1]=0.0;
287  break;
288  }
289 
290  // Calculate the global coordinates from the local ones
291  interpolated_x(s_flux_interpolation,x);
292  }

331  /// Scale the edge basis to allow arbitrary edge mappings
332  // hierher explain please
333  void scale_basis(Shape& basis) const
334  {
335 
336  // Storage for the lengths of the edges of the element
337  Vector<double> length(3,0.0);
338 
339  // Temporary storage for the vertex positions
340  double x0,y0,x1,y1;
341 
342  // loop over the edges of the element and calculate their lengths (in x-y
343  // space)
344  for(unsigned i=0;i<3;i++)
345  {
346  x0=this->node_pt(i)->x(0);
347  y0=this->node_pt(i)->x(1);
348  x1=this->node_pt((i+1)%3)->x(0);
349  y1=this->node_pt((i+1)%3)->x(1);
350 
351  length[i]=std::sqrt(std::pow(y1-y0,2)+std::pow(x1-x0,2));
352  }
353 
354  // lengths of the sides of the reference element (in the same order as the
355  // basis functions)
356  const double ref_length[3]={std::sqrt(2.0),1,1};
357 
358  // get the number of basis functions associated with the edges
359  unsigned n_q_basis_edge=nq_basis_edge();
360 
361  // rescale the edge basis functions to allow arbitrary edge mappings from
362  // element to ref. element
363  const unsigned n_index2 = basis.nindex2();
364  for(unsigned i=0;i<n_index2;i++)
365  {
366  for(unsigned l=0;l<n_q_basis_edge;l++)
367  {
368  basis(l,i)*=(length[l/(ORDER+1)]/ref_length[l/(ORDER+1)]);
369  }
370  }
371  }
