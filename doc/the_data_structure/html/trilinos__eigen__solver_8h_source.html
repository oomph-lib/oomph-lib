155  /// \brief Update \c mv with \f$ \alpha AB + \beta mv \f$.
156  static void MvTimesMatAddMv(
157  const double alpha,
158  const oomph::DoubleMultiVector& A,
159  const Teuchos::SerialDenseMatrix<int,double>& B,
160  const double beta, oomph::DoubleMultiVector& mv )
161  {
162  //For safety let's (deep) copy A
163  oomph::DoubleMultiVector C(A);
164  //First pre-multiply by the scalars
165  mv *= beta;
166  C *= alpha;
167 
168  //Find the number of vectors in each multivector
169  int mv_n_vector = mv.nvector();
170  int A_n_vector = A.nvector();
171  //Find the number of rows
172  int n_row_local = A.nrow_local();
173  //Now need to multiply by the matrix and add the result
174  //to the appropriate entry in the multivector
175  for(int i=0;i<n_row_local;++i)
176  {
177  for(int j=0;j<mv_n_vector;j++)
178  {
179  double res = 0.0;
180  for(int k=0;k<A_n_vector;k++) {res += C(k,i)*B(k,j);}
181  mv(j,i) += res;
182  }
183  }
184  }

223  /*! \brief Compute a dense matrix \c B through the matrix-matrix multiply \f$ \alpha A^Hmv \f$.
224  */
225  static void MvTransMv( const double alpha, const oomph::DoubleMultiVector& A,
226  const oomph::DoubleMultiVector& mv,
227  Teuchos::SerialDenseMatrix<int,double>& B)
228  {
229  const unsigned A_nvec = A.nvector();
230  const unsigned A_nrow_local = A.nrow_local();
231  const unsigned mv_nvec = mv.nvector();
232  //const unsigned mv_nrow_local = mv.nrow_local();
233 
234  for(unsigned v1=0;v1<A_nvec;v1++)
235  {
236  for(unsigned v2=0;v2<mv_nvec;v2++)
237  {
238  B(v1,v2) = 0.0;
239  for(unsigned i=0;i<A_nrow_local;i++)
240  {
241  B(v1,v2) += A(v1,i)*mv(v2,i);
242  }
243  B(v1,v2) *= alpha;
244  }
245  }
246 
247  //This will need to be communicated if we're in parallel and distributed
248 #ifdef OOMPH_HAS_MPI
249  if (A.distributed() &&
250  A.distribution_pt()->communicator_pt()->nproc() > 1)
251  {
252  const int n_total_val = A_nvec*mv_nvec;
253  //Pack entries into a vector
254  double b[n_total_val]; double b2[n_total_val];
255  unsigned count=0;
256  for(unsigned v1=0;v1<A_nvec;v1++)
257  {
258  for(unsigned v2=0;v2<mv_nvec;v2++)
259  {
260  b[count] = B(v1,v2);
261  b2[count] = b[count];
262  ++count;
263  }
264  }
265 
266 
267  MPI_Allreduce(&b,&b2,n_total_val,MPI_DOUBLE,MPI_SUM,
268  A.distribution_pt()->communicator_pt()->mpi_comm());
269 
270  count=0;
271  for(unsigned v1=0;v1<A_nvec;v1++)
272  {
273  for(unsigned v2=0;v2<mv_nvec;v2++)
274  {
275  B(v1,v2) = b2[count];
276  ++count;
277  }
278  }
279  }
280 #endif
281  }

455 /// Class for the shift invert operation
456 //====================================================================
457 class ProblemBasedShiftInvertOperator: public DoubleMultiVectorOperator
458 {
459 private:
460  //Pointer to the problem
461  Problem* Problem_pt;
462 
463  //Pointer to the linear solver used
464  LinearSolver* Linear_solver_pt;
465 
466  //Storage for the shift
467  double Sigma;
468 
469  //Storage for the matrices
470  CRDoubleMatrix *M_pt, *AsigmaM_pt;
471 
472 public:
473 
474  ProblemBasedShiftInvertOperator(Problem* const &problem_pt,
475  LinearSolver* const &linear_solver_pt,
476  const double &sigma=0.0) :
477  Problem_pt(problem_pt), Linear_solver_pt(linear_solver_pt), Sigma(sigma)
478  {
479  //Before we get into the Arnoldi loop solve the shifted matrix problem
480  //Allocated Row compressed matrices for the mass matrix and shifted main
481  //matrix
482  M_pt = new CRDoubleMatrix(problem_pt->dof_distribution_pt());
483  AsigmaM_pt = new CRDoubleMatrix(problem_pt->dof_distribution_pt());
484 
485  //Assemble the matrices
486  problem_pt->get_eigenproblem_matrices(*M_pt,*AsigmaM_pt,Sigma);
487 
488  //Do not report the time taken
489  Linear_solver_pt->disable_doc_time();
490  }
491 
492 
493  //Now specify how to apply the operator
494  void apply (const DoubleMultiVector &x,
495  DoubleMultiVector &y) const
496  {
497  const unsigned n_vec = x.nvector();
498  const unsigned n_row_local = x.nrow_local();
499  if(n_vec > 1) {Linear_solver_pt->enable_resolve();}
500  //Solve the first vector
501 
502  DoubleVector X(x.distribution_pt());
503  //Premultiply by mass matrix
504  M_pt->multiply(x.doublevector(0),X);
505  //For some reason I need to create a new vector and copy here
506  //This is odd and not expected, examine carefully
507  DoubleVector Y(x.distribution_pt());
508  Linear_solver_pt->solve(AsigmaM_pt,X,Y);
509  //Need to synchronise
510 //#ifdef OOMPH_HAS_MPI
511 // Problem_pt->synchronise_all_dofs();
512 //#endif
513  for(unsigned i=0;i<n_row_local;i++) {y(0,i) = Y[i];}
514 
515  //Now loop over the others and resolve
516  for(unsigned v=1;v<n_vec;++v)
517  {
518  M_pt->multiply(x.doublevector(v),X);
519  Linear_solver_pt->resolve(X,Y);
520 //#ifdef OOMPH_HAS_MPI
521 // Problem_pt->synchronise_all_dofs();
522 //#endif
523  for(unsigned i=0;i<n_row_local;i++) {y(v,i) = Y[i];}
524  }
525  }
