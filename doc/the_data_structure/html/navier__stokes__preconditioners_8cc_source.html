665 //=======================================================================
666  /// Apply preconditioner to r.
667 //=======================================================================
668  void NavierStokesSchurComplementPreconditioner::
669  preconditioner_solve(const DoubleVector &r, DoubleVector &z)
670  {
671 #ifdef PARANOID
672  if (Preconditioner_has_been_setup==false)
673  {
674  std::ostringstream error_message;
675  error_message << "setup must be called before using preconditioner_solve";
676  throw OomphLibError(
677  error_message.str(),
678  OOMPH_CURRENT_FUNCTION,
679  OOMPH_EXCEPTION_LOCATION);
680  }
681  if (z.built())
682  {
683  if (z.nrow() != r.nrow())
684  {
685  std::ostringstream error_message;
686  error_message << "The vectors z and r must have the same number of "
687  << "of global rows";
688  throw OomphLibError(
689  error_message.str(),
690  OOMPH_CURRENT_FUNCTION,
691  OOMPH_EXCEPTION_LOCATION);
692  }
693  }
694 #endif
695 
696  // if z is not setup then give it the same distribution
697  if (!z.distribution_pt()->built())
698  {
699  z.build(r.distribution_pt(),0.0);
700  }
701 
702  // Step 1 - apply approximate Schur inverse to pressure unknowns (block 1)
703  // -----------------------------------------------------------------------
704 
705  // Working vectors
706  DoubleVector temp_vec;
707  DoubleVector another_temp_vec;
708  DoubleVector yet_another_temp_vec;
709 
710  // Copy pressure values from residual vector to temp_vec:
711  // Loop over all entries in the global vector (this one
712  // includes velocity and pressure dofs in some random fashion)
713  this->get_block_vector(1,r,temp_vec);
714 
715  // NOTE: The vector temp_vec now contains the vector r_p.
716 
717  // LSC version
718  if (Use_LSC)
719  {
720  // Solve first pressure Poisson system
721 #ifdef PARANOID
722  // check a solver has been set
723  if (P_preconditioner_pt==0)
724  {
725  std::ostringstream error_message;
726  error_message << "P_preconditioner_pt has not been set.";
727  throw OomphLibError(
728  error_message.str(),
729  OOMPH_CURRENT_FUNCTION,
730  OOMPH_EXCEPTION_LOCATION);
731  }
732 #endif
733 
734  // use some Preconditioner's preconditioner_solve function
735  P_preconditioner_pt->preconditioner_solve(temp_vec, another_temp_vec);
736 
737  // NOTE: The vector another_temp_vec now contains the vector P^{-1} r_p
738 
739  // Multiply another_temp_vec by matrix E and stick the result into temp_vec
740  temp_vec.clear();
741  QBt_mat_vec_pt->multiply(another_temp_vec, temp_vec);
742  another_temp_vec.clear();
743  F_mat_vec_pt->multiply(temp_vec,another_temp_vec);
744  temp_vec.clear();
745  QBt_mat_vec_pt->multiply_transpose(another_temp_vec, temp_vec);
746 
747 
748  // NOTE: The vector temp_vec now contains E P^{-1} r_p
749 
750  // Solve second pressure Poisson system using preconditioner_solve
751  another_temp_vec.clear();
752  P_preconditioner_pt->preconditioner_solve(temp_vec, another_temp_vec);
753 
754  // NOTE: The vector another_temp_vec now contains z_p = P^{-1} E P^{-1} r_p
755  // as required (apart from the sign which we'll fix in the
756  // next step.
757  }
758  // Fp version
759  else
760  {
761 
762  // Multiply temp_vec by matrix E and stick the result into
763  // yet_another_temp_vec
764  E_mat_vec_pt->multiply(temp_vec,yet_another_temp_vec);
765 
766  // NOTE: The vector yet_another_temp_vec now contains Fp Qp^{-1} r_p
767 
768  // Solve pressure Poisson system
769 #ifdef PARANOID
770  // check a solver has been set
771  if (P_preconditioner_pt==0)
772  {
773  std::ostringstream error_message;
774  error_message << "P_preconditioner_pt has not been set.";
775  throw OomphLibError(
776  error_message.str(),
777  OOMPH_CURRENT_FUNCTION,
778  OOMPH_EXCEPTION_LOCATION);
779  }
780 #endif
781 
782  // Solve second pressure Poisson system using preconditioner_solve
783  another_temp_vec.clear();
784  P_preconditioner_pt->preconditioner_solve(yet_another_temp_vec,
785  another_temp_vec);
786 
787  // NOTE: The vector another_temp_vec now contains
788  // z_p = P^{-1} Fp Qp^{-1} r_p
789  // as required (apart from the sign which we'll fix in the
790  // next step.
791 
792  }
793 
794  // Now copy another_temp_vec (i.e. z_p) back into the global vector z.
795  // Loop over all entries in the global results vector z:
796  temp_vec.build(another_temp_vec.distribution_pt(),0.0);
797  temp_vec -= another_temp_vec;
798  return_block_vector(1,temp_vec,z);
799 
800 
801  // Step 2 - apply preconditioner to velocity unknowns (block 0)
802  // ------------------------------------------------------------
803 
804  // Recall that another_temp_vec (computed above) contains the
805  // negative of the solution of the Schur complement systen, -z_p.
806  // Multiply by G (stored in Block_matrix_pt(0,1) and store
807  // result in temp_vec (vector resizes itself).
808  temp_vec.clear();
809  Bt_mat_vec_pt->multiply(another_temp_vec, temp_vec);
810 
811  // NOTE: temp_vec now contains -G z_p
812 
813  // The vector another_temp_vec is no longer needed -- re-use it to store
814  // velocity quantities:
815  another_temp_vec.clear();
816 
817  // Loop over all enries in the global vector and find the
818  // entries associated with the velocities:
819  get_block_vector(0,r,another_temp_vec);
820  another_temp_vec += temp_vec;
821 
822  // NOTE: The vector another_temp_vec now contains r_u - G z_p
823 
824  // Solve momentum system
825 #ifdef PARANOID
826  // check a solver has been set
827  if (F_preconditioner_pt==0)
828  {
829  std::ostringstream error_message;
830  error_message << "F_preconditioner_pt has not been set.";
831  throw OomphLibError(
832  error_message.str(),
833  OOMPH_CURRENT_FUNCTION,
834  OOMPH_EXCEPTION_LOCATION);
835  }
836 #endif
837 
838  // use some Preconditioner's preconditioner solve
839  // and return
840  if (F_preconditioner_is_block_preconditioner)
841  {
842  return_block_vector(0,another_temp_vec,z);
843  F_preconditioner_pt->preconditioner_solve(z,z);
844  }
845  else
846  {
847  F_preconditioner_pt->preconditioner_solve(another_temp_vec, temp_vec);
848  return_block_vector(0,temp_vec,z);
849  }
850  }
