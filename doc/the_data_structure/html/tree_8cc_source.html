119 /// Destructor for Tree: Recursively kill all sons and the
120 /// associated objects of the non-leaf nodes. However, the objects
121 /// of the leaf nodes are not destroyed. Their destruction is handled
122 /// by the Mesh destructor.
123 //=================================================================
124 Tree::~Tree()
125 {
126  //Loop over all the sons and delete them
127  unsigned nsons=Son_pt.size();
128  for(unsigned i=0;i<nsons;i++)
129  {
130  //This will call the son's destructor (a subtle recursion)
131  delete Son_pt[i];
132  Son_pt[i]=0;
133  }
134 
135  //Delete the object only if the Tree has sons (is not a leaf node)
136  if(nsons > 0)
137  {
138  delete Object_pt;
139  Object_pt=0;
140  }
141 }

286 //================================================================
287 /// If required, kill the sons to perform unrefinement.
288 ///
289 /// Unrefinement is performed if
290 ///
291 /// object_pt()->sons_to_be_unrefined()
292 ///
293 /// returns true.
294 //=================================================================
295 void Tree::merge_sons_if_required(Mesh* &mesh_pt)
296 {
297  // Check if unrefinement is required
298  if (Object_pt->sons_to_be_unrefined())
299  {
300  // Rebuild the father from sons
301  object_pt()->rebuild_from_sons(mesh_pt);
302 
303  //Find the number of sons
304  unsigned n_sons = nsons();
305  // Kill all the sons
306  for(unsigned ison=0;ison<n_sons;ison++)
307  {
308  // Unbuild the element by marking the nodes as obsolete
309  son_pt(ison)->object_pt()->unbuild();
310 
311  // Delete the object. This must be done here, because the
312  // destructor of a tree does not delete the leaf nodes
313  // (the actual elements that are used in the mesh).
314  // In general, the destruction of the leaf nodes is handled by the
315  // mesh destructors.
316  delete son_pt(ison)->object_pt();
317 
318  // Now delete the tree representation
319  delete son_pt(ison);
320  }
321 
322  Son_pt.resize(0);
323 
324  // Have merged my sons -- can't do it again...
325  Object_pt->deselect_sons_for_unrefinement();
326  }
327 }
