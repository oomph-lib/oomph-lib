174  /// Euler timestepping x^{t+1} = x^{t} + dt M^{-1} L(x^{t})
175  //=================================================================
176  void Euler::timestep(ExplicitTimeSteppableObject* const &object_pt,
177  const double &dt)
178  {
179  object_pt->actions_before_explicit_timestep();
180  object_pt->actions_before_explicit_stage();
181 
182  //Vector that will hold the inverse mass matrix multiplied by the
183  //residuals
184  DoubleVector minv_res;
185  //Get M^{-1} R
186  object_pt->get_dvaluesdt(minv_res);
187 
188  //Add the result to the unknowns
189  object_pt->add_to_dofs(dt,minv_res);
190  //Increment the time by the appropriate amount
191  object_pt->time() += dt;
192 
193  //Call any actions required after the change in the unknowns
194  object_pt->actions_after_explicit_stage();
195  object_pt->actions_after_explicit_timestep();
196  }

216 template<>
217 void RungeKutta<4>::timestep(ExplicitTimeSteppableObject* const &object_pt,
218  const double &dt)
219 {
220  object_pt->actions_before_explicit_timestep();
221 
222  // Stage 1
223  // ============================================================
224  object_pt->actions_before_explicit_stage();
225 
226  //Store the initial values and initial time
227  DoubleVector u;
228  object_pt->get_dofs(u);
229 
230  //Now get the first unknowns
231  DoubleVector k1;
232  object_pt->get_dvaluesdt(k1);
233 
234  //Add to the residuals
235  object_pt->add_to_dofs(0.5*dt,k1);
236  //Increment the time
237  object_pt->time() += 0.5*dt;
238  object_pt->actions_after_explicit_stage();
239 
240 
241  // Stage 2
242  // ============================================================
243  object_pt->actions_before_explicit_stage();
244 
245  //Get the next unknowns
246  DoubleVector k2;
247  object_pt->get_dvaluesdt(k2);
248 
249  //Now reset the residuals
250  object_pt->set_dofs(u);
251  object_pt->add_to_dofs(0.5*dt,k2);
252  //Time remains the same
253  object_pt->actions_after_explicit_stage();
254 
255  // Stage 3
256  // ============================================================
257  object_pt->actions_before_explicit_stage();
258 
259  //Get the next unknowns
260  DoubleVector k3;
261  object_pt->get_dvaluesdt(k3);
262 
263  //Now reset the residuals
264  object_pt->set_dofs(u);
265  object_pt->add_to_dofs(dt,k3);
266  //Increment the time (now at initial_time + dt)
267  object_pt->time() += 0.5*dt;
268  object_pt->actions_after_explicit_stage();
269 
270  // Stage 4
271  // ============================================================
272  object_pt->actions_before_explicit_stage();
273 
274  //Get the final unknowns
275  DoubleVector k4;
276  object_pt->get_dvaluesdt(k4);
277 
278  //Set the final values of the unknowns
279  object_pt->set_dofs(u);
280  object_pt->add_to_dofs((dt/6.0),k1);
281  object_pt->add_to_dofs((dt/3.0),k2);
282  object_pt->add_to_dofs((dt/3.0),k3);
283  object_pt->add_to_dofs((dt/6.0),k4);
284  object_pt->actions_after_explicit_stage();
285 
286  object_pt->actions_after_explicit_timestep();
287 }

292  template<>
293  void RungeKutta<2>::timestep(ExplicitTimeSteppableObject* const &object_pt,
294  const double &dt)
295  {
296  object_pt->actions_before_explicit_timestep();
297 
298  // Stage 1
299  // ============================================================
300  object_pt->actions_before_explicit_stage();
301 
302  // Store the initial values
303  DoubleVector u;
304  object_pt->get_dofs(u);
305 
306  // Get f1 (time derivative at t0, y0) and add to dofs
307  DoubleVector f1;
308  object_pt->get_dvaluesdt(f1);
309  object_pt->add_to_dofs(dt, f1);
310 
311  // Advance time to t1 = t0 + dt
312  object_pt->time() += dt;
313 
314  object_pt->actions_after_explicit_stage();
315 
316 
317  // Stage 2
318  // ============================================================
319  object_pt->actions_before_explicit_stage();
320 
321  // get f2 (with t=t1, y = y0 + h f0)
322  DoubleVector f2;
323  object_pt->get_dvaluesdt(f2);
324 
325  // Final answer is starting dofs + h/2 * (f1 + f2)
326  object_pt->set_dofs(u);
327  object_pt->add_to_dofs(0.5*dt, f1);
328  object_pt->add_to_dofs(0.5*dt, f2);
329 
330  object_pt->actions_after_explicit_stage();
331 
332  // Done, do actions
333  object_pt->actions_after_explicit_timestep();
334  }

393 //Explicit specialisation for fourth-order RK scheme
394 template<>
395 void LowStorageRungeKutta<4>::timestep(
396  ExplicitTimeSteppableObject* const &object_pt,
397  const double &dt)
398 {
399  object_pt->actions_before_explicit_timestep();
400 
401  //Store the initial time
402  const double initial_time = object_pt->time();
403  //Temporary storage
404  DoubleVector k;
405 
406  //Temporary storage for the inverse mass matrix multiplied by the residuals
407  DoubleVector minv_res;
408 
409  //Loop over the number of steps in the scheme
410  for(unsigned i=0;i<5;i++)
411  {
412  object_pt->actions_before_explicit_stage();
413 
414  //Get the inverse mass matrix multiplied by the current value
415  //of the residuals
416  object_pt->get_dvaluesdt(minv_res);
417  //Get the values of k
418  const unsigned n_dof= minv_res.nrow();
419 
420  //First time round resize k and initialise to zero
421  if(i==0) {k.build(minv_res.distribution_pt(),0.0);}
422  //Now construct the next value of k
423  for(unsigned n=0;n<n_dof;n++)
424  {
425  k[n] *= A[i];
426  k[n] += dt*minv_res[n];
427  }
428 
429  //Add to the residuals
430  object_pt->add_to_dofs(B[i],k);
431  //Set the new time
432  object_pt->time() = initial_time + C[i]*dt;
433 
434  object_pt->actions_after_explicit_stage();
435  }
436 
437  object_pt->actions_after_explicit_timestep();
438 }

440 // ??ds this could be heavily optimised if needed. Keeping it simple for
441 // now
442 void EBDF3::timestep(ExplicitTimeSteppableObject* const &object_pt,
443  const double &dt)
444 {
445  using namespace StringConversion;
446 
447  object_pt->actions_before_explicit_timestep();
448  object_pt->actions_before_explicit_stage();
449 
450  // Storage indicies for the history values that we need
451  unsigned tn = 1;
452  unsigned tnm1 = tn+1;
453  unsigned tnm2 = tnm1+1;
454 
455  // Check dts are the same, this will need to be removed if ebdf3 is being
456  // used as something other than a predictor... But seeing as it isn't
457  // stable that isn't likely.
458 #ifdef PARANOID
459  if(std::abs(dt - object_pt->time_pt()->dt(0)) > 1e-15)
460  {
461  std::string err = "Inconsistent dts! Predictor is stepping by " + to_string(dt);
462  err += " but dt(0) = " + to_string(object_pt->time_pt()->dt(0));
463  throw OomphLibError(err, OOMPH_EXCEPTION_LOCATION,
464  OOMPH_CURRENT_FUNCTION);
465  }
466 #endif
467 
468  // Get older dt values
469  double dtnm1 = object_pt->time_pt()->dt(1);
470  double dtnm2 = object_pt->time_pt()->dt(2);
471 
472  // Calculate weights for these dts
473  set_weights(dt, dtnm1, dtnm2);
474 
475  // Get derivative value at step n (even though this uses values from t=0 =
476  // step n+1, it's ok because we haven't changed the values in that slot
477  // yet).
478  DoubleVector fn;
479  object_pt->get_dvaluesdt(fn);
480  fn *= Fn_weight;
481 
482  // Extract history values and multiply by their weights
483  DoubleVector ynp1, yn, ynm1, ynm2;
484  object_pt->get_dofs(tn, yn);
485  yn *= Yn_weight;
486  object_pt->get_dofs(tnm1, ynm1);
487  ynm1 *= Ynm1_weight;
488  object_pt->get_dofs(tnm2, ynm2);
489  ynm2 *= Ynm2_weight;
490 
491 
492  // Add everything together
493  ynp1 = yn;
494  ynp1 += ynm1;
495  ynp1 += ynm2;
496  ynp1 += fn;
497 
498  // Done, update things in the object
499  object_pt->set_dofs(ynp1);
500  object_pt->time() += dt;
501 
502  // Actions functions
503  object_pt->actions_after_explicit_stage();
504  object_pt->actions_after_explicit_timestep();
505 }

507 /// Calculate the weights for this set of step sizes.
508 void EBDF3::set_weights(const double &dtn, const double &dtnm1,
509  const double &dtnm2)
510  {
511  using namespace std;
512 
513  // If this is slow we can probably optimise by doing direct
514  // multiplication instead of using pow.
515 
516  // Generated using sympy from my python ode code.
517 
518  double denom = pow(dtnm1,4)*dtnm2 + 2*pow(dtnm1,3)*pow(dtnm2,2)
519  + pow(dtnm1,2)*pow(dtnm2,3);
520 
521  Yn_weight = -(2*pow(dtn,3)*dtnm1*dtnm2 + pow(dtn,3)*pow(dtnm2,2)
522  + 3*pow(dtn,2)*pow(dtnm1,2)*dtnm2
523  + 3*pow(dtn,2)*dtnm1*pow(dtnm2,2) + pow(dtn,2)*pow(dtnm2,3)
524  - pow(dtnm1,4)*dtnm2 - 2*pow(dtnm1,3)*pow(dtnm2,2)
525  - pow(dtnm1,2)*pow(dtnm2,3))/denom;
526 
527  Ynm1_weight = -(-pow(dtn,3)*pow(dtnm1,2) - 2*pow(dtn,3)*dtnm1*dtnm2
528  - pow(dtn,3)*pow(dtnm2,2) - pow(dtn,2)*pow(dtnm1,3)
529  - 3*pow(dtn,2)*pow(dtnm1,2)*dtnm2
530  - 3*pow(dtn,2)*dtnm1*pow(dtnm2,2)
531  - pow(dtn,2)*pow(dtnm2,3))/denom;
532 
533  Ynm2_weight = -(pow(dtn,3)*pow(dtnm1,2) + pow(dtn,2)*pow(dtnm1,3))/denom;
534 
535  Fn_weight = -(-pow(dtn,3)*pow(dtnm1,2)*dtnm2 - pow(dtn,3)*dtnm1*pow(dtnm2,2)
536  - 2*pow(dtn,2)*pow(dtnm1,3)*dtnm2
537  - 3*pow(dtn,2)*pow(dtnm1,2)*pow(dtnm2,2)
538  - pow(dtn,2)*dtnm1*pow(dtnm2,3) - dtn*pow(dtnm1,4)*dtnm2
539  - 2*dtn*pow(dtnm1,3)*pow(dtnm2,2)
540  - dtn*pow(dtnm1,2)*pow(dtnm2,3))/denom;
541  }
