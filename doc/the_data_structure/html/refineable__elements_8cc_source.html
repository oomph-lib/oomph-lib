73 /// Internal function that is used to assemble the jacobian of the mapping
74 /// from local coordinates (s) to the eulerian coordinates (x), given the
75 /// derivatives of the shape functions.
76 //=========================================================================
77  void RefineableElement::
78  assemble_local_to_eulerian_jacobian(const DShape &dpsids,
79  DenseMatrix<double> &jacobian) const
80  {
81  //Locally cache the elemental dimension
82  const unsigned el_dim = dim();
83  //The number of shape functions must be equal to the number
84  //of nodes (by definition)
85  const unsigned n_shape = nnode();
86  //The number of shape function types must be equal to the number
87  //of nodal position types (by definition)
88  const unsigned n_shape_type = nnodal_position_type();
89 
90 #ifdef PARANOID
91  //Check for dimensional compatibility
92  if(dim() != nodal_dimension())
93  {
94  std::ostringstream error_message;
95  error_message << "Dimension mismatch" << std::endl;
96  error_message << "The elemental dimension: " << dim()
97  << " must equal the nodal dimension: "
98  << nodal_dimension()
99  << " for the jacobian of the mapping to be well-defined"
100  << std::endl;
101  throw OomphLibError(
102  error_message.str(),
103  OOMPH_CURRENT_FUNCTION,
104  OOMPH_EXCEPTION_LOCATION);
105  }
106 #endif
107 
108  //Loop over the rows of the jacobian
109  for(unsigned i=0;i<el_dim;i++)
110  {
111  //Loop over the columns of the jacobian
112  for(unsigned j=0;j<el_dim;j++)
113  {
114  //Zero the entry
115  jacobian(i,j) = 0.0;
116  //Loop over the shape functions
117  for(unsigned l=0;l<n_shape;l++)
118  {
119  for(unsigned k=0;k<n_shape_type;k++)
120  {
121  //Jacobian is dx_j/ds_i, which is represented by the sum
122  //over the dpsi/ds_i of the nodal points X j
123  //Call the Hanging version of positions
124  jacobian(i,j) += nodal_position_gen(l,k,j)*dpsids(l,k,i);
125  }
126  }
127  }
128  }
129  }

138  assemble_local_to_eulerian_jacobian2(const DShape &d2psids,

140  {
141  //Find the the dimension of the element
142  const unsigned el_dim = dim();

144  //Must be equal to the number of nodes and their position types

161  //Loop over the shape functions
162  for(unsigned l=0;l<n_shape;l++)
163  {
164  //Loop over the shape function types
165  for(unsigned k=0;k<n_shape_type;k++)
166  {
167  //Add the terms to the jacobian entry
168  //Call the Hanging version of positions
169  jacobian2(i,j) += nodal_position_gen(l,k,j)*d2psids(l,k,i);

179  /// respect to the local coordinate should already have been calculated

212 //==========================================================================
213 /// Calculate the mapping from local to eulerian coordinates
214 /// assuming that the coordinates are aligned in the direction of the local
215 /// coordinates, i.e. there are no cross terms and the jacobian is diagonal.
216 /// The local derivatives are passed as dpsids and the jacobian and
217 /// inverse jacobian are returned.
218 //==========================================================================
219  double RefineableElement::
220  local_to_eulerian_mapping_diagonal(const DShape &dpsids,
221  DenseMatrix<double> &jacobian,
222  DenseMatrix<double> &inverse_jacobian)
223  const
224  {
225  //Find the dimension of the element
226  const unsigned el_dim = dim();
227  //Find the number of shape functions and shape functions types
228  //Equal to the number of nodes and their position types by definition
229  const unsigned n_shape = nnode();
230  const unsigned n_shape_type = nnodal_position_type();
231 
232 #ifdef PARANOID
233  //Check for dimension compatibility
234  if(dim() != nodal_dimension())
235  {
236  std::ostringstream error_message;
237  error_message << "Dimension mismatch" << std::endl;
238  error_message << "The elemental dimension: " << dim()
239  << " must equal the nodal dimension: "
240  << nodal_dimension()
241  << " for the jacobian of the mapping to be well-defined"
242  << std::endl;
243  throw OomphLibError(error_message.str(),
244  OOMPH_CURRENT_FUNCTION,
245  OOMPH_EXCEPTION_LOCATION);
246  }
247 #endif
248 
249  //In this case we assume that there are no cross terms, that is
250  //global coordinate 0 is always in the direction of local coordinate 0
251 
252  //Loop over the coordinates
253  for(unsigned i=0;i<el_dim;i++)
254  {
255  //Zero the jacobian and inverse jacobian entries
256  for(unsigned j=0;j<el_dim;j++)
257  {jacobian(i,j) = 0.0; inverse_jacobian(i,j) = 0.0;}
258 
259  //Loop over the shape functions
260  for(unsigned l=0;l<n_shape;l++)
261  {
262  //Loop over the types of dof
263  for(unsigned k=0;k<n_shape_type;k++)
264  {
265  //Derivatives are always dx_{i}/ds_{i}
266  jacobian(i,i) += nodal_position_gen(l,k,i)*dpsids(l,k,i);
267  }
268  }
269  }
270 
271  //Now calculate the determinant of the matrix
272  double det = 1.0;
273  for(unsigned i=0;i<el_dim;i++) {det *= jacobian(i,i);}
274 
275 //Report if Matrix is singular, or negative
276 #ifdef PARANOID
277  check_jacobian(det);
278 #endif
279 
280  //Calculate the inverse mapping (trivial in this case)
281  for(unsigned i=0;i<el_dim;i++)
282  {inverse_jacobian(i,i) = 1.0/jacobian(i,i);}
283 
284  //Return the value of the Jacobian
285  return(det);
286  }

307 /// Assign the local hang eqn.
308 //=======================================================================
309  void RefineableElement::assign_hanging_local_eqn_numbers(
310  const bool &store_local_dof_pt)
311  {
312  //Find the number of nodes
313  const unsigned n_node = nnode();
314 
315  //Check there are nodes!
316  if(n_node > 0)
317  {
318  //Find the number of continuously interpolated values
319  const unsigned n_cont_values = ncont_interpolated_values();
320 
321  //Delete the storage associated with the local hanging equation schemes
322  delete[] Local_hang_eqn;
323  //Allocate new storage
324  Local_hang_eqn = new std::map<Node*,int>[n_cont_values];
325 
326  //Boolean that is set to true if there are hanging equation numbers
327  bool hanging_eqn_numbers = false;
328 
329  //Maps that store whether the node's local equation number for a
330  //particular value has already been assigned
331  Vector<std::map<Node*,bool> > local_eqn_number_done(n_cont_values);
332 
333  //Get number of dofs assigned thus far
334  unsigned local_eqn_number = ndof();
335 
336  //A local queue to store the global equation numbers
337  std::deque<unsigned long> global_eqn_number_queue;
338 
339  //Now loop over all the nodes again to find the master nodes
340  //external to the element
341  for(unsigned n=0;n<n_node;n++)
342  {
343  //Loop over the number of continuously-interpolated values at the node
344  for(unsigned j=0;j<n_cont_values;j++)
345  {
346  //If the node is hanging node in value j
347  if(node_pt(n)->is_hanging(j))
348  {
349  //Get the pointer to the appropriate hanging info object
350  HangInfo* hang_info_pt = node_pt(n)->hanging_pt(j);
351  //Find the number of master nodes
352  unsigned n_master = hang_info_pt->nmaster();
353  //Loop over the master nodes
354  for(unsigned m=0;m<n_master;m++)
355  {
356  //Get the m-th master node
357  Node* Master_node_pt = hang_info_pt->master_node_pt(m);
358 
359  //If the master node's value has not been considered already,
360  //give it a local equation number
361  if(local_eqn_number_done[j][Master_node_pt] == false)
362  {
363 #ifdef PARANOID
364  //Check that the value is stored at the master node
365  //If not the master node must have be set up incorrectly
366  if(Master_node_pt->nvalue() < j)
367  {
368  std::ostringstream error_stream;
369  error_stream
370  << "Master node for " << j
371  << "-th value only has " << Master_node_pt->nvalue()
372  << " stored values!" << std::endl;
373 
374  throw OomphLibError(
375  error_stream.str(),
376  OOMPH_CURRENT_FUNCTION,
377  OOMPH_EXCEPTION_LOCATION);
378  }
379 #endif
380 
381  //We now need to test whether the master node is actually
382  //a local node, in which case its local equation number
383  //will already have been assigned and will be stored
384  //in nodal_local_eqn
385 
386  //Storage for the index of the local node
387  //Initialised to n_node (beyond the possible indices of
388  //"real" nodes)
389  unsigned local_node_index=n_node;
390  //Loop over the local nodes (again)
391  for(unsigned n1=0;n1<n_node;n1++)
392  {
393  //If the master node is a local node
394  //get its index and break out of the loop
395  if(Master_node_pt==node_pt(n1))
396  {
397  local_node_index=n1;
398  break;
399  }
400  }
401 
402  //Now we test whether the node was found
403  if(local_node_index < n_node)
404  {
405  //Copy the local equation number to the
406  //pointer-based look-up scheme
407  Local_hang_eqn[j][Master_node_pt] =
408  nodal_local_eqn(local_node_index,j);
409  }
410  //Otherwise it's a new master node
411  else
412  {
413  //Get the equation number
414  long eqn_number = Master_node_pt->eqn_number(j);
415  //If equation_number positive add to array
416  if(eqn_number >= 0)
417  {
418  //Add global equation number to the queue
419  global_eqn_number_queue.push_back(eqn_number);
420  //Add pointer to the dof to the queue if required
421  if(store_local_dof_pt)
422  {
423  GeneralisedElement::Dof_pt_deque.push_back(
424  Master_node_pt->value_pt(j));
425  }
426  //Add to pointer based scheme
427  Local_hang_eqn[j][Master_node_pt] = local_eqn_number;
428  //Increase number of local variables
429  local_eqn_number++;
430  }
431  //Otherwise the value is pinned
432  else
433  {
434  Local_hang_eqn[j][Master_node_pt] = Data::Is_pinned;
435  }
436  //There are now hanging equation numbers
437  }
438  //The value at this node has now been done
439  local_eqn_number_done[j][Master_node_pt] = true;
440  //There are hanging equation numbers
441  hanging_eqn_numbers = true;
442  }
443  }
444  }
445  }
446  } //End of second loop over nodes
447 
448  //Now add our global equations numbers to the internal element storage
449  add_global_eqn_numbers(global_eqn_number_queue,
450  GeneralisedElement::Dof_pt_deque);
451  //Clear the memory used in the deque
452  if(store_local_dof_pt)
453  {std::deque<double*>().swap(GeneralisedElement::Dof_pt_deque);}
454 
455 
456  //If there are no hanging_eqn_numbers delete the (empty) stored maps
457  if(!hanging_eqn_numbers)
458  {
459  delete[] Local_hang_eqn;
460  Local_hang_eqn=0;
461  }
462 
463 
464  // Setup map that associates a unique number with any of the nodes
465  // that actively control the shape of the element (i.e. they are
466  // either non-hanging nodes of this element or master nodes
467  // of hanging nodes.
468  unsigned count=0;
469  std::set<Node*> all_nodes;
470  Shape_controlling_node_lookup.clear();
471  for (unsigned j=0;j<n_node;j++)
472  {
473  // Get node
474  Node* nod_pt=node_pt(j);
475 
476  //If the node is hanging, consider master nodes
477  if(nod_pt->is_hanging())
478  {
479  HangInfo* hang_info_pt = node_pt(j)->hanging_pt();
480  unsigned n_master = hang_info_pt->nmaster();
481 
482  //Loop over the master nodes
483  for(unsigned m=0;m<n_master;m++)
484  {
485  Node* master_node_pt=hang_info_pt->master_node_pt(m);
486  // Do we have this one already?
487  unsigned old_size=all_nodes.size();
488  all_nodes.insert(master_node_pt);
489  if (all_nodes.size()>old_size)
490  {
491  Shape_controlling_node_lookup[master_node_pt]=count;
492  count++;
493  }
494  }
495  }
496  // Not hanging: Consider the node itself
497  else
498  {
499  // Do we have this one already?
500  unsigned old_size=all_nodes.size();
501  all_nodes.insert(nod_pt);
502  if (all_nodes.size()>old_size)
503  {
504  Shape_controlling_node_lookup[nod_pt]=count;
505  count++;
506  }
507  }
508  }
509 
510  } //End of if nodes
511  }

535  }
536  }

541  {

544  //Find the number of values stored at the node
545  const unsigned n_value = nod_pt->nvalue();
546

548  for(unsigned i=0;i<n_value;i++)
549  {
550  //If the node is hanging in the variable

568  }
569  }

571  else
572  {
573  //Add the index of each data value and the pointer to the set

577  }

583 //==========================================================================
584 /// Compute derivatives of elemental residual vector with respect
585 /// to nodal coordinates. Default implementation by FD can be overwritten
586 /// for specific elements.
587 /// dresidual_dnodal_coordinates(l,i,j) = d res(l) / dX_{ij}
588 /// This version is overloaded from the version in FiniteElement
589 /// and takes hanging nodes into account -- j in the above loop
590 /// loops over all the nodes that actively control the
591 /// shape of the element (i.e. they are non-hanging or master nodes of
592 /// hanging nodes in this element).
593 //==========================================================================
594  void RefineableElement::get_dresidual_dnodal_coordinates(
595  RankThreeTensor<double>& dresidual_dnodal_coordinates)
596  {
597  // Number of nodes
598  unsigned n_nod=nnode();
599 
600  // If the element has no nodes (why??!!) return straightaway
601  if (n_nod==0) return;
602 
603  // Get dimension from first node
604  unsigned dim_nod=node_pt(0)->ndim();
605 
606  // Number of dofs
607  unsigned n_dof=ndof();
608 
609  // Get reference residual
610  Vector<double> res(n_dof);
611  Vector<double> res_pls(n_dof);
612  get_residuals(res);
613 
614  // FD step
615  double eps_fd=GeneralisedElement::Default_fd_jacobian_step;
616 
617  // Do FD loop over all active nodes
618  for (std::map<Node*,unsigned>::iterator it=
619  Shape_controlling_node_lookup.begin();
620  it!=Shape_controlling_node_lookup.end();
621  it++)
622  {
623  // Get node
624  Node* nod_pt=it->first;
625 
626  // Get its number
627  unsigned node_number=it->second;
628 
629  // Loop over coordinate directions
630  for (unsigned i=0;i<dim_nod;i++)
631  {
632  // Make backup
633  double backup=nod_pt->x(i);
634 
635  // Do FD step. No node update required as we're
636  // attacking the coordinate directly...
637  nod_pt->x(i)+=eps_fd;
638 
639  // Perform auxiliary node update function
640  nod_pt->perform_auxiliary_node_update_fct();
641 
642  // Get advanced residual
643  get_residuals(res_pls);
644 
645  // Fill in FD entries [Loop order is "wrong" here as l is the
646  // slow index but this is in a function that's costly anyway
647  // and gives us the fastest loop outside where these tensor
648  // is actually used.]
649  for (unsigned l=0;l<n_dof;l++)
650  {
651  dresidual_dnodal_coordinates(l,i,node_number)=
652  (res_pls[l]-res[l])/eps_fd;

928 /// assuming that the coordinates are aligned in the direction of the local
929 /// coordinates, i.e. there are no cross terms and the jacobian is diagonal.
930 /// The local derivatives are passed as dpsids and the jacobian and
931 /// inverse jacobian are returned.

936  DenseMatrix<double> &inverse_jacobian)
937  const

942  const unsigned n_shape = nnode();
943  const unsigned n_shape_type = nnodal_lagrangian_type();

946  //global coordinate 0 is always in the direction of local coordinate 0
947 
948  //Loop over the coordinates
949  for(unsigned i=0;i<el_dim;i++)
950  {
951  //Zero the jacobian and inverse jacobian entries
952  for(unsigned j=0;j<el_dim;j++)

958  //Loop over the types of dof
959  for(unsigned k=0;k<n_shape_type;k++)

962  jacobian(i,i) += lagrangian_position_gen(l,k,i)*dpsids(l,k,i);

964  }
965  }

981  return(det);
