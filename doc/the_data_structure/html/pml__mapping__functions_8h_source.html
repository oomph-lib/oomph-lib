41 /// Class to hold the mapping function (gamma) for the Pml which defines
42 /// how the coordinates are transformed in the Pml. This class holds
43 /// the one dimensional or uniaxial case which is the most common
44 //=======================================================================
45 class PMLMapping
46 {
47 
48 public:
49 
50  /// Default constructor (empty)

53  /// \short Pure virtual to return Pml mapping gamma, where gamma is the
54  /// \f$d\tilde x / d x\f$ as function of \f$\nu\f$ where \f$\nu = x - h\f$ where h is
55  /// the vector from the origin to the start of the Pml

64 /// A mapping function propsed by Bermudez et al, appears to be the best
65 /// for the Helmholtz equations and so this will be the default mapping
66 /// (see definition of PmlHelmholtzEquations)
67 //=======================================================================
68 class BermudezPMLMapping : public PMLMapping
69 {
70 
71  public:
72 
73  /// Default constructor (empty)
74  BermudezPMLMapping(){};
75 
76  /// \short Overwrite the pure Pml mapping coefficient function to return the
77  /// coeffcients proposed by Bermudez et al
78  std::complex<double> gamma(const double& nu_i,
79  const double& pml_width_i,
80  const double& wavenumber_squared,
81  const double& alpha_shift=0.0)
82  {
83  /// return \f$\gamma=1 + (1/k)(i/|outer_boundary - x|)\f$
84  return 1.0 + MathematicalConstants::I / sqrt(wavenumber_squared)
85  *(1.0/std::fabs(pml_width_i - nu_i));
86  }
87 
88 };

104  /// \short Overwrite the pure Pml mapping coefficient function to return the
105  /// coeffcients proposed by Bermudez et al
106  std::complex<double> gamma(const double& nu_i,
107  const double& pml_width_i,
108  const double& wavenumber_squared,
109  const double& alpha_shift=0.0)
110  {
111  /// return \f$\gamma=1 + (i/k)(1/|outer_boundary - x|-1/|pml width|)\f$
112  return 1.0 + MathematicalConstants::I / sqrt(wavenumber_squared)
113  *( 1.0/std::fabs(pml_width_i - nu_i) - 1.0/std::fabs(pml_width_i));
114  }

116 };
