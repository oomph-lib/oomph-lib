84  /// Get residuals
85  virtual void fill_in_contribution_to_residuals(Vector<double>& residuals)
86  {
87  // Get pointer to one-and-only internal data object
88  Data* dat_pt = internal_data_pt(0);
89 
90  // Get it's values
91  Vector<double> u(nvalue(), 0.0);
92  dat_pt->value(u);
93 
94  // Get time stepper
95  TimeStepper* time_stepper_pt = dat_pt->time_stepper_pt();
96 
97  // Get continuous time
98  double t = time_stepper_pt->time();
99 
100  Vector<double> deriv = derivative_function(t, u);
101  for(unsigned j=0, nj=deriv.size(); j<nj; j++)
102  {
103  // Get dudt approximation from time stepper
104  double dudt = time_stepper_pt->time_derivative(1, dat_pt, j);
105 
106  // Residual is difference between the exact derivative and our
107  // time stepper's derivative estimate.
108  residuals[j] = deriv[j] - dudt;
109  }
110  }
111 
112  virtual void fill_in_contribution_to_jacobian(Vector<double>& residuals,
113  DenseMatrix<double>& jacobian)
114  {
115  // Get residuals
116  fill_in_contribution_to_residuals(residuals);
117 
118  if(Exact_solution_pt->have_jacobian() && !Use_fd_jacobian)
119  {
120  // get df/du jacobian
121  double t = internal_data_pt(0)->time_stepper_pt()->time();
122  Vector<double> dummy, u(nvalue(), 0.0);
123  internal_data_pt(0)->value(u);
124  Exact_solution_pt->jacobian(t, dummy, u, jacobian);
125 
126  // We need jacobian of residual = f - dudt so subtract diagonal
127  // (dudt)/du term.
128  const double a = internal_data_pt(0)->time_stepper_pt()->weight(1,0);
129  const unsigned n = nvalue();
130  for(unsigned i=0; i<n; i++)
131  {
132  jacobian(i, i) -= a;
133  }
134  }
135  else
136  {
137  // Use FD for jacobian
138  GeneralisedElement::fill_in_jacobian_from_internal_by_fd
139  (residuals, jacobian, true);
140  }
141  }
