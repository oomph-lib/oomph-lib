164 //======================================================================
165 /// Define the shape functions and test functions and derivatives
166 /// w.r.t. global coordinates and return Jacobian of mapping.
167 ///
168 /// Galerkin: Test functions = shape functions
169 //======================================================================
170 template<unsigned DIM, unsigned NNODE_1D>
171 double QSpectralPoissonElement<DIM,NNODE_1D>::
172  dshape_and_dtest_eulerian_poisson(
173  const Vector<double> &s,
174  Shape &psi,
175  DShape &dpsidx,
176  Shape &test,
177  DShape &dtestdx) const
178 {
179  //Call the geometrical shape functions and derivatives
180  double J = this->dshape_eulerian(s,psi,dpsidx);
181 
182  //Loop over the test functions and derivatives and set them equal to the
183  //shape functions
184  unsigned nnod=this->nnode();
185  for(unsigned i=0;i<nnod;i++)
186  {
187  test[i] = psi[i];
188  for(unsigned j=0;j<DIM;j++)
189  {
190  dtestdx(i,j) = dpsidx(i,j);
191  }
192  }
193 
194  //Return the jacobian
195  return J;
196 }

232 template<unsigned DIM, unsigned NNODE_1D>
