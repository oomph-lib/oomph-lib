113 /// This class provides a GeomObject representation of a given
114 /// finite element mesh. The Lagrangian coordinate is taken to be the
115 /// dimension of the (first) element in the mesh and the Eulerian
116 /// coordinate is taken to be the dimension of the (first) node in
117 /// the mesh. If there are no elements or nodes the appropriate dimensions
118 /// will be set to zero.
119 /// The constituent elements of the mesh must have their own
120 /// GeomObject representations, so they must be FiniteElements,
121 /// and they become sub-objects
122 /// in this compound GeomObject.
123 //========================================================================
124 class MeshAsGeomObject : public GeomObject
125 {
126 
127  private:
128 
129 
130  /// Helper function to actually build the thing
131  void build_it(SamplePointContainerParameters*
132  sample_point_container_parameters_pt)
133  {
134  Mesh_pt=sample_point_container_parameters_pt->mesh_pt();
135  if (dynamic_cast<RefineableBinArrayParameters*>(
136  sample_point_container_parameters_pt)!=0)
137  {
138  Sample_point_container_version=UseRefineableBinArray;
139  }
140  else if (dynamic_cast<NonRefineableBinArrayParameters*>(
141  sample_point_container_parameters_pt)!=0)
142  {
143  Sample_point_container_version=UseNonRefineableBinArray;
144  }
145 #ifdef OOMPH_HAS_CGAL
146  else if (dynamic_cast<CGALSamplePointContainerParameters*>(
147  sample_point_container_parameters_pt)!=0)
148  {
149  Sample_point_container_version=UseCGALSamplePointContainer;
150  }
151 #endif
152  else
153  {
154  throw OomphLibError("Wrong sample_point_container_parameters_pt",
155  OOMPH_CURRENT_FUNCTION,
156  OOMPH_EXCEPTION_LOCATION);
157  }
158 
159 #ifdef OOMPH_HAS_MPI
160 
161  // Set communicator
162  Communicator_pt=Mesh_pt->communicator_pt();
163 
164 #endif
165 
166 
167  //Storage for the Lagrangian and Eulerian dimension
168  int dim[2]={0,0};
169 
170  //Set the Lagrangian dimension from the dimension of the first element
171  //if it exists (if not the Lagrangian dimension will be zero)
172  if(Mesh_pt->nelement()!=0)
173  {
174  dim[0] = Mesh_pt->finite_element_pt(0)->dim();
175  }
176 
177  //Read out the Eulerian dimension from the first node, if it exists.
178  //(if not the Eulerian dimension will be zero);
179  if(Mesh_pt->nnode()!=0)
180  {
181  dim[1] = Mesh_pt->node_pt(0)->ndim();
182  }
183 
184  // Need to do an Allreduce to ensure that the dimension is consistent
185  // even when no elements are assigned to a certain processor
186 #ifdef OOMPH_HAS_MPI
187 
188 //Only a problem if the mesh has been distributed
189  if(Mesh_pt->is_mesh_distributed())
190  {
191  //Need a non-null communicator
192  if(Communicator_pt!=0)
193  {
194  int n_proc=Communicator_pt->nproc();
195  if (n_proc > 1)
196  {
197  int dim_reduce[2];
198  MPI_Allreduce(&dim,&dim_reduce,2,MPI_INT,
199  MPI_MAX,Communicator_pt->mpi_comm());
200 
201  dim[0] = dim_reduce[0];
202  dim[1] = dim_reduce[1];
203  }
204  }
205  }
206 #endif
207 
208  //Set the Lagrangian and Eulerian dimensions within this geometric object
209  this->set_nlagrangian_and_ndim(static_cast<unsigned>(dim[0]),
210  static_cast<unsigned>(dim[1]));
211 
212  // Create temporary storage for geometric Data (don't count
213  // Data twice!
214  std::set<Data*> tmp_geom_data;
215 
216  //Copy all the elements in the mesh into local storage
217  //N.B. elements must be able to have a geometric object representation.
218  unsigned n_sub_object = Mesh_pt->nelement();
219  Sub_geom_object_pt.resize(n_sub_object);
220  for(unsigned e=0;e<n_sub_object;e++)
221  {
222  // (Try to) cast to a finite element:
223  Sub_geom_object_pt[e]=
224  dynamic_cast<FiniteElement*>(Mesh_pt->element_pt(e));
225 
226 #ifdef PARANOID
227  if (Sub_geom_object_pt[e]==0)
228  {
229  std::ostringstream error_message;
230  error_message
231  << "Unable to dynamic cast element: " << std::endl
232  << "into a FiniteElement: GeomObject representation is not possible\n";
233  throw OomphLibError(
234  error_message.str(),
235  OOMPH_CURRENT_FUNCTION,
236  OOMPH_EXCEPTION_LOCATION);
237  }
238 #endif
239 
240  // Add the geometric Data of each element into set
241  unsigned ngeom_data=Sub_geom_object_pt[e]->ngeom_data();
242  for (unsigned i=0;i<ngeom_data;i++)
243  {
244  tmp_geom_data.insert(Sub_geom_object_pt[e]->geom_data_pt(i));
245  }
246  }
247 
248  // Now copy unique geom Data values across into vector
249  unsigned ngeom=tmp_geom_data.size();
250  Geom_data_pt.resize(ngeom);
251  typedef std::set<Data*>::iterator IT;
252  unsigned count=0;
253  for (IT it=tmp_geom_data.begin();it!=tmp_geom_data.end();it++)
254  {
255  Geom_data_pt[count]=*it;
256  count++;
257  }
258 
259  // Build the right type of bin array
260  switch (Sample_point_container_version)
261  {
262  case UseRefineableBinArray:
263 
264  Sample_point_container_pt=new RefineableBinArray(sample_point_container_parameters_pt);
265  break;
266 
267  case UseNonRefineableBinArray:
268 
269  Sample_point_container_pt=new NonRefineableBinArray(sample_point_container_parameters_pt);
270  break;
271 
272 #ifdef OOMPH_HAS_CGAL
273 
274  case UseCGALSamplePointContainer:
275 
276  Sample_point_container_pt=new CGALSamplePointContainer(sample_point_container_parameters_pt);
277  break;
278 
279 #endif
280 
281  default:
282 
283  oomph_info << "Sample_point_container_version = "
284  << Sample_point_container_version << std::endl;
285  throw OomphLibError("Sample_point_container_version",
286  OOMPH_CURRENT_FUNCTION,
287  OOMPH_EXCEPTION_LOCATION);
288  }
289  }

394  /// \short Find the sub geometric object and local coordinate therein that
395  /// corresponds to the intrinsic coordinate zeta. If sub_geom_object_pt=0
396  /// on return from this function, none of the constituent sub-objects
397  /// contain the required coordinate. Following from the general
398  /// interface to this function in GeomObjects,
399  /// setting the optional bool argument to true means that each
400  /// time the sub-object's locate_zeta function is called, the coordinate
401  /// argument "s" is used as the initial guess. However, this doesn't
402  /// make sense here and the argument is ignored (though a warning
403  /// is issued when the code is compiled in PARANOID setting)
404  void locate_zeta(const Vector<double>& zeta,
405  GeomObject*& sub_geom_object_pt,
406  Vector<double>& s,
407  const bool& use_coordinate_as_initial_guess = false)
408  {
409 #ifdef PARANOID
410  if (use_coordinate_as_initial_guess)
411  {
412  OomphLibWarning(
413  "Ignoring the use_coordinate_as_initial_guess argument.",
414  "MeshAsGeomObject::locate_zeta()",
415  OOMPH_EXCEPTION_LOCATION);
416  }
417 #endif
418 
419 
420  // Do locate in sample point container
421  Sample_point_container_pt->locate_zeta(zeta,sub_geom_object_pt,s);
422 
423  }
424 
425  /// \short Return the position as a function of the intrinsic coordinate zeta.
426  /// This provides an (expensive!) default implementation in which
427  /// we loop over all the constituent sub-objects and check if they
428  /// contain zeta and then evaluate their position() function.
429  void position(const Vector<double>& zeta, Vector<double>& r) const
430  {
431  // Call position function at current timestep:
432  unsigned t = 0;
433  position(t, zeta, r);
434  }
435 
436  /// \short Parametrised position on object: r(zeta). Evaluated at
437  /// previous timestep. t=0: current time; t>0: previous
438  /// timestep. This provides an (expensive!) default implementation in which
439  /// we loop over all the constituent sub-objects and check if they
440  /// contain zeta and then evaluate their position() function.
441  void position(const unsigned& t, const Vector<double>& zeta,
442  Vector<double>& r) const
443  {
444  // Storage for the GeomObject that contains the zeta coordinate
445  // and the intrinsic coordinate within it.
446  GeomObject* sub_geom_object_pt;
447  const unsigned n_lagrangian = this->nlagrangian();
448  Vector<double> s(n_lagrangian);
449 
450  // Find the sub object containing zeta, and the local intrinsic coordinate
451  // within it
452  const_cast<MeshAsGeomObject*>(this)->locate_zeta(zeta, sub_geom_object_pt, s);
453  if (sub_geom_object_pt == 0)
454  {
455  std::ostringstream error_message;
456  error_message << "Cannot locate zeta ";
457  for (unsigned i = 0; i < n_lagrangian; i++)
458  {
459  error_message << zeta[i] << " ";
460  }
461  error_message << std::endl;
462  Mesh_pt->output("most_recent_mesh.dat");
463  throw OomphLibError(error_message.str(),
464  OOMPH_CURRENT_FUNCTION,
465  OOMPH_EXCEPTION_LOCATION);
466  }
467  // Call that sub-object's position function
468  sub_geom_object_pt->position(t, s, r);
469 
470  }// end of position
