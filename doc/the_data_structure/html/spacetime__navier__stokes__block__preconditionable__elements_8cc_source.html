38  void BlockPrecQTaylorHoodSpaceTimeElement::
39  get_dof_numbers_for_unknowns(std::list<std::pair<unsigned long,unsigned> >&
40  dof_lookup_list) const
41  {
42  // Number of nodes
43  unsigned n_node=this->nnode();
44 
45  // Temporary pair (used to store dof lookup prior to being added to list)
46  std::pair<unsigned,unsigned> dof_lookup;
47 
48  // The number of nodes in the each direction
49  unsigned n_node_1d=this->nnode_1d();
50 
51  // Make sure that we're using quadratic interpolation otherwise this might
52  // go wrong...
53  if (n_node_1d!=3)
54  {
55  // Create an output stream
56  std::ostringstream error_message_stream;
57 
58  // Create an error message
59  error_message_stream << "Can only deal with Navier-Stokes elements which "
60  << "use quadratic interpolation at the moment. Using "
61  << n_node_1d << " nodes in each direction" << std::endl;
62 
63  // Throw an error
64  throw OomphLibError(error_message_stream.str(),
65  OOMPH_CURRENT_FUNCTION,
66  OOMPH_EXCEPTION_LOCATION);
67  }
68 
69  // Loop over the nodes
70  for (unsigned j=0;j<n_node;j++)
71  {
72  // Only bother if the node isn't a copy...
73  if (!(this->node_pt(j)->is_a_copy()))
74  {
75  // Storage for the local time slice ID (0<i_local<n_node_1d)
76  unsigned i_local=0;
77 
78  // If we're on the first time slice
79  if (j<n_node_1d*n_node_1d)
80  {
81  // In the first local time slice
82  i_local=0;
83  }
84  // If we're in the second time slice (already excluded the first time slice)
85  else if (j<2*n_node_1d*n_node_1d)
86  {
87  // In the second local time slice
88  i_local=1;
89  }
90  // If we're in the final time slice (already excluded the first/second slice)
91  else if (j<3*n_node_1d*n_node_1d)
92  {
93  // In the final local time slice
94  i_local=2;
95  }
96  // We should never get here
97  else
98  {
99  // Create an output stream
100  std::ostringstream error_message_stream;
101 
102  // Create an error message
103  error_message_stream << "Looping over too many nodes!" << std::endl;
104 
105  // Throw an error
106  throw OomphLibError(error_message_stream.str(),
107  OOMPH_CURRENT_FUNCTION,
108  OOMPH_EXCEPTION_LOCATION);
109  }
110 
111  // The number of dofs at this node
112  unsigned n_dof=this->node_pt(j)->nvalue();
113 
114  // Loop over components
115  for (unsigned i=0;i<n_dof;i++)
116  {
117  // Determine local eqn number
118  int local_eqn_number=this->nodal_local_eqn(j,i);
119 
120  // Ignore pinned values - far away degrees of freedom resulting
121  // from hanging nodes can be ignored since these are be dealt
122  // with by the element containing their master nodes
123  if (local_eqn_number>=0)
124  {
125  // The id shift (depending on which elemental time slice we're in)
126  unsigned elemental_id_shift=5*Time_slab_id;
127 
128  // Store dof lookup in temporary pair: Global equation number
129  // is the first entry in pair
130  dof_lookup.first=this->eqn_number(local_eqn_number);
131 
132  // If we're on the first time slice
133  if (i_local==0)
134  {
135  // Here i can range only between 0 and 2 (but it's also an unsigned so
136  // we only need to check if it's strictly greater than 2)
137  if (i>2)
138  {
139  // Create an output stream
140  std::ostringstream error_message_stream;
141 
142  // Create an error message
143  error_message_stream << "Don't know what to do when i=" << i
144  << ". Can only handle when i is between 0 and 2!"
145  << std::endl;
146 
147  // Throw an error
148  throw OomphLibError(error_message_stream.str(),
149  OOMPH_CURRENT_FUNCTION,
150  OOMPH_EXCEPTION_LOCATION);
151  }
152 
153  // Set dof numbers: Dof number is the second entry in pair
154  dof_lookup.second=elemental_id_shift+i;
155  }
156  // If we're on the second time slice
157  else if (i_local==1)
158  {
159  // Here i can range only between 0 and 1
160  if (i>1)
161  {
162  // Create an output stream
163  std::ostringstream error_message_stream;
164 
165  // Create an error message
166  error_message_stream << "Don't know what to do when i=" << i
167  << ". Can only handle when i is either 0 or 1!"
168  << std::endl;
169 
170  // Throw an error
171  throw OomphLibError(error_message_stream.str(),
172  OOMPH_CURRENT_FUNCTION,
173  OOMPH_EXCEPTION_LOCATION);
174  }
175 
176  // The local dof shift given which local time slice we're on (i.e.
177  // which node we're on in the time-direction). We've already covered
178  // 2 velocities and 1 pressure so the dof shift is 3.
179  unsigned local_id_shift=3;
180 
181  // Set dof numbers: Dof number is the second entry in pair
182  dof_lookup.second=elemental_id_shift+local_id_shift+i;
183  }
184  // If we're on the final time slice
185  else if (i_local==2)
186  {
187  // Here i can range only between 0 and 2
188  if (i>2)
189  {
190  // Create an output stream
191  std::ostringstream error_message_stream;
192 
193  // Create an error message
194  error_message_stream << "Don't know what to do when i=" << i
195  << ". Can only handle when i is between 0 and 2!"
196  << std::endl;
197 
198  // Throw an error
199  throw OomphLibError(error_message_stream.str(),
200  OOMPH_CURRENT_FUNCTION,
201  OOMPH_EXCEPTION_LOCATION);
202  }
203 
204  // The local dof shift given which local time slice we're on (i.e.
205  // which node we're on in the time-direction). We've already covered
206  // 4 velocities and 1 pressure so the dof shift is 5.
207  unsigned local_id_shift=5;
208 
209  // Set dof numbers: Dof number is the second entry in pair
210  dof_lookup.second=elemental_id_shift+local_id_shift+i;
211  } // if (i_local==0)
212 
213  // Add to list
214  dof_lookup_list.push_front(dof_lookup);
215  }
216  } // for (unsigned i=0;i<DIM;i++)
217  } // if (!(this->node_pt(j)->is_a_copy()))
218  } // for (unsigned j=0;j<n_node;j++)
219  } // End of get_dof_numbers_for_unknowns
220 } // End of namespace oomph
