385 /// Return pointer to greater or equal-sized edge neighbour
386 /// in specified \c direction; also provide info regarding the relative
387 /// size and orientation of neighbour:
388 /// - The vector translate_s turns the index of the local coordinate
389 /// in the present quadtree into that of the neighbour. If there are no
390 /// rotations then translate_s[i] = i, but if, for example, the neighbour's
391 /// eastern face is adjacent to our northern face translate_s[0] = 1
392 /// and translate_s[1] = 0. Of course, this could be deduced after the
393 /// fact, but it's easier to do it here.
394 /// - Let's have a look at the current quadtree's edge in the specified
395 /// direction. The edge will be parallel to one of the two
396 /// local coordinates in the element. This edge has two
397 /// vertices. The vertex at the minimum value of the local
398 /// coordinate (the "lo" vertex) is located
399 /// at the local coordinates (\c s_lo[0], \c s_lo[1]) in the neighbouring
400 /// quadtree.
401 /// - ditto with s_hi: The vertex at the maximum value of the local
402 /// coordinate located at the local coordinates (\c s_hi[0],
403 /// \c s_hi[1]) in the neighbouring quadtree.
404 /// - We're looking for a neighbour in the specified \c direction. When
405 /// viewed from the neighbouring quadtree, the edge that separates
406 /// the present quadtree from its neighbour is the neighbour's \c edge
407 /// edge. If there's no rotation between the two quadtrees, this is a
408 /// simple reflection: For instance, if we're looking
409 /// for a neighhbour in the \c N [orthern] \c direction, \c edge will
410 /// be \c S [outh]
411 /// - \c diff_level <= 0 indicates the difference in refinement levels between
412 /// the two neighbours. If \c diff_level==0, the neighbour has the
413 /// same size as the current quadtree.
414 /// - \c in_neighbouring_tree returns true is we have had to flip
415 /// to a different root, even if that root is actually the same
416 /// as it can be in periodic problems.
417 //=================================================================
418 QuadTree* QuadTree::
419 gteq_edge_neighbour(const int& direction, Vector<unsigned>& translate_s,
420  Vector<double>& s_lo, Vector<double>& s_hi,
421  int& edge, int& diff_level,
422  bool &in_neighbouring_tree) const
423 {
424  using namespace QuadTreeNames;
425 
426 #ifdef PARANOID
427  if ((direction!=S)&&(direction!=E)&&(direction!=N)&&(direction!=W))
428  {
429  std::ostringstream error_stream;
430  error_stream << "Direction " << direction
431  << " is not N, S, E, W" << std::endl;
432 
433  throw OomphLibError(error_stream.str(),
434  OOMPH_CURRENT_FUNCTION,
435  OOMPH_EXCEPTION_LOCATION);
436  }
437 #endif
438 
439  // Initialise in_neighbouring tree to false. It will be set to true
440  //during the recursion if we do actually hop over in to the neighbour
441  in_neighbouring_tree=false;
442 
443  // Maximum level to which we're allowed to descend (we only want
444  // greater-or-equal-sized neighbours)
445  int max_level=Level;
446 
447  // Current element has the following root:
448  QuadTreeRoot* orig_root_pt=dynamic_cast<QuadTreeRoot*>(Root_pt);
449 
450  // Initialise offset in local coordinate
451  double s_diff=0;
452 
453  // Initialise difference in level
454  diff_level=0;
455 
456  // Find neighbour
457  QuadTree* return_pt=gteq_edge_neighbour(direction,
458  s_diff,diff_level,
459  in_neighbouring_tree,
460  max_level,
461  orig_root_pt);
462 
463  QuadTree* neighb_pt=return_pt;
464 
465  // If neighbour exists: What's the direction of the interfacial
466  // edge when viewed from within the neighbour element?
467  if (neighb_pt!=0)
468  {
469  // Rotate things around (the orientation of N in the neighbour might be
470  // be different from that in the present element)
471  // Initialise the direction
472  int new_dir=direction;
473  // If the neighbour has a different root, then there could be a possible
474  // rotation, find it
475  if(neighb_pt->root_pt()!=Root_pt)
476  {new_dir=Rotate(orig_root_pt->north_equivalent(direction),direction);}
477 
478  s_lo[0] = S_base(0,Reflect_edge[new_dir]) +
479  S_step(0,Reflect_edge[new_dir])*s_diff;
480  s_lo[1] = S_base(1,Reflect_edge[new_dir]) +
481  S_step(1,Reflect_edge[new_dir])*s_diff;
482 
483  s_hi[0] = S_base(0,Reflect_edge[new_dir]) +
484  S_step(0,Reflect_edge[new_dir])*pow(2.0,diff_level) +
485  S_step(0,Reflect_edge[new_dir])*s_diff;
486  s_hi[1] = S_base(1,Reflect_edge[new_dir]) +
487  S_step(1,Reflect_edge[new_dir])*pow(2.0,diff_level) +
488  S_step(1,Reflect_edge[new_dir])*s_diff;
489 
490  Vector<double> s_lo_new(2), s_hi_new(2);
491 
492  // What's the direction of the interfacial edge when viewed from within
493  // the neighbour element?
494  edge=Reflect_edge[new_dir];
495 
496  //Set up the translation scheme for the local coordinates
497  {
498  bool swap=false;
499  //Do we need to switch the coordinate
500  switch(direction)
501  {
502  //If the direction is north or south,
503  //but the neighbour's coordinate s[1] is not constant, we must swap
504  case N:
505  case S:
506  if(s_lo[1] != s_hi[1]) {swap = true;}
507  break;
508  //If the direction is east or west
509  //but the neighbour's corodinate s[0] is not constant, we must swap
510  case E:
511  case W:
512  if(s_lo[0] != s_hi[0]) {swap = true;}
513  break;
514  //Catch all totally un-necessary
515  default:
516  std::ostringstream error_stream;
517  error_stream << "Direction " << direction
518  << " is not N, S, E, W" << std::endl;
519 
520  throw OomphLibError(error_stream.str(),
521  OOMPH_CURRENT_FUNCTION,
522  OOMPH_EXCEPTION_LOCATION);
523  }
524 
525  //If we must swap, then do so
526  if(swap) {translate_s[0] = 1; translate_s[1] = 0;}
527  //Othewise, it's just a straight translation
528  else {translate_s[0] = 0; translate_s[1] = 1;}
529  }
530 
531  // Reverse directions?
532  s_lo_new[0]=s_lo[0];
533  s_lo_new[1]=s_lo[1];
534  s_hi_new[0]=s_hi[0];
535  s_hi_new[1]=s_hi[1];
536 
537  if ( ( (edge==N)||(edge==S) ) &&
538  ( (Rotate_angle(direction,new_dir)==90 ) ||
539  (Rotate_angle(direction,new_dir)==180) ) )
540  {
541  s_lo_new[0]=-s_lo[0];
542  s_hi_new[0]=-s_hi[0];
543  }
544  if ( ( (edge==E)||(edge==W) ) &&
545  ( (Rotate_angle(direction,new_dir)==270 ) ||
546  (Rotate_angle(direction,new_dir)==180) ) )
547  {
548  s_lo_new[1]=-s_lo[1];
549  s_hi_new[1]=-s_hi[1];
550  }
551 
552  s_lo[0]=s_lo_new[0];
553  s_lo[1]=s_lo_new[1];
554  s_hi[0]=s_hi_new[0];
555  s_hi[1]=s_hi_new[1];
556 
557  }
558  return return_pt;
559 }

612  QuadTree* return_el_pt;
613 
614  // STEP 1: Find the neighbour's father
615  //--------
616 
617  // Does the element have a father?
618  if (Father_pt!=0)
619  {
620  // If the present quadrant (whose location inside its
621  // father element is specified by Son_type) is adjacent to the
622  // father's edge in the required direction, then its neighbour has
623  // a different father ---> we need to climb up the tree to
624  // the father and find his neighbour in the required direction
625 
626  //Note that this is the cunning recursive part. The returning may not stop
627  //until we hit the very top of the tree, when the element does NOT have
628  //a father
629  if (Is_adjacent(direction,Son_type))
630  {
631  next_el_pt=dynamic_cast<QuadTree*>(Father_pt)
632  ->gteq_edge_neighbour(direction,
633  s_diff,diff_level,
634  in_neighbouring_tree,
635  max_level,
636  orig_root_pt);
637  }
638  // If the present quadrant is not adjacent to the
639  // father's edge in the required direction, then the
640  // neighbour has the same father and is obtained
641  // by the appropriate reflection inside the father element
642  // This will only be called if we have not left the original tree.
643  else
644  {
645  next_el_pt=dynamic_cast<QuadTree*>(Father_pt);
646  }
647 
648  // We're about to ascend one level:
649  diff_level-=1;
650 
651  // Work out position of lower (or left) corner of present edge
652  // in its father element
653  s_diff += pow(0.5,-diff_level)*S_direct(direction,Son_type);
654 
655  // STEP 2: We have now located the neighbour's father and need to
656  // ------- find the appropriate son.
657 
658  // Buffer cases where the neighbour (and hence its father) lie outside
659  // the boundary
660  if (next_el_pt!=0)
661  {
662  // If the father is a leaf then we can't descend to the same
663  // level as the present node ---> simply return the father himself
664  // as the (greater) neighbour. Same applies if we are about
665  // to descend lower than the max_level (in a neighbouring tree)
666  if ((next_el_pt->Son_pt.size()==0)||(next_el_pt->Level>max_level-1))
667  {
668  return_el_pt=next_el_pt;
669  }
670  // We have located the neighbour's father: The position of the
671  // neighbour is obtained by `reflecting' the position of the
672  // node itself.
673 
674  //We know exactly how to reflect, because we know which son type we
675  //are and we have the pointer to the neighbours father
676  else
677  {
678  int son_quadrant=Reflect(direction,Son_type);
679 
680  // If the root of the neighbour's father is not our root, we
681  // might need to rotate
682  if (orig_root_pt!=next_el_pt->Root_pt)
683  {
684  //Get the north equivalent of the next element
685  int my_north=
686  dynamic_cast<QuadTreeRoot*>(Root_pt)->north_equivalent(direction);
687  son_quadrant=Rotate(my_north,son_quadrant);
688  }
689 
690  //The next element in the tree is the appropriate son of the
691  //neighbour's father
692  return_el_pt=dynamic_cast<QuadTree*>(next_el_pt->Son_pt[son_quadrant]);
693 
694  // Work out position of lower (or left) corner of present edge
695  // in next higher element
696  s_diff -= pow(0.5,-diff_level)*S_direct(direction,Son_type);
697 
698  // We have just descended one level
699  diff_level+=1;
700  }
701  }
702  // The neighbour's father lies outside the boundary --> the neighbour
703  // itself does too --> return NULL.
704  else
705  {
706  return_el_pt=0;
707  }
708  }
709  // Element does not have a father --> check if it has a neighbouring
710  // tree in the appropriate direction
711  else
712  {
713  // Find neighbouring root
714  if(Root_pt->neighbour_pt(direction)!=0)
715  {
716  //In this case we have moved to a neighbour, so set the flag
717  in_neighbouring_tree=true;
718  return_el_pt
719  =dynamic_cast<QuadTreeRoot*>(Root_pt->neighbour_pt(direction));
720  }
721  //No neighbouring tree, so there really is no neighbour --> return NULL
722  else
723  {
724  return_el_pt=0;
725  }
726  }
727 
728  return return_el_pt;
729 }
