57  /// the outer boundary of the domain and any number of internal
58  /// closed curves, specified by TetMeshFacetedSurfaces.
59  /// Also specify target volume for uniform element size.

61  TetMeshFacetedClosedSurface* const &outer_boundary_pt,
62  Vector<TetMeshFacetedSurface*>& internal_closed_surface_pt,
63  const double &element_volume,
64  TimeStepper* time_stepper_pt = &Mesh::Default_TimeStepper,
65  const bool &use_attributes = false,
66  const bool& split_corner_elements = false) :
67  TetgenMesh<ELEMENT>(outer_boundary_pt,
68  internal_closed_surface_pt,
69  element_volume,
70  time_stepper_pt,
71  use_attributes,
72  split_corner_elements), Corner_elements_must_be_split(split_corner_elements)

74  // Initialise the data associated with adaptation

76  }

81  /// \short Specialised constructor used during adaptation only.
82  /// Element sizes are specified by vector tetgen_io is passed in
83  /// from previous mesh (is then modified to build new mesh)
84  /// Ditto with use_attributes, which comes from the previous mesh
85  RefineableTetgenMesh(const Vector<double> &target_volume,
86  tetgenio* const &tetgen_io_pt,
87  TetMeshFacetedClosedSurface* const &outer_boundary_pt,
88  Vector<TetMeshFacetedSurface*>& internal_surface_pt,
89  TimeStepper* time_stepper_pt=
90  &Mesh::Default_TimeStepper,
91  const bool &use_attributes=false)
92  {
93 
94  // NOTE THERE IS A CERTAIN AMOUNT OF DUPLICATION BETWEEN THE
95  // CODE IN HERE AND THE ONE IN THE CONSTRUCTOR OF THE TetgenMesh
96  // BUT THE FACT THAT WE HAVE TO MODIFY THE TETGENIO STRUCTURE
97  // MEANS WE CAN'T QUITE RECYCLE THIS.
98 
99  // Mesh can only be built with 3D Telements.
100  MeshChecker::assert_geometric_element<TElementGeometricBase,ELEMENT>(3);
101 
102  // Initialise the data associated with adaptation
103  initialise_adaptation_data();
104 
105  // Store Timestepper used to build elements
106  this->Time_stepper_pt=time_stepper_pt;
107 
108  // Triangulation has been created -- remember to wipe it!
109  this->Tetgenio_exists =true;
110  this->Tetgenio_pt = new tetgenio;
111 
112  // Add the volume constraints to the tetgenio data object.
113  // Note that since the tetgenio structure is referred to by pointer
114  // we're also modifying the one associated with the (still existing)
115  // original mesh. Bit naughty but shouldn't cause any problems
116  // since that mesh is already built and about to go out of scope
117  // anyway.
118 
119  //Create a local copy
120  tetgenio *tetgen_input_pt = new tetgenio;;
121  this->deep_copy_of_tetgenio(tetgen_io_pt,tetgen_input_pt);
122 
123  //Add volume constraints
124  tetgen_input_pt->tetrahedronvolumelist =
125  new double[tetgen_input_pt->numberoftetrahedra];
126  for(int e=0;e<tetgen_input_pt->numberoftetrahedra;++e)
127  {
128  tetgen_input_pt->tetrahedronvolumelist[e] = target_volume[e];
129  }
130 
131  // Input string
132  std::stringstream input_string_stream;
133  input_string_stream<< "Vqra";
134 
135  // Convert to a *char
136  char tetswitches[100];
137  sprintf(tetswitches,"%s",input_string_stream.str().c_str());
138 
139  // Build triangulateio refined object
140  tetrahedralize(tetswitches, tetgen_input_pt, this->Tetgenio_pt);
141  // Build scaffold
142  this->Tmp_mesh_pt=new TetgenScaffoldMesh(*this->Tetgenio_pt);
143 
144  // Convert mesh from scaffold to actual mesh
145  this->build_from_scaffold(time_stepper_pt,use_attributes);
146 
147  // Kill the scaffold
148  delete this->Tmp_mesh_pt;
149  this->Tmp_mesh_pt=0;
150 
151  //delete the input
152  delete tetgen_input_pt;
153 
154  //Store the boundary
155  this->Outer_boundary_pt = outer_boundary_pt;
156  //Setup the reverse lookup scheme
157  this->setup_reverse_lookup_schemes_for_faceted_surface(
158  this->Outer_boundary_pt);
159  //Store the internal boundary
160  this->Internal_surface_pt = internal_surface_pt;
161  //Setup the reverse lookup schemes
162  {
163  unsigned n=this->Internal_surface_pt.size();
164  for (unsigned i=0;i<n;i++)
165  {
166  this->setup_reverse_lookup_schemes_for_faceted_surface(
167  this->Internal_surface_pt[i]);
168  }
169  }
170 
171  // Setup boundary coordinates for boundaries
172  unsigned nb=nboundary();
173  for (unsigned b=0;b<nb;b++)
174  {
175  this->template setup_boundary_coordinates<ELEMENT>(b);
176  }
177 
178  // Now snap onto geometric objects associated with triangular facets
179  // (if any!)
180  this->snap_nodes_onto_geometric_objects();
181 
182  }
