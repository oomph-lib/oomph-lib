51 /// MapMatrixMixed is a generalised, STL-map-based, sparse(ish) matrix
52 /// class with mixed indices
53 ///
54 /// The matrix is indexed by indices of type KEY_TYPE_ROW and KEY_TYPE_COL
55 /// and has entries of type VALUE_TYPE.
56 ///
57 /// Careful: If a zero entry is referenced then it is created in
58 /// memory. Therefore this isn't really a practical sparse matrix scheme.
59 /// Do not loop over `all' possible indices as even looking
60 /// at them will inflate the matrix until it occupies as much
61 /// space as a full one -- use (modification of) output routine
62 /// to retrieve all nonzero entries.
63 ///
64 /// However, this is not a serious restriction, as the main purpose
65 /// of this class is to allow non-integer indices.
66 ///
67 /// Example of usage:
68 /// \code
69 ///
70 ///
71 /// // Assume we have a Vector of pointers to objects:
72 /// Vector<Rubbish*> object_pt;
73 ///
74 /// [...]
75 ///
76 /// // Number of objects
77 /// int nentry=object_pt.size();
78 ///
79 /// // Use the pointers to the objects and associated integers as indices
80 /// // in a MapMatrixMixed whose entries are of type int
81 /// MapMatrixMixed<Rubbish*,int,int> like_a_matrix;
82 ///
83 /// for (int i=1;i<nentry;i++)
84 /// {
85 /// for (int j=1;j<nentry;j++)
86 /// {
87 /// int number=100*i+j;
88 /// like_a_matrix(object_pt[i],j)=number;
89 /// }
90 /// }
91 ///
92 /// oomph_info << "Matrix has nnz() " << like_a_matrix.nnz() <<
93 /// " and size() " << like_a_matrix.size() << std::endl;
94 ///
95 /// oomph_info << "\n\n\n Here are the nonzero entries: i, j, a(i,j)\n";
96 ///
97 /// like_a_matrix.output(oomph_info);
98 ///
99 /// // Can be used like a normal matrix:
100 /// like_a_matrix(object_pt[1],20)+=13;
101 /// like_a_matrix(object_pt[1],1)+=13;
102 ///
103 /// oomph_info << "\n\n\n Here are the nonzero entries: i, j, a(i,j)\n";
104 /// like_a_matrix.output(oomph_info);
105 ///
106 /// \endcode
107 ///
