110 /// \short Helper function to fill in contributions to residuals
111 /// (remember that part of the residual is added by the
112 /// the associated VolumeConstraintElement). This is specific for
113 /// 1D line elements that bound 2D cartesian fluid elements.
114 //====================================================================
115  void LineVolumeConstraintBoundingElement::
116  fill_in_generic_residual_contribution_volume_constraint(
117  Vector<double> &residuals)
118  {
119  //Add in the volume constraint term if required
120  const int local_eqn=this->ptraded_local_eqn();
121  if(local_eqn >=0)
122  {
123  //Find out how many nodes there are
124  const unsigned n_node = this->nnode();
125 
126  //Set up memeory for the shape functions
127  Shape psif(n_node);
128  DShape dpsifds(n_node,1);
129 
130  //Set the value of n_intpt
131  const unsigned n_intpt = this->integral_pt()->nweight();
132 
133  //Storage for the local coordinate
134  Vector<double> s(1);
135 
136  //Loop over the integration points
137  for(unsigned ipt=0;ipt<n_intpt;ipt++)
138  {
139  //Get the local coordinate at the integration point
140  s[0] = this->integral_pt()->knot(ipt,0);
141 
142  //Get the integral weight
143  double W = this->integral_pt()->weight(ipt);
144 
145  //Call the derivatives of the shape function at the knot point
146  this->dshape_local_at_knot(ipt,psif,dpsifds);
147 
148  // Get position and tangent vector
149  Vector<double> interpolated_t1(2,0.0);
150  Vector<double> interpolated_x(2,0.0);
151  for(unsigned l=0;l<n_node;l++)
152  {
153  //Loop over directional components
154  for(unsigned i=0;i<2;i++)
155  {
156  interpolated_x[i] += this->nodal_position(l,i)*psif(l);
157  interpolated_t1[i] += this->nodal_position(l,i)*dpsifds(l,0);
158  }
159  }
160 
161  //Calculate the length of the tangent Vector
162  double tlength = interpolated_t1[0]*interpolated_t1[0] +
163  interpolated_t1[1]*interpolated_t1[1];
164 
165  //Set the Jacobian of the line element
166  double J = sqrt(tlength);
167 
168  //Now calculate the normal Vector
169  Vector<double> interpolated_n(2);
170  this->outer_unit_normal(ipt,interpolated_n);
171 
172  // Assemble dot product
173  double dot = 0.0;
174  for(unsigned k=0;k<2;k++)
175  {
176  dot += interpolated_x[k]*interpolated_n[k];
177  }
178 
179  // Add to residual with sign chosen so that the volume is
180  // positive when the elements bound the fluid
181  residuals[local_eqn] += 0.5*dot*W*J;
182  }
183  }
184  }

432  /// \short Helper function to fill in contributions to residuals
433  /// (remember that part of the residual is added by the
434  /// the associated VolumeConstraintElement). This is specific for
435  /// 2D surface elements that bound 3D cartesian fluid elements.
436  //=================================================================
437  void SurfaceVolumeConstraintBoundingElement::
438  fill_in_generic_residual_contribution_volume_constraint(
439  Vector<double> &residuals)
440  {
441  //Add in the volume constraint term if required
442  const int local_eqn=this->ptraded_local_eqn();
443  if(local_eqn >=0)
444  {
445  //Find out how many nodes there are
446  const unsigned n_node = this->nnode();
447 
448  //Set up memeory for the shape functions
449  Shape psif(n_node);
450  DShape dpsifds(n_node,2);
451 
452  //Set the value of n_intpt
453  const unsigned n_intpt = this->integral_pt()->nweight();
454 
455  //Storage for the local coordinate
456  Vector<double> s(2);
457 
458  //Loop over the integration points
459  for(unsigned ipt=0;ipt<n_intpt;ipt++)
460  {
461  //Get the local coordinate at the integration point
462  for(unsigned i=0;i<2;i++) {s[i] = this->integral_pt()->knot(ipt,i);}
463 
464  //Get the integral weight
465  double W = this->integral_pt()->weight(ipt);
466 
467  //Call the derivatives of the shape function at the knot point
468  this->dshape_local_at_knot(ipt,psif,dpsifds);
469 
470  // Get position and tangent vector
471  DenseMatrix<double> interpolated_g(2,3,0.0);
472  Vector<double> interpolated_x(3,0.0);
473  for(unsigned l=0;l<n_node;l++)
474  {
475  //Loop over directional components
476  for(unsigned i=0;i<3;i++)
477  {
478  //Cache the nodal position
479  const double x_ = this->nodal_position(l,i);
480  //Calculate the position
481  interpolated_x[i] += x_*psif(l);
482  //Calculate the two tangent vecors
483  for(unsigned j=0;j<2;j++) {interpolated_g(j,i) += x_*dpsifds(l,j);}
484  }
485  }
486 
487  // Define the (non-unit) normal vector (cross product of tangent vectors)
488  Vector<double> interpolated_n(3);
489  interpolated_n[0] =
490  interpolated_g(0,1)*interpolated_g(1,2) -
491  interpolated_g(0,2)*interpolated_g(1,1);
492  interpolated_n[1] =
493  interpolated_g(0,2)*interpolated_g(1,0) -
494  interpolated_g(0,0)*interpolated_g(1,2);
495  interpolated_n[2] =
496  interpolated_g(0,0)*interpolated_g(1,1) -
497  interpolated_g(0,1)*interpolated_g(1,0);
498 
499  //The determinant of the local metric tensor is
500  //equal to the length of the normal vector, so
501  //rather than dividing and multipling, we just leave it out
502  //completely, which is why there is no J in the sum below
503 
504  //We can now find the sign to get the OUTER UNIT normal
505  for(unsigned i=0;i<3;i++) {interpolated_n[i] *= this->normal_sign();}
506 
507  // Assemble dot product
508  double dot = 0.0;
509  for(unsigned k=0;k<3;k++)
510  {
511  dot += interpolated_x[k]*interpolated_n[k];
512  }
513 
514  // Add to residual with the sign chosen such that the volume is
515  // positive when the faces are assembled such that the outer unit
516  // normal is directed out of the bulk fluid
517  residuals[local_eqn] += dot*W/3.0;
518  }
519  }
520  }
