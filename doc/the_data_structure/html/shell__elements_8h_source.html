287 //==================================================================
288 /// Matrix inversion for 2 dimensions
289 //==================================================================
290 double KirchhoffLoveShellEquations::calculate_contravariant
291 (double A[2][2], double Aup[2][2])
292 {
293  //Calculate determinant
294  double det = A[0][0]*A[1][1] - A[0][1]*A[1][0];
295 
296  //Calculate entries of the inverse
297  Aup[0][0] = A[1][1]/det;
298  Aup[0][1] = -A[0][1]/det;
299  Aup[1][0] = -A[1][0]/det;
300  Aup[1][1] = A[0][0]/det;
301 
302  //Return determinant
303  return(det);
304 }

494  /// \short Get the load vector: Pass number of the integration point,
495  /// Lagr. coordinate, Eulerian coordinate and normal vector
496  /// and return the load vector. (Not all of the input arguments will be
497  /// required for all specific load functions but the list should
498  /// cover all cases). We first evaluate the load function defined via
499  /// KirchhoffLoveShellEquations::load_vector_fct_pt() -- this
500  /// represents the non-FSI load on the shell, e.g. an external
501  /// pressure load. Then we add to this the FSI load due to
502  /// the traction exerted by the adjacent FSIFluidElements, taking
503  /// the sign of the normal into account.
504  void load_vector(const unsigned& intpt,
505  const Vector<double>& xi,
506  const Vector<double>& x,
507  const Vector<double>& N,
508  Vector<double>& load)
509  {
510  //Initially call the standard Load_vector_fct_pt
511  Load_vector_fct_pt(xi,x,N,load);
512 
513  //Memory for the FSI load
514  Vector<double> fsi_load(3);
515 
516  //Get the fluid load on the wall stress scale
517  fluid_load_vector(intpt,N,fsi_load);
518 
519  //If the normal is outer to the fluid switch the direction
520  double sign = 1.0;
521  if (!Normal_points_into_fluid) {sign = -1.0;}
522 
523  //Add the FSI load to the load vector
524  for(unsigned i=0;i<3;i++)
525  {
526  load[i] += sign*fsi_load[i];
527  }
528  }

570 /// Element that allows the imposition of boundary
571 /// conditions for a shell that is clamped to a 2D plane
572 /// that is specified by its normal. Constraint is applied by
573 /// a Lagrange multiplier.
574 /// \n
575 /// \b Note \b 1: Note that the introduction of the Lagrange
576 /// multiplier adds two additional values (relative to the number
577 /// of values before the addition of the FaceElement) to the nodes.
578 /// This ensures that nodes that are shared by adjacent FaceElements
579 /// are not resized repeatedly but also means that this won't work
580 /// if two "edges" of the shell (that share a node) are subject
581 /// to different constraints, each applied with its own
582 /// independent Lagrange multiplier. In such cases a modified
583 /// version of this class must be written.
584 /// \n
585 /// \b Note \b 2: The FaceGeometry for a HermiteShellElement is
586 /// the 1D two-node element SolidQHermiteElement<1> which has
587 /// four shape functions (two nodes, two types -- representing the
588 /// shape functions that interpolate the value and the derivative).
589 /// These are the "correct" shape functions for the interpolation
590 /// of the Lagrange multiplier and the isoparametric representation
591 /// of the geometry. However, when applying the contribution from the
592 /// constraint equation to the bulk equations, we have to take
593 /// all four types of dof into account so the element has to
594 /// reset the number of positional dofs to four. To avoid any clashes
595 /// we overload (the relevant subset of) the access functions to the
596 /// shape functions and their derivatives and set the shape functions
597 /// associated with the spurious positional dofs to zero. This is a bit
598 /// hacky but the only way (?) this can be done...
599 //======================================================================
