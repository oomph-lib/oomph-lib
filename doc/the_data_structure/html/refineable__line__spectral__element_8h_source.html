88  /// The only thing to add is rebuild from sons
89  void rebuild_from_sons(Mesh* &mesh_pt)
90  {
91  // The timestepper should be the same for all nodes and node 0 should
92  // never be deleted.
93  if(this->node_pt(0)==0)
94  {
95  throw OomphLibError("The vertex node (0) does not exist",
96  OOMPH_CURRENT_FUNCTION,
97  OOMPH_EXCEPTION_LOCATION);
98  }
99 
100  TimeStepper* time_stepper_pt = this->node_pt(0)->time_stepper_pt();
101 
102  // Determine number of history values stored
103  const unsigned ntstorage = time_stepper_pt->ntstorage();
104 
105  // Allocate storage for local coordinates
106  Vector<double> s_fraction(1), s(1);
107 
108  // Determine the number of nodes in the element
109  const unsigned n_node = this->nnode_1d();
110 
111  // Loop over the nodes in the element
112  for(unsigned n=0;n<n_node;n++)
113  {
114  // Get the fractional position of the node in the direction of s[0]
115  s_fraction[0] = this->local_one_d_fraction_of_node(n,0);
116 
117  // Determine the local coordinate in the father element
118  s[0] = -1.0 + 2.0*s_fraction[0];
119 
120  // If the node has not been built
121  if(this->node_pt(n)==0)
122  {
123  // Has the node been created by one of its neighbours?
124  bool is_periodic = false;
125  Node* created_node_pt =
126  this->node_created_by_neighbour(s_fraction,is_periodic);
127 
128  // If it has, set the pointer
129  if(created_node_pt!=0)
130  {
131  // If the node is periodic
132  if(is_periodic)
133  {
134  throw OomphLibError(
135  "Cannot handle periodic nodes in refineable spectral elements",
136  OOMPH_CURRENT_FUNCTION,
137  OOMPH_EXCEPTION_LOCATION);
138  }
139  // Non-periodic case, just set the pointer
140  else
141  {
142  this->node_pt(n) = created_node_pt;
143  }
144  }
145  // Otherwise, we need to build it
146  else
147  {
148  // First, find the son element in which the node should live
149 
150  // Find coordinate in the son
151  Vector<double> s_in_son(1);
152  using namespace BinaryTreeNames;
153  int son=-10;
154  // If s_fraction is between 0 and 0.5, we are in the left son
155  if(s_fraction[0] < 0.5)
156  {
157  son = L;
158  s_in_son[0] = -1.0 + 4.0*s_fraction[0];
159  }
160  // Otherwise we are in the right son
161  else
162  {
163  son = R;
164  s_in_son[0] = -1.0 + 4.0*(s_fraction[0]-0.5);
165  }
166 
167  // Get the pointer to the son element in which the new node
168  // would live
169  RefineableQSpectralElement<1>* son_el_pt =
170  dynamic_cast<RefineableQSpectralElement<1>*>(
171  this->tree_pt()->son_pt(son)->object_pt());
172 
173  // In 1D we should never be rebuilding an element's vertex nodes
174  // (since they will never be deleted), so throw an error if we
175  // appear to be doing so
176 #ifdef PARANOID
177  if(n==0 || n==n_node-1)
178  {
179  std::string error_message =
180  "I am trying to rebuild one of the (two) vertex nodes in\n";
181  error_message +=
182  "this 1D element. It should not have been possible to delete\n";
183  error_message +=
184  "either of these!\n";
185 
186  throw OomphLibError(
187  error_message,
188  OOMPH_CURRENT_FUNCTION,
189  OOMPH_EXCEPTION_LOCATION);
190  }
191 #endif
192 
193  // With this in mind we will always be creating normal "bulk" nodes
194  this->node_pt(n) = this->construct_node(n,time_stepper_pt);
195 
196  // Now we set the position and values at the newly created node
197 
198  // In the first instance use macro element or FE representation
199  // to create past and present nodal positions.
200  // (THIS STEP SHOULD NOT BE SKIPPED FOR ALGEBRAIC ELEMENTS AS NOT
201  // ALL OF THEM NECESSARILY IMPLEMENT NONTRIVIAL NODE UPDATE
202  // FUNCTIONS. CALLING THE NODE UPDATE FOR SUCH ELEMENTS/NODES WILL
203  // LEAVE THEIR NODAL POSITIONS WHERE THEY WERE (THIS IS APPROPRIATE
204  // ONCE THEY HAVE BEEN GIVEN POSITIONS) BUT WILL NOT ASSIGN SENSIBLE
205  // INITIAL POSITIONS!)
206 
207  // Loop over history values
208  for(unsigned t=0;t<ntstorage;t++)
209  {
210  // Allocate storage for the previous position of the node
211  Vector<double> x_prev(1);
212 
213  // Get position from son element -- this uses the macro element
214  // representation if appropriate
215  son_el_pt->get_x(t,s_in_son,x_prev);
216 
217  // Set the previous position of the new node
218  this->node_pt(n)->x(t,0) = x_prev[0];
219 
220  // Allocate storage for the previous values at the node
221  // NOTE: the size of this vector is equal to the number of values
222  // (e.g. 3 velocity components and 1 pressure, say)
223  // associated with each node and NOT the number of history values
224  // which the node stores!
225  Vector<double> prev_values;
226 
227  // Get values from son element
228  // Note: get_interpolated_values() sets Vector size itself.
229  son_el_pt->get_interpolated_values(t,s_in_son,prev_values);
230 
231  // Determine the number of values at the new node
232  const unsigned n_value = this->node_pt(n)->nvalue();
233 
234  // Loop over all values and set the previous values
235  for(unsigned v=0;v<n_value;v++)
236  {
237  this->node_pt(n)->set_value(t,v,prev_values[v]);
238  }
239  } // End of loop over history values
240 
241  // Add new node to mesh
242  mesh_pt->add_node_pt(this->node_pt(n));
243 
244  } // End of case where we build the node ourselves
245 
246  // Check if the element is an algebraic element
247  AlgebraicElementBase* alg_el_pt =
248  dynamic_cast<AlgebraicElementBase*>(this);
249 
250  // If so, throw error
251  if(alg_el_pt!=0)
252  {
253  std::string error_message =
254  "Have not implemented rebuilding from sons for";
255  error_message +=
256  "Algebraic Spectral elements yet\n";
257 
258  throw
259  OomphLibError(error_message,
260  "RefineableQSpectralElement<1>::rebuild_from_sons()",
261  OOMPH_EXCEPTION_LOCATION);
262  }
263 
264  } // End of if this node has not been built
265  } // End of loop over nodes in element
266  }
267

276  }
277  // If we get to here, OK
278  return true;
279  }
