44  /// A class for elements that solve the equations of Kirchhoff-Love
45  /// large-displacement (but linearly-elastic) thin-beam theory.
46  ///
47  /// The variational principle has the form
48  /// \f[
49  /// \int_0^{L} \left[
50  /// (\sigma_0 + \gamma) \ \delta
51  /// \gamma +
52  /// \frac{1}{12} \left(\frac{h}{R_0}\right)^2 \kappa
53  /// \ \delta \kappa -
54  /// \left( \left(\frac{R_0}{h}\right) {\bf f} - \Lambda^2
55  /// \frac{\partial^2 {\bf R}_w}{\partial t^2} \right) \cdot
56  /// \delta {\bf R}_w
57  /// \right] \ d\xi = 0,
58  /// \f]
59  /// where all lengths have been non-dimensionalised w.r.t. \f$ R_0 \f$.
60  /// The strain and and bending "tensors" \f$\gamma\f$ and \f$\kappa\f$
61  /// are computed relative to the shape of the beam's undeformed shape
62  /// which is specified as a GeomObject.
63  ///
64  /// Time is scaled on the timescale \f$T\f$ and
65  /// \f[
66  /// \Lambda = \frac{a}{T} \sqrt{\frac{\rho}{E_{eff}}},
67  /// \f]
68  /// the ratio of the timescale used in the non-dimensionalisation of the
69  /// equations to the natural timescale of the wall oscillations (in the
70  /// wall's in-plane mode). \f$ \Lambda^2 \f$ can be interpreted as
71  /// the non-dimensional wall density, therefore \f$ \Lambda=0\f$
72  /// corresponds to the case without wall inertia.
73  ///
74  ///
75  /// Note that:
76  /// - the load vector \f$ {\bf f} \f$ is scaled
77  /// on the effective elastic modulus \f$ E_{eff}=E/(1-\nu^2)\f$
78  /// (rather than the
79  /// bending stiffness). Rescale the result yourself if you prefer
80  /// another non-dimensionalisation (the current version yields the
81  /// the most compact maths).
82  /// - Poisson's ratio does not appear explicitly since it only occurs
83  /// in combination with Young's modulus \f$E\f$.
84  ///
85  /// Default values:
86  /// - the 2nd Piola Kirchhoff pre-stress \f$ \sigma_0 \f$ is zero.
87  /// - the wall thickness \f$ h/R_0\f$ is 1/20.
88  /// - the timescale ratio \f$ \Lambda^2\f$ is 1.
89  /// - the traction vector \f$ f \f$ evaluates to zero.
90  ///
91  /// Need to specify:
92  /// - the undeformed wall shape (as a GeomObject).
93  ///
94  /// The governing equations can be switched from the principle of
95  /// virtual displacements to a system of equations that forces the
96  /// beam to deform into a shape specified by a SolidInitialCondition object.
97  /// If \c SolidFiniteElement::solid_ic_pt()!=0 we solve the
98  /// the equations
99  /// \f[
100  /// \int_0^{L} \left(
101  /// \frac{\partial^i {\bf R}_{IC}}{\partial t^i} - {\bf R}_w
102  /// \right) \psi_{jk} \ d\xi = 0,
103  /// \f]
104  /// where \f$ \partial^i {\bf R}_{IC}/\partial t^i\f$ is
105  /// implemented by the SolidInitialCondition object, pointed to by
106  /// \c SolidFiniteElement::shell_ic_pt().
107  ///
108  //=======================================================================

411  /// \short Get the load vector: Pass number of the integration point,
412  /// Lagr. coordinate, Eulerian coordinate and normal vector
413  /// and return the load vector. (Not all of the input arguments will be
414  /// required for all specific load functions but the list should
415  /// cover all cases). We first evaluate the load function defined via
416  /// KirchhoffLoveBeamEquations::load_vector_fct_pt() -- this
417  /// represents the non-FSI load on the beam, e.g. an external
418  /// pressure load. Then we add to this the FSI load due to
419  /// the traction exerted by the adjacent FSIFluidElements, taking
420  /// the sign of the normal into account.
421  void load_vector(const unsigned& intpt,
422  const Vector<double>& xi,
423  const Vector<double>& x,
424  const Vector<double>& N,
425  Vector<double>& load)
426  {
427  // Initially call the standard Load_vector_fct_pt
428  Load_vector_fct_pt(xi, x, N, load);
429 
430  // Memory for the FSI load
431  Vector<double> fsi_load(2);
432 
433  // Get the fluid load on the wall stress scale
434  fluid_load_vector(intpt, N, fsi_load);
435 
436  // If the normal is outer to the fluid switch the direction
437  double sign = 1.0;
438  if (!Normal_points_into_fluid)
439  {
440  sign = -1.0;
441  }
442 
443  // Add the FSI load to the load vector
444  for (unsigned i = 0; i < 2; i++)
445  {
446  load[i] += sign * fsi_load[i];
447  }
448  }
