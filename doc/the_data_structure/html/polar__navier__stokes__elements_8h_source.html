331  /// \short i-th component of du/dt at local node n.
332  /// Uses suitably interpolated value for hanging nodes.
333  double du_dt_pnst(const unsigned &n, const unsigned &i) const
334  {
335  // Get the data's timestepper
336  TimeStepper* time_stepper_pt=node_pt(n)->time_stepper_pt();
337 
338  // Number of timsteps (past & present)
339  unsigned n_time = time_stepper_pt->ntstorage();
340 
341  //Initialise dudt
342  double dudt=0.0;
343  //Loop over the timesteps, if there is a non Steady timestepper
344  if (time_stepper_pt->type()!="Steady")
345  {
346  for(unsigned t=0;t<n_time;t++)
347  {
348  dudt+=time_stepper_pt->weight(1,t)*u_pnst(t,n,i);

569  {

573  Shape psi(n_node);
574  //Find values of shape function
575  shape(s,psi);
576 
577  for (unsigned i=0;i<2;i++)
578  {
579  //Initialise value of u

581  //Loop over the local nodes and sum

583  {

585  }
