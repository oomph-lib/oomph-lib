43  /// A base class that represents the fourth-rank elasticity tensor
44  /// \f$E_{ijkl}\f$ defined such that
45  /// \f[\tau_{ij} = E_{ijkl} e_{kl},\f]
46  /// where \f$e_{ij}\f$ is the infinitessimal (Cauchy) strain tensor
47  /// and \f$\tau_{ij}\f$ is the stress tensor. The symmetries of the
48  /// tensor are such that
49  /// \f[E_{ijkl} = E_{jikl} = E_{ijlk} = E_{klij}\f]
50  /// and thus there are relatively few independent components. These
51  /// symmetries are included in the definition of the object so that
52  /// non-physical symmetries cannot be accidentally imposed.
53  //=====================================================================
54 class ElasticityTensor
55  {
56 
57  ///\short Translation table from the four indices to the corresponding
58  ///independent component
59  static const unsigned Index[3][3][3][3];
60 
61  protected:
62 
63  ///Member function that returns the i-th independent component of the
64  ///elasticity tensor
65  virtual inline double independent_component(const unsigned &i) const
66  {return 0.0;}
67 
68 
69  ///\short Helper range checking function
70  /// (Note that this only captures over-runs in 3D but
71  /// errors are likely to be caught in evaluation of the
72  /// stress and strain tensors anyway...)
73  void range_check(const unsigned &i, const unsigned &j,
74  const unsigned &k, const unsigned &l) const
75  {
76  if((i > 2) || (j > 2) || (k> 2) || (l>2))
77  {
78  std::ostringstream error_message;
79  if(i > 2)
80  {
81  error_message << "Range Error : Index 1 " << i
82  << " is not in the range (0,2)";
83  }
84  if(j > 2)
85  {
86  error_message << "Range Error : Index 2 " << j
87  << " is not in the range (0,2)";
88  }
89 
90  if(k > 2)
91  {
92  error_message << "Range Error : Index 2 " << k
93  << " is not in the range (0,2)";
94  }
95 
96  if(l > 2)
97  {
98  error_message << "Range Error : Index 4 " << l
99  << " is not in the range (0,2)";
100  }
101 
102  //Throw the error
103  throw OomphLibError(error_message.str(),
104  OOMPH_CURRENT_FUNCTION,
105  OOMPH_EXCEPTION_LOCATION);
106  }
107  }
108 
109 
110  ///Empty Constructor
111  ElasticityTensor() {}
112 
113  public:
114 
115  ///Empty virtual Destructor
116  virtual ~ElasticityTensor() {}
117 
118  public:
119 
120  ///\short Return the appropriate independent component
121  ///via the index translation scheme (const version).
122  double operator()(const unsigned &i, const unsigned &j,
123  const unsigned &k, const unsigned &l) const
124  {
125  //Range check
126 #ifdef PARANOID
127  range_check(i,j,k,l);
128 #endif
129  return independent_component(Index[i][j][k][l]);
130  }

135 /// An isotropic elasticity tensor defined in terms of Young's modulus
136 /// and Poisson's ratio. The elasticity tensor is assumed to be
137 /// non-dimensionalised on some reference value for Young's modulus
138 /// so the value provided to the constructor (if any) is to be
139 /// interpreted as the ratio of the actual Young's modulus to the
140 /// Young's modulus used to non-dimensionalise the stresses/tractions
141 /// in the governing equations.
142 //===================================================================
143  class IsotropicElasticityTensor : public ElasticityTensor
144  {
145  //Storage for the independent components of the elasticity tensor
146  double C[4];
147 
148  //Translation scheme between the 21 independent components of the general
149  //elasticity tensor and the isotropic case
150  static const unsigned StaticIndex[21];
151 
152  public:
153 
154  /// \short Constructor. Passing in the values of the Poisson's ratio
155  /// and Young's modulus (interpreted as the ratio of the actual
156  /// Young's modulus to the Young's modulus (or other reference stiffness)
157  /// used to non-dimensionalise stresses and tractions in the governing
158  /// equations).
159  IsotropicElasticityTensor(const double &nu,
160  const double &E) : ElasticityTensor()
161  {
162  //Set the three independent components
163  C[0] = 0.0;
164  double lambda=E*nu/((1.0+nu)*(1.0-2.0*nu));
165  double mu=E/(2.0*(1.0+nu));
166  this->set_lame_coefficients(lambda,mu);
167  }
168 
169  /// \short Constructor. Passing in the value of the Poisson's ratio.
170  /// Stresses and tractions in the governing equations are assumed
171  /// to have been non-dimensionalised on Young's modulus.
172  IsotropicElasticityTensor(const double &nu) : ElasticityTensor()
173  {
174  //Set the three independent components
175  C[0] = 0.0;
176 
177  // reference value
178  double E=1.0;
179  double lambda=E*nu/((1.0+nu)*(1.0-2.0*nu));
180  double mu=E/(2.0*(1.0+nu));
181  this->set_lame_coefficients(lambda,mu);
182  }
183 
184  /// \short Constructur. Passing in the values of the two lame
185  /// coefficients directly (interpreted as the ratios of these
186  /// quantities to a reference stiffness used to non-dimensionalised
187  IsotropicElasticityTensor(const Vector<double> &lame)
188  {
189  //Set the three independent components
190  C[0] = 0.0;
191  this->set_lame_coefficients(lame[0],lame[1]);
192  }
193 
194 
195  ///Overload the independent coefficient function
196  inline double independent_component(const unsigned &i) const
197  {return C[StaticIndex[i]];}
198 
199

201

203  void set_lame_coefficients(const double &lambda, const double &mu)
204  {
205  C[1] = lambda + 2.0*mu;
206  C[2] = lambda;
207  C[3] = mu;
208  }
