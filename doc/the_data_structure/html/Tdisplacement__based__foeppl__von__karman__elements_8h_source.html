99  /// \short The number of dof types that degrees of freedom in this
100  /// element are sub-divided into
101  unsigned ndof_types() const
102  {
103  // NOTE: this assumes "clamped" bcs
104  // [0]: laplacian w interior
105  // [1]: laplacian w boundary
106  // [2]: W
107  // [3]: Ux
108  // [4]: Uy
109  return 5;

112  /// \short Create a list of pairs for all unknowns in this element,
113  /// so that the first entry in each pair contains the global
114  /// equation number of the unknown, while the second one contains
115  /// the number of the dof type that this unknown is associated with.
116  /// (Function can obviously only be called if the equation numbering
117  /// scheme has been set up.)
118  /// Dof_types
119  /// 0,1: Laplacian;
120  /// 2: Bending w
121  /// 3: Displacements Ux and Uy
122  /// The indexing of the dofs in the element is like below
123  /// [0]: w
124  /// [1]: laplacian w
125  /// [2]: U_x
126  /// [3]: U_y
127  void get_dof_numbers_for_unknowns(
128  std::list<std::pair<unsigned long,unsigned> >& dof_lookup_list) const
129  {
130  // number of nodes
131  const unsigned n_node = this->nnode();
132 
133  // temporary pair (used to store dof lookup prior to being added to list)
134  std::pair<unsigned,unsigned> dof_lookup;
135 
136  // loop over the nodes
137  for (unsigned n = 0; n < n_node; n++)
138  {
139  // Zeroth nodal value: displacement
140  //---------------------------------
141  unsigned v=0;
142 
143  // determine local eqn number
144  int local_eqn_number = this->nodal_local_eqn(n, v);
145 
146  // ignore pinned values
147  if (local_eqn_number >= 0)
148  {
149  // store dof lookup in temporary pair: Global equation
150  // number is the first entry in pair
151  dof_lookup.first = this->eqn_number(local_eqn_number);
152 
153  // set dof type numbers: Dof type is the second entry in pair
154  dof_lookup.second = 2;
155 
156  // add to list
157  dof_lookup_list.push_front(dof_lookup);
158  }
159 
160  // First nodal value: Laplacian
161  //-----------------------------
162  v=1;
163 
164  // determine local eqn number
165  local_eqn_number = this->nodal_local_eqn(n, v);
166 
167  // ignore pinned values
168  if (local_eqn_number >= 0)
169  {
170  // store dof lookup in temporary pair: Global equation
171  // number is the first entry in pair
172  dof_lookup.first = this->eqn_number(local_eqn_number);
173 
174  // Is it a boundary node? If so: It's dof type 1
175  if (node_pt(n)->is_on_boundary(0) || node_pt(n)->is_on_boundary(1))
176  {
177  dof_lookup.second = 1;
178  }
179  // otherwise it's in the interior: It's dof type 0
180  else
181  {
182  dof_lookup.second = 0;
183  }
184 
185  // add to list
186  dof_lookup_list.push_front(dof_lookup);
187  }
188 
189  // Second nodal value: U_x
190  //---------------------------------
191  v=2;
192 
193  // determine local eqn number
194  local_eqn_number = this->nodal_local_eqn(n, v);
195 
196  // ignore pinned values
197  if (local_eqn_number >= 0)
198  {
199  // store dof lookup in temporary pair: Global equation
200  // number is the first entry in pair
201  dof_lookup.first = this->eqn_number(local_eqn_number);
202 
203  // set dof type numbers: Dof type is the second entry in pair
204  dof_lookup.second = 3;
205 
206  // add to list
207  dof_lookup_list.push_front(dof_lookup);
208  }
209 
210  // Third nodal value: U_y
211  //---------------------------------
212  v=3;
213 
214  // determine local eqn number
215  local_eqn_number = this->nodal_local_eqn(n, v);
216 
217  // ignore pinned values
218  if (local_eqn_number >= 0)
219  {
220  // store dof lookup in temporary pair: Global equation
221  // number is the first entry in pair
222  dof_lookup.first = this->eqn_number(local_eqn_number);
223 
224  // set dof type numbers: Dof type is the second entry in pair
225  dof_lookup.second = 4;
226 
227  // add to list
228  dof_lookup_list.push_front(dof_lookup);
229  }
230 
231  } // for (n < n_node)
232 
233  }

334 //======================================================================
335 /// Define the shape functions and test functions and derivatives
336 /// w.r.t. global coordinates and return Jacobian of mapping.
337 ///
338 /// Galerkin: Test functions = shape functions
339 //======================================================================
340 template<unsigned NNODE_1D>
341 double TDisplacementBasedFoepplvonKarmanElement<NNODE_1D>::dshape_and_dtest_eulerian_fvk(
342  const Vector<double> &s,
343  Shape &psi,
344  DShape &dpsidx,
345  Shape &test,
346  DShape &dtestdx) const
347 {
348  unsigned n_node = this->nnode();
349 
350  //Call the geometrical shape functions and derivatives
351  double J = this->dshape_eulerian(s,psi,dpsidx);
352 
353  //Loop over the test functions and derivatives and set them equal to the
354  //shape functions
355  for(unsigned i=0;i<n_node;i++)
356  {
357  test[i] = psi[i];
358  dtestdx(i,0) = dpsidx(i,0);
359  dtestdx(i,1) = dpsidx(i,1);
360  }
361 
362  //Return the jacobian
363  return J;
364 }

367 
368 //======================================================================
369 /// Define the shape functions and test functions and derivatives
370 /// w.r.t. global coordinates and return Jacobian of mapping.
371 ///
372 /// Galerkin: Test functions = shape functions
373 //======================================================================
374 template<unsigned NNODE_1D>
375 double TDisplacementBasedFoepplvonKarmanElement<NNODE_1D>::
376 dshape_and_dtest_eulerian_at_knot_fvk(
377  const unsigned &ipt,
378  Shape &psi,
379  DShape &dpsidx,
380  Shape &test,
381  DShape &dtestdx) const
382 {
383 
384  //Call the geometrical shape functions and derivatives
385  double J = this->dshape_eulerian_at_knot(ipt,psi,dpsidx);
386 
387  //Set the pointers of the test functions
388  test = psi;
389  dtestdx = dpsidx;
390 
391  //Return the jacobian
392  return J;
393 
394 }
