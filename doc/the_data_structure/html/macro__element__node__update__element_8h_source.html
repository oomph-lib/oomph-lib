258 /// MacroElementNodeUpdate elements are elements that can not only be updated
259 /// via their MacroElement representation (in princple any FiniteElement
260 /// could do that...) but also allows the geometric Data contained
261 /// in the GeomObjects that affect the MacroElement-based node update
262 /// operations to be unknowns in the overall Problem.
263 ///
264 /// The element wraps around the ELEMENT specified by the template
265 /// parameter and computes the derivatives of the residual vector
266 /// with respect to the geometric Data (needed in the setup of
267 /// the element's Jacobian matrix) by finite differencing.
268 /// Otherwise the element behaves exactly like the templace element.
269 //========================================================================
270 template<class ELEMENT>
271 class MacroElementNodeUpdateElement :
272 public ElementWithSpecificMovingNodes<ELEMENT,MacroElementNodeUpdateNode>,
273 public MacroElementNodeUpdateElementBase

307  /// \short Set node update information:
308  /// Pass the vector of (pointers to) the geometric objects
309  /// that affect the node update. This gets passed on to all nodes in
310  /// the element.
311  void set_node_update_info(const Vector<GeomObject*>& geom_object_pt)
312  {
313  // Store local copy of geom object vector, so it can be passed on
314  // to son elements (and their nodes) during refinement
315  unsigned ngeom_object=geom_object_pt.size();
316  Geom_object_pt.resize(ngeom_object);
317  for (unsigned i=0;i<ngeom_object;i++)
318  {
319  Geom_object_pt[i]=geom_object_pt[i];
320  }
321 
322  // Loop over nodes in element
323  unsigned n_node = this->nnode();
324  for (unsigned j=0;j<n_node;j++)
325  {
326  // Get local coordinate in element (Vector sets its own size)
327  Vector<double> s_in_node_update_element;
328  this->local_coordinate_of_node(j,s_in_node_update_element);
329 
330  // Pass the lot to the node
331  static_cast<MacroElementNodeUpdateNode*>(this->node_pt(j))->
332  set_node_update_info(this,s_in_node_update_element,geom_object_pt);
333  }
334  }
335 
336 
337  /// \short Rebuild after unrefinement: Reset the node update information
338  /// for all nodes so that the nodes get updated by this element.
339  /// If we don't do that, some nodes might still want to be updated
340  /// by elements that no longer exist which leads to the most wonderful
341  /// seg fault... Afterwards, call the template element's own
342  /// rebuild_from_son() function (if it's a RefineableElement)
343  void rebuild_from_sons(Mesh* &mesh_pt)
344  {
345  // First call the element's own rebuild_from_sons() function
346  ELEMENT::rebuild_from_sons(mesh_pt);
347 
348  // Now loop over nodes in element
349  unsigned n_node =this->nnode();
350  for (unsigned j=0;j<n_node;j++)
351  {
352  // Get local coordinate in element (Vector sets its own size)
353  Vector<double> s_in_node_update_element;
354  this->local_coordinate_of_node(j,s_in_node_update_element);
355 
356  // Pass the lot to the node
357  static_cast<MacroElementNodeUpdateNode*>(this->node_pt(j))->
358  set_node_update_info(this,s_in_node_update_element,Geom_object_pt);
359  }
360  }
361 
362 
363 };

404  /// \short Update all nodal positions via sparse MacroElement-based
405  /// update functions. If a Node is hanging its position is updated
406  /// after updating the position of its masters first.
407  /// [Doesn't make sense to use this mesh with SolidElements anyway,
408  /// so we buffer the case if update_all_solid_nodes is set to
409  /// true.]
410  void node_update(const bool& update_all_solid_nodes=false)
411  {
412 #ifdef PARANOID
413  if (update_all_solid_nodes)
414  {
415  std::string error_message =
416  "Doesn't make sense to use an MacroElementNodeUpdateMesh with\n";
417  error_message +=
418  "SolidElements so specifying update_all_solid_nodes=true\n";
419  error_message += "doesn't make sense either\n";
420 
421  throw OomphLibError(error_message,
422  OOMPH_CURRENT_FUNCTION,
423  OOMPH_EXCEPTION_LOCATION);
424  }
425 #endif
426 
427  // Loop over all nodes and update their positions -- hanging nodes
428  // are updated via their masters; auxiliary update function
429  // is performed by node, too
430  unsigned n_node = nnode();
431  for (unsigned n=0;n<n_node;n++)
432  {
433  MacroElementNodeUpdateNode* nod_pt=dynamic_cast
434  <MacroElementNodeUpdateNode*>(node_pt(n));
435 #ifdef PARANOID
436  if (nod_pt==0)
437  {
438  std::ostringstream error_message;
439  error_message

441  << "Node is of type: " << typeid(node_pt(n)).name() << std::endl;
442 
443  throw OomphLibError(error_message.str(),
444  OOMPH_CURRENT_FUNCTION,
445  OOMPH_EXCEPTION_LOCATION);
446  }
447 #endif
448  nod_pt->node_update();
449  }
450 
451 #ifdef OOMPH_HAS_MPI
452  // Update positions for external halo nodes attached to this mesh
453  // Loop over processors
454  for (std::map<unsigned,Vector<Node*> >::iterator it=
455  External_halo_node_pt.begin();it!=External_halo_node_pt.end();it++)
456  {
457  int iproc=(*it).first;
458  unsigned n_ext_halo_node=nexternal_halo_node(iproc);
459  for (unsigned n=0;n<n_ext_halo_node;n++)
460  {
461  MacroElementNodeUpdateNode* nod_pt=dynamic_cast
462  <MacroElementNodeUpdateNode*>(external_halo_node_pt(iproc,n));
463 #ifdef PARANOID
464  if (nod_pt==0)
465  {
466  std::ostringstream error_message;
467  error_message
468  << "Failed to cast (ext. halo) to MacroElementNodeUpdateNode.\n"
469  << "Node is of type: " << typeid(node_pt(n)).name() << std::endl;
470 
471  throw OomphLibError(error_message.str(),
472  OOMPH_CURRENT_FUNCTION,
473  OOMPH_EXCEPTION_LOCATION);
474  }
475 #endif
476  nod_pt->node_update();
477  }
478  } // end loop over processors
479 #endif // (ifdef OOMPH_HAS_MPI)
480  }

482 #ifdef OOMPH_HAS_MPI
483  /// \short Overload the base class distribute function to deal
484  /// with halo nodes on halo elements that may have pointers
485  /// to macro elements that no longer exist
486  void distribute(OomphCommunicator* comm_pt,
487  const Vector<unsigned>& element_domain,
488  Vector<GeneralisedElement*>& deleted_element_pt,
489  DocInfo& doc_info,

500

502  // non-distributed problem may have set a macro element which no
503  // longer exists for some halo nodes which are on halo elements

524  }
