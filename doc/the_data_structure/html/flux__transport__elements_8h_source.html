42 /// Base class for the flux transport equations templated by the
43 /// dimension DIM. The equations that are solved are
44 /// \f[ \frac{\partial u_{i}}{\partial t} + \frac{\partial}{\partial x_{j}}
45 /// \left(F_{ij}(u_{k})\right), \f]
46 /// where \f$ F_{ij} \f$ is a matrix of flux components.
47 //==================================================================
48 template<unsigned DIM>
49 class FluxTransportEquations : public virtual FiniteElement
50 {
51 protected:
52 /// Return the number of fluxes (default zero)
53 virtual inline unsigned nflux() const
54 {
55 return 0;
56 }
57
58 /// Return the index at which the i-th unknown value
59 /// is stored. The default value, i, is appropriate for single-physics
60 /// problems.
61 /// In derived multi-physics elements, this function should be overloaded
62 /// to reflect the chosen storage scheme. Note that these equations require
63 /// that the unknowns are always stored at the same indices at each node.
64 virtual inline unsigned u_index_flux_transport(const unsigned& i) const
65 {
66 return i;
67 }
68
69 /// Return the flux as a function of the unknown. This interface
70 /// could (should) be generalised)
71 virtual void flux(const Vector<double>& u, DenseMatrix<double>& f)
72 {
73 std::ostringstream error_stream;
74 error_stream
75 << "Default empty flux function called\n"
76 << "This should be overloaded with a specific flux function\n"
77 << "in a derived class\n";
78 throw OomphLibError(
79 error_stream.str(), OOMPH_CURRENT_FUNCTION, OOMPH_EXCEPTION_LOCATION);
80 }
81
82 /// Return the flux derivatives as a funciton of the unknowns
83 /// Default finite-different implementation

85 RankThreeTensor<double>& df_du);

113 /// Compute the element's residual Vector
114 void fill_in_contribution_to_residuals(Vector<double>& residuals)
115 {
116 // Call the generic residuals function with flag set to 0
117 // and using a dummy matrix argument
118 fill_in_generic_residual_contribution_flux_transport(
119 residuals,
120 GeneralisedElement::Dummy_matrix,
121 GeneralisedElement::Dummy_matrix,
122 0);
123 }

125 /// Compute the element's residual Vector and the jacobian matrix
126 /// Virtual function can be overloaded by hanging-node version
127 void fill_in_contribution_to_jacobian(Vector<double>& residuals,
128 DenseMatrix<double>& jacobian)
129 {
130 // Call the generic routine with the flag set to 1
131 fill_in_generic_residual_contribution_flux_transport(
132 residuals, jacobian, GeneralisedElement::Dummy_matrix, 1);
133 }
134
135 /// Add the element's contribution to its residuals vector,
136 /// jacobian matrix and mass matrix
137 void fill_in_contribution_to_jacobian_and_mass_matrix(
138 Vector<double>& residuals,
139 DenseMatrix<double>& jacobian,
140 DenseMatrix<double>& mass_matrix)
141 {
142 // Call the generic routine with the flag set to 2
143 fill_in_generic_residual_contribution_flux_transport(
144 residuals, jacobian, mass_matrix, 2);
145 }
146
147 /// Assemble the contributions to the mass matrix and residuals
148 void fill_in_contribution_to_mass_matrix(Vector<double>& residuals,
149 DenseMatrix<double>& mass_matrix)
150 {
151 fill_in_generic_residual_contribution_flux_transport(
152 residuals, GeneralisedElement::Dummy_matrix, mass_matrix, 3);
153 }
