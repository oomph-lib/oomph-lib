43  ///Base class for the flux transport equations templated by the
44  ///dimension DIM. The equations that are solved are
45  /// \f[ \frac{\partial u_{i}}{\partial t} + \frac{\partial}{\partial x_{j}}
46  /// \left(F_{ij}(u_{k})\right), \f]
47  /// where \f$ F_{ij} \f$ is a matrix of flux components.
48  //==================================================================
49 template<unsigned DIM>
50 class FluxTransportEquations : public virtual FiniteElement
51 {
52 
53 protected:
54 
55  /// \short Return the number of fluxes (default zero)
56  virtual inline unsigned nflux() const {return 0;}
57 
58  /// \short Return the index at which the i-th unknown value
59  /// is stored. The default value, i, is appropriate for single-physics
60  /// problems.
61  /// In derived multi-physics elements, this function should be overloaded
62  /// to reflect the chosen storage scheme. Note that these equations require
63  /// that the unknowns are always stored at the same indices at each node.
64  virtual inline unsigned u_index_flux_transport(const unsigned &i)
65  const {return i;}
66 
67  /// \short Return the flux as a function of the unknown. This interface
68  /// could (should) be generalised)

70  {
71  std::ostringstream error_stream;
72  error_stream << "Default empty flux function called\n"

74  << "in a derived class\n";
75  throw OomphLibError(error_stream.str(),
76  OOMPH_CURRENT_FUNCTION,

78  }
