43 /// Set up node update info for (newly created) algebraic node: Work out its
44 /// node update information by interpolation from its father element,
45 /// based on pointer to father element and its local coordinate in
46 /// the father element. We're creating the node update info
47 /// for update functions that are shared by all nodes in the
48 /// father element.
49 //========================================================================
50 void AlgebraicElementBase::setup_algebraic_node_update(
51  Node*& node_pt, const Vector<double>& s_father, FiniteElement* father_el_pt)
52  const
53 {
54 
55  // Turn node into algebraic node
56  AlgebraicNode* alg_node_pt=dynamic_cast<AlgebraicNode*>(node_pt);
57 
58  // Get number of nodes in father element
59  unsigned nnod_father=father_el_pt->nnode();
60 
61  // Create map that stores the number of times a node update fct id
62  // has been encountered
63  std::map<int,unsigned> id_count;
64 
65  // Loop over all nodes in father element to extract node update ids
66  Vector<int> id;
67  for (unsigned j=0;j<nnod_father;j++)
68  {
69  // Get vector of ids and add them to map
70  dynamic_cast<AlgebraicNode*>(father_el_pt->node_pt(j))->
71  node_update_fct_id(id);
72  unsigned n_id=id.size();
73  for (unsigned i=0;i<n_id;i++)
74  {
75  id_count[id[i]]++;
76  }
77  }
78 
79  /// Now loop over the ids and check if they appear in all nodes
80  Vector<int> shared_ids;
81  typedef std::map<int,unsigned>::iterator IT;
82  for (IT it=id_count.begin();it!=id_count.end();it++)
83  {
84  if (it->second==nnod_father)
85  {
86  shared_ids.push_back(it->first);
87  }
88  }
89 
90 
91  // How many update functions we have?
92  unsigned n_update_id=shared_ids.size();
93 
94  // Loop over all node udate functions -- since it's shared by
95  // nodes we may as well read the required data from the first
96  // node in the father.
97  AlgebraicNode* father_node_pt=dynamic_cast<AlgebraicNode*>(father_el_pt->
98  node_pt(0));
99  for (unsigned i=0;i<n_update_id;i++)
100  {
101  // Actual id:
102  int id=shared_ids[i];
103 
104  // Is this a real update fct or the default dummy one?
105  if (id>=0)
106  {
107  // Get vector of geometric objects involved in specified node update
108  // function (create vector by copy operation)
109  Vector<GeomObject*> geom_obj_pt(father_node_pt->
110  vector_geom_object_pt(id));
111 
112  // Loop over reference values and obtain the
113  // ones for the current (son) element by interpolation from
114  // the father
115 
116  // Number of reference values for this udpate function
117  unsigned nvalue=father_node_pt->nref_value(id);
118  Vector<double> ref_value(nvalue);
119 
120  //Set up the shape functions in father element
121  Shape psi(nnod_father);
122 
123  // Get shape functions in father element
124  father_el_pt->shape(s_father,psi);
125 
126  // Initialise reference values
127  for (unsigned ivalue=0;ivalue<nvalue;ivalue++)
128  {
129  ref_value[ivalue]=0.0;
130  }
131 
132  // Loop over all nodes in father element for
133  // interpolation of nodes -- don't need to
134  // worry about hanging nodes here as reference values
135  // are only assigned once and then in a consistent way
136  for (unsigned j_father=0;j_father<nnod_father;j_father++)
137  {
138  // Get reference values at node in father by copy operation
139  Vector<double> father_ref_value(dynamic_cast<AlgebraicNode*>(
140  father_el_pt->node_pt(j_father))->
141  vector_ref_value(id));
142 
143  // Loop over reference values
144  for (unsigned ivalue=0;ivalue<nvalue;ivalue++)
145  {
146  ref_value[ivalue]+=father_ref_value[ivalue]*psi(j_father);
147  }
148  }
149 
150  // Get mesh that implements the update operation
151  AlgebraicMesh* mesh_pt=father_node_pt->mesh_pt(id);
152 
153 
154  // Setup node update info for node
155  alg_node_pt->add_node_update_info(
156  id, // id
157  mesh_pt, // mesh
158  geom_obj_pt, // vector of geom objects
159  ref_value); // vector of ref. values
160 
161  // Update the geometric references (e.g. in FSI) if required
162  mesh_pt->update_node_update(alg_node_pt);
163 
164  }// endif for real vs. default dummy node update fct
165 
166  } // end of loop over different update fcts
167 
168  // Update the node at its current and previous positions
169  bool update_all_time_levels_for_new_node=true;
170  alg_node_pt->node_update(update_all_time_levels_for_new_node);
171 
172 }

230 /// Excute the node update function: Update the current (and if
231 /// update_all_time_levels_for_new_node==true also the previous)
232 /// nodal position. Also update the current nodal values if
233 /// an auxiliary update function is defined.
234 /// Note: updating of previous positions is only required (and should
235 /// only be performed for) newly created AlgebraicNodes
236 /// i.e. when this function is called from
237 /// AlgebraicElementBase::setup_algebraic_node_update(...). We create
238 /// the history of its nodal positions from the time-dependent
239 /// version of the specific AlgebraicMesh's algebraic_node_update(...)
240 /// function.
241 //========================================================================
242  void AlgebraicNode::node_update(const bool&
243  update_all_time_levels_for_new_node)
244  {
245  // Number of time levels that need to be updated
246  unsigned ntime;
247  if (update_all_time_levels_for_new_node)
248  {
249  //Present value plus previous values
250  ntime= 1 + Position_time_stepper_pt->nprev_values();
251  }
252  else
253  {
254  //Present value only
255  ntime=1;
256  }
257 
258  // Is it a hanging node?
259  if (is_hanging())
260  {
261  // Loop over all master nodes and update their position
262  // That's all we need to update the position of hanging nodes!
263  // (Recall that for hanging nodes Node::x(...) is not
264  // guaranteed to be kept up-to-date; the (constrained!) nodal
265  // position of hanging nodes must be determined via
266  // Node::position() which determines the position
267  // via the hanging node constraints from the position of
268  // the master nodes)
269  unsigned nmaster=hanging_pt()->nmaster();
270  for (unsigned imaster=0;imaster<nmaster;imaster++)
271  {
272  dynamic_cast<AlgebraicNode*>(hanging_pt()->master_node_pt(imaster))
273  ->node_update();
274  }
275  }
276  // Node isn't hanging --> update it directly
277  else
278  {
279  // If no update function is defined, keep the nodal positions where
280  // they were (i.e. don't do anything), else update
281  if (nnode_update_fcts()!=0)
282  {
283  // Loop over time levels
284  for (unsigned t=0;t<ntime;t++)
285  {
286  // Update nodal position
287  AlgebraicMesh* mesh_pt=Mesh_pt.begin()->second;
288  // Not sure why I need this intermediate variable....
289  AlgebraicNode* node_pt=this;
290  mesh_pt->algebraic_node_update(t,node_pt);
291  }
292  }
293  }
294 
295  /// Perform auxiliary update of function values?
296  if (Aux_node_update_fct_pt!=0)
297  {
298  Aux_node_update_fct_pt(this);
299  }
300 
301  }
302

307 /// Perform self test: If the node has multiple update functions,
308 /// check that all update functions give the same result (with a tolerance of
309 /// AlgebraicNode::Max_allowed_difference_between_node_update_fcts
310 //========================================================================
311 unsigned AlgebraicNode::self_test()
312 {
313  // Initialise
314  bool passed=true;
315 
316  unsigned test=Node::self_test();
317  if (test!=0)
318  {
319  passed=false;
320  }
321 
322  // Loop over all update functions
323  unsigned nnode_update=nnode_update_fcts();
324 
325  // If there is just one (or no) update function, then no conflict
326  // can arise and the error is zero.
327  if (nnode_update<=1)
328  {
329  //return 0;
330  }
331  // Multiple update funcions: check consistency
332  else
333  {
334  // Initialise error
335  double err_max=0.0;
336 
337  //Spatial (Eulerian) position of the node
338  unsigned ndim_node=ndim();
339  Vector<double> x_0(ndim_node);
340  Vector<double> x_new(ndim_node);
341 
342  // Get vector of update fct ids
343  Vector<int> id;
344  node_update_fct_id(id);
345 
346 
347  // Quick consistency check
348 #ifdef PARANOID
349  if (id.size()!=nnode_update)
350  {
351  std::ostringstream error_stream;
352  error_stream << "Inconsistency between number of node update ids:"
353  << nnode_update << " and " << id.size()
354  << std::endl;
355  throw OomphLibError(error_stream.str(),
356  OOMPH_CURRENT_FUNCTION,
357  OOMPH_EXCEPTION_LOCATION);
358  }
359 #endif
360 
361 
362  // Update with first update function
363 
364  // Set default update function
365  set_default_node_update(id[0]);
366 
367  // Update the node:
368  node_update();
369 
370  // Store coordinates
371  for (unsigned i=0;i<ndim_node;i++)
372  {
373  x_0[i]=x(i);
374  }
375 
376 
377  // Loop over all other update functions
378  for (unsigned iupdate=1;iupdate<nnode_update;iupdate++)
379  {
380 
381  // Set default update function
382  set_default_node_update(id[iupdate]);
383 
384  // Update the node:
385  node_update();
386 
387  // Store coordinates
388  for (unsigned i=0;i<ndim_node;i++)
389  {
390  x_new[i]=x(i);
391  }
392 
393  // Check error
394  double err=0.0;
395  for (unsigned i=0;i<ndim_node;i++)
396  {
397  err+=(x_new[i]-x_0[i])*(x_new[i]-x_0[i]);
398  }
399  err=sqrt(err);
400  if (err>err_max)
401  {
402  err_max=err;
403  }
404 
405  // Dump out
406  if (err>Max_allowed_difference_between_node_update_fcts)
407  {
408  oomph_info << "Discrepancy in algebraic update function "
409  << iupdate << ": "
410  << x_0[0] << " " << x_0[1] << " "
411  << x_new[0] << " " << x_new[1] << std::endl;
412 
413  passed=false;
414  }
415 
416  }
417 
418 
419  // Update again with first update function to reset
420 
421  // Set default update function
422  set_default_node_update(id[0]);
423 
424  // Update the node:
425  node_update();
426 
427  // Return verdict
428  if (passed) {return 0;}
429  else {return 1;}
430 
431  }
432  //Catch all to remove compiler warning
433  return 0;
434 }
