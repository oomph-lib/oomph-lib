118  /// \short Fill in the residuals and jacobian for the volume constraint
119  void fill_in_contribution_to_jacobian(Vector<double> &residuals,
120  DenseMatrix<double> &jacobian)
121  {
122  //One contribution to jacobian; see comment in that function
123  this->fill_in_generic_contribution_to_residuals_normalisation(
124  residuals);
125 
126  const int local_eqn = this->eigenvalue_local_eqn(2);
127  if(local_eqn >= 0)
128  {
129  const int local_unknown = this->eigenvalue_local_eqn(0);
130  jacobian(local_eqn,local_unknown) += 1.0;
131  }
132  }

141  //No contribution to jacobian or mass matrix; see comment in that function
142  this->fill_in_generic_contribution_to_residuals_normalisation(
143  residuals);
144  }
