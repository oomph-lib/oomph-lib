127  /// \short applies the oomph-lib preconditioner. Converts the Epetra vector
128  /// applys the preconditioner by calling the oomph-lib preconditioner's
129  /// preconditioner_solve functionality.
130  /// NOTE : the oomph-lib preconditioner is setup prior to being passed to
131  /// this class
132  int ApplyInverse(const Epetra_MultiVector &epetra_r,
133  Epetra_MultiVector &epetra_z) const
134  {
135  // the oomph-lib vector for r
136  DoubleVector oomph_r;
137 
138  // copy the Epetra_MultiVector r into an oomph-lib vector
139  double** r_pt;
140  epetra_r.ExtractView(&r_pt);
141  if (Use_epetra_values)
142  {
143  oomph_r.set_external_values
144  (Oomph_lib_preconditioner_pt->distribution_pt(),*r_pt,false);
145  }
146  else
147  {
148  oomph_r.build(Oomph_lib_preconditioner_pt->distribution_pt(),0.0);
149  unsigned nrow_local =
150  Oomph_lib_preconditioner_pt->distribution_pt()->nrow_local();
151  for (unsigned i = 0; i < nrow_local; i++)
152  {
153  oomph_r[i] = r_pt[0][i];
154  }
155  }
156 
157  // oomph-lib vector for Y
158  DoubleVector oomph_z;
159  if (Use_epetra_values)
160  {
161  double** z_pt;
162  epetra_z.ExtractView(&z_pt);
163  DoubleVector oomph_z;
164  oomph_z.set_external_values(
165  Oomph_lib_preconditioner_pt->distribution_pt(),
166  *z_pt,false);
167  }
168  else
169  {
170  oomph_z.build(Oomph_lib_preconditioner_pt->distribution_pt(),0.0);
171  }
172 
173  // apply the preconditioner
174  Oomph_lib_preconditioner_pt->preconditioner_solve(oomph_r,oomph_z);
175 
176  // if not using external data copy the oomph-lib vector oomph_Y back to Y
177  if (!Use_epetra_values)
178  {
179  unsigned nrow_local =
180  Oomph_lib_preconditioner_pt->distribution_pt()->nrow_local();
181  for (unsigned i = 0; i < nrow_local; i++)
182  {
183  epetra_z.ReplaceMyValue(i,0,oomph_z[i]);
184  }
185  }
186 
187  // return 0 to indicate success
188  return 0;
189  }
