43  /// A base class that represents the fourth-rank elasticity tensor
44  /// \f$E_{ijkl}\f$ defined such that
45  /// \f[\tau_{ij} = E_{ijkl} e_{kl},\f]
46  /// where \f$e_{ij}\f$ is the infinitessimal (Cauchy) strain tensor
47  /// and \f$\tau_{ij}\f$ is the stress tensor. The symmetries of the
48  /// tensor are such that
49  /// \f[E_{ijkl} = E_{jikl} = E_{ijlk} = E_{klij}\f]
50  /// and thus there are relatively few independent components. These
51  /// symmetries are included in the definition of the object so that
52  /// non-physical symmetries cannot be accidentally imposed.
53  //=====================================================================
54 class ElasticityTensor
55  {
56 
57  protected:
58 
59  ///\short Translation table from the four indices to the corresponding
60  ///independent component
61  static const unsigned Index[3][3][3][3];
62 
63  ///Member function that returns the i-th independent component of the
64  ///elasticity tensor
65  virtual inline double independent_component(const unsigned &i) const
66  {return 0.0;}
67 
68  ///\short Helper range checking function
69  /// (Note that this only captures over-runs in 3D but
70  /// errors are likely to be caught in evaluation of the
71  /// stress and strain tensors anyway...)
72  void range_check(const unsigned &i, const unsigned &j,
73  const unsigned &k, const unsigned &l) const
74  {
75  if((i > 2) || (j > 2) || (k> 2) || (l>2))
76  {
77  std::ostringstream error_message;
78  if(i > 2)
79  {
80  error_message << "Range Error : Index 1 " << i
81  << " is not in the range (0,2)";
82  }
83  if(j > 2)
84  {
85  error_message << "Range Error : Index 2 " << j
86  << " is not in the range (0,2)";
87  }
88 
89  if(k > 2)
90  {
91  error_message << "Range Error : Index 2 " << k
92  << " is not in the range (0,2)";
93  }
94 
95  if(l > 2)
96  {
97  error_message << "Range Error : Index 4 " << l
98  << " is not in the range (0,2)";
99  }
100 
101  //Throw the error
102  throw OomphLibError(error_message.str(),
103  OOMPH_CURRENT_FUNCTION,
104  OOMPH_EXCEPTION_LOCATION);
105  }
106  }

148 /// An isotropic elasticity tensor defined in terms of Young's modulus
149 /// and Poisson's ratio. The elasticity tensor is assumed to be
150 /// non-dimensionalised on some reference value for Young's modulus
151 /// so the value provided to the constructor (if any) is to be
152 /// interpreted as the ratio of the actual Young's modulus to the
153 /// Young's modulus used to non-dimensionalise the stresses/tractions
154 /// in the governing equations.
155 //===================================================================
156  class IsotropicElasticityTensor : public ElasticityTensor
157  {
158  //Storage for the independent components of the elasticity tensor
159  double C[4];
160 
161  //Translation scheme between the 21 independent components of the general
162  //elasticity tensor and the isotropic case
163  static const unsigned StaticIndex[21];
164 
165  public:
166 
167  /// \short Constructor. Passing in the values of the Poisson's ratio
168  /// and Young's modulus (interpreted as the ratio of the actual
169  /// Young's modulus to the Young's modulus (or other reference stiffness)
170  /// used to non-dimensionalise stresses and tractions in the governing
171  /// equations).
172  IsotropicElasticityTensor(const double &nu,
173  const double &E) : ElasticityTensor()
174  {
175  //Set the three indepdent components
176  C[0] = 0.0;
177  double lambda=E*nu/((1.0+nu)*(1.0-2.0*nu));
178  double mu=E/(2.0*(1.0+nu));
179  this->set_lame_coefficients(lambda,mu);
180  }
181 
182  /// \short Constructor. Passing in the value of the Poisson's ratio.
183  /// Stresses and tractions in the governing equations are assumed
184  /// to have been non-dimensionalised on Young's modulus.
185  IsotropicElasticityTensor(const double &nu) : ElasticityTensor()
186  {
187  //Set the three indepdent components
188  C[0] = 0.0;
189 
190  // reference value
191  double E=1.0;
192  double lambda=E*nu/((1.0+nu)*(1.0-2.0*nu));
193  double mu=E/(2.0*(1.0+nu));
194  this->set_lame_coefficients(lambda,mu);
195  }
196 
197  /// \short Constructur. Passing in the values of the two lame
198  /// coefficients directly (interpreted as the ratios of these
199  /// quantities to a reference stiffness used to non-dimensionalised
200  IsotropicElasticityTensor(const Vector<double> &lame)
201  {
202  //Set the three independent componens
203  C[0] = 0.0;
204  this->set_lame_coefficients(lame[0],lame[1]);
205  }
206 
207  /// \short Update parameters: Specify values of the Poisson's ratio
208  /// and (optionally) Young's modulus (interpreted as the ratio of the actual
209  /// Young's modulus to the Young's modulus (or other reference stiffness)
210  /// used to non-dimensionalise stresses and tractions in the governing
211  /// equations).
212  void update_constitutive_parameters(const double &nu,
213  const double &E=1.0)
214  {
215  //Set the three indepdent components
216  C[0] = 0.0;
217  double lambda=E*nu/((1.0+nu)*(1.0-2.0*nu));
218  double mu=E/(2.0*(1.0+nu));
219  this->set_lame_coefficients(lambda,mu);
220  }
221 
222 
223  ///Overload the independent coefficient function
224  inline double independent_component(const unsigned &i) const
225  {return C[StaticIndex[i]];}
226 
227 
228  private:
229 
230  //Set the values of the lame coefficients
231  void set_lame_coefficients(const double &lambda, const double &mu)
232  {
233  C[1] = lambda + 2.0*mu;
234  C[2] = lambda;
235  C[3] = mu;
236  }
237 
238  };
