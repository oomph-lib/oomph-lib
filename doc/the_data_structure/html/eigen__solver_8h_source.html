56 //=======================================================================
57 /// Base class for all EigenProblem solves. This simply defines standard
58 /// interfaces so that different solvers can be used easily.
59 //=======================================================================
60 class EigenSolver : public DistributableLinearAlgebraObject
61 {
62 protected:
63 /// Double value that represents the real part of the shift in
64 /// shifted eigensolvers
65 double Sigma_real;
66
67 public:
68 /// Empty constructor
69 EigenSolver() : Sigma_real(0.0) {}
70
71 /// Empty copy constructor
72 EigenSolver(const EigenSolver&) {}
73
74 /// Empty destructor
75 virtual ~EigenSolver() {}
76
77 /// Actual eigensolver. This takes a pointer to a problem and returns
78 /// a vector of complex numbers representing the eigenvalues
79 /// and a corresponding vector of eigenvectors
80 virtual void solve_eigenproblem(Problem* const& problem_pt,
81 const int& n_eval,
82 Vector<std::complex<double>>& eigenvalue,
83 Vector<DoubleVector>& eigenvector) = 0;
84
85
86 /// Set the value of the shift
87 void set_shift(const double& shift_value)
88 {
89 Sigma_real = shift_value;
90 }
91
92 /// Return the value of the shift (const version)
93 const double& get_shift() const
94 {
95 return Sigma_real;
96 }
97 };
