35  /// Trapezoid rule time stepping scheme.
36  ///
37  /// This method requires a value of dy/dt at the initial time. The
38  /// implementation of this calculation is exactly the same as is used for
39  /// explicit time stepping.
40  ///
41  /// The function setup_initial_derivative(Problem* problem_pt) should be
42  /// called after the initial conditions have been set, but before beginning
43  /// time stepping, to compute this initial value of dy/dt.
44  ///
45  /// Warning: moving nodes not implemented (I have no test case).
46  class TR : public TimeStepper
47  {
48  // The standard trapezoid rule is:
49 
50  // y_{n+1} = y_n + dt_n (f(t_n, y_n) + f(t_{n+1}, y_{n+1}))/2
51 
52  // where y is the vector of nodal values and f is the derivative function
53  // (as in standard ODE theory). However we want to calculate time steps
54  // using only one function evaluation (i.e. f) per time step because
55  // function evaluations are expensive. So we require f(t_0, y_0) as
56  // initial input and at each step store f(t_{n+1}, y_{n+1}) as a history
57  // value for use in the calculatio of the next time step.
58 
59 
60  public:
61  /// Constructor, storage for two history derivatives (one for TR and
62  /// one for the predictor step), one history value, present value and
63  /// predicted value.
64  TR(const bool& adaptive=false) : TimeStepper(2+2+1, 1)
65  {
66  //Set the weight for the zero-th derivative
67  Weight(0,0) = 1.0;
68 
69  Initial_derivative_set = false;
70 
71  Adaptive_Flag = adaptive;
72 
73  // Initialise adaptivity stuff
74  Predictor_weight.assign(4, 0.0);
75  Error_weight = 0.0;
76 
77  Shift_f = true;
78  }

86  /// Set the weights
87  void set_weights()
88  {
89  double dt = Time_pt->dt(0);
90  Weight(1,0) = 2.0/dt;
91  Weight(1,1) = -2.0/dt;
92  Weight(1,derivative_index(0)) = -1.0; // old derivative
93  }
94 
95  void set_error_weights()
96  {
97  double dt=Time_pt->dt(0);
98  double dtprev=Time_pt->dt(1);
99  Error_weight = 1/(3*(1 + (dtprev/dt)));
100  }
101 
102  /// Function to set the predictor weights
103  void set_predictor_weights()
104  {
105  // Read the value of the time steps
106  double dt=Time_pt->dt(0);
107  double dtprev=Time_pt->dt(1);
108  double dtr = dt/dtprev;
109 
110  // y weights
111  Predictor_weight[0] = 0.0;
112  Predictor_weight[1] = 1.0;
113 
114  // dy weights
115  Predictor_weight[derivative_index(0)] = (dt/2)*(2 + dtr);
116  Predictor_weight[derivative_index(1)] = -(dt/2)*dtr;
117  }

176  void setup_initial_derivative(Problem* problem_pt)
177  {
178  oomph_info << "Solving for derivative at initial time."
179  << " Warning: if residual is not in the correct form this may fail."
180  << std::endl;
181 
182  // Get the derivative at initial time
183  DoubleVector f0;
184  problem_pt->get_dvaluesdt(f0);
185 
186  // store in derivatives slot ready for use in timestepping.
187  problem_pt->set_dofs(this->derivative_index(0), f0);
188 
189  Initial_derivative_set = true;
190  Shift_f = false;
191  }

194  /// \short This function updates the Data's time history so that
195  /// we can advance to the next timestep.
196  void shift_time_values(Data* const &data_pt)
197  {
198  //Find number of values stored
199  unsigned n_value = data_pt->nvalue();
200 
201  // Get previous step dt, time_pt has already been shifted so it's in
202  // slot 1.
203  double dtn = time_pt()->dt(1);
204 
205  //Loop over the values
206  for(unsigned j=0; j<n_value; j++)
207  {
208  //Set previous values to the previous value, if not a copy
209  if(data_pt->is_a_copy(j) == false)
210  {
211  if(Shift_f)
212  {
213  // Calculate time derivative at step n by rearranging the TR
214  // formula.
215  double fnm1 = data_pt->value(derivative_index(0), j);
216  double ynm1 = data_pt->value(1, j);
217  double yn = data_pt->value(0, j);
218  double fn = (2/dtn)*(yn - ynm1) - fnm1;
219 
220  data_pt->set_value(derivative_index(0), j, fn);
221 
222  // Shift time derivative
223  data_pt->set_value(derivative_index(1), j, fnm1);
224  }
225  else
226  {
227  std::cout << "didn't shift derivatives" << std::endl;
228  }
229 
230  // Shift value
231  data_pt->set_value(1, j,
232  data_pt->value(0, j));
233  }
234  }
235 
236  }

252  /// Function to calculate predicted positions at a node
253  void calculate_predicted_positions(Node* const &node_pt)
254  {
255  // Loop over the dimensions
256  unsigned n_dim = node_pt->ndim();
257  for(unsigned j=0;j<n_dim;j++)
258  {
259  // If the node is not copied
260  if(!node_pt->position_is_a_copy(j))
261  {
262  // Initialise the predictor to zero
263  double predicted_value = 0.0;
264 
265  // Loop over all the stored data and add appropriate values
266  // to the predictor
267  for(unsigned i=1;i<4;i++)
268  {
269  predicted_value += node_pt->x(i,j)*Predictor_weight[i];
270  }
271 
272  // Store the predicted value
273  node_pt->x(predicted_value_index(), j) = predicted_value;
274  }
275  }
276  }

278  /// Function to calculate predicted data values in a Data object
279  void calculate_predicted_values(Data* const &data_pt)
280  {
281  // Loop over the values
282  unsigned n_value = data_pt->nvalue();
283  for(unsigned j=0;j<n_value;j++)
284  {
285  // If the value is not copied
286  if(!data_pt->is_a_copy(j))
287  {
288  // Loop over all the stored data and add appropriate
289  // values to the predictor
290  double predicted_value = 0.0;
291  for(unsigned i=1;i<4;i++)
292  {
293  predicted_value += data_pt->value(i,j)*Predictor_weight[i];
294  }
295 
296  // Store the predicted value
297  data_pt->set_value(predicted_value_index(), j, predicted_value);
298  }
