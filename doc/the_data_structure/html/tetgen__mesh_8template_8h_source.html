249  /// \short Build mesh, based on a TetgenMeshFactedClosedSurface that
250  /// specifies the outer boundary of the domain and any number of internal
251  /// boundaries, specified by TetMeshFacetedSurfaces.
252  /// Also specify target size for uniform element size.
253  TetgenMesh(TetMeshFacetedClosedSurface* const& outer_boundary_pt,
254  Vector<TetMeshFacetedSurface*>& internal_surface_pt,
255  const double& element_volume,
256  TimeStepper* time_stepper_pt = &Mesh::Default_TimeStepper,
257  const bool& use_attributes = false,
258  const bool& split_corner_elements = false)
259  {
260  // Mesh can only be built with 3D Telements.
261  MeshChecker::assert_geometric_element<TElementGeometricBase, ELEMENT>(3);
262 
263  // Store the attributes
264  Use_attributes = use_attributes;
265 
266  // Store timestepper used to build elements
267  Time_stepper_pt = time_stepper_pt;
268 
269  // Copy across
270  Outer_boundary_pt = outer_boundary_pt;
271  // Setup the reverse lookup scheme
272  this->setup_reverse_lookup_schemes_for_faceted_surface(Outer_boundary_pt);
273  // Store the internal boundary
274  Internal_surface_pt = internal_surface_pt;
275  // Setup the reverse lookup schemes
276  {
277  unsigned n = this->Internal_surface_pt.size();
278  for (unsigned i = 0; i < n; i++)
279  {
280  this->setup_reverse_lookup_schemes_for_faceted_surface(
281  Internal_surface_pt[i]);
282  }
283  }
284 
285  // Tetgen data structure for the input and output
286  tetgenio in;
287  this->build_tetgenio(outer_boundary_pt, internal_surface_pt, in);
288 
289 
290  // Now tetrahedralise
291  std::stringstream input_string;
292  input_string << "pAa" << element_volume; // Andrew's parameters
293 
294  // input_string << "Vpq1.414Aa" << element_volume;
295  // << "Q"; // Q for quiet!
296  // << "V"; // V for verbose incl. quality output!
297 
298  // If any of the boundaries should not be split add the "Y" flag
299  bool can_boundaries_be_split =
300  outer_boundary_pt->boundaries_can_be_split_in_tetgen();
301  {
302  unsigned n_internal = internal_surface_pt.size();
303  for (unsigned i = 0; i < n_internal; i++)
304  {
305  can_boundaries_be_split &=
306  internal_surface_pt[i]->boundaries_can_be_split_in_tetgen();
307  }
308  }
309 
310  // If we can't split the boundaries add the flag
311  if (can_boundaries_be_split == false)
312  {
313  input_string << "Y";
314  }
315 
316  // Now convert to a C-style string
317  char tetswitches[100];
318  sprintf(tetswitches, "%s", input_string.str().c_str());
319 
320  // Make a new tetgen representation
321  this->Tetgenio_exists = true;
322  Tetgenio_pt = new tetgenio;
323  tetrahedralize(tetswitches, &in, this->Tetgenio_pt);
324 
325  // Build scaffold
326  Tmp_mesh_pt = new TetgenScaffoldMesh(*this->Tetgenio_pt);
327 
328  // If any of the objects are different regions then we need to use
329  // the atributes
330  bool regions_exist = false;
331  {
332  unsigned n_internal = internal_surface_pt.size();
333  for (unsigned i = 0; i < n_internal; i++)
334  {
335  TetMeshFacetedClosedSurface* srf_pt =
336  dynamic_cast<TetMeshFacetedClosedSurface*>(internal_surface_pt[i]);
337  if (srf_pt != 0)
338  {
339  unsigned n_int_pts = srf_pt->ninternal_point_for_tetgen();
340  for (unsigned j = 0; j < n_int_pts; j++)
341  {
342  regions_exist |=
343  srf_pt->internal_point_identifies_region_for_tetgen(j);
344  }
345  }
346  }
347  }
348 
349  // If there are regions, use the attributes
350  if (regions_exist)
351  {
352  Use_attributes = true;
353  }
354 
355  // Convert mesh from scaffold to actual mesh
356  build_from_scaffold(time_stepper_pt, Use_attributes);
357 
358  // Kill the scaffold
359  delete Tmp_mesh_pt;
360  Tmp_mesh_pt = 0;
361 
362  // Split corner elements
363  if (split_corner_elements)
364  {
365  split_elements_in_corners<ELEMENT>();
366  }
367 
368  // Setup boundary coordinates
369  unsigned nb = nboundary();
370  for (unsigned b = 0; b < nb; b++)
371  {
372  bool switch_normal = false;
373  setup_boundary_coordinates<ELEMENT>(b, switch_normal);
374  }
375 
376  // Now snap onto geometric objects associated with triangular facets
377  // (if any!)
378  snap_nodes_onto_geometric_objects();
379  }

381  ///\short Build tetgenio object from the TetMeshFacetedSurfaces
382  void build_tetgenio(TetMeshFacetedSurface* const& outer_boundary_pt,
383  Vector<TetMeshFacetedSurface*>& internal_surface_pt,
384  tetgenio& tetgen_io)
385  {
386  // Pointer to Tetgen facet
387  tetgenio::facet* f;
388  // Pointer to Tetgen polygon
389  tetgenio::polygon* p;
390 
391  // Start all indices from 1 (it's a choice and we've made it
392  tetgen_io.firstnumber = 1;
393  /// ALH: This may not be needed
394  tetgen_io.useindex = true;
395 
396  // Find the number of internal surfaces
397  const unsigned n_internal = internal_surface_pt.size();
398 
399  // Find the number of points on the outer boundary
400  const unsigned n_outer_vertex = outer_boundary_pt->nvertex();
401  tetgen_io.numberofpoints = n_outer_vertex;
402 
403  // Find the number of points on the inner boundaries and add to the totals
404  Vector<unsigned> n_internal_vertex(n_internal);
405  Vector<unsigned> internal_vertex_offset(n_internal);
406  for (unsigned h = 0; h < n_internal; ++h)
407  {
408  n_internal_vertex[h] = internal_surface_pt[h]->nvertex();
409  internal_vertex_offset[h] = tetgen_io.numberofpoints;
410  tetgen_io.numberofpoints += n_internal_vertex[h];
411  }
412 
413  // Read the data into the point list
414  tetgen_io.pointlist = new double[tetgen_io.numberofpoints * 3];
415  tetgen_io.pointmarkerlist = new int[tetgen_io.numberofpoints];
416  unsigned counter = 0;
417  for (unsigned n = 0; n < n_outer_vertex; ++n)
418  {
419  for (unsigned i = 0; i < 3; ++i)
420  {
421  tetgen_io.pointlist[counter] =
422  outer_boundary_pt->vertex_coordinate(n, i);
423  ++counter;
424  }
425  }
426  for (unsigned h = 0; h < n_internal; ++h)
427  {
428  const unsigned n_inner = n_internal_vertex[h];
429  for (unsigned n = 0; n < n_inner; ++n)
430  {
431  for (unsigned i = 0; i < 3; ++i)
432  {
433  tetgen_io.pointlist[counter] =
434  internal_surface_pt[h]->vertex_coordinate(n, i);
435  ++counter;
436  }
437  }
438  }
439 
440 
441  // Set up the pointmarkers
442  counter = 0;
443  for (unsigned n = 0; n < n_outer_vertex; ++n)
444  {
445  tetgen_io.pointmarkerlist[counter] =
446  outer_boundary_pt->one_based_vertex_boundary_id(n);
447  ++counter;
448  }
449  for (unsigned h = 0; h < n_internal; ++h)
450  {
451  const unsigned n_inner = n_internal_vertex[h];
452  for (unsigned n = 0; n < n_inner; ++n)
453  {
454  tetgen_io.pointmarkerlist[counter] =
455  internal_surface_pt[h]->one_based_vertex_boundary_id(n);
456  ++counter;
457  }
458  }
459 
460 
461  // Now the facets
462  unsigned n_outer_facet = outer_boundary_pt->nfacet();
463  tetgen_io.numberoffacets = n_outer_facet;
464  Vector<unsigned> n_inner_facet(n_internal);
465  for (unsigned h = 0; h < n_internal; ++h)
466  {
467  n_inner_facet[h] = internal_surface_pt[h]->nfacet();
468  tetgen_io.numberoffacets += n_inner_facet[h];
469  }
470 
471  tetgen_io.facetlist = new tetgenio::facet[tetgen_io.numberoffacets];
472  tetgen_io.facetmarkerlist = new int[tetgen_io.numberoffacets];
473 
474 
475  counter = 0;
476  for (unsigned n = 0; n < n_outer_facet; ++n)
477  {
478  // Set pointer to facet
479  f = &tetgen_io.facetlist[counter];
480  f->numberofpolygons = 1;
481  f->polygonlist = new tetgenio::polygon[f->numberofpolygons];
482  f->numberofholes = 0;
483  f->holelist = NULL;
484  p = &f->polygonlist[0];
485 
486  Vector<unsigned> facet = outer_boundary_pt->vertex_index_in_tetgen(n);
487 
488  p->numberofvertices = facet.size();
489  p->vertexlist = new int[p->numberofvertices];
490  for (int i = 0; i < p->numberofvertices; ++i)
491  {
492  // The offset here is because we have insisted on 1-based indexing
493  p->vertexlist[i] = facet[i] + 1;
494  }
495 
496  // Set up the boundary markers
497  tetgen_io.facetmarkerlist[counter] =
498  outer_boundary_pt->one_based_facet_boundary_id(n);
499  // Increase the counter
500  ++counter;
501  }
502 
503  // Initialise the number of holes
504  tetgen_io.numberofholes = 0;
505  // and the number of regions
506  tetgen_io.numberofregions = 0;
507 
508  // Loop over the internal stuff
509  for (unsigned h = 0; h < n_internal; ++h)
510  {
511  for (unsigned n = 0; n < n_inner_facet[h]; ++n)
512  {
513  // Set pointer to facet
514  f = &tetgen_io.facetlist[counter];
515  f->numberofpolygons = 1;
516  f->polygonlist = new tetgenio::polygon[f->numberofpolygons];
517  f->numberofholes = 0;
518  f->holelist = NULL;
519  p = &f->polygonlist[0];
520 
521  Vector<unsigned> facet =
522  internal_surface_pt[h]->vertex_index_in_tetgen(n);
523 
524  p->numberofvertices = facet.size();
525  p->vertexlist = new int[p->numberofvertices];
526  for (int i = 0; i < p->numberofvertices; ++i)
527  {
528  // Add the 1-based and vertex offsets to get these number correct
529  p->vertexlist[i] = facet[i] + internal_vertex_offset[h] + 1;
530  }
531  // Set up the boundary markers
532  tetgen_io.facetmarkerlist[counter] =
533  internal_surface_pt[h]->one_based_facet_boundary_id(n);
534  ++counter;
535  }
536 
537  // If it's a hole/region add it
538  TetMeshFacetedClosedSurface* srf_pt =
539  dynamic_cast<TetMeshFacetedClosedSurface*>(internal_surface_pt[h]);
540  if (srf_pt != 0)
541  {
542  unsigned n_int_pts = srf_pt->ninternal_point_for_tetgen();
543  for (unsigned j = 0; j < n_int_pts; j++)
544  {
545  if (srf_pt->internal_point_identifies_hole_for_tetgen(j))
546  {
547  ++tetgen_io.numberofholes;
548  }
549  // Otherwise it may be region
550  else
551  {
552  if (srf_pt->internal_point_identifies_region_for_tetgen(j))
553  {
554  ++tetgen_io.numberofregions;
555  }
556  }
557  }
558  }
559  }
560 
561  // Set storage for the holes
562  tetgen_io.holelist = new double[3 * tetgen_io.numberofholes];
563 
564  // Loop over all the internal boundaries again
565  counter = 0;
566  for (unsigned h = 0; h < n_internal; ++h)
567  {
568  TetMeshFacetedClosedSurface* srf_pt =
569  dynamic_cast<TetMeshFacetedClosedSurface*>(internal_surface_pt[h]);
570  if (srf_pt != 0)
571  {
572  unsigned n_int_pts = srf_pt->ninternal_point_for_tetgen();
573  for (unsigned j = 0; j < n_int_pts; j++)
574  {
575  if (srf_pt->internal_point_identifies_hole_for_tetgen(j))
576  {
577  for (unsigned i = 0; i < 3; ++i)
578  {
579  tetgen_io.holelist[counter] =
580  srf_pt->internal_point_for_tetgen(j, i);
581  ++counter;
582  }
583  }
584  }
585  }
586  }
587 
588  // Set storage for the regions
589  tetgen_io.regionlist = new double[5 * tetgen_io.numberofregions];
590  // Loop over all the internal boundaries again
591  counter = 0;
592  for (unsigned h = 0; h < n_internal; ++h)
593  {
594  TetMeshFacetedClosedSurface* srf_pt =
595  dynamic_cast<TetMeshFacetedClosedSurface*>(internal_surface_pt[h]);
596  if (srf_pt != 0)
597  {
598  unsigned n_int_pts = srf_pt->ninternal_point_for_tetgen();
599  for (unsigned j = 0; j < n_int_pts; j++)
600  {
601  if (srf_pt->internal_point_identifies_region_for_tetgen(j))
602  {
603  for (unsigned i = 0; i < 3; ++i)
604  {
605  tetgen_io.regionlist[counter] =
606  srf_pt->internal_point_for_tetgen(j, i);
607  ++counter;
608  }
609  tetgen_io.regionlist[counter] =
610  static_cast<double>(srf_pt->region_id_for_tetgen(j));
611  ++counter;
612  // Area target
613  tetgen_io.regionlist[counter] = 0.0;
614  ++counter;
615  }
616  }
617  }
618  }
619  }
