282  /// computes the error
283  void compute_error(std::ostream &outfile,
284  FiniteElement::SteadyExactSolutionFctPt
285  exact_soln_pt, double& error, double& norm)
286  {
287 
288  // Initialise
289  error=0.0;
290  norm=0.0;
291 
292  //Vector of local coordinates
293  Vector<double> s(DIM);
294 
295  // Vector for coordintes
296  Vector<double> x(DIM);
297 
298  //Find out how many nodes there are in the element
299  unsigned n_node = this->nnode();
300 
301  Shape psi(n_node);
302 
303  //Set the value of n_intpt
304  unsigned n_intpt = this->integral_pt()->nweight();
305 
306  // Tecplot header info
307  outfile << this->tecplot_zone_string(3);
308 
309  // Tecplot
310 // outfile << "ZONE" << std::endl;
311 
312  // Exact solution Vector (here a scalar)
313  Vector<double> exact_soln(1);
314 
315  //Loop over the integration points
316  for(unsigned ipt=0;ipt<n_intpt;ipt++)
317  {
318 
319  //Assign values of s
320  for(unsigned i=0;i<DIM;i++)
321  {
322  s[i] = this->integral_pt()->knot(ipt,i);
323  }
324 
325  //Get the integral weight
326  double w = this->integral_pt()->weight(ipt);
327 
328  // Get jacobian of mapping
329  double J=this->J_eulerian(s);
330 
331  //Premultiply the weights and the Jacobian
332  double W = w*J;
333 
334  // Get x position as Vector
335  this->interpolated_x(s,x);
336 
337  // Get FE function value
338  double u_fe=interpolated_u_biharmonic(s);
339 
340  // Get exact solution at this point
341  (*exact_soln_pt)(x,exact_soln);
342 
343  //Output x,y,...,error
344  for(unsigned i=0;i<DIM;i++)
345  {
346  outfile << x[i] << " ";
347  }
348  outfile << exact_soln[0] << " " << fabs(exact_soln[0]-u_fe)
349  << std::endl;
350 
351  // Add to error and norm
352  norm+=exact_soln[0]*exact_soln[0]*W;
353  error+=(exact_soln[0]-u_fe)*(exact_soln[0]-u_fe)*W;
354  }
355 
356  this->write_tecplot_zone_footer(outfile,3);
357  }

370  /// calculates interpolated u at s
371  double interpolated_u_biharmonic(const Vector<double>& s)
372  {
373  // initialise storage for u_interpolated
374  double uu=0;
375 
376  // number of nodes
377  unsigned n_node=this->nnode();
378 
379  // number of degrees of freedom per node
380  unsigned n_value=this->node_pt(0)->nvalue();
381 
382  // set up memory for shape functions
383  Shape psi(n_node,n_value);
384 
385  // find shape fn at position s
386  this->shape(s,psi);
387 
388  // calculate interpolated u
389  for (unsigned n = 0; n < n_node; n++)
390  {
391  for (unsigned k = 0; k < n_value; k++)
392  {
393  uu += u(n,k)*psi(n,k);
394  }
395  }
396 
397  // return interpolated_u
398  return uu;
