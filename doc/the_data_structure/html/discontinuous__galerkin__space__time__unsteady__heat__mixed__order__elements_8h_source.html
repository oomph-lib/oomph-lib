65  /// A class for all isoparametric elements that solve the
66  /// SpaceTimeUnsteadyHeatMixedOrder equations.
67  /// \f[
68  /// \frac{\partial^2 u}{\partial x_i^2}=\frac{\partial u}{\partial t}+f(t,x_j)
69  /// \f]
70  /// This contains the generic maths. Shape functions, geometric
71  /// mapping etc. must get implemented in derived class.

453  /// Get flux: flux[i]=du/dx_i
454  void get_flux(const Vector<double>& s, Vector<double>& flux) const
455  {
456  // Find out how many nodes there are in the element
457  unsigned n_node=nnode();
458 
459  // Find the index at which the variable is stored
460  unsigned u_nodal_index=u_index_ust_heat();
461 
462  // Set up memory for the shape and test functions
463  Shape psi(n_node);
464 
465  // Set up memory for the derivatives of the shape functions
466  DShape dpsidx(n_node,SPATIAL_DIM+1);
467 
468  //------------dshape_eulerian(s,psi,dpsidx)----------------------------
469  // Find the element dimension
470  const unsigned el_dim=this->dim();
471 
472  // Get the values of the shape functions and their local derivatives;
473  // temporarily stored in dpsi
474  dshape_local_ust_heat(s,psi,dpsidx);
475 
476  // Allocate memory for the inverse jacobian
477  DenseMatrix<double> inverse_jacobian(el_dim);
478 
479  // Now calculate the inverse jacobian
480  local_to_eulerian_mapping(dpsidx,inverse_jacobian);
481 
482  // Now set the values of the derivatives to be dpsidx
483  transform_derivatives(inverse_jacobian,dpsidx);
484  //------------dshape_eulerian(s,psi,dpsidx)----------------------------
485 
486  // Loop over the entries of the flux vector
487  for (unsigned j=0; j<SPATIAL_DIM; j++)
488  {
489  // Initialise j-th flux entry to zero
490  flux[j]=0.0;
491  }
492 
493  // Loop over nodes
494  for (unsigned l=0; l<n_node; l++)
495  {
496  // Loop over derivative directions
497  for (unsigned j=0; j<SPATIAL_DIM; j++)
498  {
499  // Update the flux value
500  flux[j]+=nodal_value(l,u_nodal_index)*dpsidx(l,j);
501  }
502  } // for (unsigned l=0;l<n_node;l++)
503  } // End of get_flux
