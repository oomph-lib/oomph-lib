249  template<class ELEMENT>
250  FourierDecomposedHelmholtzFluxElement<ELEMENT>::
251  FourierDecomposedHelmholtzFluxElement(FiniteElement* const &bulk_el_pt,
252  const int &face_index) :
253  FaceGeometry<ELEMENT>(), FaceElement()
254  {
255  // Let the bulk element build the FaceElement, i.e. setup the pointers
256  // to its nodes (by referring to the appropriate nodes in the bulk
257  // element), etc.
258  bulk_el_pt->build_face_element(face_index,this);
259 
260  // Initialise the prescribed-flux function pointer to zero
261  Flux_fct_pt = 0;
262 
263 
264  // Initialise index at which real and imag unknowns are stored
265  U_index_fourier_decomposed_helmholtz = std::complex<unsigned>(0,1);
266 
267  // Now read out indices from bulk element
268  FourierDecomposedHelmholtzEquations* eqn_pt =
269  dynamic_cast<FourierDecomposedHelmholtzEquations*>(bulk_el_pt);
270  //If the cast has failed die
271  if(eqn_pt==0)
272  {
273  std::string error_string =
274  "Bulk element must inherit from FourierDecomposedHelmholtzEquations.";
275  throw OomphLibError(
276  error_string,
277  OOMPH_CURRENT_FUNCTION,
278  OOMPH_EXCEPTION_LOCATION);
279  }
280  else
281  {
282  //Read the index from the (cast) bulk element
283  U_index_fourier_decomposed_helmholtz =
284  eqn_pt->u_index_fourier_decomposed_helmholtz();
285  }

290 //===========================================================================
291 /// Compute the element's residual vector and the (zero) Jacobian matrix.
292 //===========================================================================
293  template<class ELEMENT>
294  void FourierDecomposedHelmholtzFluxElement<ELEMENT>::
295  fill_in_generic_residual_contribution_fourier_decomposed_helmholtz_flux(
296  Vector<double> &residuals, DenseMatrix<double> &jacobian,
297  const unsigned& flag)
298  {
299  //Find out how many nodes there are
300  const unsigned n_node = nnode();
301 
302  //Set up memory for the shape and test functions
303  Shape psif(n_node), testf(n_node);
304 
305  //Set the value of Nintpt
306  const unsigned n_intpt = integral_pt()->nweight();
307 
308  //Set the Vector to hold local coordinates
309  Vector<double> s(1);
310 
311  //Integers to hold the local equation and unknown numbers
312  int local_eqn_real=0 ,local_eqn_imag=0;
313 
314  //Loop over the integration points
315  //--------------------------------
316  for(unsigned ipt=0;ipt<n_intpt;ipt++)
317  {
318 
319  //Assign values of s
320  for(unsigned i=0;i<1;i++) {s[i] = integral_pt()->knot(ipt,i);}
321 
322  //Get the integral weight
323  double w = integral_pt()->weight(ipt);
324 
325  //Find the shape and test functions and return the Jacobian
326  //of the mapping
327  double J = shape_and_test(s,psif,testf);
328 
329  //Premultiply the weights and the Jacobian
330  double W = w*J;
331 
332  //Need to find position to feed into flux function, initialise to zero
333  Vector<double> interpolated_x(2,0.0);
334 
335  //Calculate coordinates
336  for(unsigned l=0;l<n_node;l++)
337  {
338  //Loop over coordinates
339  for(unsigned i=0;i<2;i++)
340  {
341  interpolated_x[i] += nodal_position(l,i)*psif[l];
342  }
343  }
344 
345  //first component
346  double r = interpolated_x[0];
347 
348  //Get the imposed flux
349  std::complex<double> flux(0.0,0.0);
350  get_flux(interpolated_x,flux);
351 
352  //Now add to the appropriate equations
353  //Loop over the test functions
354  for(unsigned l=0;l<n_node;l++)
355  {
356  local_eqn_real =
357  nodal_local_eqn(l,U_index_fourier_decomposed_helmholtz.real());
358 
359  /*IF it's not a boundary condition*/
360  if(local_eqn_real >= 0)
361  {
362  //Add the prescribed flux terms
363  residuals[local_eqn_real] -= flux.real()*testf[l]*r*W;
364 
365  // Imposed traction doesn't depend upon the solution,
366  // --> the Jacobian is always zero, so no Jacobian
367  // terms are required
368  }
369  local_eqn_imag =
370  nodal_local_eqn(l,U_index_fourier_decomposed_helmholtz.imag());
371 
372  /*IF it's not a boundary condition*/
373  if(local_eqn_imag >= 0)
374  {
375  //Add the prescribed flux terms
376  residuals[local_eqn_imag] -= flux.imag()*testf[l]*r*W;
377 
378  // Imposed traction doesn't depend upon the solution,
379  // --> the Jacobian is always zero, so no Jacobian
380  // terms are required
381  }
382  }
383  }
384  }
385 
386 }
