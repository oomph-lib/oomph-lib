195 /// \short Class for a compressed-matrix coefficent (for either CC or CR
196 /// matrices). Contains the (row or column) index and value of a
197 /// coefficient in a compressed row or column.
198 /// Currently only used in ILU(0) for CCDoubleMatrices to allow the
199 /// coefficients in each compressed column [row] to be sorted by
200 /// their row [column] index.
201 //=============================================================================
202 class CompressedMatrixCoefficient
203 {
204 
205  public:
206 
207  /// Constructor (no arguments)
208  CompressedMatrixCoefficient(){}
209 
210  /// Constructor (takes the index and value as arguments)
211  CompressedMatrixCoefficient(const unsigned& index, const double& value)
212  {
213  // store the index and value
214  Index = index;
215  Value = value;
216  }
217 
218  /// Destructor (does nothing)
219  ~CompressedMatrixCoefficient(){}
220 
221  /// Copy Constructor. Not Broken. Required for STL sort function.
222  CompressedMatrixCoefficient(const CompressedMatrixCoefficient& a)
223  {
224  Index = a.index();
225  Value = a.value();
226  }
227 
228  /// Assignment Operator. Not Broken. Required for STL sort function.
229  void operator=(const CompressedMatrixCoefficient& a)
230  {
231  Index = a.index();
232  Value = a.value();
233  }
234 
235  /// Less Than Operator (for the STL sort function)
236  bool operator<(const CompressedMatrixCoefficient& a) const
237  {
238  return Index < a.index();
239  }
240 
241  /// access function for the coefficient's (row or column) index
242  unsigned& index() { return Index; }
243 
244  /// access function for the coefficient value
245  double& value() { return Value; }
246 
247  /// \short Access function for the coefficient's (row or column_ index
248  /// (const version)
249  unsigned index() const { return Index; }
250 
251  /// access function for the coefficient's value (const version)
252  double value() const { return Value; }
253 
254  private:
255 
256  /// the row or column index of the compressed-matrix coefficient
257  unsigned Index;
258 
259  /// the value of the compressed-matrix coefficient
260  double Value;
261 
262 };

387 /// \short A preconditioner for performing inner iteration preconditioner
388 /// solves. The template argument SOLVER specifies the inner iteration
389 /// solver (which must be derived from IterativeLinearSolver) and the
390 /// template argument PRECONDITIONER specifies the preconditioner for the
391 /// inner iteration iterative solver.
392 /// Note: For no preconditioning use the IdentityPreconditioner.
393 //=============================================================================
394 template <class SOLVER, class PRECONDITIONER>
395 class InnerIterationPreconditioner : public Preconditioner
396 {
397  public:
398 
399  /// Constructor
400  InnerIterationPreconditioner()
401  {
402  // create the solver
403  Solver_pt = new SOLVER;
404 
405  // create the preconditioner
406  Preconditioner_pt = new PRECONDITIONER;
407 
408 #ifdef PARANOID
409  // paranoid check that the solver is an iterative solver and
410  // the preconditioner is a preconditioner
411  if (dynamic_cast<IterativeLinearSolver* >(Solver_pt) == 0)
412  {
413  throw OomphLibError(
414  "The template argument SOLVER must be of type IterativeLinearSolver",
415  OOMPH_CURRENT_FUNCTION,
416  OOMPH_EXCEPTION_LOCATION);
417  }
418  if (dynamic_cast<Preconditioner*>(Preconditioner_pt) == 0)
419  {
420  throw OomphLibError(
421  "The template argument PRECONDITIONER must be of type Preconditioner",
422  OOMPH_CURRENT_FUNCTION,
423  OOMPH_EXCEPTION_LOCATION);
424  }
425 #endif
426 
427  // ensure the solver does not re-setup the preconditioner
428  Solver_pt->disable_setup_preconditioner_before_solve();
429 
430  // pass the preconditioner to the solver
431  Solver_pt->preconditioner_pt() = Preconditioner_pt;
432  }
433 
434  // destructor
435  ~InnerIterationPreconditioner()
436  {
437  delete Solver_pt;
438  delete Preconditioner_pt;
439  }
440 
441  // clean the memory
442  void clean_up_memory()
443  {
444  Preconditioner_pt->clean_up_memory();
445  Solver_pt->clean_up_memory();
446  }
447 
448  /// \short Preconditioner setup method. Setup the preconditioner for the inner
449  /// iteration solver.
450  void setup()
451  {
452 
453  // set the distribution
454  DistributableLinearAlgebraObject* dist_pt =
455  dynamic_cast<DistributableLinearAlgebraObject*>
456  (matrix_pt());
457  if (dist_pt != 0)
458  {
459  this->build_distribution(dist_pt->distribution_pt());
460  }
461  else
462  {
463  LinearAlgebraDistribution dist(comm_pt(),
464  matrix_pt()->nrow(),false);
465  this->build_distribution(dist);
466  }
467 
468  // Setup the inner iteration preconditioner (For some reason we need to
469  // remind the compiler that there is also a function named setup in the
470  // base class.)
471  Preconditioner_pt->Preconditioner::setup(matrix_pt());
472 
473  // setup the solverready for resolve
474  unsigned max_iter = Solver_pt->max_iter();
475  Solver_pt->max_iter() = 1;
476  DoubleVector x(this->distribution_pt(),0.0);
477  DoubleVector y(x);
478  Solver_pt->enable_resolve();
479  Solver_pt->solve(matrix_pt(),x,y);
480  Solver_pt->max_iter() = max_iter;
481  }
482 
483  /// \short Preconditioner solve method. Performs the specified number
484  /// of Krylov iterations preconditioned with the specified preconditioner
485  void preconditioner_solve(const DoubleVector &r, DoubleVector &z)
486  {
487  Solver_pt->resolve(r,z);
488  }
489 
490  /// Access to convergence tolerance of the inner iteration solver
491  double& tolerance() {return Solver_pt->tolerance();}
492 
493  /// Access to max. number of iterations of the inner iteration solver
494  unsigned& max_iter() {return Solver_pt->max_iter();}
495 
496  ///
497  SOLVER* solver_pt() { return Solver_pt; }
498 
499  ///
500  PRECONDITIONER* preconditioner_pt() { return Preconditioner_pt; }
501 
502  private:
503 
504  /// pointer to the underlying solver
505  SOLVER* Solver_pt;
506 
507  /// pointer to the underlying preconditioner
508  PRECONDITIONER* Preconditioner_pt;
509 };
510 }
