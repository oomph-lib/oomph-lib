50 /// A class for all isoparametric elements that solve the
51 /// Foeppl von Karman equations.
52 /// \f[
53 /// \nabla^4 w - \eta Diamond^4(w,\phi) = p(x,y)
54 /// \f]
55 /// and
56 /// \f[
57 /// \nabla^4 \phi + \frac{1}{2} Diamond^4(w,w) = 0
58 /// \f]
59 /// This contains the generic maths. Shape functions, geometric
60 /// mapping etc. must get implemented in derived class.

251  /// Get gradient of deflection: gradient[i] = dw/dx_i
252  void get_gradient_of_deflection(const Vector<double>& s,
253  Vector<double>& gradient) const
254  {
255  //Find out how many nodes there are in the element
256  const unsigned n_node = nnode();
257 
258  //Get the index at which the unknown is stored
259  const unsigned w_nodal_index = nodal_index_fvk(0);
260 
261  //Set up memory for the shape and test functions
262  Shape psi(n_node);
263  DShape dpsidx(n_node,2);
264 
265  //Call the derivatives of the shape and test functions
266  dshape_eulerian(s,psi,dpsidx);
267 
268  //Initialise to zero
269  for(unsigned j=0;j<2;j++)
270  {
271  gradient[j] = 0.0;
272  }
273

275  for(unsigned l=0;l<n_node;l++)
276  {
277  //Loop over derivative directions
278  for(unsigned j=0;j<2;j++)

280  gradient[j] += this->nodal_value(l,w_nodal_index)*dpsidx(l,j);
281  }
282  }

327  /// \short Return the integral of the displacement over the current
328  /// element, effectively calculating its contribution to the volume under
329  /// the membrane. Virtual so it can be overloaded in multi-physics
330  /// where the volume may incorporate an offset, say.
331  virtual double get_bounded_volume() const
332  {
333 
334  //Number of nodes and integration points for the current element
335  const unsigned n_node = nnode();
336  const unsigned n_intpt = integral_pt()->nweight();
337 
338  //Shape functions and their derivatives
339  Shape psi(n_node);
340  DShape dpsidx(n_node,2);
341 
342  //The nodal index at which the displacement is stored
343  const unsigned w_nodal_index = nodal_index_fvk();
344 
345  //Initalise the integral variable
346  double integral_w = 0;
347 
348  //Loop over the integration points
349  for(unsigned ipt=0;ipt<n_intpt;ipt++)
350  {
351  //Get the integral weight
352  double w = integral_pt()->weight(ipt);
353 
354  //Get determinant of the Jacobian of the mapping
355  double J = dshape_eulerian_at_knot(ipt,psi,dpsidx);
356 
357  //Premultiply the weight and Jacobian
358  double W = w*J;
359 
360  //Initialise storage for the w value and nodal value
361  double interpolated_w = 0;
362  double w_nodal_value;
363 
364  //Loop over the shape functions/nodes
365  for(unsigned l=0;l<n_node;l++)
366  {
367  //Get the current nodal value
368  w_nodal_value = raw_nodal_value(l,w_nodal_index);
369  //Add the contribution to interpolated w
370  interpolated_w += w_nodal_value*psi(l);
371  }
372 
373  //Add the contribution from the current integration point
374  integral_w += interpolated_w*W;
375  }
376 
377  //Return the calculated integral
378  return integral_w;
379  }

384  /// \short Sets a flag to signify that we are solving the linear, pure bending
385  /// equations, and pin all the nodal values that will not be used in this case
386  void use_linear_bending_model()
387  {
388  // Set the boolean flag
389  Linear_bending_model = true;
390 
391  // Get the index of the first FvK nodal value
392  unsigned first_fvk_nodal_index = nodal_index_fvk();
393 
394  // Get the total number of FvK nodal values (assuming they are stored
395  // contiguously) at node 0 (it's the same at all nodes anyway)
396  unsigned total_fvk_nodal_indicies = 8;
397 
398  // Get the number of nodes in this element
399  unsigned n_node = nnode();
400 
401  // Loop over the appropriate nodal indices
402  for(unsigned index=first_fvk_nodal_index+2;
403  index<first_fvk_nodal_index+total_fvk_nodal_indicies;
404  index++)
405  {
406  // Loop over the nodes in the element
407  for(unsigned inod=0;inod<n_node;inod++)
408  {
409  // Pin the nodal value at the current index
410  node_pt(inod)->pin(index);
411  }
412  }
