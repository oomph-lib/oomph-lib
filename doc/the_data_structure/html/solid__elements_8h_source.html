281  /// \short Create a list of pairs for all unknowns in this element,
282  /// so that the first entry in each pair contains the global equation
283  /// number of the unknown, while the second one contains the number
284  /// of the "DOF" that this unknown is associated with.
285  /// (Function can obviously only be called if the equation numbering
286  /// scheme has been set up.)
287  /// E.g. in a 3D problem there are 3 types of DOF:
288  /// 0 - x displacement
289  /// 1 - y displacement
290  /// 2 - z displacement
291  void get_dof_numbers_for_unknowns(
292  std::list<std::pair<unsigned long,unsigned> >& dof_lookup_list) const
293  {
294  // temporary pair (used to store dof lookup prior to being added to list
295  std::pair<unsigned,unsigned> dof_lookup;
296 
297  // number of nodes
298  const unsigned n_node = this->nnode();
299 
300  //Get the number of position dofs and dimensions at the node
301  const unsigned n_position_type = nnodal_position_type();
302  const unsigned nodal_dim = nodal_dimension();
303 
304  //Integer storage for local unknown
305  int local_unknown=0;
306 
307  //Loop over the nodes
308  for(unsigned n=0;n<n_node;n++)
309  {
310 
311  //Loop over position dofs
312  for(unsigned k=0;k<n_position_type;k++)
313  {
314  //Loop over dimension
315  for(unsigned i=0;i<nodal_dim;i++)
316  {
317  //If the variable is free
318  local_unknown = position_local_eqn(n,k,i);
319 
320  // ignore pinned values
321  if (local_unknown >= 0)
322  {
323  // store dof lookup in temporary pair: First entry in pair
324  // is global equation number; second entry is dof type
325  dof_lookup.first = this->eqn_number(local_unknown);
326  dof_lookup.second = i;
327 
328  // add to list
329  dof_lookup_list.push_front(dof_lookup);
330 
331  }
332  }
333  }
334  }
335  }

410  void fill_in_contribution_to_residuals(Vector<double> &residuals)

416  /// \short Fill in contribution to Jacobian (either by FD or analytically,
417  /// control this via evaluate_jacobian_by_fd()
418  void fill_in_contribution_to_jacobian(Vector<double> &residuals,
419  DenseMatrix<double> &jacobian)
420  {
421 
422  //Solve for the consistent acceleration in Newmark scheme?
423  if (this->Solve_for_consistent_newmark_accel_flag)
424  {
425  // Add the contribution to the residuals -- these are the
426  // full residuals of whatever solid equations we're solving
427  this->fill_in_contribution_to_residuals(residuals);
428 
429  // Jacobian is not the Jacobian associated with these
430  // residuals (treating the postions as unknowns)
431  // but the derivatives w.r.t. to the discrete generalised
432  // accelerations in the Newmark scheme -- the Jacobian
433  // is therefore the associated mass matrix, multiplier
434  // by suitable scaling factors
435  this->fill_in_jacobian_for_newmark_accel(jacobian);
436  return;
437  }
438 
439  // Are we assigning a solid initial condition?
440  if (this->Solid_ic_pt!=0)
441  {
442  this->fill_in_jacobian_for_solid_ic(residuals,jacobian);
443  return;
444  }
445 
446 
447  // Use FD
448  if ((this->Evaluate_jacobian_by_fd))
449  {
450  //Add the contribution to the residuals from this element
451  this->fill_in_generic_contribution_to_residuals_pvd(
452  residuals,GeneralisedElement::Dummy_matrix,0);
453 
454  //Get the solid entries in the jacobian using finite differences
455  this->fill_in_jacobian_from_solid_position_by_fd(jacobian);
456  }
457  // Do it analytically
458  else
459  {
460  fill_in_generic_contribution_to_residuals_pvd(residuals,jacobian,1);
461  }
462  }
