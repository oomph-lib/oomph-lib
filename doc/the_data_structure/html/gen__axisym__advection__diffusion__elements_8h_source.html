120  /// \short du/dt at local node n.
121  /// Uses suitably interpolated value for hanging nodes.
122  double du_dt_cons_axisym_adv_diff(const unsigned& n) const
123  {
124  // Get the data's timestepper
125  TimeStepper* time_stepper_pt = this->node_pt(n)->time_stepper_pt();
126 
127  // Initialise dudt
128  double dudt = 0.0;
129  // Loop over the timesteps, if there is a non Steady timestepper
130  if (!time_stepper_pt->is_steady())
131  {
132  // Find the index at which the variable is stored
133  const unsigned u_nodal_index = u_index_cons_axisym_adv_diff();
134 
135  // Number of timsteps (past & present)
136  const unsigned n_time = time_stepper_pt->ntstorage();
137 
138  for (unsigned t = 0; t < n_time; t++)
139  {
140  dudt +=
141  time_stepper_pt->weight(1, t) * nodal_value(t, n, u_nodal_index);
142  }
143  }
144  return dudt;
145  }

388  /// \short Get diffusivity tensor at (Eulerian) position
389  /// x and/or local coordinate s.
390  /// This function is
391  /// virtual to allow overloading in multi-physics problems where
392  /// the wind function might be determined by
393  /// another system of equations
394  inline virtual void get_diff_cons_axisym_adv_diff(
395  const unsigned& ipt,
396  const Vector<double>& s,
397  const Vector<double>& x,
398  DenseMatrix<double>& D) const
399  {
400  // If no wind function has been set, return identity
401  // Again three components, but not all of them matter
402  // Those in the theta direction are ignored
403  if (Diff_fct_pt == 0)
404  {
405  for (unsigned i = 0; i < 3; i++)
406  {
407  for (unsigned j = 0; j < 3; j++)
408  {
409  if (i == j)
410  {
411  D(i, j) = 1.0;
412  }
413  else
414  {
415  D(i, j) = 0.0;
416  }
417  }
418  }
419  }
420  else
421  {
422  // Get diffusivity tensor
423  (*Diff_fct_pt)(x, D);
424  }
425  }

428  /// Get flux: \f$\mbox{flux}[i] = \mbox{d}u / \mbox{d}x_i \f$
429  void get_flux(const Vector<double>& s, Vector<double>& flux) const
430  {
431  // Find out how many nodes there are in the element
432  const unsigned n_node = this->nnode();
433 
434  // Get the nodal index at which the unknown is stored
435  const unsigned u_nodal_index = this->u_index_cons_axisym_adv_diff();
436 
437  // Set up memory for the shape and test functions
438  Shape psi(n_node);
439  DShape dpsidx(n_node, 2);
440 
441  // Call the derivatives of the shape and test functions
442  dshape_eulerian(s, psi, dpsidx);
443 
444  // Initialise to zero
445  for (unsigned j = 0; j < 2; j++)
446  {
447  flux[j] = 0.0;
448  }
449 
450  // Loop over nodes
451  for (unsigned l = 0; l < n_node; l++)
452  {
453  const double u_value = this->nodal_value(l, u_nodal_index);
454  // Add in the derivative directions
455  flux[0] += u_value * dpsidx(l, 0);
456  flux[1] += u_value * dpsidx(l, 1);
457  }
458  }

460  /// Get flux: \f$\mbox{flux}[i] = \mbox{d}u / \mbox{d}x_i \f$
461  void get_total_flux(const Vector<double>& s,
462  Vector<double>& total_flux) const
463  {
464  // Find out how many nodes there are in the element
465  const unsigned n_node = nnode();
466 
467  // Get the nodal index at which the unknown is stored
468  const unsigned u_nodal_index = u_index_cons_axisym_adv_diff();
469 
470  // Set up memory for the shape and test functions
471  Shape psi(n_node);
472  DShape dpsidx(n_node, 2);
473 
474  // Call the derivatives of the shape and test functions
475  dshape_eulerian(s, psi, dpsidx);
476 
477  // Storage for the Eulerian position
478  Vector<double> interpolated_x(2, 0.0);
479  // Storage for the concentration
480  double interpolated_u = 0.0;
481  // Storage for the derivatives of the concentration
482  Vector<double> interpolated_dudx(2, 0.0);
483 
484  // Loop over nodes
485  for (unsigned l = 0; l < n_node; l++)
486  {
487  // Get the value at the node
488  const double u_value = this->nodal_value(l, u_nodal_index);
489  interpolated_u += u_value * psi(l);
490  // Loop over directions
491  for (unsigned j = 0; j < 2; j++)
492  {
493  interpolated_x[j] += this->nodal_position(l, j) * psi(l);
494  interpolated_dudx[j] += u_value * dpsidx(l, j);
495  }
496  }
497 
498  // Dummy integration point
499  unsigned ipt = 0;
500 
501  // Get the conserved wind (non-divergence free)
502  Vector<double> conserved_wind(3);
503  get_conserved_wind_cons_axisym_adv_diff(
504  ipt, s, interpolated_x, conserved_wind);
505 
506  // Get diffusivity tensor
507  DenseMatrix<double> D(3, 3);
508  get_diff_cons_axisym_adv_diff(ipt, s, interpolated_x, D);
509 
510  // Calculate the total flux made up of the diffusive flux
511  // and the conserved wind only bother with the
512  // first two components in each case becuase there can be no
513  // variation in the azimuthal direction
514  for (unsigned i = 0; i < 2; i++)
515  {
516  total_flux[i] = 0.0;
517  for (unsigned j = 0; j < 2; j++)
518  {
519  total_flux[i] += D(i, j) * interpolated_dudx[j];
520  }
521  total_flux[i] -= conserved_wind[i] * interpolated_u;
522  }
523  }
