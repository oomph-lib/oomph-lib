427  /// \short Virtual function that specifies the variation of the
428  /// zeta coordinates in the GeomObject along the
429  /// boundary connecting vertices 0 and 1 in
430  /// facet facet_id. Default implementation: Linear interpolation
431  /// between edges. zeta_boundary=0.0: we're on vertex 0;
432  /// zeta_boundary=1.0: we're on vertex 1.
433  virtual void boundary_zeta01(const unsigned& facet_id,
434  const double& zeta_boundary,
435  Vector<double>& zeta)
436  {
437  Vector<Vector<double> > zeta_vertex(2);
438  zeta_vertex[0]=Facet_pt[facet_id]->vertex_pt(0)->zeta_in_geom_object();
439  zeta_vertex[1]=Facet_pt[facet_id]->vertex_pt(1)->zeta_in_geom_object();
440  zeta[0]=zeta_vertex[0][0]+(zeta_vertex[1][0]-zeta_vertex[0][0])*zeta_boundary;
