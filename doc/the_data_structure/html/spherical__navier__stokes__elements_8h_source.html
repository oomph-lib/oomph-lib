321  /// \short i-th component of du/dt at local node n.
322  /// Uses suitably interpolated value for hanging nodes.
323  double du_dt_spherical_nst(const unsigned &n, const unsigned &i) const
324  {
325  // Get the data's timestepper
326  TimeStepper* time_stepper_pt = this->node_pt(n)->time_stepper_pt();
327 
328  //Initialise dudt
329  double dudt=0.0;
330 
331  //Loop over the timesteps, if there is a non Steady timestepper
332  if (time_stepper_pt->type()!="Steady")
333  {
334  //Find the index at which the dof is stored
335  const unsigned u_nodal_index = this->u_index_spherical_nst(i);
336 
337  // Number of timsteps (past & present)
338  const unsigned n_time = time_stepper_pt->ntstorage();
339  // Loop over the timesteps
340  for(unsigned t=0;t<n_time;t++)
341  {
342  dudt+=time_stepper_pt->weight(1,t)*nodal_value(t,n,u_nodal_index);
343  }
344  }
345 
346  return dudt;
347  }
