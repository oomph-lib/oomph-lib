55  ///\short Calculates Legendre polynomial of degree p at x
56  /// using the three term recurrence relation
57  /// \f$ (n+1) P_{n+1} = (2n+1)xP_{n} - nP_{n-1} \f$
58  inline double legendre(const unsigned &p, const double &x)
59  {
60  //Return the constant value
61  if(p==0) return 1.0;
62  //Return the linear polynomial
63  else if(p==1) return x;
64  //Otherwise use the recurrence relation
65  else{
66  //Initialise the terms in the recurrence relation, we're going
67  //to shift down before using the relation.
68  double L0 = 1.0, L1 = 1.0, L2 = x;
69  //Loop over the remaining polynomials
70  for(unsigned n=1;n<p;n++)
71  {
72  //Shift down the values
73  L0 = L1; L1 = L2;
74  //Use the three term recurrence
75  L2 = ((2*n + 1)*x*L1 - n*L0)/(n+1);
76  }
77  //Once we've finished return the final value
78  return L2;
79  }
80  }

110  /// \short Calculates first derivative of Legendre
111  /// polynomial of degree p at x
112  /// using three term recursive formula.
113  /// \f$ nP_{n+1}^{'} = (2n+1)xP_{n}^{'} - (n+1)P_{n-1}^{'} \f$
114  inline double dlegendre(const unsigned &p, const double &x)
115  {
116  double dL1 = 1.0, dL2 = 3*x, dL3=0.0;
117  if(p==0) return 0.0;
118  else if(p==1) return dL1;
119  else if(p==2) return dL2;
120  else{
121  for(unsigned n=2;n<p;n++){
122  dL3 = 1.0/n*((2.0*n+1.0)*x*dL2-(n+1.0)*dL1);
123  dL1=dL2;
124  dL2=dL3;
125  }
126  return dL3;
127  }
128  }

130  /// \short Calculates second derivative of Legendre
131  /// polynomial of degree p at x
132  /// using three term recursive formula.
133  inline double ddlegendre(const unsigned &p, const double &x)
134  {
135  double ddL2 = 3.0, ddL3 = 15*x, ddL4=0.0;
136  if(p==0) return 0.0;
137  else if(p==1) return 0.0;
138  else if(p==2) return ddL2;
139  else if(p==3) return ddL3;
140  else{
141  for(unsigned i=3;i<p;i++){
142  ddL4 = 1.0/(i-1.0)*((2.0*i+1.0)*x*ddL3-(i+2.0)*ddL2);
143  ddL2=ddL3;
144  ddL3=ddL4;
145  }
146  return ddL4;

150  /// \short Calculate the Jacobi polnomials
151  inline double jacobi(const int &alpha, const int &beta, const unsigned &p,
152  const double &x)
153  {
154  double P0 = 1.0;
155  double P1 = 0.5*(alpha-beta+(alpha+beta+2.0)*x);
156  double P2;
157  if(p==0) return P0;
158  else if(p==1) return P1;
159  else{
160  for(unsigned n=1;n<p;n++){
161  double a1n = 2*(n+1)*(n+alpha+beta+1)*(2*n+alpha+beta);
162  double a2n = (2*n+alpha+beta+1)*(alpha*alpha-beta*beta);
163  double a3n = (2*n+alpha+beta)*(2*n+alpha+beta+1)*(2*n+alpha+beta+2);
164  double a4n = 2*(n+alpha)*(n+beta)*(2*n+alpha+beta+2);
165  P2 = ( (a2n+a3n*x)*P1 - a4n*P0 ) / a1n;
166  P0 = P1;
167  P1 = P2;
168  }
169  return P2;
170  }
171  }

173  /// \short Calculate the Jacobi polnomials all in one goe
174  inline void jacobi(const int &alpha, const int &beta, const unsigned &p,
175  const double &x,
176  Vector<double> &polys)
177  {
178  //Set the constant term
179  polys[0] = 1.0;
180  //If we've only been asked for the constant term, bin out
181  if(p==0) {return;}
182  //Set the linear polynomial
183  polys[1] = 0.5*(alpha-beta+(alpha+beta+2.0)*x);
184  //Initialise the terms for the recurrence relation
185  double P0 = 1.0;
186  double P1 = 1.0;
187  double P2 = 0.5*(alpha-beta+(alpha+beta+2.0)*x);
188  //Loop over the remaining terms
189  for(unsigned n=1;n<p;n++)
190  {
191  //Shift down the terms
192  P0 = P1;
193  P1 = P2;
194  //Set the constants
195  double a1n = 2*(n+1)*(n+alpha+beta+1)*(2*n+alpha+beta);
196  double a2n = (2*n+alpha+beta+1)*(alpha*alpha-beta*beta);
197  double a3n = (2*n+alpha+beta)*(2*n+alpha+beta+1)*(2*n+alpha+beta+2);
198  double a4n = 2*(n+alpha)*(n+beta)*(2*n+alpha+beta+2);
199  //Set the latest term
200  P2 = ( (a2n+a3n*x)*P1 - a4n*P0 ) / a1n;
201  //Set the newly calculate polynomial
