60  /// Access function to assign the time slice this element lies in
61  void set_time_slab_id(const int& time_slice_id)
62  {
63  // The time slice must be greater than -1. This happens when we use
64  // an impulsive start so there are no dofs in the first time slice
65  // of the first elemental time slice of the first but there are proper
66  // dofs in the final time slice of the first elemental time slice.
67  if (time_slice_id<-1)
68  {
69  // Throw an error
70  throw OomphLibError("Time slice ID must be greater than or equal to -1!",
71  OOMPH_CURRENT_FUNCTION,
72  OOMPH_EXCEPTION_LOCATION);
73  }
74 
75  // Return the value that is currently stored
76  Time_slab_id=time_slice_id;
77  } // End of set_time_slab_id
