61  /// Returns the local form of the q basis at local coordinate s
62  template<>
63  void TPoroelasticityElement<0>::get_q_basis_local(const Vector<double> &s,
64  Shape &q_basis) const
65  {
66  // RT_0 basis functions
67  q_basis(0,0) = Sign_edge[0]*std::sqrt(2)*s[0];
68  q_basis(0,1) = Sign_edge[0]*std::sqrt(2)*s[1];
69 
70  q_basis(1,0) = Sign_edge[1]*(s[0]-1);
71  q_basis(1,1) = Sign_edge[1]*s[1];
72 
73  q_basis(2,0) = Sign_edge[2]*s[0];
74  q_basis(2,1) = Sign_edge[2]*(s[1]-1);
75  }

161  /// Returns the local form of the q basis at local coordinate s
162  template<>
163  void TPoroelasticityElement<1>::get_q_basis_local(const Vector<double> &s,
164  Shape &q_basis) const
165  {
166  // RT_1 basis functions
167 
168  double g1,g2;
169 
170  g1=edge_gauss_point(0,0);
171  g2=edge_gauss_point(0,1);
172  q_basis(0,0) = Sign_edge[0]*std::sqrt(2)*s[0]*(s[1]-g2)/(g1-g2);
173  q_basis(0,1) = Sign_edge[0]*std::sqrt(2)*s[1]*(s[1]-g2)/(g1-g2);
174 
175  q_basis(1,0) = Sign_edge[0]*std::sqrt(2)*s[0]*(s[1]-g1)/(g2-g1);
176  q_basis(1,1) = Sign_edge[0]*std::sqrt(2)*s[1]*(s[1]-g1)/(g2-g1);
177 
178  g1=edge_gauss_point(1,0);
179  g2=edge_gauss_point(1,1);
180  q_basis(2,0) = Sign_edge[1]*(s[0]-1)*(s[1]-g1)/(g2-g1);
181  q_basis(2,1) = Sign_edge[1]*s[1]*(s[1]-g1)/(g2-g1);
182 
183  q_basis(3,0) = Sign_edge[1]*(s[0]-1)*(s[1]-g2)/(g1-g2);
184  q_basis(3,1) = Sign_edge[1]*s[1]*(s[1]-g2)/(g1-g2);
185 
186  g1=edge_gauss_point(2,0);
187  g2=edge_gauss_point(2,1);
188  q_basis(4,0) = Sign_edge[2]*s[0]*(s[0]-g2)/(g1-g2);
189  q_basis(4,1) = Sign_edge[2]*(s[1]-1)*(s[0]-g2)/(g1-g2);
190 
191  q_basis(5,0) = Sign_edge[2]*s[0]*(s[0]-g1)/(g2-g1);
192  q_basis(5,1) = Sign_edge[2]*(s[1]-1)*(s[0]-g1)/(g2-g1);
193 
194  q_basis(6,0) = s[1]*s[0];
195  q_basis(6,1) = s[1]*(s[1]-1);
196 
197  q_basis(7,0) = s[0]*(s[0]-1);
198  q_basis(7,1) = s[0]*s[1];
199  }

201  /// Returns the local form of the q basis and dbasis/ds at local coordinate s
202  template<>
203  void TPoroelasticityElement<1>::get_div_q_basis_local(
204  const Vector<double> &s,
205  Shape &div_q_basis_ds) const
206  {
207  double g1,g2;
208 
209  g1=edge_gauss_point(0,0);
210  g2=edge_gauss_point(0,1);
211  div_q_basis_ds(0) = Sign_edge[0]*std::sqrt(2)*(3*s[1]-2*g2)/(g1-g2);
212  div_q_basis_ds(1) = Sign_edge[0]*std::sqrt(2)*(2*g1-3*s[1])/(g1-g2);
213 
214  g1=edge_gauss_point(1,0);
215  g2=edge_gauss_point(1,1);
216  div_q_basis_ds(2) = Sign_edge[1]*(2*g1-3*s[1])/(g1-g2);
217  div_q_basis_ds(3) = Sign_edge[1]*(3*s[1]-2*g2)/(g1-g2);
218 
219  g1=edge_gauss_point(2,0);
220  g2=edge_gauss_point(2,1);
221  div_q_basis_ds(4) = Sign_edge[2]*(3*s[0]-2*g2)/(g1-g2);
222  div_q_basis_ds(5) = Sign_edge[2]*(2*g1-3*s[0])/(g1-g2);
223 
224  div_q_basis_ds(6) = 3*s[1]-1;
225  div_q_basis_ds(7) = 3*s[0]-1;
226 
227  // Scale the basis by the ratio of the length of the edge to the length of
228  // the corresponding edge on the reference element
229  scale_basis(div_q_basis_ds);
230  }
