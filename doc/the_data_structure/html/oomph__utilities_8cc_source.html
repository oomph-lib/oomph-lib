99  // Temporary storage for the filename which can be edited

102  // The delimeter
103  std::string delimiter="_";
104 
105  // Posiiton of the delimiter
106  size_t pos=0;

108  // The string before the delimiter
109  std::string token;

111  // Get the filename (gets rid of e.g. ./ before filename)
112  while ((pos=file.find("/"))!=std::string::npos)

163  /// Compute second invariant of tensor
164  double second_invariant(const DenseMatrix<double>& tensor)
165  {
166  // get size of tensor
167  unsigned n=tensor.nrow();
168  double trace_of_tensor=0.0;
169  double trace_of_tensor_squared=0.0;
170  for (unsigned i=0; i<n; i++)
171  {
172  // Calculate the trace of the tensor: T_ii
173  trace_of_tensor += tensor(i,i);
174  // Calculate the trace of the tensor squared: T_ij T_ji
175  for (unsigned j=0; j<n; j++)
176  {
177  trace_of_tensor_squared += tensor(i,j)*tensor(j,i);
178  }
179  }
180 
181  // return the second invariant: 1.0/2.0*[(T_ii)^2 - T_ij T_ji]
182  return 0.5*(trace_of_tensor*trace_of_tensor-trace_of_tensor_squared);
183 
184  }
