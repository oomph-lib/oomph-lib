103  /// \short Constructor that builds a multivector from selected columns
104  /// of the input multivector. The boolean controls whether it is a
105  /// shallow or deep copy (default deep)
106  DoubleMultiVector(const DoubleMultiVector&old_vector,
107  const std::vector<int> &index,
108  const bool &deep_copy=true) :
109  Values(0), Nvector(0), Internal_values(deep_copy), Built(false)
110  {
111  //Build the storage based on the size of index
112  unsigned n_vector = index.size();
113  if(deep_copy)
114  {
115  //Create an entirely new data structure
116  this->build(n_vector,old_vector.distribution_pt());
117  //Now (deep) copy the data across
118  const unsigned n_row_local = this->nrow_local();
119  for(unsigned v=0;v<n_vector;v++)
120  {
121  for(unsigned i=0;i<n_row_local;i++)
122  {
123  Values[v][i] = old_vector(index[v],i);
124  }
125  }
126  }
127  //Otherwise it's a shallow copy
128  else
129  {
130  this->shallow_build(n_vector,old_vector.distribution_pt());
131  //Now shallow copy the pointers accross

133  {
134  Values[v] = old_vector.values(index[v]);
135  }
136  }
137  this->setup_doublevector_representation();
138  }

314  /// \short Assembles a DoubleMultiVector with n_vector vectors, each with a
315  /// distribution dist, if v is specified
316  /// each element is set to v, otherwise each element is set to 0.0
317  void build(const unsigned &n_vector,
318  const LinearAlgebraDistribution* const &dist_pt,
319  const double& initial_value=0.0)
320  {
321  // clean the memory
322  this->clear();
323 
324  // the vector owns the internal data
325  Internal_values = true;
326 
327  // Set the distribution
328  this->build_distribution(dist_pt);
329 
330  // update the values
331  if (dist_pt->built())
332  {
333  //Set the number of vectors
334  Nvector = n_vector;
335  //Build the array of pointers to each vector's data
336  Values = new double*[n_vector];
337  //Now build the contiguous array of real data
338  const unsigned n_row_local = this->nrow_local();
339  double *values = new double[n_row_local*n_vector];
340  // set the data
341  for(unsigned v=0;v<n_vector;v++)
342  {
343  Values[v] = &values[v*n_row_local];
344  for (unsigned i = 0; i < n_row_local; i++)
345  {
346  Values[v][i] = initial_value;
347  }
348  }
349  Built=true;
350  }

353  Built=false;
354  }
