35 /// Sparse vector -- wrapper for map (but avoids filling map during
36 /// reads of zero/empty entries). Square bracket operator is read-only.
37 //========================================================================
38 template<class T>
39 class SparseVector
40 {
41 
42  public:
43  /// Constructor
44  SparseVector()
45  {
46  // Empty instance
47  Empty_pt = new T(0);
48  }
49 
50  /// Destructor
51  ~SparseVector()
52  {
53  // Delete empty instance
54  delete Empty_pt;
55  }
56 
57  // Initialise the bin (only called by the create_bins_of_objects()
58  // method)
59  void initialise(const unsigned& size)
60  {
61  // Create a "large" bool vector indicating all entries are empty
62  Has_entry.resize(size, false);
63  }
64 
65  /// Wipe storage
66  void clear()
67  {
68  Data.clear();
69 
70  // Get current size and reset all entries
71  const unsigned size = Has_entry.size();
72  Has_entry.resize(size, false);
73  }
74 
75  /// Square bracket access (const version)
76  const T& operator[](const unsigned& i) const
77  {
78  typedef typename std::map<unsigned, T>::const_iterator IT;
79  IT it = Data.find(i);
80  if (it == Data.end())
81  {
82  return *Empty_pt;
83  }
84  return (*it).second;
85  }
86 
87  /// Set value of i-th entry
88  void set_value(const unsigned& i, const T& value)
89  {
90  Data[i] = value;
91  // Mark as having entry
92  Has_entry[i] = true;
93  }
94 
95  /// \short Number of nonzero entries stored in here (specifying the
96  /// size of the vector (as a mathematical object)
97  /// doesn't really make sense -- the thing could be infinitely
98  /// big and we wouldn't know or care)
99  unsigned nnz() const
100  {
101  return Data.size();
102  }
103 
104  /// Read-only access to underlying map

106  {

108  }
109 
110  /// Read/write access to underlying map -- dangerous!
111  std::map<unsigned,T>* map_pt()
112  {
113  return &Data;
114  }
115 
116  /// \short Check if the bin has entries
117  bool has_entry(const unsigned &nbin)
118  {
119  return Has_entry[nbin];
120  }
121 
122  /// \short Return vector containing all values
123  /// (without reference to their specific indices)
124  /// for fast direct access
125  void get_all_values(Vector<T>& all_values) const
126  {
127  all_values.clear();
128  all_values.resize(nnz());
129  typedef typename std::map<unsigned, T>::const_iterator IT;
130  unsigned count = 0;
131  for (IT it = Data.begin(); it != Data.end(); it++)
132  {
133  all_values[count++] = (*it).second;
134  }
135  }
