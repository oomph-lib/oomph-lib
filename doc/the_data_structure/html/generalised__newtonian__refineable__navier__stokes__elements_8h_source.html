141  /// \short Get 'flux' for Z2 error recovery: Upper triangular entries
142  /// in strain rate tensor.
143  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)
144  {
145 #ifdef PARANOID
146  unsigned num_entries=DIM+(DIM*(DIM-1))/2;
147  if (flux.size() < num_entries)
148  {
149  std::ostringstream error_message;
150  error_message << "The flux vector has the wrong number of entries, "
151  << flux.size() << ", whereas it should be at least "
152  << num_entries << std::endl;
153  throw OomphLibError(error_message.str(),
154  OOMPH_CURRENT_FUNCTION,
155  OOMPH_EXCEPTION_LOCATION);
156  }

159  // Get strain rate matrix
160  DenseMatrix<double> strainrate(DIM);
161  this->strain_rate(s,strainrate);
162 
163  // Pack into flux Vector
164  unsigned icount=0;
165 
166  // Start with diagonal terms
167  for(unsigned i=0;i<DIM;i++)
168  {
169  flux[icount]=strainrate(i,i);
170  icount++;
171  }
172 
173  //Off diagonals row by row
174  for(unsigned i=0;i<DIM;i++)
175  {
176  for(unsigned j=i+1;j<DIM;j++)
177  {
178  flux[icount]=strainrate(i,j);
179  icount++;

217  /// \short Compute the derivatives of the i-th component of
218  /// velocity at point s with respect
219  /// to all data that can affect its value. In addition, return the global
220  /// equation numbers corresponding to the data.
221  /// Overload the non-refineable version to take account of hanging node
222  /// information
223  void dinterpolated_u_nst_ddata(const Vector<double> &s,
224  const unsigned &i,
225  Vector<double> &du_ddata,
226  Vector<unsigned> &global_eqn_number)
227  {
228  //Find number of nodes
229  unsigned n_node = this->nnode();
230  //Local shape function
231  Shape psi(n_node);
232  //Find values of shape function at the given local coordinate
233  this->shape(s,psi);
234 
235  //Find the index at which the velocity component is stored
236  const unsigned u_nodal_index = this->u_index_nst(i);
237 
238  //Storage for hang info pointer
239  HangInfo* hang_info_pt=0;
240  //Storage for global equation
241  int global_eqn = 0;
242 
243  //Find the number of dofs associated with interpolated u
244  unsigned n_u_dof=0;
245  for(unsigned l=0;l<n_node;l++)
246  {
247  unsigned n_master = 1;
248 
249  //Local bool (is the node hanging)
250  bool is_node_hanging = this->node_pt(l)->is_hanging();
251 
252  //If the node is hanging, get the number of master nodes
253  if(is_node_hanging)
254  {
255  hang_info_pt = this->node_pt(l)->hanging_pt();
256  n_master = hang_info_pt->nmaster();
257  }
258  //Otherwise there is just one master node, the node itself
259  else
260  {
261  n_master = 1;
262  }
263 
264  //Loop over the master nodes
265  for(unsigned m=0;m<n_master;m++)
266  {
267  //Get the equation number
268  if(is_node_hanging)
269  {
270  //Get the equation number from the master node
271  global_eqn = hang_info_pt->master_node_pt(m)->
272  eqn_number(u_nodal_index);
273  }
274  else
275  {
276  // Global equation number
277  global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
278  }
279 
280  //If it's positive add to the count
281  if(global_eqn >= 0) {++n_u_dof;}
282  }
283  }
284 
285  //Now resize the storage schemes
286  du_ddata.resize(n_u_dof,0.0);
287  global_eqn_number.resize(n_u_dof,0);
288 
289  //Loop over th nodes again and set the derivatives
290  unsigned count=0;
291  //Loop over the local nodes and sum
292  for(unsigned l=0;l<n_node;l++)
293  {
294  unsigned n_master = 1;
295  double hang_weight = 1.0;
296 
297  //Local bool (is the node hanging)
298  bool is_node_hanging = this->node_pt(l)->is_hanging();
299 
300  //If the node is hanging, get the number of master nodes
301  if(is_node_hanging)
302  {
303  hang_info_pt = this->node_pt(l)->hanging_pt();
304  n_master = hang_info_pt->nmaster();
305  }
306  //Otherwise there is just one master node, the node itself
307  else
308  {
309  n_master = 1;
310  }
311 
312  //Loop over the master nodes
313  for(unsigned m=0;m<n_master;m++)
314  {
315  //If the node is hanging get weight from master node
316  if(is_node_hanging)
317  {
318  //Get the hang weight from the master node
319  hang_weight = hang_info_pt->master_weight(m);
320  }
321  else
322  {
323  // Node contributes with full weight
324  hang_weight = 1.0;
325  }
326 
327  //Get the equation number
328  if(is_node_hanging)
329  {
330  //Get the equation number from the master node
331  global_eqn = hang_info_pt->master_node_pt(m)->
332  eqn_number(u_nodal_index);
333  }
334  else
335  {
336  // Local equation number
337  global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
338  }
339 
340  if (global_eqn >= 0)
341  {
342  //Set the global equation number
343  global_eqn_number[count] = global_eqn;
344  //Set the derivative with respect to the unknown
345  du_ddata[count] = psi[l]*hang_weight;
346  //Increase the counter
347  ++count;
348  }
349  }
350  }
351  }

549  /// \short The velocity nodes are the same as the geometric nodes. The
550  /// pressure nodes must be calculated by using the same methods as
551  /// the geometric nodes, but by recalling that there are only two pressure
552  /// nodes per edge.
553  Node* get_interpolating_node_at_local_coordinate(const Vector<double> &s,
554  const int &value_id)
555  {
556  //If we are calculating pressure nodes
557  if(value_id==DIM)
558  {
559  //Storage for the index of the pressure node
560  unsigned total_index=0;
561  //The number of nodes along each 1d edge is 2.
562  unsigned NNODE_1D = 2;
563  //Storage for the index along each boundary
564  Vector<int> index(DIM);
565  //Loop over the coordinates
566  for(unsigned i=0;i<DIM;i++)
567  {
568  //If we are at the lower limit, the index is zero
569  if(s[i]==-1.0)
570  {
571  index[i] = 0;
572  }
573  //If we are at the upper limit, the index is the number of nodes minus 1
574  else if(s[i] == 1.0)
575  {
576  index[i] = NNODE_1D-1;
577  }
578  //Otherwise, we have to calculate the index in general
579  else
580  {
581  //For uniformly spaced nodes the 0th node number would be
582  double float_index = 0.5*(1.0 + s[i])*(NNODE_1D-1);
583  index[i] = int(float_index);
584  //What is the excess. This should be safe because the
585  //taking the integer part rounds down
586  double excess = float_index - index[i];
587  //If the excess is bigger than our tolerance there is no node,
588  //return null
589  if((excess > FiniteElement::Node_location_tolerance) && (
590  (1.0 - excess) > FiniteElement::Node_location_tolerance))
591  {
592  return 0;
593  }
594  }
595  ///Construct the general pressure index from the components.
596  total_index += index[i]*
597  static_cast<unsigned>(pow(static_cast<float>(NNODE_1D),
598  static_cast<int>(i)));
599  }
600  //If we've got here we have a node, so let's return a pointer to it
601  return this->pressure_node_pt(total_index);
602  }
603  //Otherwise velocity nodes are the same as pressure nodes
604  else
605  {
606  return this->get_node_at_local_coordinate(s);
607  }
608  }
