267 /// \short SubBiharmonic Preconditioner - an inexact preconditioner for the
268 /// 3x3 top left hand corner sub block matrix.
269 /// Used as part of the BiharmonicPreconditioner<MATRIX>
270 //=============================================================================
271  class InexactSubBiharmonicPreconditioner
272  : public BlockPreconditioner<CRDoubleMatrix>
273  {
274 
275  public :
276 
277  /// \short Constructor for the inexact block preconditioner. \n
278  /// This a helper class for BiharmonicPreconditioner and cannot be used
279  /// as a standalone preconditioner. \n
280  /// master_prec_pt is the pointer to the master BiharmonicPreconditioner.
281  InexactSubBiharmonicPreconditioner
282  (BiharmonicPreconditioner* master_prec_pt, const bool use_amg)
283  {
284  // Block mapping for ExactSubBiharmonicPreconditioner
285  Vector<unsigned> block_lookup(3);
286  block_lookup[0] = 0;
287  block_lookup[1] = 1;
288  block_lookup[2] = 2;
289 
290  // set as subsidiary block preconditioner
291  this->turn_into_subsidiary_block_preconditioner
292  (master_prec_pt,block_lookup);
293 
294  // zero all the preconditioner pt
295  S_00_preconditioner_pt = 0;
296  Lumped_J_11_preconditioner_pt = 0;
297  Lumped_J_22_preconditioner_pt = 0;
298 
299  // store the preconditioner type
300  Use_amg = use_amg;
301  }
302 
303  /// destructor - just calls this->clean_up_memory()
304  ~InexactSubBiharmonicPreconditioner()
305  {
306  this->clean_up_memory();
307  }
308 
309  /// clean the memory
310  void clean_up_memory()
311  {
312  // delete the sub preconditioner pt
313  delete S_00_preconditioner_pt;
314  S_00_preconditioner_pt = 0;
315 
316  // delete the lumped preconditioners
317  delete Lumped_J_11_preconditioner_pt;
318  Lumped_J_11_preconditioner_pt = 0;
319  delete Lumped_J_22_preconditioner_pt;
320  Lumped_J_22_preconditioner_pt = 0;
321 
322  // Number of block types
323  unsigned n = Matrix_of_block_pointers.nrow();
324 
325  // delete the block matrices
326  for (unsigned i = 0; i < n; i++)
327  {
328  for (unsigned j = 0; j < n; j++)
329  {
330  if (Matrix_of_block_pointers(i,j) != 0)
331  {
332  delete Matrix_of_block_pointers(i,j);
333  Matrix_of_block_pointers(i,j)=0;
334  }
335  }
336  }
337  }
338

340  InexactSubBiharmonicPreconditioner

343  BrokenCopy::broken_copy("InexactSubBiharmonicPreconditioner");
344  }

350  }
