284 template<>
285 void QHermiteElement<1>::build_face_element(const int &face_index,
286  FaceElement *face_element_pt)
287 {
288  // Set the nodal dimension from the "bulk"
289  face_element_pt->set_nodal_dimension(node_pt(0)->ndim());
290 
291  // Set the pointer to the "bulk" element
292  face_element_pt->bulk_element_pt()=this;
293 
294 #ifdef OOMPH_HAS_MPI
295  // Pass on non-halo ID
296  face_element_pt->set_halo(Non_halo_proc_ID);
297 #endif
298 
299  // Resize the storage for the original number of values at the (one and only)
300  // node of the face element.
301  face_element_pt->nbulk_value_resize(1);
302 
303  // Resize the storage for the bulk node number corresponding to the (one
304  // and only) node of the face element
305  face_element_pt->bulk_node_number_resize(1);
306 
307  //Set the face index in the face element
308  face_element_pt->face_index() = face_index;
309 
310  //Now set up the node pointer
311  //The convention is that the "normal", the coordinate, should always point
312  //out of the element
313  switch(face_index)
314  {
315  //Bottom, normal sign is negative (coordinate points into element)
316  case(-1):
317  face_element_pt->node_pt(0) = node_pt(0);
318  face_element_pt->bulk_node_number(0) = 0;
319  face_element_pt->normal_sign() = -1;
320 
321  //Set the pointer to the function that determines the bulk coordinates
322  //in the face element
323  face_element_pt->face_to_bulk_coordinate_fct_pt() =
324  &QElement1FaceToBulkCoordinates::face0;
325 
326  //Set the pointer to the function that determines the mapping of
327  //derivatives
328  face_element_pt->bulk_coordinate_derivatives_fct_pt() =
329  &QElement1BulkCoordinateDerivatives::faces0;
330 
331  //Set the number of values stored when the node is part of the "bulk"
332  //element.
333  face_element_pt->nbulk_value(0) = required_nvalue(0);
334  break;
335 
336  //Top, normal sign is positive (coordinate points out of element)
337  case(1):
338  face_element_pt->node_pt(0) = node_pt(1);
339  face_element_pt->bulk_node_number(0) = 1;
340  face_element_pt->normal_sign() = +1;
341 
342  //Set the pointer to the function that determines the bulk coordinates
343  //in the face element
344  face_element_pt->face_to_bulk_coordinate_fct_pt() =
345  &QElement1FaceToBulkCoordinates::face1;
346 
347  //Set the pointer to the function that determines the mapping of
348  //derivatives
349  face_element_pt->bulk_coordinate_derivatives_fct_pt() =
350  &QElement1BulkCoordinateDerivatives::faces0;
351 
352  //Set the number of values stored when the node is part of the "bulk"
353  //element.
354  face_element_pt->nbulk_value(0) = required_nvalue(1);
355  break;
356 
357  //Other cases
358  default:
359  std::ostringstream error_message;
360  error_message << "Face_index should only take "
361  << "the values +/-1, not " << face_index << std::endl;
362 
363  throw OomphLibError(error_message.str(),
364  OOMPH_CURRENT_FUNCTION,
365  OOMPH_EXCEPTION_LOCATION);
366  }
367 }
