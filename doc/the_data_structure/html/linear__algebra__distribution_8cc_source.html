33  /// Sets the distribution. Takes first_row, local_nrow and
34  /// global_nrow as arguments. If global_nrow is not provided or equal to
35  /// 0 then it is computed automatically

37  void LinearAlgebraDistribution::build(const OomphCommunicator*
38  const comm_pt,
39  const unsigned& first_row,
40  const unsigned& local_nrow,
41  const unsigned& global_nrow)
42  {
43  // copy the communicator
44  delete Comm_pt;
45  Comm_pt = new OomphCommunicator(*comm_pt);
46 
47  // get the rank and the number of processors
48  int my_rank = Comm_pt->my_rank();
49  int nproc = Comm_pt->nproc();
50 
51  // resize the storage
52  First_row.clear();
53  First_row.resize(nproc);
54  Nrow_local.clear();
55  Nrow_local.resize(nproc);
56 
57  // set first row and local nrow for this processor
58  First_row[my_rank] = first_row;
59  Nrow_local[my_rank] = local_nrow;
60 
61 #ifdef OOMPH_HAS_MPI
62  // gather the First_row vector
63  unsigned my_nrow_local = Nrow_local[my_rank];
64  MPI_Allgather(&my_nrow_local,1,MPI_UNSIGNED,
65  &Nrow_local[0],1,MPI_UNSIGNED,
66  Comm_pt->mpi_comm());
67 
68  // gather the Nrow_local vector
69  unsigned my_first_row = First_row[my_rank];
70  MPI_Allgather(&my_first_row,1,MPI_UNSIGNED,
71  &First_row[0],1,MPI_UNSIGNED,
72  Comm_pt->mpi_comm());
73 #endif
74 
75  // if global nrow is not provided then compute by summing local_nrow over
76  // all processors
77  if (global_nrow == 0)
78  {
79  if (nproc == 1)
80  {
81  Nrow = local_nrow;
82  }
83  else
84  {
85  Nrow = 0;
86  for (int p = 0; p < nproc; p++)
87  {
88  Nrow += Nrow_local[p];
89  }
90  }
91  }
92  else
93  {
94  Nrow = global_nrow;
95  }
96 
97  // the distribution is true, unless we only have 1 processor
98  if(nproc!=1) {Distributed = true;}
99  else {Distributed = false;}
100 
101 #ifdef OOMPH_HAS_MPI
102 #ifdef PARANOID
103  // paranoid check that the distribution works
104 
105 
106  // check that none of the processors partition overlap
107  for (int p = 0; p < nproc; p++)
108  {
109  if (Nrow_local[p] > 0)
110  {
111  for (int pp = p+1; pp < nproc; pp++)
112  {
113  if (Nrow_local[pp] > 0)
114  {
115  if ((First_row[p] >= First_row[pp] &&
116  First_row[p] < First_row[pp] + Nrow_local[pp]) ||
117  (First_row[p] + Nrow_local[p] -1 >= First_row[pp] &&
118  First_row[p] + Nrow_local[p] -1 <
119  First_row[pp] + Nrow_local[pp]))
120  {
121  std::ostringstream error_message;
122  error_message << "The distributed rows on processor " << p
123  << " and processor " << pp << " overlap.\n"
124  << "Processor " << p << " : first_row = "
125  << First_row[p] << ", nrow = "
126  << Nrow_local[p] << ".\n"
127  << "Processor " << pp << " : first_row = "
128  << First_row[pp] << ", nrow = "
129  << Nrow_local[pp] << ".\n";
130  throw OomphLibWarning(error_message.str(),
131  "LinearAlgebraDistribution::distribute(...)",
132  OOMPH_EXCEPTION_LOCATION);
133  }
134  }
135  }
136  }
137  }
138 
139  // check that no processor has a row with a global row index greater than
140  // the number of global rows
141  for (int p = 0; p < nproc; p++)
142  {
143  if (First_row[p] + Nrow_local[p] > Nrow)
144  {
145  std::ostringstream error_message;
146  error_message << "Processor " << p << " contains rows "
147  << First_row[p] << " to "
148  << First_row[p]+Nrow_local[p]-1
149  << " but there are only " << Nrow

151  throw OomphLibWarning(error_message.str(),
152  "LinearAlgebraDistribution::distribute(...)",

154  }
155  }
156 #endif
157 #endif

160  //============================================================================
161  /// \short Uniformly distribute global_nrow over all processors where
162  /// processors 0 holds approximately the first
163  /// global_nrow/n_proc, processor 1 holds the next
164  /// global_nrow/n_proc and so on...
165  //============================================================================
166  void LinearAlgebraDistribution::build(const OomphCommunicator*
167  const comm_pt,
168  const unsigned& global_nrow,
169  const bool& distribute)
170  {
171  // copy the communicator
172  delete Comm_pt;
173  Comm_pt = new OomphCommunicator(*comm_pt);
174 
175  // delete existing storage
176  First_row.clear();
177  Nrow_local.clear();
178 
179  // set global nrow
180  Nrow = global_nrow;
181 
182  // store the distributed flag
183  Distributed = distribute;
184 
185 #ifdef OOMPH_HAS_MPI
186 
187  // if distributed object then compute uniform distribution
188  if (distribute == true)
189  {
190 
191  // get the number of processors
192  int nproc = Comm_pt->nproc();
193 
194  // resize the storage
195  First_row.resize(nproc);
196  Nrow_local.resize(nproc);
197 
198  // compute first row
199  for (int p=0;p<nproc;p++)
200  {
201  First_row[p] = unsigned((double(p)/double(nproc))*double(global_nrow));
202  }
203 
204  // compute local nrow
205  for (int p=0; p<nproc-1; p++)
206  {
207  Nrow_local[p] = First_row[p+1] - First_row[p];
208  }
209  Nrow_local[nproc-1] = global_nrow -
210  First_row[nproc-1];
211  }
212 #endif
213  }

267  {

272  if (Distributed != other_dist.distributed())
273  {
274  return false;
275  }
276 
277  // if not distributed compare nrow
278  if (!Distributed)
279  {
280  if (other_dist.nrow() == Nrow)
281  {
282  return true;
283  }
284  return false;
285  }
286 
287  // compare
288  bool flag = true;
289  int nproc = Comm_pt->nproc();
290  for (int i = 0; i < nproc && flag == true; i++)
291  {
292  if (other_dist.first_row(i) != First_row[i] ||
293  other_dist.nrow_local(i) != Nrow_local[i])
294  {
295  flag = false;
296  }
297  }
298  return flag;
299 #else
300  if (other_dist.nrow() == Nrow)
301  {
302  return true;
303  }
304  return false;
305 #endif
306  }

329  /// \short Takes a vector of LinearAlgebraDistribution objects and
330  /// concatenates them such that the nrow_local of the out_distribution
331  /// is the sum of the nrow_local of all the in_distributions and the number
332  /// of global rows of the out_distribution is the sum of the number of global
333  /// rows of all the in_distributions.
334  /// This results in a permutation of the rows in the out_distribution.
335  /// Think of this in terms of DoubleVectors, if we have DoubleVectors with
336  /// distributions A and B, distributed across two processors (p0 and p1),
337  /// A: [a0] (on p0) B: [b0] (on p0)
338  /// [a1] (on p1) [b1] (on P1),
339  ///
340  /// then the out_distribution is
341  /// [a0 (on p0)
342  /// b0] (on p0)
343  /// [a1 (on p1)
344  /// b1] (on p1),
345  ///
346  /// as opposed to
347  /// [a0 (on p0)
348  /// a1] (on p0)
349  /// [b0 (on p1)
350  /// b1] (on p1).
351  ///
352  /// Note (1): The out_distribution may not be uniformly distributed even
353  /// if the in_distributions are uniform distributions.
354  /// Try this out with two distributions of global rows 3 and 5, uniformly
355  /// distributed across two processors. Compare this against a distribution
356  /// of global row 8 distributed across two processors.
357  ///
358  /// Note (2): There is no equivalent function which takes a Vector of
359  /// LinearAlgebraDistribution objects (as opposed to pointers), there should
360  /// not be one since we do not want to invoke the assignment operator when
361  /// creating the Vector of LinearAlgebraDistribution objects.
362  //===========================================================================
363  void concatenate(const Vector<LinearAlgebraDistribution*>&in_distribution_pt,
364  LinearAlgebraDistribution &out_distribution)
365  {
366  // How many distributions are in in_distribution?
367  unsigned ndistributions=in_distribution_pt.size();
368 
369 #ifdef PARANOID
370 
371  // Are any of the in_distribution pointers null?
372  // We do not want null pointers.
373  for (unsigned dist_i = 0; dist_i < ndistributions; dist_i++)
374  {
375  if(in_distribution_pt[dist_i]==0)
376  {
377  std::ostringstream error_message;
378  error_message
379  << "The pointer in_distribution_pt[" << dist_i << "] is null.\n";
380  throw OomphLibError(error_message.str(),
381  OOMPH_CURRENT_FUNCTION,
382  OOMPH_EXCEPTION_LOCATION);
383  }
384  }
385 
386  ////////////////////////////////
387 
388  // Check that all in distributions are built.
389  for (unsigned dist_i = 0; dist_i < ndistributions; dist_i++)
390  {
391  if(!in_distribution_pt[dist_i]->built())
392  {
393  std::ostringstream error_message;
394  error_message
395  << "The in_distribution_pt[" << dist_i << "] is not built.\n";
396  throw OomphLibError(error_message.str(),
397  OOMPH_CURRENT_FUNCTION,
398  OOMPH_EXCEPTION_LOCATION);
399  }
400  }
401 
402  ////////////////////////////////
403 
404  // Check that all communicators to concatenate are the same
405  // by comparing all communicators against the first one.
406  const OomphCommunicator first_comm
407  = *(in_distribution_pt[0]->communicator_pt());
408 
409  for (unsigned dist_i = 0; dist_i < ndistributions; dist_i++)
410  {
411  // Get the Communicator for the current vector.
412  const OomphCommunicator another_comm
413  = *(in_distribution_pt[dist_i]->communicator_pt());
414 
415  if(!(first_comm == another_comm))
416  {
417  std::ostringstream error_message;
418  error_message << "The communicator in position "<<dist_i<<" is \n"
419  << "not the same as the first one.\n";
420  throw OomphLibError(error_message.str(),
421  OOMPH_CURRENT_FUNCTION,
422  OOMPH_EXCEPTION_LOCATION);
423  }
424  }
425 
426  ////////////////////////////////
427 
428  // Ensure that all distributions are either distributed or not.
429  // This is because we use the distributed() function from the first
430  // distribution to indicate if the result distribution should be distributed
431  // or not.
432  bool first_distributed = in_distribution_pt[0]->distributed();
433  for (unsigned dist_i = 0; dist_i < ndistributions; dist_i++)
434  {
435  // Is the current distribution distributed?
436  bool another_distributed = in_distribution_pt[dist_i]->distributed();
437  if(first_distributed != another_distributed)
438  {
439  std::ostringstream error_message;
440  error_message
441  <<"The distribution in position "<<dist_i<<" has a different\n"
442  <<"different distributed boolean than the distribution.\n";
443  throw OomphLibError(error_message.str(),
444  OOMPH_CURRENT_FUNCTION,
445  OOMPH_EXCEPTION_LOCATION);
446  }
447  }
448 
449  ////////////////////////////////
450 
451  // Check that the out distribution is not built.
452  if(out_distribution.built())
453  {
454  std::ostringstream error_message;
455  error_message
456  << "The out distribution is built.\n"
457  << "Please clear it.\n";
458  throw OomphLibError(error_message.str(),
459  OOMPH_CURRENT_FUNCTION,
460  OOMPH_EXCEPTION_LOCATION);
461  }
462 
463 #endif
464 
465  // Get the communicator pointer
466  const OomphCommunicator* const comm_pt
467  = in_distribution_pt[0]->communicator_pt();
468 
469  // Number of processors
470  unsigned nproc = comm_pt->nproc();
471 
472  // First determine the out_nrow_local and the out_nrow (the global nrow)
473  unsigned out_nrow_local = 0;
474  unsigned out_nrow = 0;
475  for (unsigned in_dist_i = 0; in_dist_i < ndistributions; in_dist_i++)
476  {
477  out_nrow_local += in_distribution_pt[in_dist_i]->nrow_local();
478  out_nrow += in_distribution_pt[in_dist_i]->nrow();
479  }
480 
481  // Now calculate the first row for this processor. We need to know the
482  // out_nrow_local for all the other processors, MPI_Allgather must be used.
483  unsigned out_first_row = 0;
484 
485  // Distributed case: We need to gather all the out_nrow_local from all
486  // processors, the out_first_row for this processors is the partial sum up
487  // of the out_nrow_local to my_rank.
488  bool distributed = in_distribution_pt[0]->distributed();
489  if(distributed)
490  {
491 #ifdef OOMPH_HAS_MPI
492  // My rank
493  unsigned my_rank = comm_pt->my_rank();
494 
495  unsigned *out_nrow_local_all = new unsigned[nproc];
496  MPI_Allgather(&out_nrow_local,1,MPI_UNSIGNED,
497  out_nrow_local_all,1,MPI_UNSIGNED,
498  comm_pt->mpi_comm());
499  for (unsigned proc_i = 0; proc_i < my_rank; proc_i++)
500  {
501  out_first_row += out_nrow_local_all[proc_i];
502  }
503  delete [] out_nrow_local_all;
504 #endif
505  }
506  // Non-distributed case
507  else
508  {
509  out_first_row = 0;
510  }
511 
512  // Build the distribution.
513  if(nproc == 1 || !distributed)
514  {
515  // Some ambiguity here -- on a single processor it doesn't really
516  // matter if we think of ourselves as distributed or not but this
517  // follows the pattern used elsewhere.
518  out_distribution.build(comm_pt,out_nrow,false);
519  }
520  else
521  {
522  out_distribution.build(comm_pt,out_first_row,out_nrow_local,out_nrow);
523  }
524  } // function concatenate
525 
526  } // end of namespace LinearAlgebraDistributionHelpers
