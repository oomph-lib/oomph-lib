316 /// Solver: Takes pointer to problem and returns the results vector
317 /// which contains the solution of the linear system defined by
318 /// the problem's fully assembled Jacobian and residual vector.
319 void solve(Problem* const& problem_pt, DoubleVector& result);
320
321 /// Linear-algebra-type solver: Takes pointer to a matrix and rhs
322 /// vector and returns the solution of the linear system.
323 void solve(DoubleMatrixBase* const& matrix_pt,
324 const DoubleVector& rhs,
325 DoubleVector& solution)
326 {
327 // Store the matrix if required
328 if ((Enable_resolve) && (!Resolving))
329 {
330 Matrix_pt = dynamic_cast<MATRIX*>(matrix_pt);
331
332 // Matrix has been passed in from the outside so we must not
333 // delete it
334 Matrix_can_be_deleted = false;
335 }
336
337 // set the distribution
338 if (dynamic_cast<DistributableLinearAlgebraObject*>(matrix_pt))
339 {
340 // the solver has the same distribution as the matrix if possible
341 this->build_distribution(
342 dynamic_cast<DistributableLinearAlgebraObject*>(matrix_pt)
343 ->distribution_pt());
344 }
345 else
346 {
347 // the solver has the same distribution as the RHS
348 this->build_distribution(rhs.distribution_pt());
349 }
350
351 // Call the helper function
352 this->solve_helper(matrix_pt, rhs, solution);
353 }
354
355 /// Re-solve the system defined by the last assembled Jacobian
356 /// and the rhs vector specified here. Solution is returned in the
357 /// vector result.
358 void resolve(const DoubleVector& rhs, DoubleVector& result);
359
360 /// Number of iterations taken
361 unsigned iterations() const
362 {
363 return Iterations;
364 }
365
366
367 private:
368 /// General interface to solve function
369 void solve_helper(DoubleMatrixBase* const& matrix_pt,
370 const DoubleVector& rhs,
371 DoubleVector& solution);
372
373
374 /// Cleanup data that's stored for resolve (if any has been stored)
375 void clean_up_memory()
376 {
377 if ((Matrix_pt != 0) && (Matrix_can_be_deleted))
378 {
379 delete Matrix_pt;
380 Matrix_pt = 0;
381 }
382 }
383
384 /// Number of iterations taken
385 unsigned Iterations;
386
387 /// Pointer to matrix
388 MATRIX* Matrix_pt;
389
390 /// Boolean flag to indicate if the solve is done in re-solve mode,
391 /// bypassing setup of matrix and preconditioner
392 bool Resolving;
393
394 /// Boolean flag to indicate if the matrix pointed to be Matrix_pt
395 /// can be deleted.
396 bool Matrix_can_be_deleted;
397 };

560 /// the same as the max. number of iterations in the underlying

574 const DoubleVector& rhs,
575 DoubleVector& solution,
576 const double& n_dof);
577
578 protected:

629 /// Set up the smoother for the matrix specified by the pointer
630 void smoother_setup(DoubleMatrixBase* matrix_pt)
631 {
632 // Assume the matrix has been passed in from the outside so we must
633 // not delete it. This is needed to avoid pre- and post-smoothers
634 // deleting the same matrix in the MG solver. If this was originally
635 // set to TRUE then this will be sorted out in the other functions
636 // from which this was called
637 Matrix_can_be_deleted = false;
638

640 Matrix_pt = dynamic_cast<MATRIX*>(matrix_pt);
641 } // End of smoother_setup

643 /// The smoother_solve function performs fixed number of iterations
644 /// on the system A*result=rhs. The number of (smoothing) iterations is
645 /// the same as the max. number of iterations in the underlying
646 /// IterativeLinearSolver class.
647 void smoother_solve(const DoubleVector& rhs, DoubleVector& result)
648 {
649 // If you use a smoother but you don't want to calculate the residual
650 Use_as_smoother = true;
651
652 // Call the helper function
653 solve_helper(Matrix_pt, rhs, result);
654 } // End of smoother_setup
655
656 /// Solver: Takes pointer to problem and returns the results
657 /// vector which contains the solution of the linear system defined
658 /// by the problem's fully assembled Jacobian and residual vector.
659 void solve(Problem* const& problem_pt, DoubleVector& result);
660
661 /// Linear-algebra-type solver: Takes pointer to a matrix and rhs
662 /// vector and returns the solution of the linear system.
663 void solve(DoubleMatrixBase* const& matrix_pt,
664 const DoubleVector& rhs,
665 DoubleVector& solution)
666 {
667 // Reset the Use_as_smoother_flag as the solver is not being used
668 // as a smoother
669 Use_as_smoother = false;
670
671 // Set up the distribution
672 this->build_distribution(rhs.distribution_pt());
673
674 // Store the matrix if required
675 if ((Enable_resolve) && (!Resolving))
676 {
677 // Upcast to the appropriate matrix type
678 Matrix_pt = dynamic_cast<MATRIX*>(matrix_pt);
679 }
680
681 // Matrix has been passed in from the outside so we must not delete it
682 Matrix_can_be_deleted = false;
683
684 // Call the helper function
685 this->solve_helper(matrix_pt, rhs, solution);
686 } // End of solve

699 /// Re-solve the system defined by the last assembled Jacobian
700 /// and the rhs vector specified here. Solution is returned in the
701 /// vector result.
702 void resolve(const DoubleVector& rhs, DoubleVector& result)
703 {
704 // We are re-solving
705 Resolving = true;
706
707#ifdef PARANOID
708 if (Matrix_pt == 0)
709 {
710 throw OomphLibError("No matrix was stored -- cannot re-solve",
711 OOMPH_CURRENT_FUNCTION,
712 OOMPH_EXCEPTION_LOCATION);
713 }
714#endif
715
716 // Call linear algebra-style solver
717 solve(Matrix_pt, rhs, result);
718
719 // Reset re-solving flag
720 Resolving = false;
721 } // End of resolve
722

725 {
726 throw OomphLibError(
727 "Gauss Seidel is not a preconditionable iterative solver",

731 } // End of preconditioner_setup_time

743 DoubleVector& solution);
744
745 /// Cleanup data that's stored for resolve (if any has been stored)
746 void clean_up_memory()
747 {
748 // If the matrix pointer isn't null and we're allowed to delete it
749 // delete the matrix and assign the pointer the value NULL
750 if ((Matrix_pt != 0) && (Matrix_can_be_deleted))
751 {
752 // Destroy the matrix
753 delete Matrix_pt;
754
755 // Make it a null pointer
756 Matrix_pt = 0;
757 }
758 } // End of clean_up_memory
759

762
763 /// Number of iterations taken
764 unsigned Iterations;
765
766 /// Boolean flag to indicate if the solve is done in re-solve mode,
767 /// bypassing setup of matrix and preconditioner
768 bool Resolving;
769
770 /// Boolean flag to indicate if the matrix pointed to be Matrix_pt
771 /// can be deleted.
772 bool Matrix_can_be_deleted;
773 };

1366 /// Helper function to update the result vector using the result,
1367 /// x=x_0+V_m*y
1368 void update(const unsigned& k,
1369 const Vector<Vector<double>>& H,
1370 const Vector<double>& s,
1371 const Vector<DoubleVector>& v,
1372 DoubleVector& x)
1373 {
1374 // Make a local copy of s
1375 Vector<double> y(s);
1376
1377 // Backsolve:
1378 for (int i = int(k); i >= 0; i--)
1379 {
1380 // Divide the i-th entry of y by the i-th diagonal entry of H
1381 y[i] /= H[i][i];
1382
1383 // Loop over the previous values of y and update
1384 for (int j = i - 1; j >= 0; j--)
1385 {
1386 // Update the j-th entry of y
1387 y[j] -= H[i][j] * y[i];
1388 }
1389 } // for (int i=int(k);i>=0;i--)
1390
1391 // Store the number of rows in the result vector
1392 unsigned n_x = x.nrow();
1393
1394 // Build a temporary vector with entries initialised to 0.0
1395 DoubleVector temp(x.distribution_pt(), 0.0);
1396
1397 // Build a temporary vector with entries initialised to 0.0
1398 DoubleVector z(x.distribution_pt(), 0.0);
1399
1400 // Get access to the underlying values
1401 double* temp_pt = temp.values_pt();
1402
1403 // Calculate x=Vy
1404 for (unsigned j = 0; j <= k; j++)
1405 {
1406 // Get access to j-th column of v
1407 const double* vj_pt = v[j].values_pt();
1408
1409 // Loop over the entries of the vector, temp
1410 for (unsigned i = 0; i < n_x; i++)
1411 {
1412 temp_pt[i] += vj_pt[i] * y[j];
1413 }
1414 } // for (unsigned j=0;j<=k;j++)
1415
1416 // If we're using LHS preconditioning
1417 if (Preconditioner_LHS)
1418 {
1419 // Since we're using LHS preconditioning the preconditioner is applied
1420 // to the matrix and RHS vector so we simply update the value of x
1421 x += temp;
1422 }
1423 // If we're using RHS preconditioning
1424 else
1425 {
1426 // Start the timer
1427 double t_start_prec = TimingHelpers::timer();
1428
1429 // Since we're using RHS preconditioning the preconditioner is applied
1430 // to the solution vector
1431 preconditioner_pt()->preconditioner_solve(temp, z);
1432
1433 // Calculate the time taken for the preconditioner solve
1434 Preconditioner_application_time +=
1435 (TimingHelpers::timer() - t_start_prec);
1436
1437 // Use the update: x_m=x_0+inv(M)Vy [see Saad Y,"Iterative methods for
1438 // sparse linear systems", p.284]
1439 x += z;
1440 }
1441 } // End of update

1443 /// Helper function: Generate a plane rotation. This is done by
1444 /// finding the values of \f$ \cos(\theta) \f$ (i.e. cs) and \sin(\theta)
1445 /// (i.e. sn) such that:
1446 /// \f[
1447 /// \begin{bmatrix}
1448 /// \cos\theta & \sin\theta \newline
1449 /// -\sin\theta & \cos\theta
1450 /// \end{bmatrix}
1451 /// \begin{bmatrix}
1452 /// dx \newline
1453 /// dy
1454 /// \end{bmatrix}
1455 /// =
1456 /// \begin{bmatrix}
1457 /// r \newline
1458 /// 0
1459 /// \end{bmatrix},
1460 /// \f]
1461 /// where \f$ r=\sqrt{pow(dx,2)+pow(dy,2)} \f$. The values of a and b are
1462 /// given by:
1463 /// \f[
1464 /// \cos\theta&=\dfrac{dx}{\sqrt{pow(dx,2)+pow(dy,2)}},
1465 /// \f]
1466 /// and
1467 /// \f[
1468 /// \sin\theta&=\dfrac{dy}{\sqrt{pow(dx,2)+pow(dy,2)}}.
1469 /// \f]
1470 /// Taken from: Saad Y."Iterative methods for sparse linear systems", p.192
1471 void generate_plane_rotation(double& dx, double& dy, double& cs, double& sn)
1472 {
1473 // If dy=0 then we do not need to apply a rotation
1474 if (dy == 0.0)
1475 {
1476 // Using theta=0 gives cos(theta)=1
1477 cs = 1.0;
1478
1479 // Using theta=0 gives sin(theta)=0
1480 sn = 0.0;
1481 }
1482 // If dx or dy is large using the normal form of calculting cs and sn
1483 // is naive since this may overflow or underflow so instead we calculate
1484 // r=sqrt(pow(dx,2)+pow(dy,2)) by using r=|dy|sqrt(1+pow(dx/dy,2)) if
1485 // |dy|>|dx| [see <A
1486 // HREF=https://en.wikipedia.org/wiki/Hypot">Hypot</A>.].
1487 else if (fabs(dy) > fabs(dx))
1488 {
1489 // Since |dy|>|dx| calculate the ratio dx/dy
1490 double temp = dx / dy;
1491
1492 // Calculate sin(theta)=dy/sqrt(pow(dx,2)+pow(dy,2))
1493 sn = 1.0 / sqrt(1.0 + temp * temp);
1494
1495 // Calculate cos(theta)=dx/sqrt(pow(dx,2)+pow(dy,2))=(dx/dy)*sin(theta)
1496 cs = temp * sn;
1497 }
1498 // Otherwise, we have |dx|>=|dy| so to, again, avoid overflow or underflow
1499 // calculate the values of cs and sn using the method above
1500 else
1501 {
1502 // Since |dx|>=|dy| calculate the ratio dy/dx
1503 double temp = dy / dx;
1504
1505 // Calculate cos(theta)=dx/sqrt(pow(dx,2)+pow(dy,2))
1506 cs = 1.0 / sqrt(1.0 + temp * temp);
1507
1508 // Calculate sin(theta)=dy/sqrt(pow(dx,2)+pow(dy,2))=(dy/dx)*cos(theta)
1509 sn = temp * cs;
1510 }
1511 } // End of generate_plane_rotation
1512
1513 /// Helper function: Apply plane rotation. This is done using the
1514 /// update:
1515 /// \f[
1516 /// \begin{bmatrix}
1517 /// dx \newline
1518 /// dy
1519 /// \end{bmatrix}
1520 /// \leftarrow
1521 /// \begin{bmatrix}
1522 /// \cos\theta & \sin\theta \newline
1523 /// -\sin\theta & \cos\theta
1524 /// \end{bmatrix}
1525 /// \begin{bmatrix}
1526 /// dx \newline
1527 /// dy
1528 /// \end{bmatrix}.
1529 /// \f]
1530 void apply_plane_rotation(double& dx, double& dy, double& cs, double& sn)
1531 {
1532 // Calculate the value of dx but don't update it yet
1533 double temp = cs * dx + sn * dy;
1534
1535 // Set the value of dy
1536 dy = -sn * dx + cs * dy;
1537
1538 // Set the value of dx using the correct values of dx and dy
1539 dx = temp;
1540 }

1543 unsigned Iterations;
1544
1545 /// The number of iterations before the iteration proceedure is
1546 /// restarted if iteration restart is used
1547 unsigned Restart;
1548
1549 /// boolean indicating if iteration restarting is used
1550 bool Iteration_restart;
1551
1552 /// Pointer to matrix
1553 MATRIX* Matrix_pt;
1554
1555 /// Boolean flag to indicate if the solve is done in re-solve mode,
1556 /// bypassing setup of matrix and preconditioner
1557 bool Resolving;
1558
1559 /// Boolean flag to indicate if the matrix pointed to be Matrix_pt
1560 /// can be deleted.
1561 bool Matrix_can_be_deleted;
1562
1563 /// boolean indicating use of left hand preconditioning (if true)
1564 /// or right hand preconditioning (if false)
1565 bool Preconditioner_LHS;
1566
1567 /// Storage for the time spent applying the preconditioner
1568 double Preconditioner_application_time;
1569 };

1730 const DoubleVector& x,
1731 DoubleVector& soln)

1734 unsigned n_dof = matrix_pt->nrow();
1735
1736 // Allocate space for the non-augmented component of x
1737 DoubleVector x_small(this->distribution_pt(), 0.0);

1766 /// Apply the block-diagonal Schur complement preconditioner to
1767 /// compute the LHS which has size N+1 (the augmented system size)
1768 void apply_schur_complement_preconditioner(const DoubleVector& rhs,
1769 DoubleVector& soln)
1770 {
1771 // How many dofs are there in the non-augmented system?
1772 unsigned n_dof = this->distribution_pt()->nrow();
1773
1774 // Compute the final entry of r first
1775 soln[n_dof] = rhs[n_dof] / Schur_complement_scalar;
1776
1777 // Do we want to use a block-diagonal approximation? The default is to
1778 // use a block upper-triangular approximation for the preconditioner
1779 bool use_block_diagonal_preconditioner = false;
1780
1781 // Allocate space for (the non-augmented part of) r satisfying b-Jx = Mr
1782 DoubleVector r_small(this->distribution_pt(), 0.0);
1783
1784 // Allocate space for the non-augmented system part of the RHS vector
1785 DoubleVector rhs_small(this->distribution_pt(), 0.0);
1786
1787 // Loop over the first n_dof entries of the RHS vector
1788 for (unsigned i = 0; i < n_dof; i++)
1789 {
1790 // If we want to use the block-diagonal preconditioner
1791 if (use_block_diagonal_preconditioner)
1792 {
1793 // Copy the i-th entry over
1794 rhs_small[i] = rhs[i];
1795 }
1796 // If we want to use the block upper-triangular preconditioner
1797 else
1798 {
1799 // Copy the i-th entry over
1800 rhs_small[i] = rhs[i] - soln[n_dof] * (*B_pt)[i];
1801 }
1802 } // for (unsigned i=0;i<n_dof;i++)
1803
1804 // Apply the preconditioner
1805 preconditioner_pt()->preconditioner_solve(rhs_small, r_small);
1806
1807 // Loop over the first n_dof entries of the solution vector
1808 for (unsigned i = 0; i < n_dof; i++)
1809 {
1810 // Copy the i-th entry over
1811 soln[i] = r_small[i];
1812 }
1813 } // End of apply_schur_complement_preconditioner

1820 /// Helper function to update the result vector using the result,
1821 /// x=x_0+V_m*y
1822 void update(const unsigned& k,
1823 const Vector<Vector<double>>& H,
1824 const Vector<double>& s,
1825 const Vector<DoubleVector>& v,
1826 DoubleVector& x)
1827 {
1828 // Make a local copy of s
1829 Vector<double> y(s);
1830
1831 // Backsolve:
1832 for (int i = int(k); i >= 0; i--)
1833 {
1834 // Divide the i-th entry of y by the i-th diagonal entry of H
1835 y[i] /= H[i][i];
1836
1837 // Loop over the previous values of y and update
1838 for (int j = i - 1; j >= 0; j--)
1839 {
1840 // Update the j-th entry of y
1841 y[j] -= H[i][j] * y[i];
1842 }
1843 } // for (int i=int(k);i>=0;i--)
1844
1845 // Store the number of rows in the result vector
1846 unsigned n_x = x.nrow();
1847
1848 // Build a temporary vector with entries initialised to 0.0
1849 DoubleVector temp(x.distribution_pt(), 0.0);
1850
1851 // Build a temporary vector with entries initialised to 0.0
1852 DoubleVector z(x.distribution_pt(), 0.0);
1853
1854 // Get access to the underlying values
1855 double* temp_pt = temp.values_pt();
1856
1857 // Calculate x=Vy
1858 for (unsigned j = 0; j <= k; j++)
1859 {
1860 // Get access to j-th column of v
1861 const double* vj_pt = v[j].values_pt();
1862
1863 // Loop over the entries of the vector, temp
1864 for (unsigned i = 0; i < n_x; i++)
1865 {
1866 temp_pt[i] += vj_pt[i] * y[j];
1867 }
1868 } // for (unsigned j=0;j<=k;j++)
1869
1870 // If we're using LHS preconditioning
1871 if (Preconditioner_LHS)
1872 {
1873 // Since we're using LHS preconditioning the preconditioner is applied
1874 // to the matrix and RHS vector so we simply update the value of x
1875 x += temp;
1876 }
1877 // If we're using RHS preconditioning
1878 else
1879 {
1880 // Since we're using RHS preconditioning the preconditioner is applied
1881 // to the solution vector
1882 apply_schur_complement_preconditioner(temp, z);
1883
1884 // Use the update: x_m=x_0+inv(M)Vy [see Saad Y,"Iterative methods for
1885 // sparse linear systems", p.284]
1886 x += z;
1887 }
1888 } // End of update

1890 /// Helper function: Generate a plane rotation. This is done by
1891 /// finding the values of \f$ \cos(\theta) \f$ (i.e. cs) and \sin(\theta)
1892 /// (i.e. sn) such that:
1893 /// \f[
1894 /// \begin{bmatrix}
1895 /// \cos\theta & \sin\theta \newline
1896 /// -\sin\theta & \cos\theta
1897 /// \end{bmatrix}
1898 /// \begin{bmatrix}
1899 /// dx \newline
1900 /// dy
1901 /// \end{bmatrix}
1902 /// =
1903 /// \begin{bmatrix}
1904 /// r \newline
1905 /// 0
1906 /// \end{bmatrix},
1907 /// \f]
1908 /// where \f$ r=\sqrt{pow(dx,2)+pow(dy,2)} \f$. The values of a and b are
1909 /// given by:
1910 /// \f[
1911 /// \cos\theta&=\dfrac{dx}{\sqrt{pow(dx,2)+pow(dy,2)}},
1912 /// \f]
1913 /// and
1914 /// \f[
1915 /// \sin\theta&=\dfrac{dy}{\sqrt{pow(dx,2)+pow(dy,2)}}.
1916 /// \f]
1917 /// Taken from: Saad Y."Iterative methods for sparse linear systems", p.192
1918 void generate_plane_rotation(double& dx, double& dy, double& cs, double& sn)
1919 {
1920 // If dy=0 then we do not need to apply a rotation
1921 if (dy == 0.0)
1922 {
1923 // Using theta=0 gives cos(theta)=1
1924 cs = 1.0;
1925
1926 // Using theta=0 gives sin(theta)=0
1927 sn = 0.0;
1928 }
1929 // If dx or dy is large using the normal form of calculting cs and sn
1930 // is naive since this may overflow or underflow so instead we calculate
1931 // r=sqrt(pow(dx,2)+pow(dy,2)) by using r=|dy|sqrt(1+pow(dx/dy,2)) if
1932 // |dy|>|dx| [see <A
1933 // HREF=https://en.wikipedia.org/wiki/Hypot">Hypot</A>.].
1934 else if (fabs(dy) > fabs(dx))
1935 {
1936 // Since |dy|>|dx| calculate the ratio dx/dy
1937 double temp = dx / dy;
1938
1939 // Calculate sin(theta)=dy/sqrt(pow(dx,2)+pow(dy,2))
1940 sn = 1.0 / sqrt(1.0 + temp * temp);
1941
1942 // Calculate cos(theta)=dx/sqrt(pow(dx,2)+pow(dy,2))=(dx/dy)*sin(theta)
1943 cs = temp * sn;
1944 }
1945 // Otherwise, we have |dx|>=|dy| so to, again, avoid overflow or underflow
1946 // calculate the values of cs and sn using the method above
1947 else
1948 {
1949 // Since |dx|>=|dy| calculate the ratio dy/dx
1950 double temp = dy / dx;
1951
1952 // Calculate cos(theta)=dx/sqrt(pow(dx,2)+pow(dy,2))
1953 cs = 1.0 / sqrt(1.0 + temp * temp);
1954
1955 // Calculate sin(theta)=dy/sqrt(pow(dx,2)+pow(dy,2))=(dy/dx)*cos(theta)
1956 sn = temp * cs;
1957 }
1958 } // End of generate_plane_rotation

1960 /// Helper function: Apply plane rotation. This is done using the
1961 /// update:
1962 /// \f[
1963 /// \begin{bmatrix}
1964 /// dx \newline
1965 /// dy
1966 /// \end{bmatrix}
1967 /// \leftarrow
1968 /// \begin{bmatrix}
1969 /// \cos\theta & \sin\theta \newline
1970 /// -\sin\theta & \cos\theta
1971 /// \end{bmatrix}
1972 /// \begin{bmatrix}
1973 /// dx \newline
1974 /// dy
1975 /// \end{bmatrix}.
1976 /// \f]
1977 void apply_plane_rotation(double& dx, double& dy, double& cs, double& sn)
1978 {
1979 // Calculate the value of dx but don't update it yet
1980 double temp = cs * dx + sn * dy;
1981
1982 // Set the value of dy
1983 dy = -sn * dx + cs * dy;
1984
1985 // Set the value of dx using the correct values of dx and dy
1986 dx = temp;
1987 }
