44 /// Get refinement pattern of mesh: Consider the hypothetical mesh
45 /// obtained by truncating the refinement of the current mesh to a given level
46 /// (where \c level=0 is the un-refined base mesh). To advance
47 /// to the next refinement level, we need to refine (split) the
48 /// \c to_be_refined[level].size() elements identified by the
49 /// element numbers contained in \c vector to_be_refined[level][...]
50 //========================================================================
51 void TreeBasedRefineableMeshBase::get_refinement_pattern(
52  Vector<Vector<unsigned> >& to_be_refined)
53 {
54  // Extract *all* elements from current (fully refined) mesh.
55  Vector<Tree*> all_tree_nodes_pt;
56  forest_pt()->stick_all_tree_nodes_into_vector(all_tree_nodes_pt);
57 
58  // Find out maximum refinement level
59  unsigned max_level=0;
60  unsigned nnodes=all_tree_nodes_pt.size();
61  for (unsigned e=0;e<nnodes;e++)
62  {
63  unsigned level=all_tree_nodes_pt[e]->level();
64  if (level>max_level) max_level=level;
65  }
66 
67  // Assign storage for refinement pattern
68  to_be_refined.clear();
69  to_be_refined.resize(max_level);
70  Vector<unsigned> el_count(max_level);
71 
72  // Initialise count of elements that exist in mesh when refinement
73  // has proceeded to this level
74  for (unsigned l=0;l<max_level;l++)
75  {
76  el_count[l]=0;
77  }
78 
79  // Loop over all levels and extract all elements that exist
80  // in reference mesh when refinement has proceeded to this level
81  for (unsigned l=0;l<max_level;l++)
82  {
83  // Loop over all elements (tree nodes)
84  for (unsigned e=0;e<nnodes;e++)
85  {
86  // What level does this element exist on?
87  unsigned level=all_tree_nodes_pt[e]->level();
88 
89  // Element is part of the mesh at this refinement level
90  // if it exists at this level OR if it exists at a lower level
91  // and is a leaf
92  if ((level==l)||((level<l)&&(all_tree_nodes_pt[e]->is_leaf())))
93  {
94  // If element exsts at this level and is not a leaf it will
95  // be refined when we move to the next level:
96  if ((level==l)&&(!all_tree_nodes_pt[e]->is_leaf()))
97  {
98  // Add element number (in mesh at current refinement level)
99  // to the list of elements that need to be refined
100  to_be_refined[l].push_back(el_count[l]);
101  }
102  // Element exists in this mesh: Add to counter
103  el_count[l]++;
104  }
105  }
106  }
107 }

140 void TreeBasedRefineableMeshBase::refine_base_mesh(Vector<Vector<unsigned> >&
141  to_be_refined)
142 {
143  // Get mesh back to unrefined state
144  unsigned my_max,my_min;
145  get_refinement_levels(my_min,my_max);
146 
147  // Max refinement level:
148  unsigned my_max_level=to_be_refined.size();
149 
150  unsigned global_max=0;
151  unsigned global_max_level=0;
152  Vector<unsigned> data(2,0);
153  data[0]=my_max;
154  data[1]=my_max_level;
155  Vector<unsigned> global_data(2,0);
156 #ifdef OOMPH_HAS_MPI
157  if (this->is_mesh_distributed())
158  {
159  MPI_Allreduce(&data[0],&global_data[0],2,MPI_UNSIGNED,MPI_MAX,
160  Comm_pt->mpi_comm());
161  global_max=global_data[0];
162  global_max_level=global_data[1];
163 
164  }
165  else
166 #endif
167  {
168  global_max=my_max;
169  global_max_level=my_max_level;
170  }
171 
172 
173  for (unsigned i=0;i<global_max;i++)
174  {
175  unrefine_uniformly();
176  }
177 
178  // Do refinement steps in current mesh
179  for (unsigned l=0;l<global_max_level;l++)
180  {
181  // Loop over elements that need to be refined at this level
182  unsigned n_to_be_refined=0;
183  if (l<my_max_level) n_to_be_refined=to_be_refined[l].size();
184 
185  // Select relevant elements to be refined
186  for (unsigned i=0;i<n_to_be_refined;i++)
187  {
188  dynamic_cast<RefineableElement*>(
189  this->element_pt(to_be_refined[l][i]))->select_for_refinement();
190  }
191 
192  // Now do the actual mesh refinement
193  adapt_mesh();
194  }

300 //========================================================================
301 /// Do adaptive refinement for mesh.
302 /// - Pass Vector of error estimates for all elements.
303 /// - Refine those whose errors exceeds the threshold
304 /// - (Try to) unrefine those whose errors is less than
305 /// threshold (only possible if the three brothers also want to be
306 /// unrefined, of course.)
307 /// - Update the nodal positions in the whole lot
308 /// - Store # of refined/unrefined elements.
309 /// - Doc refinement process (if required)
310 //========================================================================
311 void TreeBasedRefineableMeshBase::adapt(const Vector<double>& elemental_error)
312  {
313  //Set the refinement tolerance to be the max permissible error
314  double refine_tol=max_permitted_error();
315 
316  //Set the unrefinement tolerance to be the min permissible error
317  double unrefine_tol=min_permitted_error();
318 
319  // Setup doc info
320  DocInfo local_doc_info;
321  if (doc_info_pt()==0) {local_doc_info.disable_doc();}
322  else {local_doc_info=doc_info();}
323 
324 
325  // Check that the errors make sense
326  if (refine_tol<=unrefine_tol)
327  {
328  std::ostringstream error_stream;
329  error_stream << "Refinement tolerance <= Unrefinement tolerance"
330  << refine_tol << " " << unrefine_tol << std::endl
331  << "doesn't make sense and will almost certainly crash"
332  << std::endl
333  << "this beautiful code!" << std::endl;
334 
335  throw OomphLibError(error_stream.str(),
336  OOMPH_CURRENT_FUNCTION,
337  OOMPH_EXCEPTION_LOCATION);
338  }
339 
340 
341  // Select elements for refinement and unrefinement
342  //================================================
343  // Reset counter for number of elements that would like to be
344  // refined further but can't
345  nrefinement_overruled()=0;
346 
347  // Note: Yes, this needs to be a map because we'll have to check
348  // the refinement wishes of brothers (who we only access via pointers)
349  std::map<RefineableElement*,bool> wants_to_be_unrefined;
350 
351  // Initialise a variable to store the number of elements for refinment
352  unsigned n_refine=0;
353 
354  // Loop over all elements and mark them according to the error criterion
355  unsigned long Nelement=this->nelement();
356  for (unsigned long e=0;e<Nelement;e++)
357  {
358  // (Cast) pointer to the element
359  RefineableElement* el_pt =
360  dynamic_cast<RefineableElement*>(this->element_pt(e));
361 
362  // Initially element is not to be refined
363  el_pt->deselect_for_refinement();
364 
365  // If the element error exceeds the threshold ...
366  if(elemental_error[e] > refine_tol)
367  {
368  // ... and its refinement level is less than the maximum desired level
369  // mark is to be refined
370  if ((el_pt->refinement_is_enabled())&&
371  (el_pt->refinement_level() < max_refinement_level()))
372  {
373  el_pt->select_for_refinement();
374  n_refine++;
375  }
376  // ... otherwise mark it as having been over-ruled
377  else
378  {
379  nrefinement_overruled()+=1;
380  }
381  }
382 
383  // Now worry about unrefinement (first pass):
384 
385  // Is my error too small AND do I have a father?
386  if ((elemental_error[e]<unrefine_tol)&&
387  (el_pt->tree_pt()->father_pt()!=0))
388  {
389  // Flag to indicate whether to unrefine
390  bool unrefine=true;
391  unsigned n_sons = el_pt->tree_pt()->father_pt()->nsons();
392 
393  // Are all brothers leaf nodes?
394  for (unsigned ison=0;ison<n_sons;ison++)
395  {
396  // (At least) one brother is not a leaf: end of story; we're not doing
397  // it (= the unrefinement)
398  if (!(el_pt->tree_pt()->father_pt()->son_pt(ison)->is_leaf()))
399  {unrefine=false;}
400  }
401 
402  // Don't allow unrefinement of elements that would become larger
403  // than the minimum legal refinement level
404  if (el_pt->refinement_level()-1<min_refinement_level())
405  {unrefine=false;}
406 
407  // So, all things considered, is the element eligbible for refinement?
408  if(unrefine) {wants_to_be_unrefined[el_pt]=true;}
409  else {wants_to_be_unrefined[el_pt]=false;}
410  }
411  }
412 
413  oomph_info << " \n Number of elements to be refined: "
414  << n_refine << std::endl;
415  oomph_info << " \n Number of elements whose refinement was overruled: "
416  << nrefinement_overruled() << std::endl;
417 
418  // Second pass for unrefinement --- an element cannot be unrefined unless
419  // all brothers want to be unrefined.
420  // Loop over all elements again and let the first set of sons check if their
421  // brothers also want to be unrefined
422  unsigned n_unrefine=0;
423  for (unsigned long e=0;e<Nelement;e++)
424  {
425  //(Cast) pointer to the element
426  RefineableElement* el_pt =
427  dynamic_cast<RefineableElement*>(this->element_pt(e));
428 
429  // hierher: This is a bit naughty... We want to put the
430  // first son in charge -- the statement below assumes (correctly) that the
431  // enumeration of all (!) trees starts with son types.
432  // This is correct for oc and quadtrees but will bite us if we
433  // ever introduce other trees if/when we accidentally break this
434  // tacit assumption. Not sure what to do about it for
435  // now other than leaving it hierher-ed...
436  if (el_pt->tree_pt()->son_type()==OcTreeNames::LDB)
437  {
438  // Do all sons want to be unrefined?
439  bool unrefine=true;
440  unsigned n_sons = el_pt->tree_pt()->father_pt()->nsons();
441  for (unsigned ison=0;ison<n_sons;ison++)
442  {
443  if (!(wants_to_be_unrefined[
444  dynamic_cast<RefineableElement*>
445  (el_pt->tree_pt()->father_pt()->
446  son_pt(ison)->object_pt())]))
447  {
448  // One guy isn't cooperating and spoils the party.
449  unrefine=false;
450  }
451  }
452 
453  // Tell father that his sons need to be merged
454  if(unrefine)
455  {
456  el_pt->tree_pt()->father_pt()->object_pt()->
457  select_sons_for_unrefinement();
458  n_unrefine += n_sons;
459  }
460  //Otherwise mark the sons as not to be touched
461  else
462  {
463  el_pt->tree_pt()->father_pt()->object_pt()->
464  deselect_sons_for_unrefinement();
465  }
466  }
467  }
468  oomph_info << " \n Number of elements to be merged : "
469  << n_unrefine << std::endl << std::endl;
470 
471 
472  //Now do the actual mesh adaptation
473  //---------------------------------
474 
475  // Check whether its worth our while
476  // Either some elements want to be refined,
477  // or the number that want to be unrefined are greater than the
478  // specified tolerance
479 
480  // In a parallel job, it is possible that one process may not have
481  // any elements to refine, BUT a neighbouring process may refine an
482  // element which changes the hanging status of a node that is on
483  // both processes (i.e. a halo(ed) node). To get around this issue,
484  // ALL processes need to call adapt_mesh if ANY refinement is to
485  // take place anywhere.
486 
487  unsigned total_n_refine=0;
488 #ifdef OOMPH_HAS_MPI
489  // Sum n_refine across all processors
490  if (this->is_mesh_distributed())
491  {
492  MPI_Allreduce(&n_refine,&total_n_refine,1,MPI_UNSIGNED,MPI_SUM,
493  Comm_pt->mpi_comm());
494  }
495  else
496  {
497  total_n_refine=n_refine;
498  }
499 #else
500  total_n_refine=n_refine;
501 #endif
502 
503  // There may be some issues with unrefinement too, but I have not
504  // been able to come up with an example (either in my head or in a
505  // particular problem) where anything has arisen. I can see that
506  // there may be an issue if n_unrefine differs across processes so
507  // that (total_n_unrefine > max_keep_unrefined()) on some but not
508  // all processes. I haven't seen any examples of this yet so the
509  // following code may or may not work! (Andy, 06/03/08)
510 
511  unsigned total_n_unrefine=0;
512 #ifdef OOMPH_HAS_MPI
513  // Sum n_unrefine across all processors
514  if (this->is_mesh_distributed())
515  {
516  MPI_Allreduce(&n_unrefine,&total_n_unrefine,1,MPI_UNSIGNED,MPI_SUM,
517  Comm_pt->mpi_comm());
518  }
519  else
520  {
521  total_n_unrefine=n_unrefine;
522  }
523 #else
524  total_n_unrefine=n_unrefine;
525 #endif
526 
527  oomph_info << "---> " << total_n_refine << " elements to be refined, and "
528  << total_n_unrefine << " to be unrefined, in total.\n" << std::endl;
529 
530  if ((total_n_refine > 0) || (total_n_unrefine > max_keep_unrefined()))
531  {
532 
533 #ifdef PARANOID
534 #ifdef OOMPH_HAS_MPI
535 
536 
537  // Sanity check: Each processor checks if the enforced unrefinement of
538  // its haloed element is matched by enforced unrefinement of the
539  // corresponding halo elements on the other processors.
540  if (this->is_mesh_distributed())
541  {
542  // Store number of processors and current process
543  MPI_Status status;
544  int n_proc=Comm_pt->nproc();
545  int my_rank=Comm_pt->my_rank();
546

548  // for halo elements with processor d to processor d where it's
549  // compared against the unrefinement pattern for the corresponding
550  // haloed elements

552  {
553 
554  // No halo with self: Send unrefinement info to proc d
555  if (d!=my_rank)
556  {
557  // Get the vector of halo elements whose non-halo counterpart
558  // are on processor d
559  Vector<GeneralisedElement*> halo_elem_pt(this->halo_element_pt(d));
560 
561  // Create vector containing (0)1 to indicate that
562  // halo element is (not) to be unrefined
563  unsigned nhalo=halo_elem_pt.size();

566  {

569  {
570  halo_to_be_unrefined[e]=1;
571  }
572  }
573 
574  //Trap the case when there are no halo elements
575  //so that we don't get a segfault in the MPI send
576  if(nhalo > 0)
577  {
578  // Send it across to proc d
579  MPI_Send(&halo_to_be_unrefined[0],nhalo,MPI_INT,
580  d,0,Comm_pt->mpi_comm());
581  }
582  }
583  // else (d=my_rank): Receive unrefinement pattern from all
584  // other processors (dd)
585  else
586  {
587  // Loop over other processors
588  for (int dd=0; dd<n_proc; dd++)
589  {
590  // No halo with yourself
591  if (dd!=d)
592  {
593  // Get the vector of haloed elements on current processor
594  // with processor dd
595  Vector<GeneralisedElement*>
596  haloed_elem_pt(this->haloed_element_pt(dd));
597 
598  // Ask processor dd to send vector containing (0)1 for
599  // halo element with current processor to be (not)unrefined

609

647  {

650  if (d!=my_rank)

653  // are on processor d
654  Vector<GeneralisedElement*> halo_elem_pt(this->halo_element_pt(d));
655

657  // halo element is (not) to be refined
658  unsigned nhalo=halo_elem_pt.size();
659  Vector<int> halo_to_be_refined(nhalo,0);
660  for (unsigned e=0;e<nhalo;e++)
661  {
662  if (dynamic_cast<RefineableElement*>(halo_elem_pt[e])
663  ->to_be_refined())
664  {
665  halo_to_be_refined[e]=1;
666  }
667  }
668

670  //so that we don't get a segfault in the MPI send
671  if(nhalo > 0)
672  {
673  // Send it across to proc d
674  MPI_Send(&halo_to_be_refined[0],nhalo,MPI_INT,
675  d,0,Comm_pt->mpi_comm());
676  }
677  }
678  // else (d=my_rank): Receive refinement pattern from all
679  // other processors (dd)
680  else
681  {
682  // Loop over other processors
683  for (int dd=0; dd<n_proc; dd++)
684  {
685  // No halo with yourself
686  if (dd!=d)
687  {
688  // Get the vector of haloed elements on current processor
689  // with processor dd
690  Vector<GeneralisedElement*>
691  haloed_elem_pt(this->haloed_element_pt(dd));
692 
693  // Ask processor dd to send vector containing (0)1 for
694  // halo element with current processor to be (not)refined
695  unsigned nhaloed=haloed_elem_pt.size();
696  Vector<int> halo_to_be_refined(nhaloed);
697  //Trap to catch the case that there are no haloed elements
698  if(nhaloed > 0)
699  {
700  // Receive unrefinement pattern of haloes from proc dd
701  MPI_Recv(&halo_to_be_refined[0],nhaloed,MPI_INT,dd,0,
702  Comm_pt->mpi_comm(),&status);
703  }
704 
705  // Check it
706  for (unsigned e=0;e<nhaloed;e++)
707  {
708  if ( ( (halo_to_be_refined[e]==0)&&
709  (dynamic_cast<RefineableElement*>(haloed_elem_pt[e])->
710  to_be_refined()) ) ||
711  ( (halo_to_be_refined[e]==1)&&
712  (!dynamic_cast<RefineableElement*>(haloed_elem_pt[e])->
713  to_be_refined()) ) )

716  error_message

718  << "Haloed element: " << e << " on proc " << my_rank<< " \n"
719  << "wants to be refined whereas its halo counterpart on\n"
720  << "proc " << dd << " doesn't (or vice versa)...\n"
721  << "This is most likely because the error estimator\n"
722  << "has not assigned the same errors to halo and haloed\n"
723  << "elements -- it ought to!\n";

725  error_message.str(),
726  OOMPH_CURRENT_FUNCTION,
727  OOMPH_EXCEPTION_LOCATION);
728  }
729  }
730  }
731  }
732  }
733  }
734  }
735 
736 #endif
737 #endif
738 
739  // Perform the actual adaptation
740  adapt_mesh(local_doc_info);
741 
742  // The number of refineable elements is still local to each process
743  Nunrefined=n_unrefine;
744  Nrefined=n_refine;
745  }
746  // If not worthwhile, say so but still reorder nodes and kill
747  // external storage for consistency in parallel computations
748  else
749  {
750 
751 #ifdef OOMPH_HAS_MPI
752  // Delete any external element storage - any interaction will still
753  // be set up on the fly again, so we need to get rid of old information.
754  // This particularly causes problems in multi-domain examples where
755  // we decide not to refine one of the meshes
756  this->delete_all_external_storage();
757 #endif
758 
759  // Reorder the nodes within the mesh's node vector
760  // to establish a standard ordering regardless of the sequence
761  // of mesh refinements -- this is required to allow dump/restart
762  // on refined meshes
763  this->reorder_nodes();
764 
765 #ifdef OOMPH_HAS_MPI
766

768  // nodes
769  // This is required in cases where delete_all_external_storage()

773  DocInfo doc_info;
774  doc_info.disable_doc();
775  classify_halo_and_haloed_nodes(doc_info,doc_info.is_doc_enabled());
776  }
777 
778 #endif
779 
780  if (n_refine==0)
781  {
782  oomph_info
783  << " Not enough benefit in adapting mesh."
784  << std::endl << std::endl;
785  }
786  Nunrefined=0;
787  Nrefined=0;
788  }

827 //================================================================
828 /// Adapt mesh, which exists in two representations,
829 /// namely as:
830 /// - a FE mesh
831 /// - a forest of Oc or QuadTrees
832 ///
833 /// Refinement/derefinement process is documented (in tecplot-able form)
834 /// if requested.
835 ///
836 /// Procedure:
837 /// - Loop over all elements and do the refinement for those who want to
838 /// be refined. Note: Refinement/splitting only allocates elements but
839 /// doesn't build them.
840 /// - Build the new elements (i.e. give them nodes (create new ones where
841 /// necessary), assign boundary conditions, and add nodes to mesh
842 /// and mesh boundaries.
843 /// - For all nodes that were hanging on the previous mesh (and are still
844 /// marked as such), fill in their nodal values (consistent
845 /// with the current hanging node scheme) to make sure they are fully
846 /// functional, should they have become non-hanging during the
847 /// mesh-adaptation. Then mark the nodes as non-hanging.
848 /// - Unrefine selected elements (which may cause nodes to be re-built).
849 /// - Add the new elements to the mesh (by completely overwriting
850 /// the old Vector of elements).
851 /// - Delete any nodes that have become obsolete.
852 /// - Mark up hanging nodes and setup hanging node scheme (incl.
853 /// recursive cleanup for hanging nodes that depend on other
854 /// hanging nodes).
855 /// - Adjust position of hanging nodes to make sure their position
856 /// is consistent with the FE-based represenetation of their larger
857 /// neighbours.
858 /// - run a quick self-test on the neighbour finding scheme and
859 /// check the integrity of the elements (if PARANOID)
860 /// - doc hanging node status, boundary conditions, neighbour
861 /// scheme if requested.
862 ///
863 ///
864 /// After adaptation, all nodes (whether new or old) have up-to-date
865 /// current and previous values.
866 ///
867 /// If refinement process is being documented, the following information
868 /// is documented:
869 /// - The files
870 /// - "neighbours.dat"
871 /// - "all_nodes.dat"
872 /// - "new_nodes.dat"
873 /// - "hang_nodes_*.dat"
874 /// where the * denotes a direction (n,s,e,w) in 2D
875 /// or (r,l,u,d,f,b) in 3D
876 ///.
877 /// can be viewed with
878 /// - QHangingNodes.mcr
879 /// .
880 /// - The file
881 /// - "hangnodes_withmasters.dat"
882 /// .
883 /// can be viewed with
884 /// - QHangingNodesWithMasters.mcr
885 /// .
886 /// to check the hanging node status.
887 /// - The neighbour status of the elements is documented in
888 /// - "neighbours.dat"
889 /// .
890 /// and can be viewed with
891 /// - QuadTreeNeighbours.mcr
892 /// .
893 //=================================================================
