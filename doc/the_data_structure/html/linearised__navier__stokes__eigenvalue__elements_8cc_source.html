36 /// \short Fill in the residuals for the volume constraint
37 //====================================================================
38  void LinearisedNavierStokesEigenfunctionNormalisationElement::
39  fill_in_generic_contribution_to_residuals_normalisation(
40  Vector<double> &residuals)
41  {
42  //Note that this element can only be used with our linearised navier
43  //stokes elements
44  //Read part
45  {
46  const int local_eqn = this->eigenvalue_local_eqn(0);
47  if(local_eqn >= 0)
48  {
49  residuals[local_eqn] -= (*Normalisation_pt).real();
50  }
51  }
52  //Imaginary part
53  {
54  const int local_eqn = this->eigenvalue_local_eqn(1);
55  if(local_eqn >= 0)
56  {
57  residuals[local_eqn] -= (*Normalisation_pt).imag();
58  }
59  }
60  //Bifurcation constraint
61  const int local_eqn = this->eigenvalue_local_eqn(2);
62  if(local_eqn >= 0)
63  {
64  residuals[local_eqn] += this->eigenvalue(0);
65  }
66 
67  }
