87 /// Initialise instance of mumps data structure
88 //=============================================================================
89  void MumpsSolver::initialise_mumps()
90  {
91 
92  // Make new instance of Mumps data structure
93  Mumps_struc_pt=new DMUMPS_STRUC_C;
94 
95  // Mumps' hack to indicate that mpi_comm_world is used. Conversion of any
96  // other communicators appears to be non-portable, so we simply
97  // issue a warning if we later discover that oomph-lib's communicator
98  // is not MPI_COMM_WORLD
99  Mumps_struc_pt->comm_fortran = -987654;
100 
101  // Root processor participates in solution
102  Mumps_struc_pt->par = 1;
103 
104  // Matrix is unsymmetric
105  Mumps_struc_pt->sym = 0;
106 
107  // First call does the initialise phase
108  Mumps_struc_pt->job = -1;
109 
110  // Call c interface to double precision mumps to initialise
111  dmumps_c(Mumps_struc_pt);
112  Mumps_is_initialised=true;
113 
114  //Output stream for global info on host. Negative value suppresses printing
115  Mumps_struc_pt->icntl[2]=-1;
116 
117  //Only show error messages and stats
118  Mumps_struc_pt->icntl[3]=2;// 4 for full doc; creates huge amount of output
119 
120  //Write matrix
121  // sprintf(Mumps_struc_pt->write_problem,"/work/e173/e173/mheil/matrix");
122 
123  // Assembled matrix (rather than element-by_element)
124  Mumps_struc_pt->icntl[4]=0;
125 
126  // Distributed problem with user-specified distribution
127  Mumps_struc_pt->icntl[17]=3;
128 
129  // Dense RHS
130  Mumps_struc_pt->icntl[19]=0;
131 
132  // Non-distributed solution
133  Mumps_struc_pt->icntl[20]=0;

177 /// LU decompose the matrix addressed by matrix_pt using
178 /// mumps. The resulting matrix factors are stored internally.
179 /// Note: if Delete_matrix_data is true the function
180 /// matrix_pt->clean_up_memory() will be used to wipe the matrix data.
181 //=============================================================================
182 void MumpsSolver::factorise(DoubleMatrixBase* const &matrix_pt)
183 {
184 
185 
186  // Initialise timer
187  double t_start = TimingHelpers::timer();
188 
189  // set the distribution
190  DistributableLinearAlgebraObject* dist_matrix_pt=
191  dynamic_cast<DistributableLinearAlgebraObject*>(matrix_pt);
192  if (dist_matrix_pt)
193  {
194  // the solver has the same distribution as the matrix if possible
195  this->build_distribution(dist_matrix_pt->distribution_pt());
196  }
197  else
198  {
199  throw OomphLibError("Matrix must be distributable",
200  OOMPH_CURRENT_FUNCTION,
201  OOMPH_EXCEPTION_LOCATION);
202  }
203 
204 
205  //Check that we have a square matrix
206 #ifdef PARANOID
207  int n = matrix_pt->nrow();
208  int m = matrix_pt->ncol();
209  if(n != m)
210  {
211  std::ostringstream error_message_stream;
212  error_message_stream << "Can only solve for square matrices\n"
213  << "N, M " << n << " " << m << std::endl;
214 
215  throw OomphLibError(error_message_stream.str(),
216  OOMPH_CURRENT_FUNCTION,
217  OOMPH_EXCEPTION_LOCATION);
218  }
219  if (!Suppress_warning_about_MPI_COMM_WORLD)
220  {
221  if (this->distribution_pt()->communicator_pt()->mpi_comm()!=
222  MPI_COMM_WORLD)
223  {
224  std::ostringstream error_message_stream;
225  error_message_stream
226  << "Warning: Mumps wrapper assumes that communicator is MPI_COMM_WORLD\n"
227  << " which is not the case, so mumps may die...\n"
228  << " If it does initialise oomph-lib's mpi without requesting\n"
229  << " the communicator to be a duplicate of MPI_COMM_WORLD\n"
230  << " (done via an optional boolean to MPI_Helpers::init(...)\n\n"
231  << " (You can suppress this warning by recompiling without\n"
232  << " paranoia or calling \n\n"
233  << " MumpsSolver::enable_suppress_warning_about_MPI_COMM_WORLD()\n\n"
234  << " \n";
235  OomphLibWarning(error_message_stream.str(),
236  "MumpsSolver::factorise()",
237  OOMPH_EXCEPTION_LOCATION);
238  }
239  }
240 
241 #endif
242 
243  // Initialise
244  if (Mumps_is_initialised)
245  {
246  shutdown_mumps();
247  }
248  initialise_mumps();
249 
250 
251  // Doc stats?
252  if (Doc_stats)
253  {
254  //Output stream for global info on host. Negative value suppresses printing
255  Mumps_struc_pt->icntl[2]=6;
256  }
257 
258  // number of processors
259  unsigned nproc = 1;
260  if (dist_matrix_pt != 0)
261  {
262  nproc = dist_matrix_pt->distribution_pt()->communicator_pt()->nproc();
263  }
264 
265  // Is it a CRDoubleMatrix?
266  CRDoubleMatrix* cr_matrix_pt = dynamic_cast<CRDoubleMatrix*>(matrix_pt);
267  if (cr_matrix_pt != 0)
268  {
269 #ifdef PARANOID
270  // paranoid check that the matrix has been set up
271  if (!cr_matrix_pt->built())
272  {
273  throw OomphLibError
274  ("To apply MumpsSolver to a CRDoubleMatrix - it must be built",
275  OOMPH_CURRENT_FUNCTION,OOMPH_EXCEPTION_LOCATION);
276  }
277 #endif
278 
279  // if the matrix is distributed then set up solver
280  if ((nproc==1)||(cr_matrix_pt->distributed()))
281  {
282  double t_start_copy = TimingHelpers::timer();
283 
284  // Find the number of rows and non-zero entries in the matrix
285  const int nnz_loc = int(cr_matrix_pt->nnz());
286  const int n = matrix_pt->nrow();
287 
288  // Create mumps storage
289 
290  // Create vector for row numbers
291  Irn_loc.resize(nnz_loc);
292 
293  // Create vector for column numbers
294  Jcn_loc.resize(nnz_loc);
295 
296  // Vector for entries
297  A_loc.resize(nnz_loc);
298 
299  // First row held on this processor
300  int first_row = cr_matrix_pt->first_row();
301 
302  // Copy into coordinate storage scheme using pointer arithmetic
303  double* matrix_value_pt = cr_matrix_pt->value();
304  int* matrix_index_pt = cr_matrix_pt->column_index();
305  int* matrix_start_pt = cr_matrix_pt->row_start();
306  int i_row=0;
307  for(int count=0;count<nnz_loc;count++)
308  {
309  A_loc[count] = matrix_value_pt[count];
310  Jcn_loc[count] = matrix_index_pt[count]+1;
311  if (count<matrix_start_pt[i_row+1])
312  {
313  Irn_loc[count] = first_row+i_row+1;
314  }
315  else
316  {
317  i_row++;
318  Irn_loc[count] = first_row+i_row+1;
319  }
320  }
321 
322  // Now delete the matrix if we are allowed
323  if (Delete_matrix_data==true)
324  {
325  cr_matrix_pt->clear();
326  }
327 
328  if ((Doc_time) &&
329  (this->distribution_pt()->communicator_pt()->my_rank()==0))
330  {
331  double t_end_copy = TimingHelpers::timer();
332  oomph_info
333  << "Time for copying matrix into MumpsSolver data structure [sec] : "
334  << t_end_copy-t_start_copy << std::endl;
335  }
336 
337  // Call mumps factorisation
338  //-------------------------
339 
340  //Specify size of system
341  Mumps_struc_pt->n=n;
342 
343  // Number of nonzeroes
344  Mumps_struc_pt->nz_loc=nnz_loc;
345 
346  //The entries
347  Mumps_struc_pt->irn_loc=&Irn_loc[0];
348  Mumps_struc_pt->jcn_loc=&Jcn_loc[0];
349  Mumps_struc_pt->a_loc=&A_loc[0];
350 
351  double t_start_analyse = TimingHelpers::timer();
352 
353  // Do analysis
354  Mumps_struc_pt->job = 1;
355  dmumps_c(Mumps_struc_pt);
356 
357 
358  if ((Doc_time) &&
359  (this->distribution_pt()->communicator_pt()->my_rank()==0))
360  {
361  double t_end_analyse = TimingHelpers::timer();
362  oomph_info
363  << "Time for mumps analysis stage in MumpsSolver [sec] : "
364  << t_end_analyse-t_start_analyse << std::endl;
365  }
366 
367 
368  int my_rank=this->distribution_pt()->communicator_pt()->my_rank();
369 
370  // Document estimate for size of workspace
371  if (my_rank==0)
372  {
373  if(!Suppress_mumps_info_during_solve)
374  {
375  oomph_info << "Estimated max. workspace in MB: "
376  << Mumps_struc_pt->infog[25] << std::endl;
377  }
378  }
379 
380  double t_start_factor = TimingHelpers::timer();
381 
382  // Loop until successfully factorised
383  bool factorised=false;
384  while (!factorised)
385  {
386 
387  // Set workspace to multiple of that -- ought to be "significantly
388  // larger than infog(26)", according to the manual :(
389  Mumps_struc_pt->icntl[22]=Workspace_scaling_factor*
390  Mumps_struc_pt->infog[25];
391 
392  if(!Suppress_mumps_info_during_solve)
393  {
394  oomph_info << "Attempting factorisation with workspace estimate: "
395  << Mumps_struc_pt->icntl[22] << " MB\n";
396  oomph_info << "corresponding to Workspace_scaling_factor = "
397  << Workspace_scaling_factor << "\n";
398  }
399 
400  // Do factorisation
401  Mumps_struc_pt->job = 2;
402  dmumps_c(Mumps_struc_pt);
403 
404  // Check for error
405  if (Mumps_struc_pt->infog[0]!=0)
406  {
407  if(!Suppress_mumps_info_during_solve)
408  {
409  oomph_info << "Error during mumps factorisation!\n";
410  oomph_info << "Error codes: "
411  << Mumps_struc_pt->info[0] << " "
412  << Mumps_struc_pt->info[1] << std::endl;
413  }
414 
415  // Increase scaling factor for workspace and run again
416  Workspace_scaling_factor*=2;
417 
418  if(!Suppress_mumps_info_during_solve)
419  {
420  oomph_info << "Increasing workspace_scaling_factor to "
421  << Workspace_scaling_factor << std::endl;
422  }
423  }
424  else
425  {
426  factorised=true;
427  }
428  }
429 
430 
431  if ((Doc_time) &&
432  (this->distribution_pt()->communicator_pt()->my_rank()==0))
433  {
434  double t_end_factor = TimingHelpers::timer();
435  oomph_info
436  << "Time for actual mumps factorisation in MumpsSolver [sec] : "
437  << t_end_factor-t_start_factor<< std::endl;
438  }
439  }
440  // else the CRDoubleMatrix is not distributed
441  else
442  {
443  std::ostringstream error_message_stream;
444  error_message_stream << "MumpsSolver only works for a "
445  << " distributed CRDoubleMatrix\n";
446  throw OomphLibError(error_message_stream.str(),
447  OOMPH_CURRENT_FUNCTION,
448  OOMPH_EXCEPTION_LOCATION);
449 
450  }
451  }
452  // Otherwise throw an error
453  else
454  {
455  std::ostringstream error_message_stream;
456  error_message_stream << "MumpsSolver implemented only for "
457  << "distributed CRDoubleMatrix. \n";
458  throw OomphLibError(error_message_stream.str(),
459  OOMPH_CURRENT_FUNCTION,
460  OOMPH_EXCEPTION_LOCATION);
461  }
462 
463 
464  if ((Doc_time) && (this->distribution_pt()->communicator_pt()->my_rank()==0))
465  {
466 
467  double t_end = TimingHelpers::timer();
468  oomph_info << "Time for MumpsSolver factorisation [sec] : "
469  << t_end-t_start << std::endl;
470  }
471 
472  // Switch off docing again by setting output stream for global info on
473  // to negative number
474  Mumps_struc_pt->icntl[2]=-1;
475 
476 }

479 /// Do the backsubstitution for mumps solver.
480 /// This does not make any assumption about the distribution of the
481 /// vectors
482 //=============================================================================
483  void MumpsSolver::backsub(const DoubleVector &rhs,
484  DoubleVector &result)
485  {
486 
487  double t_start = TimingHelpers::timer();
488 
489 #ifdef PARANOID
490  if (!Suppress_warning_about_MPI_COMM_WORLD)
491  {
492  if (this->distribution_pt()->communicator_pt()->mpi_comm()!=MPI_COMM_WORLD)
493  {
494  std::ostringstream error_message_stream;
495  error_message_stream
496  << "Warning: Mumps wrapper assumes that communicator is MPI_COMM_WORLD\n"
497  << " which is not the case, so mumps may die...\n"
498  << " If it does initialise oomph-lib's mpi without requesting\n"
499  << " the communicator to be a duplicate of MPI_COMM_WORLD\n"
500  << " (done via an optional boolean to MPI_Helpers::init(...)\n\n"
501  << " (You can suppress this warning by recompiling without\n"
502  << " paranoia or calling \n\n"
503  << " MumpsSolver::enable_suppress_warning_about_MPI_COMM_WORLD()\n\n"
504  << " \n";
505  OomphLibWarning(error_message_stream.str(),
506  "MumpsSolver::backsub()",
507  OOMPH_EXCEPTION_LOCATION);
508  }
509  }
510 
511  // Initialised?
512  if (!Mumps_is_initialised)
513  {
514  std::ostringstream error_message_stream;
515  error_message_stream
516  << "Mumps has not been initialised.";
517  throw OomphLibError(error_message_stream.str(),
518  OOMPH_CURRENT_FUNCTION,
519  OOMPH_EXCEPTION_LOCATION);
520  }
521 
522  // check that the rhs vector is setup
523  if (!rhs.distribution_pt()->built())
524  {
525  std::ostringstream error_message_stream;
526  error_message_stream
527  << "The vectors rhs must be setup";
528  throw OomphLibError(error_message_stream.str(),
529  OOMPH_CURRENT_FUNCTION,
530  OOMPH_EXCEPTION_LOCATION);
531  }
532 
533 #endif
534 
535  // Check that the rhs distribution is the same as the distribution as this
536  // solver. If not redistribute and issue a warning
537  LinearAlgebraDistribution rhs_distribution(rhs.distribution_pt());
538  if (!(*rhs.distribution_pt() == *this->distribution_pt()))
539  {
540  if(!Suppress_incorrect_rhs_distribution_warning_in_resolve)
541  {
542  std::ostringstream warning_stream;
543  warning_stream
544  << "The distribution of rhs vector does not match that of the solver.\n";
545  warning_stream
546  << "The rhs may have to be redistributed but we're not doing this because\n"
547  << "I'm no longer convinced it's necessary. Keep an eye on this...\n";
548  warning_stream
549  << "To remove this warning you can either:\n"
550  << " i) Ensure that the rhs vector has the correct distribution\n"
551  << " before calling the resolve() function\n"
552  << "or ii) Set the flag \n"
553  << " MumpsSolver::Suppress_incorrect_rhs_distribution_warning_in_resolve\n"
554  << " to be true\n\n";
555 
556  OomphLibWarning(warning_stream.str(),
557  "MumpsSolver::resolve()",
558  OOMPH_EXCEPTION_LOCATION);
559  }
560 
561  // //Have to cast away const-ness (which tells us that we shouldn't really
562  // //be doing this!)
563  // const_cast<DoubleVector&>(rhs).redistribute(this->distribution_pt());
564  }
565 
566 
567 
568 #ifdef PARANOID
569  // if the result vector is setup then check it has the same distribution
570  // as the rhs
571  if (result.distribution_built())
572  {
573  if (!(*result.distribution_pt() == *rhs.distribution_pt()))
574  {
575  std::ostringstream error_message_stream;
576  error_message_stream
577  << "The result vector distribution has been setup; it must have the "
578  << "same distribution as the rhs vector.";
579  throw OomphLibError(error_message_stream.str(),
580  OOMPH_CURRENT_FUNCTION,
581  OOMPH_EXCEPTION_LOCATION);
582  }
583  }
584 #endif
585 
586 
587  // Doc stats?
588  if (Doc_stats)
589  {
590  //Output stream for global info on host. Negative value suppresses printing
591  Mumps_struc_pt->icntl[2]=6;
592  }
593 
594  // number of DOFs
595  int ndof = this->distribution_pt()->nrow();
596 
597  // Make backup to avoid over-writing
598  DoubleVector tmp_rhs;
599  tmp_rhs=rhs;
600 
601  // Now turn this into a global (non-distributed) vector
602  // because that's what mumps needs
603 
604  // Make a global distribution (i.e. one that isn't distributed)
605  LinearAlgebraDistribution global_distribution(
606  this->distribution_pt()->communicator_pt(),ndof,false);
607 
608  // Redistribute the tmp_rhs vector with this distribution -- it's
609  // now "global", as required for mumps
610  tmp_rhs.redistribute(&global_distribution);
611 
612  // Do the backsubsitution phase -- overwrites the tmp_rhs vector with the
613  // solution
614  Mumps_struc_pt->rhs=&tmp_rhs[0];
615  Mumps_struc_pt->job=3;
616  dmumps_c(Mumps_struc_pt);
617 
618  // Copy back
619  unsigned n=Mumps_struc_pt->n;
620  for (unsigned j=0;j<n;j++)
621  {
622  tmp_rhs[j]=Mumps_struc_pt->rhs[j];
623  }
624 
625  // Broadcast the result which is only held on root
626  MPI_Bcast(&tmp_rhs[0],ndof,MPI_DOUBLE,0,
627  this->distribution_pt()->communicator_pt()->mpi_comm());
628 
629  // If the result vector is distributed, re-distribute the
630  // non-distributed tmp_rhs vector to match
631  if (result.built())
632  {
633  tmp_rhs.redistribute(result.distribution_pt());
634  }
635  else
636  {
637  tmp_rhs.redistribute(this->distribution_pt());
638  }
639 
640  // Now copy the tmp_rhs vector into the (matching) result
641  result = tmp_rhs;
642 
643  if ((Doc_time) && (this->distribution_pt()->communicator_pt()->my_rank()==0))
644  {
645  double t_end = TimingHelpers::timer();
646  oomph_info << "Time for MumpsSolver backsub [sec] : "
647  << t_end-t_start << std::endl;
648  }
649 
650  // Switch off docing again by setting output stream for global info on
651  // to negative number
652  Mumps_struc_pt->icntl[2]=-1;
653 
654 }

666 /// Linear-algebra-type solver: Takes pointer to a matrix and rhs
667 /// vector and returns the solution of the linear system. Problem pointer
668 /// defaults to NULL and can be omitted. The function returns the global
669 /// result vector. Matrix must be CRDoubleMatrix.
670 /// Note: if Delete_matrix_data is true the function
671 /// matrix_pt->clean_up_memory() will be used to wipe the matrix data.
672 //=========================================================================
673 void MumpsSolver::solve(DoubleMatrixBase* const &matrix_pt,
674  const DoubleVector &rhs,
675  DoubleVector &result)
676 {
677 #ifdef PARANOID
678  if (!Suppress_warning_about_MPI_COMM_WORLD)
679  {
680  if (dynamic_cast<DistributableLinearAlgebraObject*>(matrix_pt)
681  ->distribution_pt()->communicator_pt()->mpi_comm()!=
682  MPI_COMM_WORLD)
683  {
684  std::ostringstream error_message_stream;
685  error_message_stream
686  << "Warning: Mumps wrapper assumes that communicator is MPI_COMM_WORLD\n"
