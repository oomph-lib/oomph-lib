643  /// Classify all halo and haloed information in the mesh (overloaded
644  /// version from Mesh base class. Calls that one first, then synchronises
645  /// hanging nodes)
646  void classify_halo_and_haloed_nodes(DocInfo& doc_info,
647  const bool& report_stats)
648  {
649  // Call version in base class but don't bother to call
650  // resize_halo_nodes() -- we'll do it ourselves below
651  bool backup = Resize_halo_nodes_not_required;
652  Resize_halo_nodes_not_required = false;
653  Mesh::classify_halo_and_haloed_nodes(doc_info, report_stats);
654  Resize_halo_nodes_not_required = backup;
655 
656  double t_start = 0.0;
657  if (Global_timings::Doc_comprehensive_timings)
658  {
659  t_start = TimingHelpers::timer();
660  }
661 
662  // Communicate positions of non-hanging nodes that depend on non-existent
663  // neighbours (e.g. h-refinement of neighbouring elements with different
664  // p-orders where the shared edge is on the outer edge of the halo layer)
665  synchronise_nonhanging_nodes();
666 
667  // Get number of continously interpolated variables
668  unsigned local_ncont_interpolated_values = 0;
669 
670  // Bypass if we have no elements and then get overall
671  // value by reduction
672  if (nelement() > 0)
673  {
674  local_ncont_interpolated_values =
675  dynamic_cast<RefineableElement*>(element_pt(0))
676  ->ncont_interpolated_values();
677  }
678  unsigned ncont_interpolated_values = 0;
679  MPI_Allreduce(&local_ncont_interpolated_values,
680  &ncont_interpolated_values,
681  1,
682  MPI_UNSIGNED,
683  MPI_MAX,
684  Comm_pt->mpi_comm());
685 
686  // Synchronise the hanging nodes
687  synchronise_hanging_nodes(ncont_interpolated_values);
688 
689  if (Global_timings::Doc_comprehensive_timings)
690  {
691  double t_end = TimingHelpers::timer();
692  oomph_info
693  << "Time for "

934  /// \short Compute target volume based on the elements' error and the
935  /// error target; return max edge ratio
936  double compute_volume_target(const Vector<double>& elem_error,
937  Vector<double>& target_volume)
938  {
939  double max_edge_ratio = 0.0;
940  unsigned count_unrefined = 0;
941  unsigned count_refined = 0;
942  this->Nrefinement_overruled = 0;
943 
944  unsigned nel = this->nelement();
945  for (unsigned e = 0; e < nel; e++)
946  {
947  // Get element
948  FiniteElement* el_pt = this->finite_element_pt(e);
949 
950  // Calculate the volume of the element
951  double volume = el_pt->size();
952 
953  // Find the vertex coordinates
954  // (vertices are enumerated first)
955  double vertex[4][3];
956  for (unsigned n = 0; n < 4; ++n)
957  {
958  for (unsigned i = 0; i < 3; ++i)
959  {
960  vertex[n][i] = el_pt->node_pt(n)->x(i);
961  }
962  }
963 
964  // Compute the radius of the circumsphere of the tetrahedron
965  // Algorithm stolen from tetgen for consistency
966  DenseDoubleMatrix A(3);
967  for (unsigned i = 0; i < 3; ++i)
968  {
969  A(0, i) = vertex[1][i] - vertex[0][i];
970  A(1, i) = vertex[2][i] - vertex[0][i];
971  A(2, i) = vertex[3][i] - vertex[0][i];
972  }
973 
974  Vector<double> rhs(3);
975  // Compute the right hand side vector b (3x1).
976  for (unsigned i = 0; i < 3; ++i)
977  {
978  rhs[i] = 0.0;
979  for (unsigned k = 0; k < 3; ++k)
980  {
981  rhs[i] += A(i, k) * A(i, k);
982  }
983  rhs[i] *= 0.5;
984  }
985 
986  // Solve the linear system, in which the rhs is over-written with
987  // the solution
988  A.solve(rhs);
989 
990  // Calculate the circum-radius
991  double circum_radius =
992  sqrt(rhs[0] * rhs[0] + rhs[1] * rhs[1] + rhs[2] * rhs[2]);
993 
994  // Now find the shortest edge length
995  Vector<double> edge(3);
996  double min_length = DBL_MAX;
997  for (unsigned start = 0; start < 4; ++start)
998  {
999  for (unsigned end = start + 1; end < 4; ++end)
1000  {
1001  for (unsigned i = 0; i < 3; ++i)
1002  {
1003  edge[i] = vertex[start][i] - vertex[end][i];
1004  }
1005  double length =
1006  sqrt(edge[0] * edge[0] + edge[1] * edge[1] + edge[2] * edge[2]);
1007  if (length < min_length)
1008  {
1009  min_length = length;
1010  }
1011  }
1012  }
1013 
1014  // Now calculate the minimum edge ratio for this element
1015  double local_max_edge_ratio = circum_radius / min_length;
1016  if (local_max_edge_ratio > max_edge_ratio)
1017  {
1018  max_edge_ratio = local_max_edge_ratio;
1019  }
1020 
1021  // Mimick refinement in tree-based procedure: Target volumes
1022  // for elements that exceed permitted error is 1/4 of their
1023  // current volume, corresponding to a uniform sub-division.
1024  if (elem_error[e] > this->max_permitted_error())
1025  {
1026  target_volume[e] = std::max(volume / 4.0, Min_element_size);
1027  if (target_volume[e] != Min_element_size)
1028  {
1029  count_refined++;
1030  }
1031  else
1032  {
1033  this->Nrefinement_overruled++;
1034  }
1035  }
1036  else if (elem_error[e] < this->min_permitted_error())
1037  {
1038  target_volume[e] = std::min(4.0 * volume, Max_element_size);
1039  if (target_volume[e] != Max_element_size)
1040  {
1041  count_unrefined++;
1042  }
1043  }
1044  else
1045  {
1046  // Leave it alone
1047  target_volume[e] = std::max(volume, Min_element_size);
1048  }
1049 
1050  } // End of loop over elements
1051 
1052  // Tell everybody
1053  this->Nrefined = count_refined;
1054  this->Nunrefined = count_unrefined;
1055 
1056  return max_edge_ratio;
1057  }
1058 
1059  /// Max permitted element size
1060  double Max_element_size;
1061 
1062  /// Min permitted element size
1063  double Min_element_size;
1064 
1065  /// Max edge ratio before remesh gets triggered
1066  double Max_permitted_edge_ratio;
1067  };
