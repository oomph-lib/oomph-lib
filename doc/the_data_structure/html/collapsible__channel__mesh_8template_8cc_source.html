37 /// Constructor: Pass number of elements in upstream/collapsible/downstream
38 /// segment and across the channel; lengths of upstream/collapsible/downstream
39 /// segments and width of channel, pointer to GeomObject that defines
40 /// the collapsible segment and pointer to TimeStepper (defaults to the
41 /// default timestepper, Steady).
42 //========================================================================
43 template <class ELEMENT>
44 CollapsibleChannelMesh<ELEMENT>::CollapsibleChannelMesh(
45  const unsigned& nup,
46  const unsigned& ncollapsible,
47  const unsigned& ndown,
48  const unsigned& ny,
49  const double& lup,
50  const double& lcollapsible,
51  const double& ldown,
52  const double& ly,
53  GeomObject* wall_pt,
54  TimeStepper* time_stepper_pt)
55  : SimpleRectangularQuadMesh<ELEMENT>(nup+ncollapsible+ndown, ny,
56  lup+lcollapsible+ldown, ly,
57  time_stepper_pt),
58  Nup(nup), Ncollapsible(ncollapsible), Ndown(ndown), Ny(ny),
59  Wall_pt(wall_pt)
60 {
61  // Mesh can only be built with 2D Qelements.
62  MeshChecker::assert_geometric_element<QElementGeometricBase,ELEMENT>(2);
63 
64  //Create the CollapsibleChannelDomain with the wall represented
65  //by the geometric object
66  Domain_pt = new CollapsibleChannelDomain(nup, ncollapsible, ndown, ny,
67  lup,lcollapsible, ldown, ly,
68  wall_pt);
69 
70  // Total number of (macro/finite)elements
71  unsigned nmacro=(nup+ncollapsible+ndown)*ny;
72 
73  // Loop over all elements and set macro element pointer
74  for (unsigned e=0;e<nmacro;e++)
75  {
76  // Get pointer to finite element
77  FiniteElement* el_pt=this->finite_element_pt(e);
78 
79  // Set pointer to macro element so the curvlinear boundaries
80  // of the mesh/domain can get picked up during adaptive
81  // mesh refinement
82  el_pt->set_macro_elem_pt(this->Domain_pt->macro_element_pt(e));
83  }
84 
85  // Update the nodal positions corresponding to their
86  // macro element representations.
87  this->node_update();
88 
89  // Update the boundary numbering scheme and set boundary coordinate
90  //-----------------------------------------------------------------
91 
92  // (Note: The original SimpleRectangularQuadMesh had four boundaries.
93  // We need to overwrite the boundary lookup scheme for the current
94  // mesh so that the collapsible segment becomes identifiable).
95  // While we're doing this, we're also setting up a boundary
96  // coordinate for the nodes located on the collapsible segment.
97  // The boundary coordinate can be used to setup FSI.
98 
99  // How many boundaries does the mesh have now?
100  unsigned nbound=this->nboundary();
101  for (unsigned b=0;b<nbound;b++)
102  {
103  // Remove all nodes on this boundary from the mesh's lookup scheme
104  // and also delete the reverse lookup scheme held by the nodes
105  this->remove_boundary_nodes(b);
106  }
107 
108 #ifdef PARANOID
109  // Sanity check
110  unsigned nnod=this->nnode();
111  for (unsigned j=0;j<nnod;j++)
112  {
113  if (this->node_pt(j)->is_on_boundary())
114  {
115  std::ostringstream error_message;
116  error_message << "Node " << j << "is still on boundary " << std::endl;
117 
118  throw OomphLibError(error_message.str(),
119  OOMPH_CURRENT_FUNCTION,
120  OOMPH_EXCEPTION_LOCATION);
121  }
122  }
123 #endif
124 
125  //Change the numbers of boundaries
126  this->set_nboundary(6);
127 
128  // Get the number of nodes along the element edge from first element
129  unsigned nnode_1d=this->finite_element_pt(0)->nnode_1d();
130 
131  // Vector of Lagrangian coordinates used as boundary coordinate
132  Vector<double> zeta(1);
133 
134  // Index of first element underneath the collapsible bit
135  unsigned first_collapsible=(ny-1)*(nup+ncollapsible+ndown)+nup;
136 
137  // Zeta increment over elements (used for assignment of
138  // boundary coordinate)
139  double dzeta= lcollapsible/double(ncollapsible);
140 
141  // Manually loop over the elements near the boundaries and
142  // assign nodes to boundaries. Also set up boundary coordinate
143  unsigned nelem=this->nelement();
144  for (unsigned e=0;e<nelem;e++)
145  {
146  // Bottom row of elements
147  if (e<nup+ncollapsible+ndown)
148  {
149  for (unsigned i=0;i<nnode_1d;i++)
150  {
151  this->add_boundary_node(0, this->finite_element_pt(e)->node_pt(i));
152  }
153  }
154  // Upstream upper rigid bit
155  if ((e>(ny-1)*(nup+ncollapsible+ndown)-1)&&
156  (e<(ny-1)*(nup+ncollapsible+ndown)+nup))
157  {
158  for (unsigned i=0;i<nnode_1d;i++)
159  {
160  this->add_boundary_node(4,
161  this->finite_element_pt(e)->node_pt((nnode_1d-1)*nnode_1d+i));
162  }
163  }
164  // Collapsible bit
165  if ((e>(ny-1)*(nup+ncollapsible+ndown)+nup-1)&&
166  (e<(ny-1)*(nup+ncollapsible+ndown)+nup+ncollapsible))
167  {
168  for (unsigned i=0;i<nnode_1d;i++)
169  {
170  this->add_boundary_node(3,
171  this->finite_element_pt(e)->node_pt((nnode_1d-1)*nnode_1d+i));
172 
173  // What column of elements are we in?
174  unsigned ix=e-first_collapsible;
175 
176  // Zeta coordinate
177  zeta[0]=double(ix)*dzeta+double(i)*dzeta/double(nnode_1d-1);
178 
179  // Set boundary coordinate
180  this->finite_element_pt(e)->node_pt((nnode_1d-1)*nnode_1d+i)->
181  set_coordinates_on_boundary(3,zeta);
182  }
183  }
184  // Downstream upper rigid bit
185  if ((e>(ny-1)*(nup+ncollapsible+ndown)+nup+ncollapsible-1)&&
186  (e<ny*(nup+ncollapsible+ndown)))
187  {
188  for (unsigned i=0;i<nnode_1d;i++)
189  {
190  this->add_boundary_node(2,
191  this->finite_element_pt(e)->node_pt((nnode_1d-1)*nnode_1d+i));
192  }
193  }
194  // Left end
195  if (e%(nup+ncollapsible+ndown)==0)
196  {
197  for (unsigned i=0;i<nnode_1d;i++)
198  {
199  this->add_boundary_node(5,
200  this->finite_element_pt(e)->node_pt(i*nnode_1d));
201  }
202  }
203  // Right end
204  if (e%(nup+ncollapsible+ndown)==(nup+ncollapsible+ndown)-1)
205  {
206  for (unsigned i=0;i<nnode_1d;i++)
207  {
208  this->add_boundary_node(1,
209  this->finite_element_pt(e)->node_pt((i+1)*nnode_1d-1));
210  }
211  }
212  }
213 
214  // Re-setup lookup scheme that establishes which elements are located
215  // on the mesh boundaries (doesn't need to be wiped)
216  this->setup_boundary_element_info();
217 
218  //We have only bothered to parametrise boundary 3
219  this->Boundary_coordinate_exists[3] = true;
220 }

232 /// Perform algebraic mesh update at time level t (t=0: present;
233 /// t>0: previous)
234 //=================================================================
235 template<class ELEMENT>
236 void AlgebraicCollapsibleChannelMesh<ELEMENT>::algebraic_node_update(
237  const unsigned& t, AlgebraicNode*& node_pt)
238 {
239 
240 
241 #ifdef PARANOID
242  // We're updating the nodal positions (!) at time level t
243  // and determine them by evaluating the wall GeomObject's
244  // position at that gime level. I believe this only makes sense
245  // if the t-th history value in the positional timestepper
246  // actually represents previous values (rather than some
247  // generalised quantity). Hence if this function is called with
248  // t>nprev_values(), we issue a warning and terminate the execution.
249  // It *might* be possible that the code still works correctly
250  // even if this condition is violated (e.g. if the GeomObject's
251  // position() function returns the appropriate "generalised"
252  // position value that is required by the timestepping scheme but it's
253  // probably worth flagging this up and forcing the user to manually switch
254  // off this warning if he/she is 100% sure that this is kosher.
255  if (t>node_pt->position_time_stepper_pt()->nprev_values())
256  {
257  std::string error_message =
258  "Trying to update the nodal position at a time level";
259  error_message += "beyond the number of previous values in the nodes'";
260  error_message += "position timestepper. This seems highly suspect!";
261  error_message += "If you're sure the code behaves correctly";
262  error_message += "in your application, remove this warning ";
263  error_message += "or recompile with PARNOID switched off.";
264 
265  std::string function_name =
266  "AlgebraicCollapsibleChannelMesh::";
267  function_name += "algebraic_node_update()";
268 
269  throw OomphLibError(error_message,
270  OOMPH_CURRENT_FUNCTION,
271  OOMPH_EXCEPTION_LOCATION);
272  }
273 #endif
274 
275  // Extract references for update by copy construction
276  Vector<double> ref_value(node_pt->vector_ref_value());
277 
278  // First reference value: Original x-position. Used as the start point
279  // for the lines connecting the nodes in the vertical direction
280  double x_bottom=ref_value[0];
281 
282  // Second reference value: Fractional position along
283  // straight line from the bottom (at the original x position)
284  // to the reference point on the upper wall
285  double fract=ref_value[1];
286 
287  // Third reference value: Reference local coordinate
288  // in GeomObject that represents the upper wall (local coordinate
289  // in finite element if the wall GeomObject is a finite element mesh)
290  Vector<double> s(1);
291  s[0]=ref_value[2];
292 
293  // Fourth reference value: zeta coordinate on the upper wall
294  // If the wall is a simple GeomObject, zeta[0]=s[0]
295  // but if it's a compound GeomObject (e.g. a finite element mesh)
296  // zeta scales during mesh refinement, whereas s[0] and the
297  // pointer to the geom object have to be re-computed.
298  // double zeta=ref_value[3]; // not needed here
299 
300  // Extract geometric objects for update by copy construction
301  Vector<GeomObject*> geom_object_pt(node_pt->vector_geom_object_pt());
302 
303  // Pointer to actual wall geom object (either the same as the wall object
304  // or the pointer to the actual finite element)
305  GeomObject* geom_obj_pt=geom_object_pt[0];
306 
307  // Get position vector to wall at previous timestep t!
308  Vector<double> r_wall(2);
309  geom_obj_pt->position(t,s,r_wall);
310 
311  // Assign new nodal coordinate
312  node_pt->x(t,0)=x_bottom+fract*(r_wall[0]-x_bottom);
313  node_pt->x(t,1)=fract*r_wall[1];
314 
315 }

322 /// Setup algebraic mesh update -- assumes that mesh has
323 /// initially been set up with a flush upper wall
324 //=================================================================
325 template<class ELEMENT>
326 void AlgebraicCollapsibleChannelMesh<ELEMENT>::setup_algebraic_node_update()
327 {
328  // Shorthand for some geometric data:
329  double l_up=this->domain_pt()->l_up();
330  double l_collapsible=this->domain_pt()->l_collapsible();
331 
332  // Loop over all nodes in mesh
333  unsigned nnod=this->nnode();
334  for (unsigned j=0;j<nnod;j++)
335  {
336  // Get pointer to node -- recall that that Mesh::node_pt(...) has been
337  // overloaded in the AlgebraicMesh class to return a pointer to
338  // an AlgebraicNode.
339  AlgebraicNode* nod_pt=node_pt(j);
340 
341  // Get coordinates
342  double x=nod_pt->x(0);
343  double y=nod_pt->x(1);
344 
345  // Check if it's in the collapsible part:
346  if ( (x>=l_up) && (x<=(l_up+l_collapsible)) )
347  {
348 
349  // Get zeta coordinate on the undeformed wall
350  Vector<double> zeta(1);
351  zeta[0]=x-l_up;
352 
353  // Get pointer to geometric (sub-)object and Lagrangian coordinate
354  // on that sub-object. For a wall that is represented by
355  // a single geom object, this simply returns the input.
356  // If the geom object consists of sub-objects (e.g.
357  // if it is a finite element mesh representing a wall,
358  // then we'll obtain the pointer to the finite element
359  // (in its incarnation as a GeomObject) and the
360  // local coordinate in that element.
361  GeomObject* geom_obj_pt;
362  Vector<double> s(1);
363  this->Wall_pt->locate_zeta(zeta,geom_obj_pt,s);
364 
365  // Get position vector to wall:
366  Vector<double> r_wall(2);
367  geom_obj_pt->position(s,r_wall);
368 
369  // Sanity check: Confirm that the wall is in its undeformed position
370 #ifdef PARANOID
371  if ((std::fabs(r_wall[0]-x)>1.0e-15)&&(std::fabs(r_wall[1]-y)>1.0e-15))
372  {
373  std::ostringstream error_stream;
374  error_stream
375  << "Wall must be in its undeformed position when\n"
376  << "algebraic node update information is set up!\n "
377  << "x-discrepancy: " << std::fabs(r_wall[0]-x) << std::endl
378  << "y-discrepancy: " << std::fabs(r_wall[1]-y) << std::endl;
379 
380  throw OomphLibError(
381  error_stream.str(),
382  OOMPH_CURRENT_FUNCTION,
383  OOMPH_EXCEPTION_LOCATION);
384  }
385 #endif
386 
387 
388 
389  // One geometric object is involved in update operation
390  Vector<GeomObject*> geom_object_pt(1);
391 
392  // The actual geometric object (If the wall is simple GeomObject
393  // this is the same as Wall_pt; if it's a compound GeomObject
394  // this points to the sub-object)
395  geom_object_pt[0]=geom_obj_pt;
396 
397  // The update function requires four parameters:
398  Vector<double> ref_value(4);
399 
400  // First reference value: Original x-position
401  ref_value[0]=r_wall[0];
402 
403  // Second reference value: fractional position along
404  // straight line from the bottom (at the original x position)
405  // to the point on the wall)
406  ref_value[1]=y/r_wall[1];
407 
408  // Third reference value: Reference local coordinate
409  // in wall element (local coordinate in FE if we're dealing
410  // with a wall mesh)
411  ref_value[2]=s[0];
412 
413  // Fourth reference value: zeta coordinate on wall
414  // If the wall is a simple GeomObject, zeta[0]=s[0]
415  // but if it's a compound GeomObject (e.g. a finite element mesh)
416  // zeta scales during mesh refinement, whereas s[0] and the
417  // pointer to the geom object have to be re-computed.
418  ref_value[3]=zeta[0];
419 
420  // Setup algebraic update for node: Pass update information
421  nod_pt->add_node_update_info(
422  this, // mesh
423  geom_object_pt, // vector of geom objects
424  ref_value); // vector of ref. values
425  }
426 
427  }
428 
429 } //end of setup_algebraic_node_update
