76 /// Auxiliary Problem to smooth a SolidMesh by adjusting the internal
77 /// nodal positions via the solution of a nonlinear solid mechanics problem.
78 /// The mesh will typically have been created with an unstructured
79 /// mesh generator that uses a low-order (simplex) representation of the
80 /// element geometry; some of the nodes, typically non-vertex nodes on
81 /// the domain's curvilinear boundaries, were then moved to their new
82 /// position to provide a more accurate representation of the geometry.
83 /// This class should be used to deal with elements that may have
84 /// become inverted during the node motion.
85 /// \n
86 /// \b Important \b assumption:
87 /// - It is important that the Lagrangian coordinates of all nodes still
88 /// indicate their original position, i.e. their position before (some of)
89 /// them were moved to their new position. This is because
90 /// we apply the boundary displacements in small increments.
91 /// .
92 /// Template argument specifies type of element. It must be a pure
93 /// solid mechanics element! This shouldn't cause any problems
94 /// since mesh smoothing operations tend to be performed off-line
95 /// so the mesh may as well be built with pure solid elements
96 /// even if it is ultimately to be used with other element types.
97 /// (This restriction could easily be avoided but would
98 /// require double templating and would generally be messy...)
99 //====================================================================
100 template<class ELEMENT>
101 class NonLinearElasticitySmoothMesh: public Problem
102 {

107  /// orig_mesh_pt in response to the displacement of some of its
108  /// nodes relative to their original position which must still be indicated
109  /// by the nodes' Lagrangian position. copy_of_mesh_pt must be a deep copy
110  /// of orig_mesh_pt, with the same boundary coordinates etc. This mesh
111  /// is used as workspace and can be deleted afterwards.
112  /// The vector controlled_boundary_id contains the ids of
113  /// the mesh boundaries in orig_mesh_pt whose position is supposed
114  /// to remain fixed (while the other nodes are re-positioned to avoid
115  /// the inversion of elements). The final optional argument
116  /// specifies the max. number of increments in which the mesh
117  /// boundary is deformed.

137  /// by the nodes' Lagrangian position. copy_of_mesh_pt must be a deep copy

141  /// the mesh boundaries in orig_mesh_pt whose position is supposed
142  /// to remain fixed (while the other nodes are re-positioned to avoid
143  /// the inversion of elements). The DocInfo allows allows the output
144  /// of the intermediate meshes. The final optional argument
145  /// specifies the max. number of increments in which the mesh

148  const Vector<unsigned>& controlled_boundary_id,

158  unsigned nbound=orig_mesh_pt->nboundary();
159  unsigned dim=orig_mesh_pt->node_pt(0)->ndim();

162  add_sub_mesh(Dummy_mesh_pt);
163 
164  // Backup original nodal positions with boundary nodes snapped
165  // into quadratic position; will soon move these back to
166  // undeformed positon and gently move them back towards

170  for (unsigned j=0;j<nnod;j++)
171  {

173  SolidNode* nod_pt=dynamic_cast<SolidNode*>(Orig_mesh_pt->node_pt(j));
174  for (unsigned i=0;i<dim;i++)
175  {
176  Orig_node_pos[j][i]=nod_pt->x(i);
177  }
178  }
179 
180  // Meshes containing the face elements that represent the
181  // quadratic surface
182  Vector<SolidMesh*> quadratic_surface_mesh_pt(nbound);
183 
184  // GeomObject incarnations
185  Vector<MeshAsGeomObject*> quadratic_surface_geom_obj_pt(nbound);
186 
187 
188  // Create FaceElements on original mesh to define
189  //-------------------------------------------------
190  // the desired boundary shape
191  //---------------------------
192 
193  unsigned n=controlled_boundary_id.size();
194  for (unsigned i=0;i<n;i++)
195  {
196  // Get boundary ID
197  unsigned b=controlled_boundary_id[i];
198 
199  // Create mesh for surface elements
200  quadratic_surface_mesh_pt[b]=new SolidMesh;
201 
202  // How many bulk elements are adjacent to boundary b?
203  unsigned n_element = Orig_mesh_pt->nboundary_element(b);
204 
205  // Loop over the bulk elements adjacent to boundary b
206  for(unsigned e=0;e<n_element;e++)
207  {
208  // Get pointer to the bulk element that is adjacent to boundary b
209  ELEMENT* bulk_elem_pt = dynamic_cast<ELEMENT*>(
210  Orig_mesh_pt->boundary_element_pt(b,e));
211 
212  //What is the index of the face of the element e along boundary b
213  int face_index = Orig_mesh_pt->face_index_at_boundary(b,e);
214 
215  // Create new element
216  SolidTractionElement<ELEMENT>* el_pt=
217  new SolidTractionElement<ELEMENT>(bulk_elem_pt,face_index);
218 
219  // Add it to the mesh
220  quadratic_surface_mesh_pt[b]->add_element_pt(el_pt);
221 
222  // Specify boundary number
223  el_pt->set_boundary_number_in_bulk_mesh(b);
224  }
225 
226  // Create GeomObject incarnation
227  quadratic_surface_geom_obj_pt[b]=
228  new MeshAsGeomObject(quadratic_surface_mesh_pt[b]);

233  //-------------------------------------------------------

242 
243  // How many bulk elements are adjacent to boundary b?
244  unsigned n_element = Dummy_mesh_pt->nboundary_element(b);
245 
246  // Loop over the bulk fluid elements adjacent to boundary b?
247  for(unsigned e=0;e<n_element;e++)
248  {

266  // the boundary coordinate from the bulk nodes)
267  el_pt->set_boundary_shape_geom_object_pt(
268  quadratic_surface_geom_obj_pt[b],b);
269  }
270 
271  // Add sub mesh
272  add_sub_mesh(dummy_lagrange_multiplier_mesh_pt[i]);
273  }

276  // Combine the lot
277  build_global_mesh();

279  oomph_info << "Number of equations for nonlinear smoothing problem: "
280  << assign_eqn_numbers() << std::endl;
281 
282 
283  // Complete the build of the elements so they are fully functional
284  //----------------------------------------------------------------
285  unsigned n_element = Dummy_mesh_pt->nelement();
286  for(unsigned e=0;e<n_element;e++)
287  {
288  // Upcast from GeneralisedElement to the present element
289  ELEMENT* el_pt =
290  dynamic_cast<ELEMENT*>(Dummy_mesh_pt->element_pt(e));
291 
292  // Set the constitutive law for pseudo-elastic mesh deformation
293  el_pt->constitutive_law_pt() =
294  Helper_namespace_for_mesh_smoothing::Constitutive_law_pt;
295 
296  } // end loop over elements

298 
299  //Output initial configuration
300  doc_solution(doc_info);
301  doc_info.number()++;
302 
303  // Initial scale
304  Helper_namespace_for_mesh_smoothing::Scale=0.0;
305  Helper_namespace_for_mesh_smoothing::Scale_increment=0.1;
306 
307 
308  // Increase scale of deformation until full range is reached
309  //----------------------------------------------------------
310  bool done=false;
311  unsigned count=0;
312  while (!done)
313  {
314 
315  // Increase scale
316  Helper_namespace_for_mesh_smoothing::Scale+=
317  Helper_namespace_for_mesh_smoothing::Scale_increment;
318 
319  // Backup current nodal positions in dummy mesh
320  backup();
321 
322  // Try it...
323  bool success=true;
324  try
325  {
326  // Avoid overshoot
327  if (Helper_namespace_for_mesh_smoothing::Scale>1.0)
328  {
329  Helper_namespace_for_mesh_smoothing::Scale=1.0;
330  }
331

340  Helper_namespace_for_mesh_smoothing::Scale_increment/=2.0;
341 
342  // Reset current nodal positions in dummy mesh
343  reset();
344  }
345 
346  //Output solution
347  if (success)

349  count++;
350  doc_solution(doc_info);
351  doc_info.number()++;

363  << Helper_namespace_for_mesh_smoothing::Scale << std::endl;

366  for (unsigned j=0;j<nnode;j++)
367  {
368  // Get nodes
369  Node* orig_node_pt=orig_mesh_pt->node_pt(j);

371

374  {
375  orig_node_pt->x(i)=new_node_pt->x(i);
376  }
377  }

519 //====================================================================
520 /// Auxiliary Problem to smooth a SolidMesh by adjusting the internal
521 /// nodal positions by solving a LINEAR solid mechanics problem for the
522 /// nodal displacements between the specified displacements of certain
523 /// pinned nodes (usually located on boundaries). The template
524 /// parameter specifies the linear elasticity element that must have
525 /// the same shape (geometric element type) as the elements contained
526 /// in the mesh that's to be smoothed. So, e.g. for the ten-noded
527 /// three-dimensional tetrahedral TTaylorHoodElement<3>, it would be
528 /// a TLinearElasticityElement<3,3>, etc.
529 /// \b Important \b assumptions:
530 /// - It is important that the Lagrangian coordinates of all nodes
531 /// still indicate their original position, i.e. their position
532 /// before (some of) them were moved to their new position.
533 /// - It is assumed that in its original state, the mesh does not contain
534 /// any inverted elements.
535 /// .
536 //====================================================================
537 template<class LINEAR_ELASTICITY_ELEMENT>
538 class LinearElasticitySmoothMesh : public Problem
539 {
540 
541  public:
542 
543  /// \short Constructor: Specify SolidMesh whose nodal positions are to
544  /// be adjusted, and set of nodes in that mesh whose position
545  /// are to remain fixed.
546  void operator()(SolidMesh* orig_mesh_pt,
547  std::set<Node*> pinned_nodes)
548  {
549  // Create new mesh and read out node/element numbers from old one
550  mesh_pt()=new Mesh;
551  unsigned nelem=orig_mesh_pt->nelement();
552  unsigned nnode=orig_mesh_pt->nnode();
553 
554  // Have we already created that node?
555  std::map<Node*,Node*> new_node;
556 
557  // Create new elements
558  for (unsigned e=0;e<nelem;e++)
559  {
560 
561  // Make/add new element
562  LINEAR_ELASTICITY_ELEMENT* el_pt=new LINEAR_ELASTICITY_ELEMENT;
563  mesh_pt()->add_element_pt(el_pt);
564 
565  //Set elasticity tensor
566  el_pt->elasticity_tensor_pt()=
567  &Helper_namespace_for_mesh_smoothing::Isotropic_elasticity_tensor;
568 
569  // Find corresponding original element
570  SolidFiniteElement* orig_elem_pt=
571  dynamic_cast<SolidFiniteElement*>(orig_mesh_pt->finite_element_pt(e));
572  unsigned nnod=orig_elem_pt->nnode();
573 
574  // Create nodes
575  for (unsigned j=0;j<nnod;j++)
576  {
577  // Does it not exist yet?
578  if (new_node[orig_elem_pt->node_pt(j)]==0)
579  {
580  Node* new_nod_pt=mesh_pt()->finite_element_pt(e)->construct_node(j);
581  new_node[orig_elem_pt->node_pt(j)]=new_nod_pt;
582  mesh_pt()->add_node_pt(new_nod_pt);
583  unsigned dim=new_nod_pt->ndim();
584  for (unsigned i=0;i<dim;i++)
585  {
586  // Set new nodal position to be the old one in the
587  // SolidMesh (assumed to contain no inverted elements)
588  new_nod_pt->x(i)=
589  dynamic_cast<SolidNode*>(orig_elem_pt->node_pt(j))->xi(i);
590  }
591  }
592  // It already exists -- copy across
593  else
594  {
595  mesh_pt()->finite_element_pt(e)->node_pt(j)=
596  new_node[orig_elem_pt->node_pt(j)];
597  }
598  }
599  }
600 
601  // Loop over pinned nodes -- pin their positions and assign updated nodal
602  // positions
603  double scale=1.0;
604  for (std::set<Node*>::iterator it=pinned_nodes.begin();
605  it!=pinned_nodes.end();it++)
606  {
607  unsigned dim=(*it)->ndim();
608  for (unsigned i=0;i<dim;i++)
609  {
610  new_node[*it]->pin(i);
611  new_node[*it]->set_value(i,scale*
612  (dynamic_cast<SolidNode*>(*it)->x(i)-
613  dynamic_cast<SolidNode*>(*it)->xi(i)));
614  }
615  }
616 
617  oomph_info << "Number of equations for smoothing problem: "
618  << assign_eqn_numbers() << std::endl;
619 
620 
621  // Solve
622  newton_solve();
623 
624  // Loop over nodes and assign displacement difference
625  for (unsigned j=0;j<nnode;j++)
626  {
627  // Get nodes
628  SolidNode* orig_node_pt=orig_mesh_pt->node_pt(j);
629  Node* new_node_pt=new_node[orig_node_pt];
630 
631  // Assign displacement difference
632  unsigned dim=new_node_pt->ndim();
633  for (unsigned i=0;i<dim;i++)
634  {
635  orig_node_pt->x(i)=orig_node_pt->xi(i)+new_node_pt->value(i);
636  }
637  }
638 
639  // Now re-assign undeformed position
640  orig_mesh_pt->set_lagrangian_nodal_coordinates();
641 
642  // Clean up -- mesh deletes nodes and elements
643  delete mesh_pt();
644  }
645 
646  /// Destructor (empty)
647  ~LinearElasticitySmoothMesh(){}
648

657 //====================================================================
658 /// Functor to smooth a SolidMesh by adjusting the internal
659 /// nodal positions by solving a Poisson problem for the
660 /// nodal displacements in the interior. The displacements of the specified
661 /// pinned nodes (usually located on boundaries) remain fixed (their
662 /// displacements are computed from the difference between their
663 /// Lagrangian and Eulerian coordinates). The assumptions is
664 /// that the Lagrangian coordinates in the SolidMesh still reflect
665 /// the original nodal positions before the boundary nodes were
666 /// moved.
667 /// \n
668 /// The template parameter specifies the Poisson element that must have
669 /// the same shape (geometric element type) as the elements contained
670 /// in the mesh that's to be smoothed. So, e.g. for the ten-noded
671 /// three-dimensional tetrahedral TTaylorHoodElement<3>, it would be
672 /// a TPoissonElement<3,3>, etc.
673 //====================================================================
674 template<class POISSON_ELEMENT>
675 class PoissonSmoothMesh : public Problem
676 {

680  /// \short Functor: Specify SolidMesh whose nodal positions are to
681  /// be adjusted, and set of nodes in that mesh whose position
682  /// are to remain fixed.
683  void operator()(SolidMesh* orig_mesh_pt,
684  std::set<Node*> pinned_nodes)
685  {
686  // Create new mesh and read out node/element numbers from old one
687  mesh_pt()=new Mesh;
688  unsigned nelem=orig_mesh_pt->nelement();
689  unsigned nnode=orig_mesh_pt->nnode();
690 
691  // Have we already created that node?
692  std::map<Node*,Node*> new_node;
693 
694  // Create new elements
695  for (unsigned e=0;e<nelem;e++)
696  {
697  mesh_pt()->add_element_pt(new POISSON_ELEMENT);
698 
699  // Find corresponding original element
700  SolidFiniteElement* orig_elem_pt=
701  dynamic_cast<SolidFiniteElement*>(orig_mesh_pt->finite_element_pt(e));
702  unsigned nnod=orig_elem_pt->nnode();
703 
704  // Create nodes
705  for (unsigned j=0;j<nnod;j++)
706  {
707  // Does it not exist yet?
708  if (new_node[orig_elem_pt->node_pt(j)]==0)
709  {
710  Node* new_nod_pt=mesh_pt()->finite_element_pt(e)->construct_node(j);
711  new_node[orig_elem_pt->node_pt(j)]=new_nod_pt;
712  mesh_pt()->add_node_pt(new_nod_pt);
713  unsigned dim=new_nod_pt->ndim();
714  for (unsigned i=0;i<dim;i++)
715  {
716  // Set new nodal position to be the old one in the
717  // SolidMesh (assumed to contain no inverted elements)
718  new_nod_pt->x(i)=
719  dynamic_cast<SolidNode*>(orig_elem_pt->node_pt(j))->xi(i);
720  }
721  }
722  // It already exists -- copy across
723  else
724  {
725  mesh_pt()->finite_element_pt(e)->node_pt(j)=
726  new_node[orig_elem_pt->node_pt(j)];
727  }
728  }
729  }
730 
731 
732  // Loop over pinned nodes
733  for (std::set<Node*>::iterator it=pinned_nodes.begin();
734  it!=pinned_nodes.end();it++)
735  {
736  new_node[*it]->pin(0);
737  }
738 
739  oomph_info << "Number of equations for Poisson displacement smoothing: "
740  << assign_eqn_numbers() << std::endl;
741 
742  // Solve separate displacement problems
743  unsigned dim=orig_mesh_pt->node_pt(0)->ndim();
744  for (unsigned i=0;i<dim;i++)
745  {
746  // Loop over nodes and assign displacement difference
747  for (unsigned j=0;j<nnode;j++)
748  {
749  // Get nodes
750  SolidNode* orig_node_pt=orig_mesh_pt->node_pt(j);
751  Node* new_node_pt=new_node[orig_node_pt];
752 
753  // Assign displacement difference
754  new_node_pt->set_value(0,orig_node_pt->x(i)-orig_node_pt->xi(i));
755  }
756 
757  // Solve
758  newton_solve();
759 
760  // Loop over nodes and assign displacement difference
761  for (unsigned j=0;j<nnode;j++)
762  {
763  // Get nodes
764  SolidNode* orig_node_pt=orig_mesh_pt->node_pt(j);
765  Node* new_node_pt=new_node[orig_node_pt];
766 
767  // Assign displacement difference
768  orig_node_pt->x(i)=orig_node_pt->xi(i)+new_node_pt->value(0);
769  }
770  }
771 
772  // Now re-assign undeformed position
773  orig_mesh_pt->set_lagrangian_nodal_coordinates();
774 
775  // Clean up -- mesh deletes nodes and elements
776  delete mesh_pt();
777 
778  }
