104  /// \short du/dt at local node n.
105  /// Uses suitably interpolated value for hanging nodes.
106  double du_dt_adv_diff(const unsigned& n) const
107  {
108  // Get the data's timestepper
109  TimeStepper* time_stepper_pt = this->node_pt(n)->time_stepper_pt();
110 
111  // Initialise dudt
112  double dudt = 0.0;
113  // Loop over the timesteps, if there is a non Steady timestepper
114  if (!time_stepper_pt->is_steady())
115  {
116  // Find the index at which the variable is stored
117  const unsigned u_nodal_index = u_index_adv_diff();
118 
119  // Number of timsteps (past & present)
120  const unsigned n_time = time_stepper_pt->ntstorage();
121 
122  for (unsigned t = 0; t < n_time; t++)
123  {
124  dudt +=
125  time_stepper_pt->weight(1, t) * nodal_value(t, n, u_nodal_index);
126  }
127  }
128  return dudt;
129  }

349  /// \short Get source term at (Eulerian) position x. This function is
350  /// virtual to allow overloading in multi-physics problems where
351  /// the strength of the source function might be determined by

353  inline virtual void get_source_adv_diff(const unsigned& ipt,
354  const Vector<double>& x,
355  double& source) const

394  /// Get flux: \f$\mbox{flux}[i] = \mbox{d}u / \mbox{d}x_i \f$
395  void get_flux(const Vector<double>& s, Vector<double>& flux) const
396  {
397  // Find out how many nodes there are in the element
398  unsigned n_node = nnode();
399 
400  // Get the nodal index at which the unknown is stored
401  unsigned u_nodal_index = u_index_adv_diff();
402 
403  // Set up memory for the shape and test functions
404  Shape psi(n_node);
405  DShape dpsidx(n_node, DIM);
406 
407  // Call the derivatives of the shape and test functions
408  dshape_eulerian(s, psi, dpsidx);
409 
410  // Initialise to zero
411  for (unsigned j = 0; j < DIM; j++)
412  {
413  flux[j] = 0.0;
414  }
415 
416  // Loop over nodes
417  for (unsigned l = 0; l < n_node; l++)
418  {
419  // Loop over derivative directions
420  for (unsigned j = 0; j < DIM; j++)
421  {
422  flux[j] += nodal_value(l, u_nodal_index) * dpsidx(l, j);
423  }
424  }
425  }

465  /// Return FE representation of function value u(s) at local coordinate s
466  inline double interpolated_u_adv_diff(const Vector<double>& s) const
467  {
468  // Find number of nodes
469  unsigned n_node = nnode();
470 
471  // Get the nodal index at which the unknown is stored
472  unsigned u_nodal_index = u_index_adv_diff();
473 
474  // Local shape function
475  Shape psi(n_node);
476 
477  // Find values of shape function
478  shape(s, psi);
479 
480  // Initialise value of u
481  double interpolated_u = 0.0;
482 
483  // Loop over the local nodes and sum
484  for (unsigned l = 0; l < n_node; l++)
485  {
486  interpolated_u += nodal_value(l, u_nodal_index) * psi[l];
487  }
488 
489  return (interpolated_u);
490  }

493  ///\short Return derivative of u at point s with respect to all data
494  /// that can affect its value.
495  /// In addition, return the global equation numbers corresponding to the
496  /// data. This is virtual so that it can be overloaded in the
497  /// refineable version
498  virtual void dinterpolated_u_adv_diff_ddata(
499  const Vector<double>& s,
500  Vector<double>& du_ddata,
501  Vector<unsigned>& global_eqn_number)
502  {
503  // Find number of nodes
504  const unsigned n_node = nnode();
505 
506  // Get the nodal index at which the unknown is stored
507  const unsigned u_nodal_index = u_index_adv_diff();
508 
509  // Local shape function
510  Shape psi(n_node);
511 
512  // Find values of shape function
513  shape(s, psi);
514 
515  // Find the number of dofs associated with interpolated u
516  unsigned n_u_dof = 0;
517  for (unsigned l = 0; l < n_node; l++)
518  {
519  int global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
520  // If it's positive add to the count
521  if (global_eqn >= 0)
522  {
523  ++n_u_dof;
524  }
525  }
526 
527  // Now resize the storage schemes
528  du_ddata.resize(n_u_dof, 0.0);
529  global_eqn_number.resize(n_u_dof, 0);
530 
531  // Loop over the nodes again and set the derivatives
532  unsigned count = 0;
533  for (unsigned l = 0; l < n_node; l++)
534  {
535  // Get the global equation number
536  int global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
537  // If it's positive
538  if (global_eqn >= 0)
539  {
540  // Set the global equation number
541  global_eqn_number[count] = global_eqn;
542  // Set the derivative with respect to the unknown
543  du_ddata[count] = psi[l];
544  // Increase the counter
545  ++count;
546  }
547  }
548  }
