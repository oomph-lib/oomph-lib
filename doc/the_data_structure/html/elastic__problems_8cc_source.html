42 /// Setup IC problem by:
43 /// - Pinning all nodal values in the mesh
44 /// - Pinning the internal data of all elements.
45 /// - Freeing/unpinnning all positional data.
46 /// - Flushing the pointers to the elements' external data.
47 /// - Setting the pointer to the IC object for all elements to
48 /// ensure that the correct residuals/Jacobians are computed.
49 //======================================================================

51 {

53  //Find out how many nodes there are
54  unsigned long n_node = mesh_pt()->nnode();
55 
56  //Loop over all the nodes

58  {

61

355 void SolidICProblem::reset_original_state()
356 {
357 
358  //Find out how many nodes there are
359  unsigned long n_node = mesh_pt()->nnode();
360 
361  // Initialise counter for backed up dofs
362  unsigned count=0;
363 
364  //Loop over all the nodes
365  for(unsigned n=0;n<n_node;n++)
366  {
367  // Cast to an elastic node
368  SolidNode* node_pt=dynamic_cast<SolidNode*>(mesh_pt()->node_pt(n));
369 
370 #ifdef PARANOID
371  if (node_pt==0)
372  {
373  throw OomphLibError("Wasn't able to cast to SolidNode\n",
374  OOMPH_CURRENT_FUNCTION,
375  OOMPH_EXCEPTION_LOCATION);
376  }
377 #endif
378 
379  // Get spatial dimension of node
380  unsigned ndim=node_pt->ndim();
381 
382  //Find out how many positional dofs there are
383  SolidFiniteElement* elem_pt=
384  dynamic_cast<SolidFiniteElement*>(mesh_pt()->element_pt(0));
385 
386 #ifdef PARANOID
387  if (elem_pt==0)
388  {
389  throw OomphLibError("Wasn't able to cast to SolidFiniteElement\n",
390  OOMPH_CURRENT_FUNCTION,
391  OOMPH_EXCEPTION_LOCATION);
392  }
393 #endif
394 
395  unsigned ntype = elem_pt->nnodal_position_type();
396 
397  //Loop over coordinate directions
398  for(unsigned i=0;i<ndim;i++)
399  {
400  //Loop over type of dof
401  for(unsigned k=0;k<ntype;k++)
402  {
403  // Reset pinned status (positional dofs were all unpinned)
404  if (Backup_pinned[count])
405  {
406  node_pt->pin_position(k,i);
407  }
408  count++;
409  }
410  }
411 
412  // Loop over nodal values
413  unsigned nval=node_pt->nvalue();
414  for (unsigned ival=0;ival<nval;ival++)
415  {
416  // Reset pinned status (nodal values were all pinned)
417  if (Backup_pinned[count])
418  {
419  node_pt->unpin(ival);
420  }
421  }
422  }
423 
424 
425  // Loop over the elements
426  unsigned Nelement=mesh_pt()->nelement();
427  for(unsigned i=0;i<Nelement;i++)
428  {
429 
430  //Cast to proper element type
431  SolidFiniteElement *elem_pt = dynamic_cast<SolidFiniteElement*>(
432  mesh_pt()->element_pt(i));
433 
434 #ifdef PARANOID
435  if (elem_pt==0)
436  {
437  throw OomphLibError("Wasn't able to cast to SolidFiniteElement\n",
438  OOMPH_CURRENT_FUNCTION,
439  OOMPH_EXCEPTION_LOCATION);
440  }
441 #endif
442 
443  // Switch back to normal Jacobian
444  dynamic_cast<SolidFiniteElement*>(elem_pt)->
445  disable_solve_for_consistent_newmark_accel();
446 
447  // Switch off flag for setting initial condition
448  dynamic_cast<SolidFiniteElement*>(elem_pt)->solid_ic_pt() = 0;
449 
450  //IF it's an FSI wall element then turn on external stuff again
451  if(FSIWallElement* fsi_elem_pt = dynamic_cast<FSIWallElement*>(elem_pt))
452  {
453  fsi_elem_pt->include_external_load_data();
454  }
455 
456  // Find out number of external data
457  unsigned next=Backup_ext_data[i].size();
458 
459  // Loop over external data
460  for (unsigned j=0;j<next;j++)
461  {
462  // Backed up external data
463  Data* data_pt=Backup_ext_data[i][j];
464 
465  // Add external data
466  elem_pt->add_external_data(data_pt);
467  }
468 
469  // Find out number of internal data
470  unsigned nint=elem_pt->ninternal_data();
471 
472  // Loop over internal data
473  for (unsigned j=0;j<nint;j++)
474  {
475  Data* data_pt=elem_pt->internal_data_pt(j);
476 
477  // Loop over internal values
478  unsigned nval=data_pt->nvalue();
479  for (unsigned ival=0;ival<nval;ival++)
480  {
481  // Restore pinned status (values were all pinned)
482  if (!Backup_pinned[count])
483  {
484  data_pt->unpin(ival);
485  }
486  }
487  }
488 
489 
490 #ifdef PARANOID
491  //If there is internal solid data, complain
492  if (elem_pt->has_internal_solid_data())
493  {
494  std::string error_message =
495  "Automatic assignment of initial conditions doesn't work yet\n";
496  error_message +=
497  "for elasticity elements with internal solid dofs (pressures)\n";
498 
499  throw OomphLibError(error_message,
500  OOMPH_CURRENT_FUNCTION,
501  OOMPH_EXCEPTION_LOCATION);
502  }
503 #endif

507 
508 // // Loop over values

527

529  Backup_ext_data.clear();

535 /// IC problem for wall: Deform wall into the static initial shape
536 /// described by the IC object at given time.
537 //======================================================================
538 void SolidICProblem::set_static_initial_condition(Problem* problem_pt,
539  Mesh* wall_mesh_pt,
540  SolidInitialCondition* ic_pt,
541  const double& time)
542 {
543  // Tell this sub-problem it is distributed if the main problem is distributed
544 #ifdef OOMPH_HAS_MPI
545  if (problem_pt->problem_has_been_distributed())
546  {
547  Problem_has_been_distributed=true;
548  }
549  // This (sub-)problem needs to know about the oomph communicator
550  delete Communicator_pt;
551  Communicator_pt=new OomphCommunicator(problem_pt->communicator_pt());
552 #endif
553 
554 
555  // Backup value of time
556  double backup_time=0.0;
557 
558  // Set value of time for IC object (needs to be backed up and
559  // restored since it might be pointed to by other objects)
560  TimeStepper* timestepper_pt=ic_pt->geom_object_pt()->time_stepper_pt();
561  if (timestepper_pt!=0)
562  {
563  backup_time=timestepper_pt->time_pt()->time();
564  timestepper_pt->time_pt()->time()=time;
565  }
566 
567  // Delete dummy mesh
568  delete mesh_pt();
569 
570  // Set pointer to mesh
571  mesh_pt()=wall_mesh_pt;
572 
573  // Set pointer to initial condition object
574  IC_pt=ic_pt;
575 
576  // Backup the pinned status of all dofs and remove external data
577  // of all elements
578  backup_original_state();
579 
580  // Now alter the pinned status so that the IC problem for the
581  // positional variables can be solved; setup equation numbering
582  // scheme
583  setup_problem();
584 
585  // Assign displacements
586  IC_pt->ic_time_deriv()=0;
587 
588  // Solve the problem for initial shape
589  newton_solve();
590 
591  // Impulsive start
592  assign_initial_values_impulsive();
593 
594  // Reset the pinned status and re-attach the external data to the elements
595  reset_original_state();
596 
597  // Reset time
598  if (timestepper_pt!=0)
599  {
600  timestepper_pt->time_pt()->time()=backup_time;
601  }
602 
603  // Set pointer to dummy mesh so there's something that can be deleted
604  // when static problem finally goes out of scope.
605  mesh_pt()=new DummyMesh;
606 
607  // We have temporarily over-written equation numbers -- need
608  // to reset them now
609  oomph_info << "Number of equations in big problem: "
610  << problem_pt->assign_eqn_numbers() << std::endl;
611 }
