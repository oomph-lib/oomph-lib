103  ///Recursively add any master nodes (and their master nodes etc) of
104  /// external nodes
105  //========================================================================
106  void recursively_add_masters_of_external_haloed_node(
107  int& iproc, Node* nod_pt,
108  Mesh* const &mesh_pt,
109  int& n_cont_inter_values,
110  Vector<unsigned>& send_unsigneds,
111  Vector<double>& send_doubles)
112  {
113 
114  // Loop over continuously interpolated values and add masters
115  for (int i_cont=-1;i_cont<n_cont_inter_values;i_cont++)
116  {
117  if (nod_pt->is_hanging(i_cont))
118  {
119  // Indicate that this node is a hanging node so the other
120  // process knows to create HangInfo and masters, etc.
121  send_unsigneds.push_back(1);
122 #ifdef ANNOTATE_MISSING_MASTERS_COMMUNICATION
123  Flat_packed_unsigneds_string.push_back("Is hanging");
124 #endif
125  // If this is a hanging node then add all its masters as
126  // external halo nodes if they have not yet been added
127  HangInfo* hang_pt=nod_pt->hanging_pt(i_cont);
128  // Loop over masters
129  unsigned n_master=hang_pt->nmaster();
130 
131  // Indicate number of master nodes to add on other process
132  send_unsigneds.push_back(n_master);
133 #ifdef ANNOTATE_MISSING_MASTERS_COMMUNICATION
134  Flat_packed_unsigneds_string.push_back("nmaster");
135 #endif
136  for (unsigned m=0;m<n_master;m++)
137  {
138  Node* master_nod_pt=hang_pt->master_node_pt(m);
139 
140  // Call the helper function for master nodes
141  add_external_haloed_master_node_helper(iproc,master_nod_pt,
142  mesh_pt,
143  n_cont_inter_values,
144  send_unsigneds,
145  send_doubles);
146 
147  // Indicate the weight of this master
148  send_doubles.push_back(hang_pt->master_weight(m));
149 
150  // Recursively add any master nodes (and their master nodes etc)
151  recursively_add_masters_of_external_haloed_node(iproc, master_nod_pt,
152  mesh_pt,
153  n_cont_inter_values,
154  send_unsigneds,
155  send_doubles);
156  }
157  }
158  else
159  {
160  // Indicate that it's not a hanging node in this variable
161  send_unsigneds.push_back(0);
162 #ifdef ANNOTATE_MISSING_MASTERS_COMMUNICATION
163  Flat_packed_unsigneds_string.push_back("Not hanging");
164 #endif
165  }
166  } // end loop over continously interpolated values

430 //========start of get_required_nodal_information_helper==================
431 /// Helper function to get the required nodal information from an
432 /// external haloed node so that a fully-functional external halo
433 /// node (and therefore element) can be created on the receiving process
434 //========================================================================
435  void get_required_nodal_information_helper(int& iproc, Node* nod_pt,
436  Mesh* const &mesh_pt,
437  int& n_cont_inter_values,
438  Vector<unsigned>& send_unsigneds,
439  Vector<double>& send_doubles)
440  {
441  // Tell the halo copy of this node how many values there are
442  // [NB this may be different for nodes within the same element, e.g.
443  // when using Lagrange multipliers]
444  unsigned n_val=nod_pt->nvalue();
445  send_unsigneds.push_back(n_val);
446 #ifdef ANNOTATE_MISSING_MASTERS_COMMUNICATION
447  Flat_packed_unsigneds_string.push_back("Number of values");
448 #endif
449 
450  unsigned n_dim=nod_pt->ndim();
451  TimeStepper* time_stepper_pt=nod_pt->time_stepper_pt();
452 
453  // Default number of previous values to 1
454  unsigned n_prev=1;
455  if (time_stepper_pt!=0)
456  {
457  // Add number of history values to n_prev
458  n_prev=time_stepper_pt->ntstorage();
459  }
460 
461  // Is the node on any boundaries?
462  if (nod_pt->is_on_boundary())
463  {
464  send_unsigneds.push_back(1);
465 #ifdef ANNOTATE_MISSING_MASTERS_COMMUNICATION
466  Flat_packed_unsigneds_string.push_back("Node is on boundary");
467 #endif
468 
469  // Loop over the boundaries of the external mesh
470  Vector<unsigned> boundaries;
471  unsigned n_bnd=mesh_pt->nboundary();
472  for (unsigned i_bnd=0;i_bnd<n_bnd;i_bnd++)
473  {
474  // Which boundaries (could be more than one) is it on?
475  if (nod_pt->is_on_boundary(i_bnd))
476  {
477  boundaries.push_back(i_bnd);
478  }
479  }
480  unsigned nb=boundaries.size();
481  send_unsigneds.push_back(nb);
482 #ifdef ANNOTATE_MISSING_MASTERS_COMMUNICATION
483  std::stringstream junk;
484  junk << "Node is on "<< nb << " boundaries";
485  Flat_packed_unsigneds_string.push_back(junk.str());
486 #endif
487  for (unsigned i=0;i<nb;i++)
488  {
489  send_unsigneds.push_back(boundaries[i]);
490 #ifdef ANNOTATE_MISSING_MASTERS_COMMUNICATION
491  std::stringstream junk;
492  junk << "Node is on boundary " << boundaries[i] << " of " << n_bnd;
493  Flat_packed_unsigneds_string.push_back(junk.str());
494 #endif

498  // additional values created by face elements

520 #endif

530  Flat_packed_unsigneds_string.push_back(junk.str());

538 #ifdef ANNOTATE_MISSING_MASTERS_COMMUNICATION

545 #endif

547  }
548  }
