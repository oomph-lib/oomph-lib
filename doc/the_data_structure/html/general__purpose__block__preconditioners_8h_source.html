64  /// Base class for general purpose block preconditioners. Deals with
65  /// setting subsidiary preconditioners and dof to block maps.
66  /// Subsidiary preconditioners can be set in two ways:
67  /// 1) A pointer to a subsidiary preconditioner for block i can be passed
68  /// to set_subsidiary_preconditioner_pt(prec, i).
69  /// 2) A default subsidiary preconditioner can be set up by providing a
70  /// function pointer to a function which creates a preconditioner. During
71  /// setup() all unset subsidiary preconditioner pointers will be filled in
72  /// using this function. By default this uses SuperLU.
73  //============================================================================
74  template<typename MATRIX>

81  /// employed to solve the subsidiary linear systems.
82  /// The function should return a pointer to the required subsidiary
83  /// preconditioner generated using new. This preconditioner is responsible
84  /// for the destruction of the subsidiary preconditioners.
85  typedef Preconditioner* (*SubsidiaryPreconditionerFctPt)();

152  /// \short Set the subsidiary preconditioner to use for block i. The
153  /// subsidiary preconditioner should have been created using new (the
154  /// general purpose block preconditioner will delete it later). If null
155  /// the general purpose block preconditioner will use the
156  /// Subsidiary_preconditioner_creation_function_pt to create the
157  /// preconditioner during setup().
158  void set_subsidiary_preconditioner_pt(Preconditioner* prec, const unsigned &i)
159  {
160  // If the vector is currently too small to hold that many
161  // preconditioners then expand it and fill with nulls.
162  if(Subsidiary_preconditioner_pt.size() < i +1)
163  {
164  Subsidiary_preconditioner_pt.resize(i+1, 0);
165  }
166  // Note: the size of the vector is checked by
167  // fill_in_subsidiary_preconditioners(..) when we know what size it
168  // should be.
169 
170  // I'm assuming that the number of preconditioners is always "small"
171  // compared to Jacobian size, so a resize doesn't waste much time.
172 
173  // Put the pointer in the vector
174  Subsidiary_preconditioner_pt[i] = prec;
175  }

257  /// \short Create any subsidiary preconditioners needed. Usually
258  /// nprec_needed = nblock_types, except for the ExactBlockPreconditioner
259  /// which only requires one preconditioner.
260  void fill_in_subsidiary_preconditioners(const unsigned &nprec_needed)
261  {
262 
263  // If it's empty then fill it in with null pointers.
264  if(Subsidiary_preconditioner_pt.empty())
265  {
266  Subsidiary_preconditioner_pt.assign(nprec_needed, 0);
267  }
268  else
269  {
270  // Otherwise check we have the right number of them
271 #ifdef PARANOID
272  if(Subsidiary_preconditioner_pt.size() != nprec_needed)
273  {
274  using namespace StringConversion;
275  std::string error_msg = "Wrong number of precondtioners in";
276  error_msg += "Subsidiary_preconditioner_pt, should have ";
277  error_msg += to_string(nprec_needed) + " but we actually have ";
278  error_msg += to_string(Subsidiary_preconditioner_pt.size());
279  throw OomphLibError(error_msg, OOMPH_CURRENT_FUNCTION,
280  OOMPH_EXCEPTION_LOCATION);
281  }
282 #endif
283  }
284 
285 
286  // Now replace any null pointers with new preconditioners
287  for(unsigned j=0, nj=Subsidiary_preconditioner_pt.size(); j<nj; j++)
288  {
289  if(Subsidiary_preconditioner_pt[j] == 0)
290  {
291  Subsidiary_preconditioner_pt[j] =
292  (*Subsidiary_preconditioner_creation_function_pt)();
293  }
294  }
295 
296  }
