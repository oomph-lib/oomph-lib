176  /// Further build: Copy source function pointer from father element
177  void further_build()
178  {

180  = dynamic_cast<RefineableAdvectionDiffusionEquations<DIM>*>(

193  /// \short Compute the derivatives of the i-th component of
194  /// velocity at point s with respect
195  /// to all data that can affect its value. In addition, return the global
196  /// equation numbers corresponding to the data.
197  /// Overload the non-refineable version to take account of hanging node
198  /// information
199  void dinterpolated_u_adv_diff_ddata(const Vector<double> &s,
200  Vector<double> &du_ddata,
201  Vector<unsigned> &global_eqn_number)
202  {
203  //Find number of nodes
204  unsigned n_node = this->nnode();
205  //Local shape function
206  Shape psi(n_node);
207  //Find values of shape function at the given local coordinate
208  this->shape(s,psi);
209 
210  //Find the index at which the velocity component is stored
211  const unsigned u_nodal_index = this->u_index_adv_diff();
212 
213  //Storage for hang info pointer
214  HangInfo* hang_info_pt=0;
215  //Storage for global equation
216  int global_eqn = 0;
217 
218  //Find the number of dofs associated with interpolated u
219  unsigned n_u_dof=0;
220  for(unsigned l=0;l<n_node;l++)
221  {
222  unsigned n_master = 1;
223 
224  //Local bool (is the node hanging)
225  bool is_node_hanging = this->node_pt(l)->is_hanging();
226 
227  //If the node is hanging, get the number of master nodes
228  if(is_node_hanging)
229  {
230  hang_info_pt = this->node_pt(l)->hanging_pt();
231  n_master = hang_info_pt->nmaster();
232  }
233  //Otherwise there is just one master node, the node itself
234  else
235  {
236  n_master = 1;
237  }
238 
239  //Loop over the master nodes
240  for(unsigned m=0;m<n_master;m++)
241  {
242  //Get the equation number
243  if(is_node_hanging)
244  {
245  //Get the equation number from the master node
246  global_eqn = hang_info_pt->master_node_pt(m)->
247  eqn_number(u_nodal_index);
248  }
249  else
250  {
251  // Global equation number
252  global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
253  }
254 
255  //If it's positive add to the count
256  if (global_eqn >= 0) {++n_u_dof;}
257  }
258  }
259 
260  //Now resize the storage schemes
261  du_ddata.resize(n_u_dof,0.0);
262  global_eqn_number.resize(n_u_dof,0);
263 
264  //Loop over th nodes again and set the derivatives
265  unsigned count=0;
266  //Loop over the local nodes and sum
267  for(unsigned l=0;l<n_node;l++)
268  {
269  unsigned n_master = 1;
270  double hang_weight = 1.0;
271 
272  //Local bool (is the node hanging)
273  bool is_node_hanging = this->node_pt(l)->is_hanging();
274 
275  //If the node is hanging, get the number of master nodes
276  if(is_node_hanging)
277  {
278  hang_info_pt = this->node_pt(l)->hanging_pt();
279  n_master = hang_info_pt->nmaster();
280  }
281  //Otherwise there is just one master node, the node itself
282  else
283  {
284  n_master = 1;
285  }
286 
287  //Loop over the master nodes
288  for(unsigned m=0;m<n_master;m++)
289  {
290  //If the node is hanging get weight from master node
291  if(is_node_hanging)
292  {
293  //Get the hang weight from the master node
294  hang_weight = hang_info_pt->master_weight(m);
295  }
296  else
297  {
298  // Node contributes with full weight
299  hang_weight = 1.0;
300  }
301 
302  //Get the equation number
303  if(is_node_hanging)
304  {
305  //Get the equation number from the master node
306  global_eqn = hang_info_pt->master_node_pt(m)->
307  eqn_number(u_nodal_index);
308  }
309  else
310  {
311  // Global equation number
312  global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
313  }
314 
315  if (global_eqn >= 0)
316  {
317  //Set the global equation number
318  global_eqn_number[count] = global_eqn;
319  //Set the derivative with respect to the unknown
320  du_ddata[count] = psi[l]*hang_weight;
321  //Increase the counter
322  ++count;
323  }
324  }
325  }
326  }
