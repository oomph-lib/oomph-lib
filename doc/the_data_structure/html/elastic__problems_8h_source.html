70 /// \short IC problem for an elastic body discretised on a given (sub)-mesh.
71 /// We switch the elements' residuals and Jacobians to the system of
72 /// equations that forces the wall shape to become that of
73 /// a specified "initial condition object".
74 ///
75 ///
76 /// Boundary conditions for all data items associated with the mesh's
77 /// elements' are temporarily overwritten so that all positional dofs (and
78 /// only those!) become free while all other data (nodal data and the element's
79 /// internal and external data) are either pinned (for nodal and internal
80 /// data) or completely removed (for pointers to external data).
81 /// The complete removal of the (pointers to the) external data is necessary
82 /// because in FSI problems, positional data of certain elements
83 /// can feature as external data of others. Hence pinning them
84 /// in their incarnation as external data would also pin them
85 /// in their incarnation as positional data.
86 ///
87 ///
88 /// All data and its pinned-status are restored at the end of the
89 /// call to setup_ic().
90 //======================================================================

248  timestepper_pt->time_pt()->dt()=dt;
249 
250  // Set the weights
251  timestepper_pt->set_weights();
252 
253  // Delete dummy mesh
254  delete mesh_pt();
255 
256  // Set pointer to mesh
257  mesh_pt()=wall_mesh_pt;
258 
259  // Set pointer to initial condition object
260  IC_pt=ic_pt;
261 
262  // Backup the pinned status of all dofs and remove external data
263  // of all elements
264  backup_original_state();

267  // positional variables can be solved; setup equation numbering
268  // scheme
269  setup_problem();
270 
271 
272  // Store times at which we need to assign ic:
273  double current_time=timestepper_pt->time_pt()->time();
274  double previous_time=timestepper_pt->time_pt()->time(1);
275 
276  // Stage 1: Set values and time derivs at current time
277  //----------------------------------------------------
278 
279  // [Note: this acts on time everywhere!]
280  IC_pt->geom_object_pt()->time_stepper_pt()->time_pt()->time()=
281  current_time;
282 
283  // Loop over time-derivatives
284  for (unsigned t_deriv=0;t_deriv<=2;t_deriv++)
285  {
286 
287  // Set flag to ensure that the t_deriv-th time derivative
288  // of the prescribed solution gets stored in displacements
289  IC_pt->ic_time_deriv()=t_deriv;
290 
291  // Solve the problem for initial shape
292  newton_solve();
293 
294  //Loop over all the nodes
295  unsigned n_node=mesh_pt()->nnode();
296  for(unsigned n=0;n<n_node;n++)
297  {
298  // Assign current time derivative in its temporary storage
299  // position
300  timestepper_pt->assign_initial_data_values_stage1(
301  t_deriv,dynamic_cast<SolidNode*>(
302  mesh_pt()->node_pt(n))->variable_position_pt());
303  }
304  }
305 
306 
307 
308  // Stage 2: Now get position at previous time and adjust previous
309  //---------------------------------------------------------------
310  // values of veloc and accel in Newmark scheme so that current
311  //---------------------------------------------------------------
312  // veloc and accel (specified in step 1) are represented exactly.
313  //---------------------------------------------------------------
314 
315  // [Note: this acts on time everywhere and needs to be reset!]
316  IC_pt->geom_object_pt()->time_stepper_pt()->time_pt()->time()=
317  previous_time;
318 
319  // Set flag to ensure that the t_deriv-th time derivative
320  // of the prescribed solution gets stored in displacements
321  IC_pt->ic_time_deriv()=0;
322 
323  // Solve the problem for initial shape
324  newton_solve();
325 
326  //Loop over all the nodes and make the final adjustments
327  unsigned n_node=mesh_pt()->nnode();
328  for(unsigned n=0;n<n_node;n++)
329  {
330  timestepper_pt->assign_initial_data_values_stage2(
331  dynamic_cast<SolidNode*>(mesh_pt()->node_pt(n))
332  ->variable_position_pt());
333  }
334 
335  // Reset time
336  IC_pt->geom_object_pt()->time_stepper_pt()->time_pt()->time()=
337  current_time;
338 
339  // Reset the pinned status and re-attach the external data to the elements
340  reset_original_state();
341 
342  // Set pointer to dummy mesh so there's something that can be deleted
343  // when static problem finally goes out of scope.
344  mesh_pt()=new DummyMesh;
345 
346  // We have temporarily over-written equation numbers -- need
347  // to reset them now
348  oomph_info << "Number of equations in big problem: "
349  << problem_pt->assign_eqn_numbers() << std::endl;
350
