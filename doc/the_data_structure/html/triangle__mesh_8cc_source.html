108 #ifdef OOMPH_HAS_MPI

415 /// Setup lookup schemes which establish which elements are located
416 /// next to which boundaries (Doc to outfile if it's open).
417 //================================================================
418  void TriangleMeshBase::setup_boundary_element_info(std::ostream
419  &outfile)
420  {
421  // Should we document the output here
422  bool doc = false;
423 
424  if(outfile) doc = true;
425 
426  // Number of boundaries
427  unsigned nbound=nboundary();
428 
429  // Wipe/allocate storage for arrays
430  Boundary_element_pt.clear();
431  Face_index_at_boundary.clear();
432  Boundary_element_pt.resize(nbound);
433  Face_index_at_boundary.resize(nbound);
434 
435  // Temporary vector of vectors of pointers to elements on the boundaries:
436  // This is a vector to ensure that order is strictly preserved
437  Vector<Vector<FiniteElement*> > vector_of_boundary_element_pt;
438  vector_of_boundary_element_pt.resize(nbound);
439 
440  // Matrix map for working out the fixed face for elements on boundary
441  MapMatrixMixed<unsigned,FiniteElement*, int >
442  face_identifier;
443 
444  // Loop over elements
445  //-------------------
446  unsigned nel=nelement();
447 
448  // Get pointer to vector of boundaries that the
449  // node lives on
450  Vector<std::set<unsigned>*> boundaries_pt(3,0);
451 
452  // Data needed to deal with edges through the
453  // interior of the domain
454  std::map<Edge,unsigned > edge_count;
455  std::map<Edge,TriangleBoundaryHelper::BCInfo>
456  edge_bcinfo;
457  std::map<Edge,TriangleBoundaryHelper::BCInfo>
458  face_info;
459  MapMatrixMixed<unsigned,FiniteElement*, int > face_count;
460  Vector<unsigned> bonus(nbound);
461 
462  // When using internal boundaries, an edge can be related to more than
463  // one element (because of both sides of the internal boundaries)
464  std::map<Edge,Vector<TriangleBoundaryHelper::BCInfo> > edge_internal_bnd;
465 
466  for (unsigned e=0;e<nel;e++)
467  {
468  // Get pointer to element
469  FiniteElement* fe_pt=finite_element_pt(e);
470 
471  if (doc) {outfile << "Element: " << e << " " << fe_pt << std::endl;}
472 
473  // Only include 2D elements! Some meshes contain interface elements too.
474  if (fe_pt->dim()==2)
475  {
476  // Loop over the element's nodes and find out which boundaries they're on
477  // ----------------------------------------------------------------------
478 
479  //We need only loop over the corner nodes
480  for(unsigned i=0;i<3;i++)
481  {
482  fe_pt->node_pt(i)->get_boundaries_pt(boundaries_pt[i]);
483  }
484 
485  //Find the common boundaries of each edge
486  Vector<std::set<unsigned> > edge_boundary(3);
487 
488  //Edge 0 connects points 1 and 2
489  //-----------------------------
490 
491  if(boundaries_pt[1] && boundaries_pt[2])
492  {
493  // Create the corresponding edge
494  Edge edge0(fe_pt->node_pt(1),fe_pt->node_pt(2));
495 
496  // Update infos about this edge
497  TriangleBoundaryHelper::BCInfo info;
498  info.Face_id=0;
499  info.FE_pt = fe_pt;
500 
501  std::set_intersection(boundaries_pt[1]->begin(),boundaries_pt[1]->end(),
502  boundaries_pt[2]->begin(),boundaries_pt[2]->end(),
503  std::insert_iterator<std::set<unsigned> >(
504  edge_boundary[0],edge_boundary[0].begin()));
505  std::set<unsigned>::iterator it0=edge_boundary[0].begin();
506 
507  // Edge does exist:
508  if ( edge_boundary[0].size() > 0 )
509  {
510  info.Boundary=*it0;
511 
512  // How many times this edge has been visited
513  edge_count[edge0]++;
514 
515  // Update edge_bcinfo
516  edge_bcinfo.insert(std::make_pair(edge0,info));
517 
518  // ... and also update the info associated with internal bnd
519  edge_internal_bnd[edge0].push_back(info);
520  }
521  }
522 
523  //Edge 1 connects points 0 and 2
524  //-----------------------------
525 
526  if(boundaries_pt[0] && boundaries_pt[2])
527  {
528  std::set_intersection(boundaries_pt[0]->begin(),boundaries_pt[0]->end(),
529  boundaries_pt[2]->begin(),boundaries_pt[2]->end(),
530  std::insert_iterator<std::set<unsigned> >(
531  edge_boundary[1],edge_boundary[1].begin()));
532 
533  // Create the corresponding edge
534  Edge edge1(fe_pt->node_pt(0),fe_pt->node_pt(2));
535 
536  // Update infos about this edge
537  TriangleBoundaryHelper::BCInfo info;
538  info.Face_id=1;
539  info.FE_pt = fe_pt;
540  std::set<unsigned>::iterator it1=edge_boundary[1].begin();
541 
542  // Edge does exist:
543  if ( edge_boundary[1].size() > 0)
544  {
545  info.Boundary=*it1;
546 
547  // How many times this edge has been visited
548  edge_count[edge1]++;
549 
550  // Update edge_bcinfo
551  edge_bcinfo.insert(std::make_pair(edge1,info));
552 
553  // ... and also update the info associated with internal bnd
554  edge_internal_bnd[edge1].push_back(info);
555  }
556  }
557 
558  //Edge 2 connects points 0 and 1
559  //-----------------------------
560 
561  if(boundaries_pt[0] && boundaries_pt[1])
562  {
563  std::set_intersection(boundaries_pt[0]->begin(),boundaries_pt[0]->end(),
564  boundaries_pt[1]->begin(),boundaries_pt[1]->end(),
565  std::insert_iterator<std::set<unsigned> >(
566  edge_boundary[2],edge_boundary[2].begin()));
567 
568  // Create the corresponding edge
569  Edge edge2(fe_pt->node_pt(0),fe_pt->node_pt(1));
570 
571  // Update infos about this edge
572  TriangleBoundaryHelper::BCInfo info;
573  info.Face_id=2;
574  info.FE_pt = fe_pt;
575  std::set<unsigned>::iterator it2=edge_boundary[2].begin();
576 
577  // Edge does exist:
578  if ( edge_boundary[2].size() > 0)
579  {
580  info.Boundary=*it2;
581 
582  // How many times this edge has been visited
583  edge_count[edge2]++;
584 
585  // Update edge_bcinfo
586  edge_bcinfo.insert(std::make_pair(edge2,info));
587 
588  // ... and also update the info associated with internal bnd
589  edge_internal_bnd[edge2].push_back(info);
590  }
591  }
592 
593 
594 #ifdef PARANOID
595 
596  // Check if edge is associated with multiple boundaries
597 
598  //We now know whether any edges lay on the boundaries
599  for(unsigned i=0;i<3;i++)
600  {
601  //How many boundaries are there
602  unsigned count = 0;
603 
604  //Loop over all the members of the set and add to the count
605  //and set the boundary
606  for(std::set<unsigned>::iterator it=edge_boundary[i].begin();
607  it!=edge_boundary[i].end();++it)
608  {
609  ++count;
610  }
611 
612  //If we're on more than one boundary, this is weird, so die
613  if(count > 1)
614  {
615  std::ostringstream error_stream;
616  error_stream << "Edge " << i << " is located on " <<
617  count << " boundaries.\n";
618  error_stream << "This is rather strange, so I'm going to die\n";
619  throw OomphLibError(
620  error_stream.str(),
621  OOMPH_CURRENT_FUNCTION,
622  OOMPH_EXCEPTION_LOCATION);
623  }
624  }
625 
626 #endif
627 
628  // Now we set the pointers to the boundary sets to zero
629  for(unsigned i=0;i<3;i++) {boundaries_pt[i] = 0;}
630 
631  }
632  } //end of loop over all elements
633 
634  // Loop over all edges that are located on a boundary
635  typedef std::map<Edge,TriangleBoundaryHelper::BCInfo>::iterator ITE;
636  for (ITE it=edge_bcinfo.begin();
637  it!=edge_bcinfo.end();
638  it++)
639  {
640  Edge current_edge = it->first;
641  unsigned bound=it->second.Boundary;
642 
643  // If the edge has been visited only once
644  if(edge_count[current_edge]==1)
645  {
646  // Count the edges that are on the same element and on the same boundary
647  face_count(static_cast<unsigned>(bound),it->second.FE_pt)=
648  face_count(static_cast<unsigned>(bound),it->second.FE_pt) + 1;
649 
650  //If such edges exist, let store the corresponding element
651  if( face_count(bound,it->second.FE_pt) > 1)
652  {
653  // Update edge's infos
654  TriangleBoundaryHelper::BCInfo info;
655  info.Face_id=it->second.Face_id;
656  info.FE_pt = it->second.FE_pt;
657  info.Boundary=it->second.Boundary;
658 
659  // Add it to FIinfo, that stores infos of problematic elements
660  face_info.insert(std::make_pair(current_edge,info));
661 
662  //How many edges on which boundary have to be added
663  bonus[bound]++;
664  }
665  else
666  {
667  //Add element and face to the appropriate vectors
668  // Does the pointer already exits in the vector
669  Vector<FiniteElement*>::iterator b_el_it =
670  std::find(vector_of_boundary_element_pt[
671  static_cast<unsigned>(bound)].begin(),
672  vector_of_boundary_element_pt[
673  static_cast<unsigned>(bound)].end(),
674  it->second.FE_pt);
675 
676  //Only insert if we have not found it (i.e. got to the end)
677  if(b_el_it == vector_of_boundary_element_pt[
678  static_cast<unsigned>(bound)].end())
679  {
680  vector_of_boundary_element_pt[static_cast<unsigned>(bound)].
681  push_back(it->second.FE_pt);
682  }
683 
684  //set_of_boundary_element_pt[static_cast<unsigned>(bound)].insert(
