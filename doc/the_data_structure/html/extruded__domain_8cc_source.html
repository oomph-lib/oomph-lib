137  //=================================================================
138  /// \short Vector representation of the i_macro-th macro element
139  /// boundary i_direct (e.g. N/S/W/E in 2D spatial = 3D space-time).
140  /// NOTE: Some extra care has to be taken here to translate the
141  /// OcTree enumeration to the QuadTree enumeration (in the
142  /// appropriate manner) so that the original Domain object can be
143  /// used to calculate the global coordinate associated with the
144  /// provided local coordinates.
145  //=================================================================
146  void ExtrudedDomain::macro_element_boundary(const unsigned& time,
147  const unsigned& i_macro,
148  const unsigned& i_direct,
149  const Vector<double>& s,
150  Vector<double>& x)
151  {
152  // Make sure that time=0 otherwise this doesn't make sense
153  if (time!=0)
154  {
155  // Create an output stream
156  std::ostringstream error_message_stream;
157 
158  // Create an error message
159  error_message_stream << "This output function outputs a space-time\n"
160  << "representation of the domain. As such, it\n"
161  << "does not make sense to output the domain\n"
162  << "at a previous time level!" << std::endl;
163 
164  // Throw an error
165  throw OomphLibError(error_message_stream.str(),
166  OOMPH_CURRENT_FUNCTION,
167  OOMPH_EXCEPTION_LOCATION);
168  }
169 
170  // The number of dimensions
171  unsigned n_dim=3;
172 
173  // The number of macro elements in the original spatial Domain
174  unsigned n_macro=Domain_pt->nmacro_element();
175 
176  // Calculate the shifted macro element number (to correspond to the
177  // correct macro element in the original spatial Domain)
178  unsigned true_macro_id=i_macro%n_macro;
179 
180  // Which layer of the extruded domain are we in? Layer 0 corresponds to
181  // elements closest to t=T_min and the layer N_extruded_element-1
182  // corresponds to elements closest to t=T_max
183  unsigned i_layer=(i_macro-true_macro_id)/n_macro;
184 
185  // Calculate the width of an extrusion layer
186  double layer_width=(T_max-T_min)/double(N_extruded_element);
187 
188  // Storage for the global (spatial) coordinates
189  Vector<double> x_project(n_dim-1,0.0);
190 
191  // Calculate the time value associated with the start of the i-layer-th
192  // extrusion layer
193  double t_lower=(T_min+i_layer*layer_width);
194 
195  // Initialise the time value
196  double t=t_lower;
197 
198  // If we're on one of the edges (once projected to 2D in the t-direction)
199  if ((i_direct==OcTreeNames::L)||(i_direct==OcTreeNames::R)||
200  (i_direct==OcTreeNames::D)||(i_direct==OcTreeNames::U))
201  {
202  // Update the time value to get the time value associated
203  // with the input surface coordinates
204  t+=0.5*(1.0+s[1])*layer_width;
205 
206  // Get the local coordinate associated with the surface coordinate
207  // in the projected Domain. This is the first surface coordinate
208  // on the left, right, down and up faces of the 3D macro element.
209  Vector<double> s_project(1,s[0]);
210 
211  // If we're on the left face (or the West face once projected to 2D)
212  if (i_direct==OcTreeNames::L)
213  {
214  // Call the corresponding function through the provided Domain object
215  Domain_pt->macro_element_boundary(t,true_macro_id,QuadTreeNames::W,
216  s_project,x_project);
217  }
218  // If we're on the right face (or the East face once projected to 2D)
219  else if (i_direct==OcTreeNames::R)
220  {
221  // Call the corresponding function through the provided Domain object
222  Domain_pt->macro_element_boundary(t,true_macro_id,QuadTreeNames::E,
223  s_project,x_project);
224  }
225  // If we're on the down face (or the South face once projected to 2D)
226  else if (i_direct==OcTreeNames::D)
227  {
228  // Call the corresponding function through the provided Domain object
229  Domain_pt->macro_element_boundary(t,true_macro_id,QuadTreeNames::S,
230  s_project,x_project);
231  }
232  // If we're on the up face (or the North face once projected to 2D)
233  else if (i_direct==OcTreeNames::U)
234  {
235  // Call the corresponding function through the provided Domain object
236  Domain_pt->macro_element_boundary(t,true_macro_id,QuadTreeNames::N,
237  s_project,x_project);
238  }
239  }
240  else if ((i_direct==OcTreeNames::B)||(i_direct==OcTreeNames::F))
241  {
242  // Grab the i_macro-th macro element from the Domain object
243  MacroElement* macro_element_pt=Domain_pt->macro_element_pt(true_macro_id);
244 
245  // If we're on the back face (or the whole element at t=T_max)
246  if (i_direct==OcTreeNames::B)
247  {
248  // Call the macro_map function to calculate the global (spatial)
249  // coordinates associated with the given surface coordinates
250  macro_element_pt->macro_map(t,s,x_project);
251  }
252  // If we're on the front face (or the whole element at t=T_max)
253  else if (i_direct==OcTreeNames::F)
254  {
255  // Update the time value to get the time value associated
256  // with the input surface coordinates
257  t+=layer_width;
258 
259  // Call the macro_map function to calculate the global (spatial)
260  // coordinates associated with the given surface coordinates
261  macro_element_pt->macro_map(t,s,x_project);
262  }
263  }
264  else
265  {
266  // Create an output stream
267  std::ostringstream error_message_stream;
268 
269  // Create an error message
270  error_message_stream << "Incorrect face enumeration input! Should either "
271  << "be L,R,D,U,B or F. You input "
272  << OcTree::Direct_string[i_direct]
273  << "!" << std::endl;
274 
275  // We should never get here so throw an error if we do
276  throw OomphLibError(error_message_stream.str(),
277  OOMPH_CURRENT_FUNCTION,
278  OOMPH_EXCEPTION_LOCATION);
279  } // if ((i_direct==OcTreeNames::L)||(i_direct==OcTreeNames::R)||...
280 
281  // Loop over the global (spatial) coordinates
282  for (unsigned i=0; i<n_dim-1; i++)
283  {
284  // Copy the global coordinates over
285  x[i]=x_project[i];
286  }
287 
288  // Finally, add in the final coordinate (the time value)
289  x[n_dim-1]=t;
290  } // End of macro_element_boundary
291 } // End of namespace oomph
