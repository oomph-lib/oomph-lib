197  /// \short General purpose block triangular preconditioner. By default this
198  /// operates as an upper triangular preconditioner. Also, by default
199  /// SuperLUPreconditioner (or SuperLUDistPreconditioner) is used to solve the
200  /// subsidiary systems, but other preconditioners can be used by setting them
201  /// using passing a pointer to a function of type SubsidiaryPreconditionerFctPt
202  /// to the method subsidiary_preconditioner_function_pt().
203  //=============================================================================
204  template<typename MATRIX>
205  class BandedBlockTriangularPreconditioner :
206  public GeneralPurposeBlockPreconditioner<MATRIX>
207  {
208 
209  public :
210 
211  /// Constructor. (By default this preconditioner is upper triangular).
212  BandedBlockTriangularPreconditioner() :
213  GeneralPurposeBlockPreconditioner<MATRIX>()
214  {
215  // Default to upper triangular
216  Upper_triangular=true;
217 
218  // Default is to assume every block above the diagonal is non-empty
219  Block_bandwidth=-1;
220 
221  // The preconditioner has been created but it hasn't been set up yet
222  Preconditioner_has_been_setup=false;
223 
224  // By default, don't store the memory statistics of this preconditioner
225  Compute_memory_statistics=false;
226 
227  // Initialise the value of Memory_usage_in_bytes
228  Memory_usage_in_bytes=0.0;
229  } // End of BandedBlockTriangularPreconditioner

237  } // End of ~BandedBlockTriangularPreconditioner

239 
240  /// Clean up the memory
241  virtual void clean_up_memory()
242  {
243  // Delete anything in Off_diagonal_matrix_vector_products
244  for (unsigned i=0,ni=Off_diagonal_matrix_vector_products.nrow(); i<ni; i++)
245  {
246  for (unsigned j=0,nj=Off_diagonal_matrix_vector_products.ncol(); j<nj; j++)
247  {
248  // Delete the matrix-vector product
249  delete Off_diagonal_matrix_vector_products(i,j);
250 
251  // Make it a null pointer
252  Off_diagonal_matrix_vector_products(i,j)=0;
253  }
254  } // for (unsigned i=0,ni=Off_diagonal_matrix_vector_products.nrow();...
255 
256  // Clean up the base class too
257  GeneralPurposeBlockPreconditioner<MATRIX>::clean_up_memory();
258  } // End of clean_up_memory
