34 /// Compute element residual vector taking hanging nodes into account
35 //======================================================================
36 void RefineableYoungLaplaceEquations::
37 fill_in_contribution_to_residuals(Vector<double> &residuals)
38 {
39 
40  //Find out how many nodes there are
41  unsigned n_node = nnode();
42 
43  //Set up memory for the shape (test) functions
44  Shape psi(n_node);
45  DShape dpsidzeta(n_node,2);
46 
47  //Set the value of n_intpt
48  unsigned n_intpt = integral_pt()->nweight();
49 
50  //Integers to store the local equation numbers
51  int local_eqn=0;
52 
53  //Loop over the integration points
54  for(unsigned ipt=0;ipt<n_intpt;ipt++)
55  {
56  //Get the integral weight
57  double w = integral_pt()->weight(ipt);
58 
59  //Call the derivatives of the shape (test) functions
60  double J = dshape_eulerian_at_knot(ipt,psi,dpsidzeta);
61 
62  //Premultiply the weights and the Jacobian
63  double W = w*J;
64 
65  //Calculate local values of the pressure and velocity components
66  //Allocate and initialise to zero
67  double interpolated_u=0.0;
68  Vector<double> interpolated_zeta(2,0.0);
69  Vector<double> interpolated_du_dzeta(2,0.0);
70 
71  //Calculate function value and derivatives:
72  //-----------------------------------------
73  // Loop over nodes
74  for(unsigned l=0;l<n_node;l++)
75  {
76  interpolated_u += u(l)*psi(l);
77  // Loop over directions
78  for(unsigned j=0;j<2;j++)
79  {
80  interpolated_zeta[j] += nodal_position(l,j)*psi(l);
81  interpolated_du_dzeta[j] += u(l)*dpsidzeta(l,j);
82  }
83  }
84 
85  // Allocation and definition of variables necessary for
86  // further calculations
87 
88  /// "Simple" case
89  ///--------------
90  double nonlinearterm=1.0;
91  double sqnorm=0.0;
92 
93  /// Spine case
94  ///-----------
95 
96  // Derivs of position vector w.r.t. global intrinsic coords
97  Vector<Vector<double> > dRdzeta;
98  allocate_vector_of_vectors(2,3,dRdzeta);
99 
100  // Unnormalised normal
101  Vector<double> N_unnormalised(3,0.0);
102 
103  // Spine and spine basis vectors, entries initialised to zero
104  Vector<double> spine_base(3,0.0), spine(3,0.0);
105 
106  // Derivative of spine basis vector w.r.t to the intrinsic
107  // coordinates: dspine_base[i,j] = j-th component of the deriv.
108  // of the spine basis vector w.r.t. to the i-th global intrinsic
109  // coordinate
110  Vector< Vector<double> > dspine_base;
111  allocate_vector_of_vectors(2,3,dspine_base);
112 
113  // Derivative of spine vector w.r.t to the intrinsic
114  // coordinates: dspine[i,j] = j-th component of the deriv.
115  // of the spine vector w.r.t. to the i-th global intrinsic
116  // coordinate
117  Vector< Vector<double> > dspine;
118  allocate_vector_of_vectors(2,3,dspine);
119 
120  // Vector v_\alpha contains the numerator of the variations of the
121  // area element {\cal A}^{1/2} w.r.t. the components of dR/d\zeta_\alpha
122  Vector<double> area_variation_numerator_0(3,0.0);
123  Vector<double> area_variation_numerator_1(3,0.0);
124 
125  // Vector position
126  Vector<double> r(3,0.0);
127 
128  //No spines
129  //---------
130  if (!use_spines())
131  {
132 
133  for (unsigned j=0; j<2; j++)
134  sqnorm += interpolated_du_dzeta[j]*interpolated_du_dzeta[j];
135 
136  nonlinearterm=1.0/sqrt(1.0+sqnorm);
137  }
138 
139  //Spines
140  //------
141  else
142  {
143  // Get the spines
144  get_spine_base(interpolated_zeta, spine_base, dspine_base);
145  get_spine(interpolated_zeta, spine, dspine);
146 
147  // calculation of dR/d\zeta_\alpha
148  for (unsigned alpha=0;alpha<2;alpha++)
149  {
150  // Product rule for d(u {\bf S} ) / d \zeta_\alpha
151  Vector<double> dudzeta_times_spine(3,0.0);
152  scalar_times_vector(interpolated_du_dzeta[alpha],
153  spine,dudzeta_times_spine);
154 
155  Vector<double> u_times_dspinedzeta(3,0.0);
156  scalar_times_vector(interpolated_u,dspine[alpha],u_times_dspinedzeta);
157 
158  Vector<double> d_u_times_spine_dzeta(3,0.0);
159  vector_sum(dudzeta_times_spine,
160  u_times_dspinedzeta,
161  d_u_times_spine_dzeta);
162 
163  // Add derivative of spine base
164  vector_sum(d_u_times_spine_dzeta,dspine_base[alpha],dRdzeta[alpha]);
165  }
166 
167  /// Get the unnormalized normal
168  cross_product(dRdzeta[0],dRdzeta[1],N_unnormalised);
169 
170  /// Tmp storage
171  Vector<double> v_tmp_1(3,0.0);
172  Vector<double> v_tmp_2(3,0.0);
173 
174  // Calculation of
175  // |dR/d\zeta_1|^2 dR/d\zeta_0 - <dR/d\zeta_0,dR/d\zeta_1>dR/d\zeta_1
176  scalar_times_vector(pow(two_norm(dRdzeta[1]),2), dRdzeta[0], v_tmp_1);
177  scalar_times_vector(-1*scalar_product(dRdzeta[0],dRdzeta[1]),
178  dRdzeta[1], v_tmp_2);
179  vector_sum(v_tmp_1,v_tmp_2,area_variation_numerator_0);
180 
181  // Calculation of
182  // |dR/d\zeta_0|^2 dR/d\zeta_1 - <dR/d\zeta_0,dR/d\zeta_1>dR/d\zeta_0
183  scalar_times_vector(pow(two_norm(dRdzeta[0]),2), dRdzeta[1], v_tmp_1);
184  scalar_times_vector(-1*scalar_product(dRdzeta[0],dRdzeta[1]),
185  dRdzeta[0], v_tmp_2);
186  vector_sum(v_tmp_1,v_tmp_2,area_variation_numerator_1);
187 
188  // Global Eulerian cooordinates
189  for (unsigned j=0; j<3; j++)
190  {
191  r[j]=spine_base[j]+interpolated_u*spine[j];
192  }
193 
194  }
195 
196 
197 
198  // Assemble residuals
199  //-------------------
200 
201  // Loop over the nodes for the test functions
202  for(unsigned l=0;l<n_node;l++)
203  {
204  //Local variables used to store the number of master nodes and the
205  //weight associated with the shape function if the node is hanging
206  unsigned n_master=1; double hang_weight=1.0;
207 
208  //If the node is hanging, get the number of master nodes
209  if(node_pt(l)->is_hanging())
210  {
211  n_master = node_pt(l)->hanging_pt()->nmaster();
212  }
213  //Otherwise there is just one master node, the node itself
214  else
215  {
216  n_master = 1;
217  }
218 
219  //Loop over the master nodes
220  for(unsigned m=0;m<n_master;m++)
221  {
222  //Get the local equation number and hang_weight
223  //If the node is hanging
224  if(node_pt(l)->is_hanging())
225  {
226  //Read out the local equation from the master node
227  local_eqn =
228  local_hang_eqn(node_pt(l)->hanging_pt()->master_node_pt(m),0);
229  //Read out the weight from the master node
230  hang_weight = node_pt(l)->hanging_pt()->master_weight(m);
231  }
232  //If the node is not hanging
233  else
234  {
235  //The local equation number comes from the node itself
236  local_eqn = this->u_local_eqn(l);
237  //The hang weight is one
238  hang_weight = 1.0;
239  }
240 
241  /*IF it's not a boundary condition*/
242  if(local_eqn >= 0)
243  {
244 
245  // "simple" calculation case
246  if (!use_spines())
247  {
248  // Add source term: The curvature
249  residuals[local_eqn] += get_kappa()*psi(l)*W*hang_weight;
250 
251  // The YoungLaplace bit itself
252  for(unsigned k=0;k<2;k++)
253  {
254  residuals[local_eqn] += nonlinearterm*
255  interpolated_du_dzeta[k]*dpsidzeta(l,k)*W*hang_weight;
256  }
257  }
258 
259  // Spine calculation case
260  else
261  {
262 
263  // Calculation of d(u S)/d\zeta_0
264  //-------------------------------
265  Vector<double> v_tmp_1(3,0.0);
266  scalar_times_vector(dpsidzeta(l,0), spine, v_tmp_1);
267 
268  Vector<double> v_tmp_2(3,0.0);
269  scalar_times_vector(psi(l),dspine[0], v_tmp_2);
270 
271  Vector<double> d_uS_dzeta0(3,0.0);
272  vector_sum(v_tmp_1,v_tmp_2,d_uS_dzeta0);
273 
274  // Add contribution to residual
275  residuals[local_eqn] +=
276  W*hang_weight*
277  scalar_product(area_variation_numerator_0,d_uS_dzeta0)
278  /two_norm(N_unnormalised);
279 
280  // Calculation of d(u S)/d\zeta_1
281  scalar_times_vector(dpsidzeta(l,1), spine, v_tmp_1);
282  scalar_times_vector(psi(l),dspine[1], v_tmp_2);
283  Vector<double> d_uS_dzeta1(3,0.0);
284  vector_sum(v_tmp_1,v_tmp_2,d_uS_dzeta1);
285 
286  // Add contribution to residual
287  residuals[local_eqn] +=
288  W*hang_weight*
289  scalar_product(area_variation_numerator_1,d_uS_dzeta1)
290  /two_norm(N_unnormalised);
291 
292  // Curvature contribution to the residual : kappa N S test
293  residuals[local_eqn] += W*hang_weight*(get_kappa())*
294  scalar_product(N_unnormalised,spine)*psi(l);
295  }
296  }
297 
298  } //End of loop over master nodes for residual
299 
300  } //End of loop over nodes
301 
302  } // End of loop over integration points
303 
304 }
