65  /// \short Function to set up a preconditioner for the linear
66  /// system defined by matrix_pt. This function must be called
67  /// before using preconditioner_solve.
68  /// Note: matrix_pt must point to an object of class
69  /// CRDoubleMatrix or CCDoubleMatrix
70  void setup()
71  {
72  oomph_info << "Setting up SuperLU (exact) preconditioner" << std::endl;
73  if (dynamic_cast<DistributableLinearAlgebraObject*>(matrix_pt()) != 0)
74  {
75  LinearAlgebraDistribution dist(
76  (dynamic_cast<DistributableLinearAlgebraObject*>(matrix_pt()))
77  ->distribution_pt());
78  this->build_distribution(dist);
79  Solver.factorise(matrix_pt());
80  }
81  else
82  {
83  std::ostringstream error_message_stream;
84  error_message_stream
85  << "SuperLUPreconditioner can only be applied to matrices derived \n"
86  << "DistributableLinearAlgebraObject.\n"
87  << "You are most likely to be here because you are using the\n "
88  << "soon to be obsolete CCDoubleMatrix\n";
89  throw OomphLibError(error_message_stream.str(),
90  OOMPH_CURRENT_FUNCTION,
91  OOMPH_EXCEPTION_LOCATION);
92  }
93  }
94 
95  /// \short Function applies SuperLU to vector r for (exact) preconditioning,
96  /// this requires a call to setup(...) first.
97  void preconditioner_solve(const DoubleVector& r, DoubleVector& z)
98  {
99  Solver.resolve(r, z);
100  }

102  /// \short Function applies SuperLU to vector r for (exact) preconditioning

104  /// first.
105  void preconditioner_solve_transpose(const DoubleVector& r, DoubleVector& z)
106  {
107  Solver.resolve_transpose(r, z);

136  /// \short Get the amount of memory taken up by SuperLU. The first entry
137  /// of the returned result contains the memory used to store the LU
138  /// factors and the second entry contains the total memory used to
139  /// store AND calculate the LU factors
140  Vector<double> get_memory_usage_for_superlu()
141  {
142  // Allocate storage for the memory statistics
143  Vector<double> memory_usage(2, 0.0);
144 
145  // The first entry contains the memory used to store the LU factors
146  memory_usage[0] = Solver.get_memory_usage_for_lu_factors();
147 
148  // The second entry contains the total memory used to both calculate
149  // and store the LU factors
150  memory_usage[1] = Solver.get_total_needed_memory();
151 
152  // Now return the calculated result
153  return memory_usage;
154  } // End of get_memory_usage_for_superlu
