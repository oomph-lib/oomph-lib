47 /// \short Pseudo buckling ring: Circular ring deformed by the
48 /// N-th buckling mode of a thin-wall elastic ring.
49 /// \f[
50 /// x = R_0 \cos(\zeta) +
51 /// \epsilon \left( \cos(N \zeta) \cos(\zeta) - A \sin(N \zeta) \sin(\zeta)
52 /// \right) sin(2 \pi t/T)
53 /// \f]
54 /// \f[
55 /// y = R_0 \sin(\zeta) +
56 /// \epsilon \left( \cos(N \zeta) \sin(\zeta) + A \sin(N \zeta) \cos(\zeta)
57 /// \right) sin(2 \pi t/T)
58 /// \f]
59 /// where A is the ratio of the aziumuthal to the radial buckling
60 /// amplitude (A=-1/N for statically buckling rings) and epsilon
61 /// is the buckling amplitude.
62 ///

173  /// \short Constructor: 1 Lagrangian coordinate, 2 Eulerian coords. Pass
174  /// buckling amplitude, h/R, buckling wavenumbe and pointer
175  /// to global timestepper. Other parameters get set up to represent
176  /// oscillating ring with mode imode (1 or 2). All geometric data is
177  /// pinned by default.
178  PseudoBucklingRing(const double& eps_buckl,
179  const double& HoR,
180  const unsigned& n_buckl,
181  const unsigned& imode,
182  TimeStepper* time_stepper_pt) :
183  GeomObject(1,2,time_stepper_pt)
184  {
185  // Constants in Soedel solution:
186  double K1=(pow(double(n_buckl),2)+1.0)*
187  (1.0/12.0*pow(double(n_buckl),2)*pow(HoR,2)+1.0);
188 
189  double K2oK1sq=1.0/12.0*pow(HoR,2)*pow(double(n_buckl),2)*
190  pow( (pow(double(n_buckl),2)-1.0) ,2)/
191  ( pow((pow(double(n_buckl),2)+1.0),2) *
192  pow((1.0/12.0*pow(double(n_buckl),2)*pow(HoR,2)+1.0),2) );
193 
194  // The two fundamental frequencies:
195  double omega1=sqrt(0.5*K1*(1.0+sqrt(1.0-4*K2oK1sq)));
196  double omega2=sqrt(0.5*K1*(1.0-sqrt(1.0-4*K2oK1sq)));
197 
198  // The two amplitude ratios:
199  double ampl_ratio1=
200  (double(n_buckl)*(1.0/12.0*pow(double(n_buckl),2)*pow(HoR,2)+1.0))/
201  (pow(omega1,2)-pow(double(n_buckl),2)*(1.0/12.0*pow(HoR,2)+1.0));
202 
203  double ampl_ratio2=
204  double(n_buckl)*(1.0/12.0*pow(double(n_buckl),2)*pow(HoR,2)+1.0)/
205  (pow(omega2,2)-pow(double(n_buckl),2)*(1.0/12.0*pow(HoR,2)+1.0));
206 
207  double T;
208  double ampl_ratio;
209 
210  if (n_buckl>1)
211  {
212  T=2.0*MathematicalConstants::Pi/omega2;
213  ampl_ratio=ampl_ratio2;
214  if (imode==1)
215  {
216  T= 2.0*MathematicalConstants::Pi/omega1;
217  ampl_ratio=ampl_ratio1;
218  }
219  else if (imode==2)
220  {
221  T= 2.0*MathematicalConstants::Pi/omega2;
222  ampl_ratio=ampl_ratio2;
223  }
224  else
225  {
226  oomph_info << "wrong imode " << imode << std::endl;
227  }
228  }
229  else
230  {
231  T=2.0*MathematicalConstants::Pi/omega1;
232  ampl_ratio=ampl_ratio1;
233  }
234 
235  // Number of previous timesteps that need to be stored
236  unsigned n_time=time_stepper_pt->nprev_values();
237 
238  // Setup geometric data for element: Five items of data live in
239  // in the one and only geometric data. Also store time history
240  Geom_data_pt.resize(1);
241  Geom_data_pt[0]=new Data(time_stepper_pt,5);
242 
243  // I've created the data, I need to clean up
244  Must_clean_up=true;
245 
246  for (unsigned itime=0;itime<=n_time;itime++)
247  {
248 
249  // Buckling amplitude
250  Geom_data_pt[0]->set_value(itime,0,eps_buckl);
251  Geom_data_pt[0]->pin(0);
252

548 //=========================================================================
549 /// \short Pseudo buckling ring: Circular ring deformed by the
550 /// N-th buckling mode of a thin-wall elastic ring.
551 /// \f[
552 /// x = R_0 \cos(\zeta) +
553 /// \epsilon \left( \cos(N \zeta) \cos(\zeta) - A \sin(N \zeta) \sin(\zeta)
554 /// \right) sin(2 \pi t/T)
555 /// \f]
556 /// \f[
557 /// y = R_0 \sin(\zeta) +
558 /// \epsilon \left( \cos(N \zeta) \sin(\zeta) + A \sin(N \zeta) \cos(\zeta)
559 /// \right) sin(2 \pi t/T)
560 /// \f]
561 /// where A is the ratio of the aziumuthal to the radial buckling
562 /// amplitude (A=-1/N for statically buckling rings) and epsilon
563 /// is the buckling amplitude.
564 /// Scale R_0 is adjusted to ensure conservation of (computational)
565 /// volume/area. This is implemented by a
566 /// pseudo-elasticity approach: The governing equation for \f$ R_0 \f$ is:
567 /// \f[
568 /// p_{ref} = R_0 - 1.0
569 /// \f]
570 /// The pointer to the reference pressure needs to be set with
571 /// reference_pressure_pt().
572 //=========================================================================
573 class PseudoBucklingRingElement : public GeneralisedElement,
574  public PseudoBucklingRing
575 {
576  private:
577

579  /// become an unknown
580  unsigned Internal_geometric_variable_index;
581 
582  /// \short The Data object that represents the reference pressure is stored
583  /// at the location indexed by this integer in the external data storage.
584  unsigned External_reference_pressure_index;
585 
586  /// Pointer to the data object that represents the external reference pressure
587  Data* External_reference_pressure_pt;
588 
589  /// Return the local equation number of the internal geometric variable
590  inline int geometric_local_eqn()
591  {return internal_local_eqn(0,Internal_geometric_variable_index);}
