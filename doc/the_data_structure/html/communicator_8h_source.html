289  /// broadcast function for Vector<int>
290  void broadcast(const int& source, Vector<int>& x)
291  {
292  int n;
293 
294  // Get number of entries on processor source (where the vector exists)
295  unsigned long n_long;
296  if (this->my_rank()==source)
297  {
298  n_long=x.size();
299 
300  // Convert to int
301  n=static_cast<int>(n_long);
302  }
303 
304  // Broadcast to everybody how many entries to expect
305  MPI_Bcast(&n,1,MPI_INT,source,this->mpi_comm());
306 
307  // Resize Vector everywhere else in preparation
308  if (this->my_rank()!=source)
309  {
310  x.resize(n);
311  }
312 
313  // Broadcast the Vector directly
314  MPI_Bcast(&x[0],n,MPI_INT,source,this->mpi_comm());
315  }
