43  /// A class which allows the user to specify a prescribed
44  /// volume (as opposed to a prescribed pressure) for in the region
45  /// bounded by the membrane.
46  /// Effectively adds an equation to the system for pressure.
47  /// There would usually only be a single instance of this
48  /// element in a problem.
49  //=============================================================
50  template<class ELEMENT, template<class> class MESH>
51  class FoepplvonKarmanVolumeConstraintElement
52  : public virtual GeneralisedElement
53  {
54  public:
55  /// \short Constructor. Takes pointer to mesh of Foeppl von Karman
56  /// elements and a vector of unsigneds which identifies the
57  /// regions within it that contribute to the controlled volume
58  /// defined as int w dA (i.e. the "volume under the membrane").
59  /// The optional final argument specifies the initial value
60  /// for the pressure that is "traded" in exchange for the
61  /// volume constraint.
62  FoepplvonKarmanVolumeConstraintElement(
63  MESH<ELEMENT>* bounding_mesh_pt,
64  const Vector<unsigned>& contributing_region,
65  const double& pressure = 0.0)
66  : Bounding_mesh_pt(bounding_mesh_pt),
67  Contributing_region(contributing_region),
68  Prescribed_volume_pt(0)
69  {
70  // Create instance of pressure that is traded for volume constraint
71  Volume_control_pressure_pt = new Data(1);
72  Volume_control_pressure_pt->set_value(0, pressure);
73 
74  // Add the data object as internal data for this element
75  Pressure_data_index = add_internal_data(Volume_control_pressure_pt);

94  /// \short Returns the volume "under the elements" in the constrained
95  /// regions
96  double get_bounded_volume()
97  {
98  double bounded_volume = 0.0;
99 
100  // Loop over the bounded regions
101  unsigned n_contributing_regions = Contributing_region.size();
102  for (unsigned r = 0; r < n_contributing_regions; r++)
103  {
104  // Loop over the elements in the bounded regions
105  unsigned n_inner_el =
106  Bounding_mesh_pt->nregion_element(Contributing_region[r]);
107  for (unsigned e = 0; e < n_inner_el; e++)
108  {
109  // Add the contribution
110  ELEMENT* el_pt = dynamic_cast<ELEMENT*>(
111  Bounding_mesh_pt->region_element_pt(Contributing_region[r], e));
112  bounded_volume += el_pt->get_bounded_volume();
113  }
114  }
115  return bounded_volume;
116  }
