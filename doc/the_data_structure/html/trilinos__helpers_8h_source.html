71 /// \short Helper namespace for use with the Trilinos Epetra package.
72 /// Contains functions to generate two Epetra containers (Epetra_Vector and
73 /// Epetra_CrsMatrix) and provides access to the trilinos matrix-matrix
74 /// and matrix-vector product routines.
75 //=============================================================================
76 namespace TrilinosEpetraHelpers
77 {
78 
79  // VECTOR METHODS ============================================================
80 
81  /// \short create an Epetra_Vector from an oomph-lib DoubleVector.
82  /// If oomph_vec is NOT distributed (i.e. locally replicated) and
83  /// on more than one processor, then the returned Epetra_Vector will be
84  /// uniformly distributed. If the oomph_vec is distributed then the
85  /// Epetra_Vector returned will have the same distribution as oomp_vec.
86  Epetra_Vector* create_distributed_epetra_vector(const DoubleVector&
87  oomph_vec);
88 
89  /// \short create an Epetra_Vector based on the argument oomph-lib
90  /// LinearAlgebraDistribution
91  /// If dist is NOT distributed and
92  /// on more than one processor, then the returned Epetra_Vector will be
93  /// uniformly distributed. If dist is distributed then the Epetra_Vector
94  /// returned will have the same distribution as dist.
95  /// The coefficient values are not set.
96  Epetra_Vector* create_distributed_epetra_vector
97  (const LinearAlgebraDistribution* dist_pt);
98 
99  /// \short create an Epetra_Vector equivalent of DoubleVector
100  /// The argument DoubleVector must be built.
101  /// The Epetra_Vector will point to, and NOT COPY the underlying data in the
102  /// DoubleVector.
103  /// The oomph-lib DoubleVector and the returned Epetra_Vector will have the
104  /// the same distribution.
105  Epetra_Vector* create_epetra_vector_view_data(DoubleVector& oomph_vec);
106 
107  /// \short Helper function to copy the contents of a Trilinos vector to an
108  /// oomph-lib distributed vector. The distribution of the two vectors must
109  /// be identical
110  void copy_to_oomphlib_vector(const Epetra_Vector* epetra_vec_pt,
111  DoubleVector& oomph_vec);
112 
113  // MATRIX METHODS ============================================================
114 
115  /// \short create an Epetra_CrsMatrix from an oomph-lib CRDoubleMatrix.
116  /// If oomph_matrix_pt is NOT distributed (i.e. locally replicated) and
117  /// on more than one processor, then the returned Epetra_Vector will be
118  /// uniformly distributed. If the oomph_matrix_pt is distributed then the
119  /// Epetra_CrsMatrix returned will have the same distribution as
120  /// oomph_matrix_pt.
121  /// The LinearAlgebraDistribution argument dist_pt should specify the
122  /// distribution of the object this matrix will operate on.
123  Epetra_CrsMatrix* create_distributed_epetra_matrix
124  (const CRDoubleMatrix* oomph_matrix_pt,
125  const LinearAlgebraDistribution* dist_pt);
126 
127  /// \short create and Epetra_CrsMatrix from an oomph-lib CRDoubleMatrix.
128  /// Specialisation for Trilinos AztecOO.
129  /// If oomph_matrix_pt is NOT distributed (i.e. locally replicated) and
130  /// on more than one processor, then the returned Epetra_Vector will be
131  /// uniformly distributed. If the oomph_matrix_pt is distributed then the
132  /// Epetra_CrsMatrix returned will have the same distribution as
133  /// oomph_matrix_pt.
134  /// For AztecOO, the column map is ordered such that the local rows are
135  /// first.
136  Epetra_CrsMatrix* create_distributed_epetra_matrix_for_aztecoo
137  (CRDoubleMatrix* oomph_matrix_pt);
138 
139  // MATRIX OPERATION METHODS ==================================================
140 
141  /// \short Function to perform a matrix-vector multiplication on a
142  /// oomph-lib matrix and vector using Trilinos functionality.
143  /// NOTE 1. the matrix and the vectors must have the same communicator.
144  /// NOTE 2. The vector will be returned with the same distribution
145  /// as the matrix, unless a distribution is predefined in the solution
146  /// vector in which case the vector will be returned with that distribution.
147  void multiply(const CRDoubleMatrix* matrix,
148  const DoubleVector& x,
149  DoubleVector &soln);
150 
151  /// \short Function to perform a matrix-matrix multiplication on oomph-lib
152  /// matrices by using Trilinos functionality.
153  /// \b NOTE 1. There are two Trilinos matrix-matrix multiplication methods
154  /// available, using either the EpetraExt::MatrixMatrix class (if use_ml ==
155  /// false) or using ML (Epetra_MatrixMult method)
156  /// \b NOTE 2. the solution matrix (matrix_soln) will be returned with the
157  /// same distribution as matrix1
158  /// \b NOTE 3. All matrices must share the same communicator.
159  void multiply(const CRDoubleMatrix &matrix_1,
160  const CRDoubleMatrix &matrix_2,
161  CRDoubleMatrix &matrix_soln,
162  const bool& use_ml = false);
163
