48 /// Generic class for numerical integration schemes:
49 /// \f[
50 /// \int f(x_0,x_1...)\ dx_0 \ dx_1... =
51 /// \sum_{i_{int}=0}^{\mbox{\tt nweight()} }
52 /// f( x_j=\mbox{\tt knot($i_{int}$,j)} ) \ \ \ \mbox{\tt weight($i_{int}$)}
53 /// \f]
54 //=========================================================
55 class Integral
56 {
57  public:
58

60  Integral(){};

70  {
71  BrokenCopy::broken_assign("Integral");
72  }
73 
74  /// Virtual destructor (empty)
75  virtual ~Integral(){};
76 
77  /// Return the number of integration points of the scheme.

79

82
