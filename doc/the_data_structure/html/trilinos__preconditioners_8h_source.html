57  {
58  public:
59 
60  /// Constructor.

65  Epetra_matrix_pt=0;
66  }
67

69  virtual ~TrilinosPreconditionerBase()
70  {
71  clean_up_memory();
72  }
73 
74  /// \short Static double that accumulates the preconditioner
75  /// solve time of all instantiations of this class. Reset
76  /// it manually, e.g. after every Newton solve.
77  static double Cumulative_preconditioner_solve_time;
78 
79  /// deletes the preconditioner, matrices and maps
80  void clean_up_memory()
81  {
82  // delete the Epetra preconditioner
83  delete Epetra_preconditioner_pt;
84  Epetra_preconditioner_pt = 0;
85 
86  // delete the epetra matrix
87  delete Epetra_matrix_pt;
88  Epetra_matrix_pt = 0;
89  }
90 
91  /// Broken copy constructor.
92  TrilinosPreconditionerBase(const TrilinosPreconditionerBase&)
93  {
94  BrokenCopy::broken_copy("TrilinosPreconditionerBase");
95  }
96 
97 
98  /// Broken assignment operator.
99 //Commented out broken assignment operator because this can lead to a conflict warning
100 //when used in the virtual inheritence hierarchy. Essentially the compiler doesn't
101 //realise that two separate implementations of the broken function are the same and so,
102 //quite rightly, it shouts.
103  /*void operator=(const TrilinosPreconditionerBase&)
104  {
105  BrokenCopy::broken_assign("TrilinosPreconditionerBase");
106  }*/
107 
108  /// \short Function to set up a preconditioner for the linear system
109  /// defined by matrix_pt. This function must be called before using
110  /// preconditioner_solve.
111  /// \b NOTE 1. matrix_pt must point to an object of class CRDoubleMatrix or
112  /// DistributedCRDoubleMatrix
113  /// This method should be called by oomph-lib solvers and preconditioners
114  void setup();
115 
116  /// \short Function to setup a preconditioner for the linear system defined
117  /// by the oomph-lib oomph_matrix_pt and Epetra epetra_matrix_pt matrices.
118  /// This method is called by Trilinos solvers.
119  void setup(Epetra_CrsMatrix* epetra_matrix_pt);
120 
121  /// \short applies the preconditioner
122  void preconditioner_solve(const DoubleVector& r,
123  DoubleVector &z);
124 
125  /// Access function to Epetra_preconditioner_pt.
126  /// For use with \c TrilinosAztecOOSolver
127  Epetra_Operator*& epetra_operator_pt()
128  {
129  return Epetra_preconditioner_pt;
130  }
131 
132  /// Access function to Epetra_preconditioner_pt (const version)
133  /// For use with \c TrilinosAztecOOSolver
134  Epetra_Operator* epetra_operator_pt() const
135  {
136  return Epetra_preconditioner_pt;
137  }
138 
139  protected:
140 
141  /// \short Function to set up a specific Trilinos preconditioner.
142  /// This is called by setup(...).
143  virtual void setup_trilinos_preconditioner
144  (Epetra_CrsMatrix* epetra_matrix_pt)=0;
145 
146  /// \short The preconditioner which will be set up using function

148  Epetra_Operator* Epetra_preconditioner_pt;
149 
150  /// \short Pointer used to store the epetra matrix - only used when this
151  /// preconditioner is setup using the oomph-lib interface
152  Epetra_CrsMatrix* Epetra_matrix_pt;
153  };

262 
263 
264  /// \short Default number of V cycles (one to be consistent with
