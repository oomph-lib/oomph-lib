33 /// Add the element's contribution to the elemental residual vector
34 /// and/or elemental jacobian matrix.
35 /// This function overloads the standard version so that the possible
36 /// presence of hanging nodes is taken into account.
37 //=========================================================================
38 template<unsigned DIM>
39 void RefineableGeneralisedAdvectionDiffusionEquations<DIM>::
40 fill_in_generic_residual_contribution_cons_adv_diff(
41  Vector<double> &residuals,
42  DenseMatrix<double> &jacobian,
43  DenseMatrix<double>
44  &mass_matrix,
45  unsigned flag)
46 {
47 
48 //Find out how many nodes there are in the element
49 unsigned n_node = nnode();
50 
51 //Get the nodal index at which the unknown is stored
52  unsigned u_nodal_index = this->u_index_cons_adv_diff();
53 
54 //Set up memory for the shape and test functions
55  Shape psi(n_node), test(n_node);
56  DShape dpsidx(n_node,DIM), dtestdx(n_node,DIM);
57 
58 //Set the value of n_intpt
59 unsigned n_intpt = integral_pt()->nweight();
60 
61 //Set the Vector to hold local coordinates
62 Vector<double> s(DIM);
63 
64 //Get Peclet number
65 double peclet = this->pe();
66 
67 //Get the Peclet multiplied by the Strouhal number
68 double peclet_st = this->pe_st();
69 
70 //Integers used to store the local equation number and local unknown
71 //indices for the residuals and jacobians
72 int local_eqn=0, local_unknown=0;
73 
74 // Local storage for pointers to hang_info objects
75  HangInfo *hang_info_pt=0, *hang_info2_pt=0;
76 
77 //Local variable to determine the ALE stuff
78  bool ALE_is_disabled_flag = this->ALE_is_disabled;
79 
80 //Loop over the integration points
81 for(unsigned ipt=0;ipt<n_intpt;ipt++)
82 {
83 
84  //Assign values of s
85  for(unsigned i=0;i<DIM;i++) s[i] = integral_pt()->knot(ipt,i);
86 
87  //Get the integral weight
88  double w = integral_pt()->weight(ipt);
89 
90  //Call the derivatives of the shape and test functions
91  double J =
92  this->dshape_and_dtest_eulerian_at_knot_cons_adv_diff(ipt,psi,dpsidx,
93  test,dtestdx);
94 
95  //Premultiply the weights and the Jacobian
96  double W = w*J;
97 
98  //Calculate local values of the function, initialise to zero
99  double dudt=0.0;
100  double interpolated_u=0.0;
101 
102  //These need to be a Vector to be ANSI C++, initialise to zero
103  Vector<double> interpolated_x(DIM,0.0);
104  Vector<double> interpolated_dudx(DIM,0.0);
105  Vector<double> mesh_velocity(DIM,0.0);
106 
107  //Calculate function value and derivatives:
108  //-----------------------------------------
109 
110  // Loop over nodes
111  for(unsigned l=0;l<n_node;l++)
112  {
113  //Get the value at the node
114  double u_value = this->nodal_value(l,u_nodal_index);
115  interpolated_u += u_value*psi(l);
116  dudt += this->du_dt_cons_adv_diff(l)*psi(l);
117  // Loop over directions
118  for(unsigned j=0;j<DIM;j++)
119  {
120  interpolated_x[j] += nodal_position(l,j)*psi(l);
121  interpolated_dudx[j] += u_value*dpsidx(l,j);
122  }
123  }
124 
125  //Get the mesh velocity, if required
126  if (!ALE_is_disabled_flag)
127  {
128  for(unsigned l=0;l<n_node;l++)
129  {
130  // Loop over directions
131  for(unsigned j=0;j<DIM;j++)
132  {
133  mesh_velocity[j] += dnodal_position_dt(l,j)*psi(l);
134  }
135  }
136  }
137 
138 
139  //Get body force
140  double source;
141  this->get_source_cons_adv_diff(ipt,interpolated_x,source);
142 
143 
144  //Get wind
145  //--------
146  Vector<double> wind(DIM);
147  this->get_wind_cons_adv_diff(ipt,s,interpolated_x,wind);
148 
149  //Get the conserved wind (non-divergence free)
150  Vector<double> conserved_wind(DIM);
151  this->get_conserved_wind_cons_adv_diff(ipt,s,interpolated_x,conserved_wind);
152 
153  //Get diffusivity tensor
154  DenseMatrix<double> D(DIM,DIM);
155  this->get_diff_cons_adv_diff(ipt,s,interpolated_x,D);
156 
157  // Assemble residuals and Jacobian
158  //================================
159 
160  // Loop over the nodes for the test functions
161  for(unsigned l=0;l<n_node;l++)
162  {
163  //Local variables to store the number of master nodes and
164  //the weight associated with the shape function if the node is hanging
165  unsigned n_master=1; double hang_weight=1.0;
166  //Local bool (is the node hanging)
167  bool is_node_hanging = this->node_pt(l)->is_hanging();
168 
169  //If the node is hanging, get the number of master nodes
170  if(is_node_hanging)
171  {
172  hang_info_pt = this->node_pt(l)->hanging_pt();
173  n_master = hang_info_pt->nmaster();
174  }
175  //Otherwise there is just one master node, the node itself
176  else
177  {
178  n_master = 1;
179  }
180 
181  //Loop over the number of master nodes
182  for(unsigned m=0;m<n_master;m++)
183  {
184  //Get the local equation number and hang_weight
185  //If the node is hanging
186  if(is_node_hanging)
187  {
188  //Read out the local equation from the master node
189  local_eqn = this->local_hang_eqn(hang_info_pt->master_node_pt(m),
190  u_nodal_index);
191  //Read out the weight from the master node
192  hang_weight = hang_info_pt->master_weight(m);
193  }
194  //If the node is not hanging
195  else
196  {
197  //The local equation number comes from the node itself
198  local_eqn = this->nodal_local_eqn(l,u_nodal_index);
199  //The hang weight is one
200  hang_weight = 1.0;
201  }
202 
203  //If the nodal equation is not a boundary conditino
204  if(local_eqn >= 0)
205  {
206  //Add du/dt and body force/source term here
207  residuals[local_eqn] -=
208  (peclet_st*dudt + source)*test(l)*W*hang_weight;
209 
210  //The Mesh velocity and GeneralisedAdvection--Diffusion bit
211  for(unsigned k=0;k<DIM;k++)
212  {
213  //Terms that multiply the test function
214  double tmp = peclet*wind[k];
215  //If the mesh is moving need to subtract the mesh velocity
216  if(!ALE_is_disabled_flag)
217  {tmp -= peclet_st*mesh_velocity[k];}
218  tmp *= interpolated_dudx[k];
219 
220  //Terms that multiply the derivative of the test function
221  //Advective term
222  double tmp2 = - conserved_wind[k]*interpolated_u;
223  //Now the diuffusive term
224  for(unsigned j=0;j<DIM;j++)
225  {
226  tmp2 += D(k,j)*interpolated_dudx[j];
227  }
228  //Now construct the contribution to the residuals
229  residuals[local_eqn] -= (tmp*test(l) + tmp2*dtestdx(l,k))*W*
230  hang_weight;
231  }
232 
233  // Calculate the Jacobian
234  if(flag)
235  {
236  //Local variables to store the number of master nodes
237  //and the weights associated with each hanging node
238  unsigned n_master2=1; double hang_weight2=1.0;
239  //Loop over the nodes for the variables
240  for(unsigned l2=0;l2<n_node;l2++)
241  {
242  //Local bool (is the node hanging)
243  bool is_node2_hanging = this->node_pt(l2)->is_hanging();
244  //If the node is hanging, get the number of master nodes
245  if(is_node2_hanging)
246  {
247  hang_info2_pt = this->node_pt(l2)->hanging_pt();
248  n_master2 = hang_info2_pt->nmaster();
249  }
250  //Otherwise there is one master node, the node itself
251  else
252  {
253  n_master2 = 1;
254  }
255 
256  //Loop over the master nodes
257  for(unsigned m2=0;m2<n_master2;m2++)
258  {
259  //Get the local unknown and weight
260  //If the node is hanging
261  if(is_node2_hanging)
262  {
263  //Read out the local unknown from the master node
264  local_unknown =
265  this->local_hang_eqn(hang_info2_pt->master_node_pt(m2),
266  u_nodal_index);
267  //Read out the hanging weight from the master node
268  hang_weight2 = hang_info2_pt->master_weight(m2);
269  }
270  //If the node is not hanging
271  else
272  {
273  //The local unknown number comes from the node
274  local_unknown = this->nodal_local_eqn(l2,u_nodal_index);
275  //The hang weight is one
276  hang_weight2 = 1.0;
277  }
278 
279  //If the unknown is not pinned
280  if(local_unknown >= 0)
281  {
282  //Add contribution to Elemental Matrix
283  // Mass matrix du/dt term
284  jacobian(local_eqn,local_unknown)
285  -= peclet_st*test(l)*psi(l2)*
286  this->node_pt(l2)->time_stepper_pt()->weight(1,0)
287  *W*hang_weight*hang_weight2;
288 
289  //Add contribution to mass matrix
290  if(flag==2)
291  {
292  mass_matrix(local_eqn,local_unknown) +=
293  peclet_st*test(l)*psi(l2)*W*hang_weight*hang_weight2;
294  }
295 
296  //Add contribution to Elemental Matrix
297  for(unsigned k=0;k<DIM;k++)
298  {
299  //Temporary term used in assembly
300  double tmp = peclet*wind[k];
301  if(!ALE_is_disabled_flag)
302  {tmp -= peclet_st*mesh_velocity[k];}
303  tmp *= dpsidx(l2,k);
304 
305  double tmp2 = - conserved_wind[k]*psi(l2);
306  //Now the diffusive term
307  for(unsigned j=0;j<DIM;j++)
308  {
309  tmp2 += D(k,j)*dpsidx(l2,j);
310  }
311 
312  //Now assemble Jacobian term
313  jacobian(local_eqn,local_unknown)
314  -= (tmp*test(l) + tmp2*dtestdx(l,k))*W*
315  hang_weight*hang_weight2;
316  }
317  }
318  } //End of loop over master nodes
319  } //End of loop over nodes
320  } //End of Jacobian calculation
321 
322  } //End of non-zero equation
323 
324  } //End of loop over the master nodes for residual
325  } //End of loop over nodes
326 
327  } // End of loop over integration points
328 }
