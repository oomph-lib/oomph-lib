252 /// An interface to allow Mumps to be used as an (exact) Preconditioner
253 //====================================================================
254 class NewMumpsPreconditioner : public Preconditioner
255 {
256  public:
257 
258  /// Constructor.
259  NewMumpsPreconditioner()
260  {}
261 
262  /// Destructor.
263  ~NewMumpsPreconditioner()
264  {}
265 
266  /// Broken copy constructor.
267  NewMumpsPreconditioner(const NewMumpsPreconditioner&)
268  {
269  BrokenCopy::broken_copy("NewMumpsPreconditioner");
270  }
271 
272 
273  /// Broken assignment operator.
274  void operator=(const NewMumpsPreconditioner&)
275  {
276  BrokenCopy::broken_assign("NewMumpsPreconditioner");
277  }
278 
279  /// \short Function to set up a preconditioner for the linear
280  /// system defined by matrix_pt. This function must be called
281  /// before using preconditioner_solve.
282  /// Note: matrix_pt must point to an object of class
283  /// CRDoubleMatrix or CCDoubleMatrix
284  void setup()
285  {
286  oomph_info << "Setting up Mumps (exact) preconditioner"
287  << std::endl;
288 
289  DistributableLinearAlgebraObject* dist_matrix_pt=
290  dynamic_cast<DistributableLinearAlgebraObject*>(matrix_pt());
291  if (dist_matrix_pt!=0)
292  {
293  LinearAlgebraDistribution dist(dist_matrix_pt->distribution_pt());
294  this->build_distribution(dist);
295  Solver.factorise(matrix_pt());
296  }
297  else
298  {
299  std::ostringstream error_message_stream;
300  error_message_stream
301  << "NewMumpsPreconditioner can only be applied to matrices derived \n"
302  << "DistributableLinearAlgebraObject.\n";
303  throw OomphLibError(error_message_stream.str(),
304  OOMPH_CURRENT_FUNCTION,
305  OOMPH_EXCEPTION_LOCATION);
306  }
307  }
308 
309  /// \short Function applies Mumps to vector r for (exact)
310  /// preconditioning, this requires a call to setup(...) first.
311  void preconditioner_solve(const DoubleVector &r, DoubleVector &z)
312  {
313  Solver.resolve(r, z);
314  }
315 
316 
317  /// \short Clean up memory -- forward the call to the version in
318  /// Mumps in its LinearSolver incarnation.
319  void clean_up_memory()
320  {
321  Solver.clean_up_memory();
322  }
323 
324 
325  /// Enable documentation of timings
326  void enable_doc_time() {Solver.enable_doc_time();}
327 
328  /// Disable the documentation of timings
329  void disable_doc_time() {Solver.disable_doc_time();}
330 
331  private:
332 
333  /// \short the Mumps solver emplyed by this preconditioner
334  MumpsSolver Solver;
335 
336 };
