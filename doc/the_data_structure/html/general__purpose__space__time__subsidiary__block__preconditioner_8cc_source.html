479  /// Preconditioner solve for the block triangular preconditioner
480  //=============================================================================
481  void SpaceTimeNavierStokesSubsidiaryPreconditioner::
482  preconditioner_solve(const DoubleVector& r,DoubleVector& z)
483  {
484  // If z is not set up then give it the same distribution
485  if (!z.distribution_pt()->built())
486  {
487  // Build z with the same distribution as r
488  z.build(r.distribution_pt(),0.0);
489  }
490 
491  // -----------------------------------------------------------------------
492  // Step 1 - apply approximate Schur inverse to pressure unknowns (block 1)
493  // -----------------------------------------------------------------------
494  // First working vectors
495  DoubleVector temp_vec;
496 
497  // Second working vector
498  DoubleVector another_temp_vec;
499 
500  // Copy pressure values from residual vector to temp_vec:
501  // Loop over all entries in the global vector (this one
502  // includes velocity and pressure dofs in some random fashion)
503  this->get_block_vector(1,r,temp_vec);
504 
505  // Compute the vector P^{-1}r_p
506  P_preconditioner_pt->preconditioner_solve(temp_vec,another_temp_vec);
507 
508  // Now clear the vector storing r_p
509  temp_vec.clear();
510 
511  // Now compute G(P^{-1}r_p)
512  G_mat_vec_pt->multiply(another_temp_vec,temp_vec);
513 
514  // Clear the vector storing P^{-1}r_p
515  another_temp_vec.clear();
516 
517  // Now update to get F(GP^{-1}r_p)
518  F_mat_vec_pt->multiply(temp_vec,another_temp_vec);
519 
520  // Clear the vector storing GP^{-1}r_p
521  temp_vec.clear();
522 
523  // Now compute D(FGP^{-1}r_p)=G^{T}(FGP^{-1}r_p)
524  G_mat_vec_pt->multiply_transpose(another_temp_vec,temp_vec);
525 
526  // Clear the vector storing FGP^{-1}r_p
527  another_temp_vec.clear();
528 
529  // Finally, compute the full approximation to -z_p; -z_p=P^{-1}(DFG)P^{-1}r_p
530  // NOTE: We'll move the sign over in the next step
531  P_preconditioner_pt->preconditioner_solve(temp_vec,another_temp_vec);
532 
533  // Rebuild temp_vec with another_temp_vec's distribution with all
534  // entries initialised to zero
535  temp_vec.build(another_temp_vec.distribution_pt(),0.0);
536 
537  // Now fix the sign and store the result in temp_vec
538  temp_vec-=another_temp_vec;
539 
540  // Now copy temp_vec (i.e. z_p) back into the global vector z
541  return_block_vector(1,temp_vec,z);
542 
543  // ------------------------------------------------------------
544  // Step 2 - apply preconditioner to velocity unknowns (block 0)
545  // ------------------------------------------------------------
546  // Clear the vector storing z_p=-P^{-1}(DFG)P^{-1}r_p
547  temp_vec.clear();
548 
549  // Recall that another_temp_vec (computed above) contains the negative
550  // of the solution of the Schur complement system, -z_p. Multiply by G
551  // and store the result in temp_vec (vector resizes itself).
552  G_mat_vec_pt->multiply(another_temp_vec,temp_vec);
553 
554  // Now clear the vector storing P^{-1}(DFG)P^{-1}r_p
555  another_temp_vec.clear();
556 
557  // Get the residual vector entries associated with the velocities, r_u
558  get_block_vector(0,r,another_temp_vec);
559 
560  // Update the result to get r_u-Gz_p
561  another_temp_vec+=temp_vec;
562 
563  // Compute the solution z_u which satisfies z_u=F^{-1}(r_u-Gz_p)
564  F_preconditioner_pt->preconditioner_solve(another_temp_vec,temp_vec);
565 
566  // Now store the result in the global solution vector
567  return_block_vector(0,temp_vec,z);
568  } // End of preconditioner_solve

647  double block_setup_time=t_block_setup_finish-t_block_setup_start;
648 
649  // Document the time taken
650  oomph_info << "Time for block_setup(...) [sec]: "
651  << block_setup_time << std::endl;
652 
653  // Get the number of block types
654  unsigned n_block_type=this->nblock_types();
655 
656  // How many block types are there? Should only be 2...
657  oomph_info << "Number of block types: " << n_block_type << std::endl;
658 
659  // Space for the concatenated block matrix
660  Matrix_pt=new CRDoubleMatrix;
661 
662  // Get the (one and only) block to be used by this block preconditioner
663  this->get_block(0,0,*Matrix_pt);
664 
665  // Create a new instance of the NS subsidiary preconditioner
666  Navier_stokes_subsidiary_preconditioner_pt=
667  new SpaceTimeNavierStokesSubsidiaryPreconditioner;
668 
669  // Do we need to doc. the memory statistics?
670  if (Compute_memory_statistics)
671  {
672  // How many rows are there in this block?
673  unsigned n_row=Matrix_pt->nrow();
674 
675  // How many nonzeros are there in this block?
676  unsigned n_nnz=Matrix_pt->nnz();
677 
678  // Compute the memory usage. The only memory overhead here (for the
679  // setup phase) is storing the system matrix
680  Memory_usage_in_bytes=((2*((n_row+1)*sizeof(int)))+
681  (n_nnz*(sizeof(int)+sizeof(double))));
682 
683  // We might as well document the memory statistics of the Navier-Stokes
684  // subsidiary block preconditioner while we're at it...
685  Navier_stokes_subsidiary_preconditioner_pt->enable_doc_memory_usage();
686  }
687 
688  // Create a map to declare which of the 3 dof types in the GMRES
689  // subsidiary preconditioner correspond to the dof types in the NS
690  // subsidiary block preconditioner
691  Vector<unsigned> dof_map(n_dof_types);
692 
693  // Loop over the dof types associated with the GMRES subsidiary block
694  // preconditioner
695  for (unsigned i=0; i<n_dof_types; i++)
696  {
697  // There is, essentially, an identity mapping between the GMRES subsidiary
698  // block preconditioner and the NSSP w.r.t. the dof types
699  dof_map[i]=i;
700  }
701 
702  // Now turn it into an "proper" subsidiary preconditioner
703  Navier_stokes_subsidiary_preconditioner_pt->
704  turn_into_subsidiary_block_preconditioner(this,dof_map);
705 
706  // Setup: The NS subsidiary preconditioner is itself a block preconditioner
707  // so we need to pass it a pointer to full-size matrix!
708  Navier_stokes_subsidiary_preconditioner_pt->setup(this->matrix_pt());
709 
710  // Remember that the preconditioner has been setup so the stored
711  // information can be wiped when we come here next...
712  Preconditioner_has_been_setup=true;
713 
714  // If we're meant to build silently, reassign the oomph stream pointer
715  if (this->Silent_preconditioner_setup==true)
716  {
717  // Store the output stream pointer
718  oomph_info.stream_pt()=this->Stream_pt;
719 
720  // Reset our own stream pointer
721  this->Stream_pt=0;
722  }
723  } // End of setup

725  //=============================================================================
726  /// Preconditioner solve for the GMRES block preconditioner
727  //=============================================================================
728  void GMRESBlockPreconditioner::
729  preconditioner_solve(const DoubleVector& rhs,DoubleVector& solution)
730  {
731  // Get the number of block types
732  unsigned n_block_types=this->nblock_types();
733 
734  // If there is more than one block type (there shouldn't be!)
735  if (n_block_types!=1)
736  {
737  // Throw an error
738  throw OomphLibError("Can only deal with one dof type!",
739  OOMPH_CURRENT_FUNCTION,
740  OOMPH_EXCEPTION_LOCATION);
741  }
742 
743  // Allocate space for the solution blocks
744  Vector<DoubleVector> block_z(n_block_types);
745 
746  // Allocate space for the residual blocks
747  Vector<DoubleVector> block_r(n_block_types);
748 
749  // Get the reordered RHS vector
750  this->get_block_vectors(rhs,block_r);
751 
752  // Get the reordered solution vector
753  this->get_block_vectors(solution,block_z);
754 
755  // Initialise the entries in the solution vector to zero
756  block_z[0].initialise(0.0);
757 
758  // Get number of dofs
759  unsigned n_dof=block_r[0].nrow();
760 
761  // Time solver
762  double t_start=TimingHelpers::timer();
763 
764  // Relative residual
765  double resid;
766 
767  // Iteration counter
768  unsigned iter=1;
769 
770  // Initialise vectors
771  Vector<double> s(n_dof+1,0);
772  Vector<double> cs(n_dof+1);
773  Vector<double> sn(n_dof+1);
774  DoubleVector w(block_r[0].distribution_pt(),0.0);
775 
776  // Storage for the time taken for all preconditioner applications
777  double t_prec_application_total=0.0;
778 
779  // Storage for the RHS vector
780  DoubleVector r(block_r[0].distribution_pt(),0.0);
781 
782  // If we're using LHS preconditioning then solve Mr=b-Jx for r (assumes x=0)
783  if (Preconditioner_LHS)
784  {
785  // Initialise timer
786  double t_prec_application_start=TimingHelpers::timer();
787 
788  // This is pretty inefficient but there is no other way to do this with
789  // block subsidiary preconditioners as far as I can tell because they
790  // demand to have the full r and z vectors...
791  DoubleVector block_z_with_size_of_full_z(rhs.distribution_pt(),0.0);
792  DoubleVector r_updated(rhs.distribution_pt(),0.0);
793 
794  // Reorder the entries in block_r[0] and put them back into the
795  // global-sized vector. The subsidiary preconditioner should only
796  // operate on the dofs that this preconditioner operates on so
797  // don't worry about all the other entries
798  this->return_block_vector(0,block_r[0],r_updated);
799 
800  // Solve on the global-sized vectors
801  Navier_stokes_subsidiary_preconditioner_pt->
802  preconditioner_solve(r_updated,block_z_with_size_of_full_z);
803 
804  // Now grab the part of the solution associated with this preconditioner
805  this->get_block_vector(0,block_z_with_size_of_full_z,block_z[0]);
806 
807  // Doc time for setup of preconditioner
808  double t_prec_application_end=TimingHelpers::timer();
809 
810  // Update the preconditioner application time total
811  t_prec_application_total+=(t_prec_application_end-
812  t_prec_application_start);
813  }
814  // If we're using RHS preconditioning, do nothing to the actual RHS vector
815  else
816  {
817  // Copy the entries into r
818  r=block_r[0];
819  }
820 
821  // Calculate the initial residual norm (assumes x=0 initially)
822  double normb=r.norm();
823 
824  // Set beta (the initial residual)
825  double beta=normb;
826 
827  // Compute initial relative residual
828  if (normb==0.0)
829  {
830  // To avoid dividing by zero, set normb to 1
831  normb=1;
832  }
833 
834  // Now calculate the relative residual norm
835  resid=beta/normb;
836 
837  // If required, document convergence history to screen or file (if
838  // stream is open)
839  if (Doc_convergence_history)
840  {
841  // If a file has not been provided to output to
842  if (!Output_file_stream.is_open())
843  {
844  // Output the initial relative residual norm to screen
845  oomph_info << 0 << " " << resid << std::endl;
846  }
847  // If there is a file provided to output to
848  else
849  {
850  // Output the initial relative residual norm to file
851  Output_file_stream << 0 << " " << resid <<std::endl;
852  }
853  } // if (Doc_convergence_history)
854 
855  // If GMRES converges immediately
856  if (resid<=Tolerance)
857  {
858  // If the user wishes GMRES to output information about the solve to screen
859  if (Doc_time)
860  {
861  // Tell the user
862  oomph_info << "GMRES block preconditioner converged immediately. "
863  << "Normalised residual norm: " << resid << std::endl;
864  }
865 
866  // Now reorder the values in block_z[0] and put them back into the
867  // global solution vector
868  this->return_block_vector(0,block_z[0],solution);
869 
870  // Doc time for solver
871  double t_end=TimingHelpers::timer();
872  Solution_time=t_end-t_start;
873 
874  // If the user wishes GMRES to output information about the solve to screen
875  if (Doc_time)
876  {
877  // Tell the user
878  oomph_info << "Time for all preconditioner applications [sec]: "
879  << t_prec_application_total
880  << "\nTotal time for solve with GMRES block preconditioner "
881  << "[sec]: " << Solution_time << std::endl;
882  }
883 
884  // We're finished; end here!
885  return;
886  } // if (resid<=Tolerance)
887 
888  // Initialise vector of orthogonal basis vectors, v
889  Vector<DoubleVector> v(n_dof+1);
890 
891  // Initialise the upper Hessenberg matrix H.
892  // NOTE: For implementation purposes the upper hessenberg matrix indices
893  // are swapped so the matrix is effectively transposed
894  Vector<Vector<double> > H(n_dof+1);
895 
896  // Loop as many times as we're allowed
897  while (iter<=Max_iter)
898  {
899  // Copy the entries of the residual vector, r, into v[0]
900  v[0]=r;
901 
902  // Now update the zeroth basis vector, v[0], to have values r/beta
903  v[0]/=beta;
904 
905  // Storage for ...?
906  s[0]=beta;
907 
908  // Inner iteration counter for restarted version (we don't actually use
909  // a restart in this implementation of GMRES...)
910  unsigned iter_restart;
911 
912  // Perform iterations
913  for (iter_restart=0; iter_restart<n_dof&&iter<=Max_iter; iter_restart++,iter++)
914  {
915  // Resize next column of the upper Hessenberg matrix
916  H[iter_restart].resize(iter_restart+2);
917 
918  // Solve for w inside the scope of these braces
919  {
920  // Temporary vector
921  DoubleVector temp(block_r[0].distribution_pt(),0.0);
922 
923  // If we're using LHS preconditioning solve Mw=Jv[i] for w
924  if (Preconditioner_LHS)
925  {
926  // Calculate temp=Jv[i]
927  Matrix_pt->multiply(v[iter_restart],temp);
928 
929  // Get the current time
930  double t_prec_application_start=TimingHelpers::timer();
931 
932  // This is pretty inefficient but there is no other way to do this with
933  // block sub preconditioners as far as I can tell because they demand
934  // to have the full r and z vectors...
935  DoubleVector block_z_with_size_of_full_z(rhs.distribution_pt(),0.0);
936  DoubleVector r_updated(rhs.distribution_pt(),0.0);
937 
938  // Put the values in temp into the global-sized vector
939  this->return_block_vector(0,temp,r_updated);
940 
941  // Apply the NSSP to the global-sized vectors (this will extract
942  // the appropriate sub-vectors and solve on the reduced system)
943  Navier_stokes_subsidiary_preconditioner_pt->
944  preconditioner_solve(r_updated,block_z_with_size_of_full_z);
945 
946  // Now grab the part of the solution associated with this preconditioner
947  this->get_block_vector(0,block_z_with_size_of_full_z,w);
948 
949  // End timer
950  double t_prec_application_end=TimingHelpers::timer();
951 
952  // Update the preconditioner application time total
953  t_prec_application_total+=(t_prec_application_end-
954  t_prec_application_start);
955  }
956  // If we're using RHS preconditioning solve
957  else
958  {
959  // Initialise timer
960  double t_prec_application_start=TimingHelpers::timer();
961 
962  // This is pretty inefficient but there is no other way to do this with
963  // block sub preconditioners as far as I can tell because they demand
964  // to have the full r and z vectors...
965  DoubleVector block_z_with_size_of_full_z(rhs.distribution_pt());
966  DoubleVector r_updated(rhs.distribution_pt());
967 
968  // Put the values in v[iter_restart] back into the global-sized vector
969  this->return_block_vector(0,v[iter_restart],r_updated);
970 
971  // Solve on the global-sized vectors
972  Navier_stokes_subsidiary_preconditioner_pt->
973  preconditioner_solve(r_updated,block_z_with_size_of_full_z);
974 
975  // Now grab the part of the solution associated with this preconditioner
976  this->get_block_vector(0,block_z_with_size_of_full_z,temp);
977 
978  // Doc time for setup of preconditioner
979  double t_prec_application_end=TimingHelpers::timer();
980 
981  // Update the preconditioner application time total
982  t_prec_application_total+=(t_prec_application_end-
983  t_prec_application_start);
984 
985  // Calculate w=Jv_m where v_m=M^{-1}v
986  Matrix_pt->multiply(temp,w);
987  }
988  } // End of solve for w
989 
990  // Get a pointer to the values in w
991  double* w_pt=w.values_pt();
992 
993  // Loop over the columns of the (transposed) Hessenberg matrix
994  for (unsigned k=0; k<=iter_restart; k++)
995  {
996  // Initialise the entry in the upper Hessenberg matrix
997  H[iter_restart][k]=0.0;
998 
999  // Get the pointer to the values of the k-th basis vector
1000  double* vk_pt=v[k].values_pt();
1001 
1002  // Update H with the dot product of w and v[k]
1003  H[iter_restart][k]+=w.dot(v[k]);
1004 
1005  // Loop over the entries in w and v[k] again
1006  for (unsigned i=0; i<n_dof; i++)
1007  {
1008  // Now update the values in w
1009  w_pt[i]-=H[iter_restart][k]*vk_pt[i];
1010  }
1011  } // for (unsigned k=0;k<=iter_restart;k++)
1012 
1013  // Calculate the (iter_restart+1,iter_restart)-th entry in the upper
1014  // Hessenberg matrix (remember, H is actually the transpose of the
1015  // proper upper Hessenberg matrix)
1016  H[iter_restart][iter_restart+1]=w.norm();
1017 
1018  // Build the (iter_restart+1)-th basis vector by copying w
1019  v[iter_restart+1]=w;
1020 
1021  // Now rescale the basis vector
1022  v[iter_restart+1]/=H[iter_restart][iter_restart+1];
1023 
1024  // Loop over the columns of the transposed upper Hessenberg matrix
1025  for (unsigned k=0; k<iter_restart; k++)
1026  {
1027  // Apply a Givens rotation to entries of H
1028  apply_plane_rotation(H[iter_restart][k],H[iter_restart][k+1],cs[k],sn[k]);
1029  }
1030 
1031  // Now generate the cs and sn entries for a plane rotation
1032  generate_plane_rotation(H[iter_restart][iter_restart],
1033  H[iter_restart][iter_restart+1],
1034  cs[iter_restart],
1035  sn[iter_restart]);
1036 
1037  // Use the newly generated entries in cs and sn to apply a Givens
1038  // rotation to those same entries
1039  apply_plane_rotation(H[iter_restart][iter_restart],
1040  H[iter_restart][iter_restart+1],
1041  cs[iter_restart],
1042  sn[iter_restart]);
1043 
1044  // Now apply the Givens rotation to the entries in s
1045  apply_plane_rotation(s[iter_restart],s[iter_restart+1],
1046  cs[iter_restart],sn[iter_restart]);
1047 
1048  // Compute the current residual
1049  beta=std::fabs(s[iter_restart+1]);
1050 
1051  // Compute the relative residual norm
1052  resid=beta/normb;
1053 
1054  // If required, document the convergence history to screen or file
1055  if (Doc_convergence_history)
1056  {
1057  // If an output file has not been provided
1058  if (!Output_file_stream.is_open())
1059  {
1060  // Output the convergence history to screen
1061  oomph_info << iter << " "<< resid << std::endl;
1062  }
1063  // An output file has been provided so output to that
1064  else
1065  {
1066  // Output the convergence history to file
1067  Output_file_stream << iter << " " << resid <<std::endl;
1068  }
1069  } // if (Doc_convergence_history)
1070 
1071  // If the required tolerance was met
1072  if (resid<Tolerance)
1073  {
1074  // Storage for the global-sized solution vector. Strictly speaking
1075  // we could actually just use the vector, solution but this can be
1076  // done after we know we've implemented everything correctly...
1077  DoubleVector block_z_with_size_of_full_z(rhs.distribution_pt(),0.0);
1078 
1079  // Take the current solution, reorder the entries appropriately
1080  // and stick them into the global sized vector
1081  this->return_block_vector(0,block_z[0],block_z_with_size_of_full_z);
1082 
1083  // This will update block_z[0] and won't touch the global-size vector
1084  // block_x_with_size_of_full_x. We're in charge of reordering the
1085  // entries and putting it in solution
1086  update(iter_restart,H,s,v,block_z_with_size_of_full_z,block_z[0]);
1087 
1088  // Now go into block_z[0], reorder the entries in the appropriate manner
1089  // and put them into the vector, solution
1090  this->return_block_vector(0,block_z[0],solution);
1091 
1092  // If the user wishes GMRES to document the convergence
1093  if (Doc_time)
1094  {
1095  // Document the convergence to screen
1096  oomph_info << "\nGMRES block preconditioner converged (1). Normalised "
1097  << "residual norm: " << resid
1098  << "\nNumber of iterations to convergence: " << iter << "\n"
1099  << std::endl;
1100  }
1101 
1102  // Get the current time
1103  double t_end=TimingHelpers::timer();
1104 
1105  // Calculate the time difference, i.e. the time taken for the whole solve
1106  Solution_time=t_end-t_start;
1107 
1108  // Storage the iteration count
1109  Iterations=iter;
1110 
1111  // If the user wishes GMRES to document the convergence
1112  if (Doc_time)
1113  {
1114  // Document the convergence to screen
1115  oomph_info << "Time for all preconditioner applications [sec]: "
1116  << t_prec_application_total
1117  << "\nTotal time for solve with GMRES block preconditioner "
1118  << "[sec]: " << Solution_time << std::endl;
1119  }
1120 
1121  // We're done now; finish here
1122  return;
1123  } // if (resid<Tolerance)
1124  } // for (iter_restart=0;iter_restart<n_dof&&iter<=Max_iter;...
1125 
1126  // Update
1127  if (iter_restart>0)
1128  {
1129  // Storage for the global-sized solution vector
1130  DoubleVector block_z_with_size_of_full_z(rhs.distribution_pt(),0.0);
1131 
1132  // Take the current solution, reorder the entries appropriately
1133  // and stick them into the global sized vector
1134  this->return_block_vector(0,block_z[0],block_z_with_size_of_full_z);
1135 
1136  // Update the (reordered block) solution vector
1137  update((iter_restart-1),H,s,v,block_z_with_size_of_full_z,block_z[0]);
1138 
1139  // Now go into block_z[0], reorder the entries in the appropriate manner
1140  // and put them into the vector, solution
1141  this->return_block_vector(0,block_z[0],solution);
1142  }
1143 
1144  // Solve Mr=(b-Jx) for r
1145  {
1146  // Temporary vector to store b-Jx
1147  DoubleVector temp(block_r[0].distribution_pt(),0.0);
1148 
1149  // Calculate temp=b-Jx
1150  Matrix_pt->residual(block_z[0],block_r[0],temp);
1151 
1152  // If we're using LHS preconditioning
1153  if (Preconditioner_LHS)
1154  {
1155  // Initialise timer
1156  double t_prec_application_start=TimingHelpers::timer();
1157 
1158  // This is pretty inefficient but there is no other way to do this with
1159  // block sub preconditioners as far as I can tell because they demand
1160  // to have the full r and z vectors...
1161  DoubleVector block_z_with_size_of_full_z(rhs.distribution_pt());
1162  DoubleVector r_updated(rhs.distribution_pt());
1163 
1164  // Reorder the entries of temp and put them into the global-sized vector
1165  this->return_block_vector(0,temp,r_updated);
1166 
1167  // Solve on the global-sized vectors
1168  Navier_stokes_subsidiary_preconditioner_pt->
1169  preconditioner_solve(r_updated,block_z_with_size_of_full_z);
1170 
1171  // Now grab the part of the solution associated with this preconditioner
1172  this->get_block_vector(0,block_z_with_size_of_full_z,r);
1173 
1174  // Doc time for setup of preconditioner
1175  double t_prec_application_end=TimingHelpers::timer();
1176 
1177  // Update the preconditioner application time total
1178  t_prec_application_total+=(t_prec_application_end-
1179  t_prec_application_start);
1180  }
1181  } // End of solve Mr=(b-Jx) for r
1182 
1183  // Compute the current residual norm
1184  beta=r.norm();
1185 
1186  // if relative residual within tolerance
1187  resid=beta/normb;
1188  if (resid<Tolerance)
1189  {
1190  // Now store the result in the global solution vector
1191  this->return_block_vector(0,block_z[0],solution);
1192 
1193  if (Doc_time)
1194  {
1195  oomph_info << "\nGMRES block preconditioner converged (2). Normalised "
1196  << "residual norm: " << resid
1197  << "\nNumber of iterations to convergence: " << iter << "\n"
1198  << std::endl;
1199  }
1200 
1201  // Doc time for solver
1202  double t_end=TimingHelpers::timer();
1203  Solution_time=t_end-t_start;
1204 
1205  Iterations=iter;
1206 
1207  if (Doc_time)
1208  {
1209  oomph_info << "Time for all preconditioner applications [sec]: "
1210  << t_prec_application_total
1211  << "\nTotal time for solve with GMRES block preconditioner "
1212  << "[sec]: " << Solution_time << std::endl;
1213  }
1214  return;
1215  }
1216  }
1217 
1218 
1219  // otherwise GMRES failed convergence
1220  oomph_info << "\nGMRES block preconditioner did not converge to required "
1221  << "tolerance! \nReturning with normalised residual norm: "
1222  << resid << "\nafter " << Max_iter << " iterations.\n"
1223  << std::endl;
1224 
1225  if (Throw_error_after_max_iter)
1226  {
1227  std::ostringstream error_message_stream;
1228  error_message_stream << "Solver failed to converge and you requested an "
1229  << "error on convergence failures.";
1230  throw OomphLibError(error_message_stream.str(),
1231  OOMPH_EXCEPTION_LOCATION,
1232  OOMPH_CURRENT_FUNCTION);
1233  }
1234  } // End of solve_helper
