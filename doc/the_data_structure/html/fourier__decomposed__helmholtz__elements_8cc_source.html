50 //========================================================================
51 /// Legendre polynomials depending on one parameter
52 //========================================================================
53  double plgndr1(const unsigned& n, const double& x)
54  {
55  unsigned i;
56  double pmm,pmm1;
57  double pmm2=0;
58 
59 #ifdef PARANOID
60  // Shout if things went wrong
61  if (std::fabs(x) > 1.0)
62  {
63  std::ostringstream error_stream;
64  error_stream << "Bad arguments in routine plgndr1: x=" << x
65  << " but should be less than 1 in absolute value.\n";
66  throw OomphLibError(error_stream.str(),
67  OOMPH_CURRENT_FUNCTION,
68  OOMPH_EXCEPTION_LOCATION);
69  }
70 #endif
71 
72  // Compute pmm : if l=m it's finished
73  pmm=1.0;
74  if (n == 0)
75  {
76  return pmm;
77  }
78 
79  pmm1=x*1.0;
80  if (n == 1)
81  {
82  return pmm1;
83  }
84  else
85  {
86  for (i=2;i<=n;i++)
87  {
88  pmm2=(x*(2*i-1)*pmm1-(i-1)*pmm)/i;
89  pmm=pmm1;
90  pmm1=pmm2;
91  }
92  return pmm2;
93  }
94 
95  }//end of plgndr1

99 //========================================================================
100 // Legendre polynomials depending on two parameters
101 //========================================================================
102  double plgndr2(const unsigned& l, const unsigned& m, const double& x)
103  {
104  unsigned i,ll;
105  double fact,pmm,pmmp1,somx2;
106  double pll=0.0;
107 
108 #ifdef PARANOID
109  // Shout if things went wrong
110  if (std::fabs(x) > 1.0)
111  {
112  std::ostringstream error_stream;
113  error_stream << "Bad arguments in routine plgndr2: x=" << x
114  << " but should be less than 1 in absolute value.\n";
115  throw OomphLibError(error_stream.str(),
116  OOMPH_CURRENT_FUNCTION,
117  OOMPH_EXCEPTION_LOCATION);
118  }
119 #endif
120 
121  // This one is easy...
122  if (m > l)
123  {
124  return 0.0;
125  }
126 
127  // Compute pmm : if l=m it's finished
128  pmm=1.0;
129  if (m > 0)
130  {
131  somx2=sqrt((1.0-x)*(1.0+x));
132  fact=1.0;
133  for (i=1;i<=m;i++)
134  {
135  pmm *= -fact*somx2;
136  fact += 2.0;
137  }
138  }
139  if (l == m) return pmm;
140 
141  // Compute pmmp1 : if l=m+1 it's finished
142  else
143  {
144  pmmp1=x*(2*m+1)*pmm;
145  if (l == (m+1))
146  {
147  return pmmp1;
148  }
149  // Compute pll : if l>m+1 it's finished
150  else
151  {
152  for (ll=m+2;ll<=l;ll++)
153  {
154  pll=(x*(2*ll-1)*pmmp1-(ll+m-1)*pmm)/(ll-m);
155  pmm=pmmp1;
156  pmmp1=pll;
157  }
158  return pll;
159  }
160  }
161  }//end of plgndr2

180 //======================================================================
181 /// Compute element residual Vector and/or element Jacobian matrix
182 ///
183 /// flag=1: compute both
184 /// flag=0: compute only residual Vector
185 ///
186 /// Pure version without hanging nodes
187 //======================================================================
188  void FourierDecomposedHelmholtzEquations::
189  fill_in_generic_residual_contribution_fourier_decomposed_helmholtz(
190  Vector<double> &residuals,
191  DenseMatrix<double> &jacobian,
192  const unsigned& flag)
193  {
194  //Find out how many nodes there are
195  const unsigned n_node = nnode();
196 
197  //Set up memory for the shape and test functions
198  Shape psi(n_node), test(n_node);
199  DShape dpsidx(n_node,2), dtestdx(n_node,2);
200 
201  //Set the value of n_intpt
202  const unsigned n_intpt = integral_pt()->nweight();
203 
204  //Integers to store the local equation and unknown numbers
205  int local_eqn_real=0, local_unknown_real=0;
206  int local_eqn_imag=0, local_unknown_imag=0;
207 
208  //Loop over the integration points
209  for(unsigned ipt=0;ipt<n_intpt;ipt++)
210  {
211  //Get the integral weight
212  double w = integral_pt()->weight(ipt);
213 
214  //Call the derivatives of the shape and test functions
215  double J = dshape_and_dtest_eulerian_at_knot_fourier_decomposed_helmholtz
216  (ipt,psi,dpsidx,test,dtestdx);
217 
218  //Premultiply the weights and the Jacobian
219  double W = w*J;
220 
221  //Calculate local values of unknown
222  //Allocate and initialise to zero
223  std::complex<double> interpolated_u(0.0,0.0);
224  Vector<double> interpolated_x(2,0.0);
225  Vector< std::complex<double> > interpolated_dudx(2);
226 
227  //Calculate function value and derivatives:
228  //-----------------------------------------
229  // Loop over nodes
230  for(unsigned l=0;l<n_node;l++)
231  {
232  // Loop over directions
233  for(unsigned j=0;j<2;j++)
234  {
235  interpolated_x[j] += raw_nodal_position(l,j)*psi(l);
236  }
237 
238  //Get the nodal value of the helmholtz unknown
239  const std::complex<double> u_value(
240  raw_nodal_value(l,u_index_fourier_decomposed_helmholtz().real()),
241  raw_nodal_value(l,u_index_fourier_decomposed_helmholtz().imag()));
242 
243  //Add to the interpolated value
244  interpolated_u += u_value*psi(l);
245 
246  // Loop over directions
247  for(unsigned j=0;j<2;j++)
248  {
249  interpolated_dudx[j] += u_value*dpsidx(l,j);
250  }
251  }
252 
253  //Get source function
254  //-------------------
255  std::complex<double> source(0.0,0.0);
256  get_source_fourier_decomposed_helmholtz(ipt,interpolated_x,source);
257 
258  double r = interpolated_x[0];
259  double rr = r*r;
260  double n = (double)fourier_wavenumber();
261  double n_squared = n*n;
262 
263  // Assemble residuals and Jacobian
264  //--------------------------------
265 
266  // Loop over the test functions
267  for(unsigned l=0;l<n_node;l++)
268  {
269 
270  // first, compute the real part contribution
271  //-------------------------------------------
272 
273  //Get the local equation
274  local_eqn_real =
275  nodal_local_eqn(l,u_index_fourier_decomposed_helmholtz().real());
276 
277  /*IF it's not a boundary condition*/
278  if(local_eqn_real >= 0)
279  {
280  // Add body force/source term and Helmholtz bit
281 
282  residuals[local_eqn_real] +=
283  (source.real()-((-n_squared/rr)+k_squared())*
284  interpolated_u.real())*test(l)*r*W;
285 
286  // The Helmholtz bit itself
287  for(unsigned k=0;k<2;k++)
288  {
289  residuals[local_eqn_real] +=
290  interpolated_dudx[k].real()*dtestdx(l,k)*r*W;
291  }
292 
293  // Calculate the jacobian
294  //-----------------------
295  if(flag)
296  {
297  //Loop over the velocity shape functions again
298  for(unsigned l2=0;l2<n_node;l2++)
299  {
300  local_unknown_real =
301  nodal_local_eqn(l2,u_index_fourier_decomposed_helmholtz().real());
302 
303  //If at a non-zero degree of freedom add in the entry
304  if(local_unknown_real >= 0)
305  {
306  //Add contribution to elemental Matrix
307  for(unsigned i=0;i<2;i++)
308  {
309  jacobian(local_eqn_real,local_unknown_real)
310  += dpsidx(l2,i)*dtestdx(l,i)*r*W;
311  }
312 
313  // Add the helmholtz contribution
314  jacobian(local_eqn_real,local_unknown_real)
315  -= ((-n_squared/rr)+k_squared())*psi(l2)*test(l)*r*W;
316 
317  }//end of local_unknown
318  }
319  }
320  }
321 
322  // Second, compute the imaginary part contribution
323  //------------------------------------------------
324 
325  //Get the local equation
326  local_eqn_imag = nodal_local_eqn
327  (l,u_index_fourier_decomposed_helmholtz().imag());
328 
329  /*IF it's not a boundary condition*/
330  if(local_eqn_imag >= 0)
331  {
332  // Add body force/source term and Helmholtz bit
333  residuals[local_eqn_imag] +=
334  (source.imag() - ( (-n_squared/rr) + k_squared() ) *
335  interpolated_u.imag() )*test(l)*r*W;
336 
337  // The Helmholtz bit itself
338  for(unsigned k=0;k<2;k++)
339  {
340  residuals[local_eqn_imag] += interpolated_dudx[k].imag()*
341  dtestdx(l,k)*r*W;
342  }
343 
344  // Calculate the jacobian
345  //-----------------------
346  if(flag)
347  {
348  //Loop over the velocity shape functions again
349  for(unsigned l2=0;l2<n_node;l2++)
350  {
351  local_unknown_imag = nodal_local_eqn
352  (l2,u_index_fourier_decomposed_helmholtz().imag());
353 
354  //If at a non-zero degree of freedom add in the entry
355  if(local_unknown_imag >= 0)
356  {
357  //Add contribution to Elemental Matrix
358  for(unsigned i=0;i<2;i++)
359  {
360  jacobian(local_eqn_imag,local_unknown_imag)
361  += dpsidx(l2,i)*dtestdx(l,i)*r*W;
362  }
363  // Add the helmholtz contribution
364  jacobian(local_eqn_imag,local_unknown_imag)
365  -= ((-n_squared/rr)+k_squared())*psi(l2)*test(l)*r*W;
366  }
367  }
368  }
369  }
370 
371  }
372  } // End of loop over integration points
373  }
