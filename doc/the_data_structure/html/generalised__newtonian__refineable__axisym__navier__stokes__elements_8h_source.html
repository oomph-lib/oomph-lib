80  /// \short Get 'flux' for Z2 error recovery: Upper triangular entries
81  /// in strain rate tensor.
82  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)
83  {
84  //Specify the number of velocity dimensions
85  unsigned DIM=3;
86 
87 #ifdef PARANOID
88  unsigned num_entries=DIM+((DIM*DIM)-DIM)/2;
89  if (flux.size() < num_entries)
90  {
91  std::ostringstream error_message;
92  error_message << "The flux vector has the wrong number of entries, "
93  << flux.size() << ", whereas it should be "
94  << num_entries << std::endl;
95  throw OomphLibError(
96  error_message.str(),
97  OOMPH_CURRENT_FUNCTION,
98  OOMPH_EXCEPTION_LOCATION);
99  }

103  // Get strain rate matrix
104  DenseMatrix<double> strainrate(DIM);
105  this->strain_rate(s,strainrate);
106 
107  // Pack into flux Vector
108  unsigned icount=0;
109 
110  // Start with diagonal terms
111  for(unsigned i=0;i<DIM;i++)
112  {
113  flux[icount]=strainrate(i,i);
114  icount++;
115  }
116 
117  //Off diagonals row by row
118  for(unsigned i=0;i<DIM;i++)
119  {
120  for(unsigned j=i+1;j<DIM;j++)
121  {
122  flux[icount]=strainrate(i,j);
123  icount++;
124  }

168  /// \short Compute the derivatives of the i-th component of
169  /// velocity at point s with respect
170  /// to all data that can affect its value. In addition, return the global
171  /// equation numbers corresponding to the data.
172  /// Overload the non-refineable version to take account of hanging node
173  /// information
174  void dinterpolated_u_axi_nst_ddata(const Vector<double> &s,
175  const unsigned &i,
176  Vector<double> &du_ddata,
177  Vector<unsigned> &global_eqn_number)
178  {
179  //Find number of nodes
180  unsigned n_node = this->nnode();
181  //Local shape function
182  Shape psi(n_node);
183  //Find values of shape function at the given local coordinate
184  this->shape(s,psi);
185 
186  //Find the index at which the velocity component is stored
187  const unsigned u_nodal_index = this->u_index_axi_nst(i);
188 
189  //Storage for hang info pointer
190  HangInfo* hang_info_pt=0;
191  //Storage for global equation
192  int global_eqn = 0;
193 
194  //Find the number of dofs associated with interpolated u
195  unsigned n_u_dof=0;
196  for(unsigned l=0;l<n_node;l++)
197  {
198  unsigned n_master = 1;
199 
200  //Local bool (is the node hanging)
201  bool is_node_hanging = this->node_pt(l)->is_hanging();
202 
203  //If the node is hanging, get the number of master nodes
204  if(is_node_hanging)
205  {
206  hang_info_pt = this->node_pt(l)->hanging_pt();
207  n_master = hang_info_pt->nmaster();
208  }
209  //Otherwise there is just one master node, the node itself
210  else
211  {
212  n_master = 1;
213  }
214 
215  //Loop over the master nodes
216  for(unsigned m=0;m<n_master;m++)
217  {
218  //Get the equation number
219  if(is_node_hanging)
220  {
221  //Get the equation number from the master node
222  global_eqn = hang_info_pt->master_node_pt(m)->
223  eqn_number(u_nodal_index);
224  }
225  else
226  {
227  // Global equation number
228  global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
229  }
230 
231  //If it's positive add to the count
232  if(global_eqn >= 0) {++n_u_dof;}
233  }
234  }
235 
236  //Now resize the storage schemes
237  du_ddata.resize(n_u_dof,0.0);
238  global_eqn_number.resize(n_u_dof,0);
239 
240  //Loop over th nodes again and set the derivatives
241  unsigned count=0;
242  //Loop over the local nodes and sum
243  for(unsigned l=0;l<n_node;l++)
244  {
245  unsigned n_master = 1;
246  double hang_weight = 1.0;
247 
248  //Local bool (is the node hanging)
249  bool is_node_hanging = this->node_pt(l)->is_hanging();
250 
251  //If the node is hanging, get the number of master nodes
252  if(is_node_hanging)
253  {
254  hang_info_pt = this->node_pt(l)->hanging_pt();
255  n_master = hang_info_pt->nmaster();
256  }
257  //Otherwise there is just one master node, the node itself
258  else
259  {
260  n_master = 1;
261  }
262 
263  //Loop over the master nodes
264  for(unsigned m=0;m<n_master;m++)
265  {
266  //If the node is hanging get weight from master node
267  if(is_node_hanging)
268  {
269  //Get the hang weight from the master node
270  hang_weight = hang_info_pt->master_weight(m);
271  }
272  else
273  {
274  // Node contributes with full weight
275  hang_weight = 1.0;
276  }
277 
278  //Get the equation number
279  if(is_node_hanging)
280  {
281  //Get the equation number from the master node
282  global_eqn = hang_info_pt->master_node_pt(m)->
283  eqn_number(u_nodal_index);
284  }
285  else
286  {
287  // Local equation number
288  global_eqn = this->node_pt(l)->eqn_number(u_nodal_index);
289  }
290 
291  if (global_eqn >= 0)
292  {
293  //Set the global equation number
294  global_eqn_number[count] = global_eqn;
295  //Set the derivative with respect to the unknown
296  du_ddata[count] = psi[l]*hang_weight;
297  //Increase the counter
298  ++count;
299  }
300  }
301  }
302  }

577  /// \short The velocity nodes are the same as the geometric nodes. The
578  /// pressure nodes must be calculated by using the same methods as
579  /// the geometric nodes, but by recalling that there are only two pressure
580  /// nodes per edge.
581  Node* get_interpolating_node_at_local_coordinate(const Vector<double> &s,
582  const int &n_value)
583  {
584  int DIM=3;
585  //If we are calculating pressure nodes
586  if(n_value==static_cast<int>(DIM))
587  {
588  //Storage for the index of the pressure node
589  unsigned total_index=0;
590  //The number of nodes along each 1d edge is 2.
591  unsigned NNODE_1D = 2;
592  //Storage for the index along each boundary
593  //Note that it's only a 2D spatial element
594  Vector<int> index(2);
595  //Loop over the coordinates
596  for(unsigned i=0;i<2;i++)
597  {
598  //If we are at the lower limit, the index is zero
599  if(s[i]==-1.0)
600  {
601  index[i] = 0;
602  }
603  //If we are at the upper limit, the index is the number of nodes minus 1
604  else if(s[i] == 1.0)
605  {
606  index[i] = NNODE_1D-1;
607  }
608  //Otherwise, we have to calculate the index in general
609  else
610  {
611  //For uniformly spaced nodes the 0th node number would be
612  double float_index = 0.5*(1.0 + s[i])*(NNODE_1D-1);
613  index[i] = int(float_index);
614  //What is the excess. This should be safe because the
615  //taking the integer part rounds down
616  double excess = float_index - index[i];
617  //If the excess is bigger than our tolerance there is no node,
618  //return null
619  if((excess > FiniteElement::Node_location_tolerance) && (
620  (1.0 - excess) > FiniteElement::Node_location_tolerance))
621  {
622  return 0;
623  }
624  }
625  ///Construct the general pressure index from the components.
626  total_index += index[i]*
627  static_cast<unsigned>(pow(static_cast<float>(NNODE_1D),
628  static_cast<int>(i)));
629  }
630  //If we've got here we have a node, so let's return a pointer to it
631  return this->node_pt(this->Pconv[total_index]);
632  }
633  //Otherwise velocity nodes are the same as pressure nodes
634  else
635  {
636  return this->get_node_at_local_coordinate(s);
637  }
638  }
