80  /// \short Get 'flux' for Z2 error recovery: Upper triangular entries
81  /// in strain rate tensor.
82  void get_Z2_flux(const Vector<double>& s, Vector<double>& flux)
83  {
84  //Specify the number of velocity dimensions
85  unsigned DIM=3;
86 #ifdef PARANOID
87  unsigned num_entries=DIM+((DIM*DIM)-DIM)/2;
88  if (flux.size() < num_entries)
89  {
90  std::ostringstream error_message;
91  error_message << "The flux vector is too small, size "
92  << flux.size() << ", whereas it should be "
93  << num_entries << std::endl;
94  throw OomphLibError(
95  error_message.str(),
96  OOMPH_CURRENT_FUNCTION,
97  OOMPH_EXCEPTION_LOCATION);
98  }
99 #endif
100  // Get strain rate matrix
101  DenseMatrix<double> strainrate(DIM);
102  this->strain_rate(s,strainrate);
103 
104  // Pack into flux Vector
105  unsigned icount=0;
106 
107  // Start with diagonal terms
108  for(unsigned i=0;i<DIM;i++)
109  {
110  flux[icount]=strainrate(i,i);
111  icount++;
112  }
113 
114  //Off diagonals row by row
115  for(unsigned i=0;i<DIM;i++)
116  {
117  for(unsigned j=i+1;j<DIM;j++)
118  {
119  flux[icount]=strainrate(i,j);
120  icount++;
121  }
122  }
123  }

308 /// \short Get the function value u in Vector.
309 /// Note: Given the generality of the interface (this function
310 /// is usually called from black-box documentation or interpolation routines),
311 /// the values Vector sets its own size in here.
312  void get_interpolated_values(const unsigned& t, const Vector<double>&s,
313  Vector<double>& values)
314  {
315  // Set size of Vector: u,w,v,p
316  values.resize(4);
317 
318  // Initialise
319  for(unsigned i=0;i<4;i++) {values[i]=0.0;}
320 
321  //Find out how many nodes there are
322  unsigned n_node = nnode();
323 
324  // Shape functions
325  Shape psif(n_node);
326  shape(s,psif);
327 
328  //Calculate velocities: values[0],...
329  for(unsigned i=0;i<3;i++)
330  {

332  const unsigned u_nodal_index = u_index_spherical_nst(i);

384  /// \short The velocity nodes are the same as the geometric nodes. The
385  /// pressure nodes must be calculated by using the same methods as
386  /// the geometric nodes, but by recalling that there are only two pressure
387  /// nodes per edge.
388  Node* get_interpolating_node_at_local_coordinate(const Vector<double> &s,
389  const int &n_value)
390  {
391  //If we are calculating pressure nodes
392  if(n_value==3)
393  {
394  //Storage for the index of the pressure node
395  unsigned total_index=0;
396  //The number of nodes along each 1d edge is 2.
397  unsigned NNODE_1D = 2;
398  //Storage for the index along each boundary
399  //Note that it's only a 2D spatial element
400  Vector<int> index(2);
401  //Loop over the coordinates
402  for(unsigned i=0;i<2;i++)
403  {
404  //If we are at the lower limit, the index is zero
405  if(s[i]==-1.0)
406  {
407  index[i] = 0;
408  }
409  //If we are at the upper limit, the index is the number of nodes minus 1
410  else if(s[i] == 1.0)
411  {
412  index[i] = NNODE_1D-1;
413  }
414  //Otherwise, we have to calculate the index in general
415  else
416  {
417  //For uniformly spaced nodes the 0th node number would be
418  double float_index = 0.5*(1.0 + s[i])*(NNODE_1D-1);
419  index[i] = int(float_index);
420  //What is the excess. This should be safe because the
421  //taking the integer part rounds down
422  double excess = float_index - index[i];
423  //If the excess is bigger than our tolerance there is no node,
424  //return null
425  if((excess > FiniteElement::Node_location_tolerance) && (
426  (1.0 - excess) > FiniteElement::Node_location_tolerance))
427  {
428  return 0;
429  }
430  }
431  ///Construct the general pressure index from the components.
432  total_index += index[i]*
433  static_cast<unsigned>(pow(static_cast<float>(NNODE_1D),
434  static_cast<int>(i)));
435  }
436  //If we've got here we have a node, so let's return a pointer to it
437  return this->node_pt(this->Pconv[total_index]);
438  }
439  //Otherwise velocity nodes are the same as pressure nodes
440  else
441  {
442  return this->get_node_at_local_coordinate(s);
443  }
444  }

636  values.resize(3,0.0);
637

639  for(unsigned i=0;i<3;i++) {values[i] = interpolated_u_spherical_nst(s,i);}

642  /// \short Get all function values [u,v..,p] at previous timestep t
643  /// (t=0: present; t>0: previous timestep).
644  /// \n
645  /// Note: Given the generality of the interface (this function is
646  /// usually called from black-box documentation or interpolation routines),
647  /// the values Vector sets its own size in here.
648  /// \n
649  /// Note: No pressure history is kept, so pressure is always
650  /// the current value.
651  void get_interpolated_values(const unsigned& t, const Vector<double>&s,
652  Vector<double>& values)
653  {
654  // Set size of Vector: u,w,v
655  values.resize(3);
656 
657  // Initialise
658  for(unsigned i=0;i<3;i++) {values[i]=0.0;}
659

661  unsigned n_node = nnode();
662 
663  // Shape functions
664  Shape psif(n_node);
665  shape(s,psif);
666 
667  //Calculate velocities: values[0],...

669  {
670  //Get the local index at which the i-th velocity is stored
671  unsigned u_local_index = u_index_spherical_nst(i);
672  for(unsigned l=0;l<n_node;l++)

675  }
676  }
677  }

680  /// that are not interpolated by all nodes. Empty
681  void further_setup_hanging_nodes() {}

684  /// pressure dofs from father element: Make sure pressure values and
685  /// dp/ds agree between fathers and sons at the midpoints of the son
686  /// elements.
687  void further_build()
688  {

691 
692  using namespace QuadTreeNames;
693 
694  // What type of son am I? Ask my quadtree representation...
695  int son_type=quadtree_pt()->son_type();
696 
697  // Pointer to my father (in element impersonation)
698  RefineableElement* father_el_pt=
699  quadtree_pt()->father_pt()->object_pt();
700 
701  Vector<double> s_father(2);
702 
703  // Son midpoint is located at the following coordinates in father element:
704 
705  // South west son
706  if (son_type==SW)
707  {
708  s_father[0]=-0.5;
709  s_father[1]=-0.5;
710  }
711  // South east son
712  else if (son_type==SE)
713  {
714  s_father[0]= 0.5;
715  s_father[1]=-0.5;
716  }
717  // North east son
718  else if (son_type==NE)
719  {
720  s_father[0]=0.5;
721  s_father[1]=0.5;
722  }
723 
724  // North west son
725  else if (son_type==NW)
726  {
727  s_father[0]=-0.5;
728  s_father[1]= 0.5;
729  }
730 
731  // Pressure value in father element
732  RefineableQSphericalCrouzeixRaviartElement* cast_father_el_pt=
733  dynamic_cast<RefineableQSphericalCrouzeixRaviartElement*>(father_el_pt);
734  double press=cast_father_el_pt->interpolated_p_spherical_nst(s_father);
735 
736  // Pressure value gets copied straight into internal dof:
737  internal_data_pt(P_spherical_nst_internal_index)->set_value(0,press);
738 
739 
740  // The slopes get copied from father
741  internal_data_pt(P_spherical_nst_internal_index)->
742  set_value(1,0.5*cast_father_el_pt
743  ->internal_data_pt(P_spherical_nst_internal_index)
744  ->value(1));
745 
746  internal_data_pt(P_spherical_nst_internal_index)->
747  set_value(2,0.5*cast_father_el_pt
748  ->internal_data_pt(P_spherical_nst_internal_index)
749  ->value(2));
750 }
