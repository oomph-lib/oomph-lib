41  TetMeshFacetedClosedSurfaceForRemesh::TetMeshFacetedClosedSurfaceForRemesh(
42  Vector<Node*> const &vertex_node_pt,
43  Vector<Vector<unsigned> > const &facet_connectivity,
44  Vector<unsigned> const &facet_boundary_id) : TetMeshFacetedClosedSurface()
45  {
46  //Create the vertices
47  unsigned n_vertex = vertex_node_pt.size();
48  Vertex_pt.resize(n_vertex);
49  for(unsigned v=0;v<n_vertex;++v)
50  {
51  Vertex_pt[v] = new TetMeshVertex(vertex_node_pt[v]);
52  }
53 
54  //Create the facets
55  unsigned n_facet = facet_connectivity.size();
56  Facet_pt.resize(n_facet);
57  for(unsigned f=0;f<n_facet;++f)
58  {
59  unsigned n_vertex_on_facet = facet_connectivity[f].size();
60  Facet_pt[f] = new TetMeshFacet(n_vertex_on_facet);
61  for(unsigned i=0;i<n_vertex_on_facet;++i)
62  {
63  Facet_pt[f]->set_vertex_pt(i,Vertex_pt[facet_connectivity[f][i]]);
64  }
65  //Add in the boundary id
66  Facet_pt[f]->set_one_based_boundary_id(facet_boundary_id[f]);
67  }

91 /// Setup lookup schemes which establish which elements are located
92 /// next to which boundaries (Doc to outfile if it's open).
93 //================================================================
94 void TetMeshBase::setup_boundary_element_info(std::ostream &outfile)
95 {
96 
97  //Should we document the output here
98  bool doc = false;
99 
100  if (outfile) doc = true;
101 
102  // Number of boundaries
103  unsigned nbound=nboundary();
104 
105  // Wipe/allocate storage for arrays
106  Boundary_element_pt.clear();
107  Face_index_at_boundary.clear();
108  Boundary_element_pt.resize(nbound);
109  Face_index_at_boundary.resize(nbound);
110  Lookup_for_elements_next_boundary_is_setup=false;
111 
112  // Temporary vector of vectors of pointers to elements on the boundaries:
113  // This is a vector to ensure UNIQUE ordering in all processors
114  Vector<Vector<FiniteElement*> > vector_of_boundary_element_pt;
115  vector_of_boundary_element_pt.resize(nbound);
116 
117  // Matrix map for working out the fixed face for elements on boundary
118  MapMatrixMixed<unsigned,FiniteElement*, int >
119  face_identifier;
120 
121  // Loop over elements
122  //-------------------
123  unsigned nel=nelement();
124 
125 
126  // Get pointer to vector of boundaries that the
127  // node lives on
128  Vector<std::set<unsigned>*> boundaries_pt(4,0);
129 
130  for (unsigned e=0;e<nel;e++)
131  {
132  // Get pointer to element
133  FiniteElement* fe_pt=finite_element_pt(e);
134 
135  if (doc) outfile << "Element: " << e << " " << fe_pt << std::endl;
136 
137  // Only include 3D elements! Some meshes contain interface elements too.
138  if (fe_pt->dim()==3)
139  {
140  // Loop over the element's nodes and find out which boundaries they're on
141  // ----------------------------------------------------------------------
142  //We need only loop over the corner nodes
143  for(unsigned i=0;i<4;i++)
144  {
145  fe_pt->node_pt(i)->get_boundaries_pt(boundaries_pt[i]);
146  }
147 
148  //Find the common boundaries of each face
149  Vector<std::set<unsigned> > face(4);
150 
151  // NOTE: Face indices defined in Telements.h
152 
153  //Face 3 connnects points 0, 1 and 2
154  if(boundaries_pt[0] && boundaries_pt[1] && boundaries_pt[2])
155  {
156  std::set<unsigned> aux;
157 
158  std::set_intersection(boundaries_pt[0]->begin(),boundaries_pt[0]->end(),
159  boundaries_pt[1]->begin(),boundaries_pt[1]->end(),
160  std::insert_iterator<std::set<unsigned> >(
161  aux,aux.begin()));
162 
163  std::set_intersection(aux.begin(),aux.end(),
164  boundaries_pt[2]->begin(),boundaries_pt[2]->end(),
165  std::insert_iterator<std::set<unsigned> >(
166  face[3],face[3].begin()));
167  }
168 
169  //Face 2 connects points 0, 1 and 3
170  if(boundaries_pt[0] && boundaries_pt[1] && boundaries_pt[3])
171  {
172  std::set<unsigned> aux;
173 
174  std::set_intersection(boundaries_pt[0]->begin(),boundaries_pt[0]->end(),
175  boundaries_pt[1]->begin(),boundaries_pt[1]->end(),
176  std::insert_iterator<std::set<unsigned> >(
177  aux,aux.begin()));
178 
179  std::set_intersection(aux.begin(),aux.end(),
180  boundaries_pt[3]->begin(),boundaries_pt[3]->end(),
181  std::insert_iterator<std::set<unsigned> >(
182  face[2],face[2].begin()));
183  }
184 
185  //Face 1 connects points 0, 2 and 3
186  if(boundaries_pt[0] && boundaries_pt[2] && boundaries_pt[3])
187  {
188  std::set<unsigned> aux;
189 
190  std::set_intersection(boundaries_pt[0]->begin(),boundaries_pt[0]->end(),
191  boundaries_pt[2]->begin(),boundaries_pt[2]->end(),
192  std::insert_iterator<std::set<unsigned> >(
193  aux,aux.begin()));
194 
195  std::set_intersection(aux.begin(),aux.end(),
196  boundaries_pt[3]->begin(),boundaries_pt[3]->end(),
197  std::insert_iterator<std::set<unsigned> >(
198  face[1],face[1].begin()));
199  }
200 
201  //Face 0 connects points 1, 2 and 3
202  if(boundaries_pt[1] && boundaries_pt[2] && boundaries_pt[3])
203  {
204  std::set<unsigned> aux;
205 
206  std::set_intersection(boundaries_pt[1]->begin(),boundaries_pt[1]->end(),
207  boundaries_pt[2]->begin(),boundaries_pt[2]->end(),
208  std::insert_iterator<std::set<unsigned> >(
209  aux,aux.begin()));
210 
211  std::set_intersection(aux.begin(),aux.end(),
212  boundaries_pt[3]->begin(),boundaries_pt[3]->end(),
213  std::insert_iterator<std::set<unsigned> >(
214  face[0],face[0].begin()));
215  }
216 
217 
218 
219  //We now know whether any faces lay on the boundaries
220  for(unsigned i=0;i<4;i++)
221  {
222  //How many boundaries are there
223  unsigned count = 0;
224 
225  //The number of the boundary
226  int boundary=-1;
227 
228  //Loop over all the members of the set and add to the count
229  //and set the boundary
230  for(std::set<unsigned>::iterator it=face[i].begin();
231  it!=face[i].end();++it)
232  {
233  ++count;
234  boundary = *it;
235  }
236 
237  //If we're on more than one boundary, this is weird, so die
238  if(count > 1)
239  {
240  std::ostringstream error_stream;
241  fe_pt->output(error_stream);
242  error_stream << "Face " << i << " is on " <<
243  count << " boundaries.\n";
244  error_stream << "This is rather strange.\n";
245  error_stream << "Your mesh may be too coarse or your tet mesh\n";
246  error_stream << "may be screwed up. I'm skipping the automated\n";
247  error_stream << "setup of the elements next to the boundaries\n";
248  error_stream << "lookup schemes.\n";
249  OomphLibWarning(
250  error_stream.str(),
251  OOMPH_CURRENT_FUNCTION,
252  OOMPH_EXCEPTION_LOCATION);
253  }
254 
255  //If we have a boundary then add this to the appropriate set
256  if(boundary >= 0)
257  {
258 
259  // Does the pointer already exits in the vector
260  Vector<FiniteElement*>::iterator b_el_it =
261  std::find(vector_of_boundary_element_pt[
262  static_cast<unsigned>(boundary)].begin(),
263  vector_of_boundary_element_pt[
264  static_cast<unsigned>(boundary)].end(),
265  fe_pt);
266 
267  //Only insert if we have not found it (i.e. got to the end)
268  if(b_el_it == vector_of_boundary_element_pt[
269  static_cast<unsigned>(boundary)].end())
270  {
271  vector_of_boundary_element_pt[static_cast<unsigned>(boundary)].
272  push_back(fe_pt);
273  }
274 
275  //Also set the fixed face
276  face_identifier(static_cast<unsigned>(boundary),fe_pt) = i;
277  }
278  }
279 
280  //Now we set the pointers to the boundary sets to zero
281  for(unsigned i=0;i<4;i++) {boundaries_pt[i] = 0;}
282 
283  }
284  }
285 
286  // Now copy everything across into permanent arrays
287  //-------------------------------------------------
288 
289  // Loop over boundaries
290  //---------------------
291  for (unsigned i=0;i<nbound;i++)
292  {
293  // Number of elements on this boundary (currently stored in a set)
294  unsigned nel=vector_of_boundary_element_pt[i].size();
295 
296  // Allocate storage for the coordinate identifiers
297  Face_index_at_boundary[i].resize(nel);
298 
299  unsigned e_count=0;
300  typedef Vector<FiniteElement*>::iterator IT;
301  for (IT it=vector_of_boundary_element_pt[i].begin();
302  it!=vector_of_boundary_element_pt[i].end();
303  it++)
304  {
305  // Recover pointer to element
306  FiniteElement* fe_pt=*it;
307 
308  // Add to permanent storage
309  Boundary_element_pt[i].push_back(fe_pt);
310 
311  Face_index_at_boundary[i][e_count] = face_identifier(i,fe_pt);
312 
313  // Increment counter
314  e_count++;
315  }
316  }
317 
318 
319 
320  // Doc?
321  //-----
322  if (doc)
323  {
324  // Loop over boundaries
325  for (unsigned i=0;i<nbound;i++)
326  {
327  unsigned nel=Boundary_element_pt[i].size();
328  outfile << "Boundary: " << i
329  << " is adjacent to " << nel
330  << " elements" << std::endl;
331 
332  // Loop over elements on given boundary
333  for (unsigned e=0;e<nel;e++)
334  {
335  FiniteElement* fe_pt=Boundary_element_pt[i][e];
336  outfile << "Boundary element:" << fe_pt
337  << " Face index of boundary is "
338  << Face_index_at_boundary[i][e] << std::endl;
339  }
340  }
341  }
342 
343 
344  // Lookup scheme has now been setup yet
345  Lookup_for_elements_next_boundary_is_setup=true;
346 
347 }

354 /// Assess mesh quality: Ratio of max. edge length to min. height,
355 /// so if it's very large it's BAAAAAD.
356 //======================================================================
357 void TetMeshBase::assess_mesh_quality(std::ofstream& some_file)
358 {
359  Vector<Vector<double> > edge(6);
360  for (unsigned e=0;e<6;e++)
361  {
362  edge[e].resize(3);
363  }
364  unsigned nel=this->nelement();
365  for (unsigned e=0;e<nel;e++)
366  {
367  FiniteElement* fe_pt=this->finite_element_pt(e);
368  for (unsigned i=0;i<3;i++)
369  {
370  edge[0][i]=fe_pt->node_pt(2)->x(i)-fe_pt->node_pt(1)->x(i);
371  edge[1][i]=fe_pt->node_pt(0)->x(i)-fe_pt->node_pt(2)->x(i);
372  edge[2][i]=fe_pt->node_pt(1)->x(i)-fe_pt->node_pt(0)->x(i);
373  edge[3][i]=fe_pt->node_pt(3)->x(i)-fe_pt->node_pt(0)->x(i);
374  edge[4][i]=fe_pt->node_pt(3)->x(i)-fe_pt->node_pt(1)->x(i);
375  edge[5][i]=fe_pt->node_pt(3)->x(i)-fe_pt->node_pt(2)->x(i);
376  }
377 
378  double max_length=0.0;
379  for (unsigned j=0;j<6;j++)
380  {
381  double length=0.0;
382  for (unsigned i=0;i<3;i++)
383  {
384  length+=edge[j][i]*edge[j][i];
385  }
386  length=sqrt(length);
387  if (length>max_length) max_length=length;
388  }
389 
390 
391  double min_height=DBL_MAX;
392  for (unsigned j=0;j<4;j++)
393  {
394  Vector<double> normal(3);
395  unsigned e0=0;
396  unsigned e1=0;
397  unsigned e2=0;
398  switch(j)
399  {
400  case 0:
401  e0=4;
402  e1=5;
403  e2=1;
404  break;
405 
406  case 1:
407  e0=1;
408  e1=3;
409  e2=2;
410  break;
411 
412  case 2:
413  e0=3;
414  e1=4;
415  e2=1;
416  break;
417 
418  case 3:
419  e0=1;
420  e1=2;
421  e2=3;
422  break;
423 
424  default:
425 
426  oomph_info << "never get here\n";
427  abort();
428  }
429 
430  normal[0]=edge[e0][1]*edge[e1][2]-edge[e0][2]*edge[e1][1];
431  normal[1]=edge[e0][2]*edge[e1][0]-edge[e0][0]*edge[e1][2];
432  normal[2]=edge[e0][0]*edge[e1][1]-edge[e0][1]*edge[e1][0];
433  double norm=
434  normal[0]*normal[0]+
435  normal[1]*normal[1]+
436  normal[2]*normal[2];
437  double inv_norm=1.0/sqrt(norm);
438  normal[0]*=inv_norm;
439  normal[1]*=inv_norm;
440  normal[2]*=inv_norm;
441 
442  double height=fabs(
443  edge[e2][0]*normal[0]+
444  edge[e2][1]*normal[1]+
445  edge[e2][2]*normal[2]);
446 
447  if (height<min_height) min_height=height;
448  }
449 
450  double aspect_ratio=max_length/min_height;
451 
452  some_file << "ZONE N=4, E=1, F=FEPOINT, ET=TETRAHEDRON\n";
453  for (unsigned j=0;j<4;j++)
454  {
455  for (unsigned i=0;i<3;i++)
456  {
457  some_file << fe_pt->node_pt(j)->x(i) << " ";
458  }
459  some_file << aspect_ratio << std::endl;
460  }
461  some_file << "1 2 3 4" << std::endl;
462  }
463 }

468  /// Move the nodes on boundaries with associated Geometric Objects (if any)
469  /// so that they exactly coincide with the geometric object. This requires
470  /// that the boundary coordinates are set up consistently
471  //======================================================================
472 void TetMeshBase::snap_nodes_onto_geometric_objects()
473  {
474 
475  // Backup in case elements get inverted
476  std::map<Node*,Vector<double> > old_nodal_posn;
477  std::map<Node*,Vector<double> > new_nodal_posn;
478  unsigned nnod=nnode();
479  for (unsigned j=0;j<nnod;j++)
480  {
481  Node* nod_pt=node_pt(j);
482  Vector<double> x(3);
483  nod_pt->position(x);
484  old_nodal_posn[nod_pt]=x;
485  }
486 
487  // Loop over all boundaries
488  unsigned n_bound = this->nboundary();
489  for (unsigned b = 0; b < n_bound; b++)
490  {
491  bool do_it=true;
492 
493  // Accumulate reason for not snapping
494  std::stringstream reason;
495  reason << "Can't snap nodes on boundary " << b
496  << " onto geom object because: \n";
497 
498  TetMeshFacetedSurface* faceted_surface_pt=0;
499  std::map<unsigned,TetMeshFacetedSurface*>::iterator it=
500  Tet_mesh_faceted_surface_pt.find(b);
501  if (it!=Tet_mesh_faceted_surface_pt.end())
502  {
503  faceted_surface_pt=(*it).second;
504  }
505 
506  // Facet associated with this boundary?
507  if (faceted_surface_pt==0)
508  {
509  reason << "-- no facets asssociated with boundary\n";
510  do_it=false;
511  }
512 
513  // Get geom object associated with facet
514  GeomObject* geom_obj_pt=0;
515  if (do_it)
516  {
517  geom_obj_pt=
518  faceted_surface_pt->geom_object_with_boundaries_pt();
519  if (geom_obj_pt==0)
520  {
521  reason << "-- no geom object associated with boundary\n";
522  do_it=false;
523  }
524  }
525 
526  // Triangular facet?
527  if (Triangular_facet_vertex_boundary_coordinate[b].size()==0)
528  {
529  reason << "-- facet has to be triangular and vertex coordinates have\n"
530  << " to have been set up\n";
531  do_it=false;
532  }
533 
534  // We need boundary coordinates!
535  if (!Boundary_coordinate_exists[b])
536  {
537  reason << "-- no boundary coordinates were set up\n";
538  do_it=false;
539  }
540 
541 
542 
543  // Which facet is associated with this boundary?
544  unsigned facet_id_of_boundary=0;
545  TetMeshFacet* f_pt=0;
546  if (do_it)
547  {
548  unsigned nf=faceted_surface_pt->nfacet();
549  for (unsigned f=0;f<nf;f++)
550  {
551  if ((faceted_surface_pt->
552  one_based_facet_boundary_id(f)-1)==b)
553  {
554  facet_id_of_boundary=f;
555  break;
556  }
557  }
558  f_pt=faceted_surface_pt->
559  facet_pt(facet_id_of_boundary);
560 
561 
562  // Three vertices?
563  unsigned nv=f_pt->nvertex();
564  if (nv!=3)
565  {
566  reason << "-- number of facet vertices is " << nv << " rather than 3\n";
567  do_it=false;
568  }
569 
570  // Have we set up zeta coordinates in geometric object?
571  if ((f_pt->vertex_pt(0)->zeta_in_geom_object().size()!=2)||
572  (f_pt->vertex_pt(1)->zeta_in_geom_object().size()!=2)||
573  (f_pt->vertex_pt(2)->zeta_in_geom_object().size()!=2))
574  {
575  reason << "-- no boundary coordinates were set up\n";
576  do_it=false;
577  }
578  }
579 
580 
581  // Are we ready to go?
582  if (!do_it)
583  {
584  const bool tell_us_why=false;
585  if (tell_us_why)
586  {
587  oomph_info << reason.str() << std::endl;
588  }
589  }
590  else
591  {
592 
593  // Setup area coordinantes in triangular facet
594  double x1=Triangular_facet_vertex_boundary_coordinate[b][0][0];
595  double y1=Triangular_facet_vertex_boundary_coordinate[b][0][1];
596 
597  double x2=Triangular_facet_vertex_boundary_coordinate[b][1][0];
598  double y2=Triangular_facet_vertex_boundary_coordinate[b][1][1];
599 
600  double x3=Triangular_facet_vertex_boundary_coordinate[b][2][0];
601  double y3=Triangular_facet_vertex_boundary_coordinate[b][2][1];
602 
603  double detT=(y2-y3)*(x1-x3)+(x3-x2)*(y1-y3);
604 
605 
606  // Boundary coordinate (cartesian coordinates inside facet)
607  Vector<double> zeta(2);
608 
609  // Loop over all nodes on that boundary
610  const unsigned n_boundary_node = this->nboundary_node(b);
611  for (unsigned n = 0; n < n_boundary_node; ++n)
612  {
613  // Get the boundary node and coordinates
614  Node* const nod_pt = this->boundary_node_pt(b,n);
615  nod_pt->get_coordinates_on_boundary(b,zeta);
616 
617  // Now we have zeta, the cartesian boundary coordinates
618  // in the (assumed to be triangular) boundary facet; let's
619  // work out the area coordinates
620  // Notation as in
621  // https://en.wikipedia.org/wiki/Barycentric_coordinate_system
622  double s0=((y2-y3)*(zeta[0]-x3)+(x3-x2)*(zeta[1]-y3))/detT;
623  double s1=((y3-y1)*(zeta[0]-x3)+(x1-x3)*(zeta[1]-y3))/detT;
624  double s2=1.0-s0-s1;
625 
626  Vector<double> zeta_in_geom_obj(2,0.0);
627  Vector<double> position_from_geom_obj(3,0.0);
628 
629  // Vertex zeta coordinates
630  Vector<double> zeta_0(2);
631  zeta_0=f_pt->vertex_pt(0)->zeta_in_geom_object();
632 
633  Vector<double> zeta_1(2);
634  zeta_1=f_pt->vertex_pt(1)->zeta_in_geom_object();
635 
636  Vector<double> zeta_2(2);
637  zeta_2=f_pt->vertex_pt(2)->zeta_in_geom_object();
638 
639 
640 #ifdef PARANOID
641 
642  // Compute zeta values of the vertices from parametrisation of
643  // boundaries
644  double tol=1.0e-12;
645  Vector<double> zeta_from_boundary(2);
646  Vector<double> zeta_vertex(2);
647  for (unsigned v=0;v<3;v++)
648  {
649  zeta_vertex=f_pt->vertex_pt(v)->zeta_in_geom_object();
650  for (unsigned alt=0;alt<2;alt++)
651  {
652  switch (v)
653  {
654 
655  case 0:
656 
657  if (alt==0)
658  {
659  faceted_surface_pt->
