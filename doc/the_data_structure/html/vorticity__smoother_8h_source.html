148  /// \short Helper function that determines the number of n-th order partial
149  /// derivatives in d-dimensions. Specifically there are (n+d-1)(choose)(d-1)
150  /// possible n-th order partial derivatives in d-dimensions. Implementation
151  /// makes use of the code found at:
152  /// www.geeksforgeeks.org/space-and-time-efficient-binomial-coefficient/
153  unsigned npartial_derivative(const unsigned& n) const
154  {
155  // This will only work in 2D so n_dim is always 2
156  unsigned n_dim=2;
157 
158  // Calculate m
159  unsigned n_bins=n+n_dim-1;
160 
161  // Calculate k
162  unsigned k=n_dim-1;
163 
164  // Initialise the result
165  unsigned value=1;
166 
167  // Since C(n_bins,k)=C(n_bins,n_bins-k)
168  if (k>n_bins-k)
169  {
170  // Replace k
171  k=n_bins-k;
172  }
173 
174  // Calculate [n_bins*(n_bins-1)*...*(n_bins-k+1)]/[k*(k-1)*...*1]
175  for (unsigned i=0;i<k;++i)
176  {
177  // First update
178  value*=(n_bins-i);
179 
180  // Second update
181  value/=(i+1);
182  }
183 
184  // Return the result
185  return value;
186  } // End of npartial_derivative

273  /// \short Helper function to create a container for the vorticity and
274  /// its partial derivatives. If the user wishes to output everything then
275  /// this also creates space for the velocity derivatives too. This function
276  /// has been written to allow for generalisation of the storage without
277  /// (hopefully!) affecting too much
278  Vector<Vector<double> > create_container_for_vorticity_and_derivatives() const
279  {
280  // Maximum number of vorticity derivatives
281  unsigned n_vort_deriv=Maximum_order_of_vorticity_derivative;
282 
283  // Maximum number of velocity derivatives
284  unsigned n_veloc_deriv=Maximum_order_of_velocity_derivative;
285 
286  // The number of vectors in the output
287  unsigned n_vector=n_vort_deriv+n_veloc_deriv+1;
288

290  Vector<Vector<double> > vort_and_derivs(n_vector);
291 
292  // Loop over the entries of the vector
293  for (unsigned i=0;i<n_vort_deriv+1;i++)
294  {
295  // Get the number of partial derivatives of the vorticity
296  vort_and_derivs[i].resize(npartial_derivative(i),0.0);

310  /// \short Helper function that, given the local dof number of the i-th
311  /// vorticity or velocity derivative, returns the index in the container
312  /// that stores the corresponding value.
313  /// Note 1: this function goes hand-in-hand with
314  /// create_container_for_vorticity_and_derivatives()
315  /// Note 2: i=0: vorticity itself;
316  /// i>0: vorticity derivatives
317  std::pair<unsigned,unsigned>
318  vorticity_dof_to_container_id(const unsigned& i) const
319  {
320  // Maximum number of vorticity derivatives (that we actually want)
321  unsigned n_vort_deriv=Maximum_order_of_vorticity_derivative;
322 
323  // Maximum number of velocity derivatives (that we actually want)
324  unsigned n_veloc_deriv=Maximum_order_of_velocity_derivative;
325 
326 #ifdef PARANOID
327  // We cannot calculate this if we're not recovering anything
328  if (n_vort_deriv+n_veloc_deriv+1==0)
329  {
330  // Throw an error
331  throw OomphLibError("Not recovering anything so this shouldn't be called.",
332  OOMPH_CURRENT_FUNCTION,
333  OOMPH_EXCEPTION_LOCATION);
334  }
335 #endif
336 
337  // Maximum order of vorticity derivative (that can be recovered)
338  unsigned max_vort_order=Maximum_order_of_recoverable_vorticity_derivatives;
339 
340  // Maximum order of velocity derivative (that can be recovered)
341  unsigned max_veloc_order=Maximum_order_of_recoverable_velocity_derivatives;
342 
343  // Maximum number of recoverable vorticity terms
344  unsigned max_vort_recov=0;
345 
346  // Maximum number of recoverable velocity terms
347  unsigned max_veloc_recov=0;
348 
349  // Loop over the entries of the vector
350  for (unsigned j=0;j<max_vort_order+1;j++)
351  {
352  // Get the number of partial derivatives of the vorticity
353  max_vort_recov+=npartial_derivative(j);
354  }
355 
356  // Loop over the entries of the vector
357  for (unsigned j=1;j<max_veloc_order+1;j++)
358  {
359  // Get the number of partial derivatives of the vorticity
360  max_veloc_recov+=2*npartial_derivative(j);
361  }
362 
363  // Create a pair to store the output
364  std::pair<unsigned,unsigned> container_id;
365 
366  // Store the number of derivatives we've gone over
367  unsigned nprev_deriv=0;
368 
369  // The number of derivatives available of a given order
370  unsigned n_deriv=0;
371 
372  // If the dof number i is associated with a vorticity derivative
373  if (i<max_vort_recov)
374  {
375  // Loop over the vorticity vectors
376  for (unsigned jj=0;jj<n_vort_deriv+1;jj++)
377  {
378  // Number of derivatives of order jj
379  n_deriv=npartial_derivative(jj);
380 
381  // Update nprev_derivs
382  nprev_deriv+=n_deriv;
383 
384  // If this is true then we need the jj-th vector
385  if (i<nprev_deriv)
386  {
387  // Assign the first index
388  container_id.first=jj;
389 
390  // Index of the entry
391  container_id.second=i-(nprev_deriv-n_deriv);
392 
393  // We're done
394  return container_id;
395  }
396  } // for (unsigned jj=0;jj<n_vort_deriv+1;jj++)
397  }
398  // If the dof number i is associated with a velocity derivative
399  else if (i<max_vort_recov+max_veloc_recov)
400  {
401  // Initialise the value of nprev_deriv (depending on how many
402  // vorticity derivatives we can recover)
403  nprev_deriv=max_vort_recov;
404 
405  // Loop over the velocity vectors
406  for (unsigned jj=n_vort_deriv+1;jj<n_vort_deriv+n_veloc_deriv+1;jj++)
407  {
408  // Number of velocity derivatives
409  n_deriv=2*npartial_derivative(jj-n_vort_deriv);
410 
411  // Update nprev_derivs
412  nprev_deriv+=n_deriv;
413 
414  // If this is true then we need the jj-th vector
415  if (i<nprev_deriv)
416  {
417  // Assign the first index
418  container_id.first=jj;
419 
420  // Index of the entry
421  container_id.second=i-(nprev_deriv-n_deriv);
422 
423  // We're done
424  return container_id;
425  }
426  } // for (unsigned jj=n_vort_deriv+1;jj<n_veloc_deriv;jj++)
427  }
428  else
429  {
430  // Create an ostringstream object to create an error message
431  std::ostringstream error_message_stream;
432 
433  // Create an error message
434  error_message_stream << "Dof number " << i << " not associated "
435  << "with a vorticity or velocity derivative!"
436  << std::endl;
437 
438  // Throw an error
439  throw OomphLibError(error_message_stream.str(),
440  OOMPH_CURRENT_FUNCTION,
441  OOMPH_EXCEPTION_LOCATION);
442  } // if (i<max_vort_recov)
443 
444  // We'll never get here but need to return something
445  return container_id;
446  } // End of vorticity_dof_to_container_id

449  /// \short Helper function that, given the local dof number of the i-th
450  /// vorticity or velocity derivative, returns the index in the container
451  /// that stores the corresponding value.
452  /// Note 1: this function goes hand-in-hand with
453  /// create_container_for_vorticity_and_derivatives()
454  /// Note 2: The input to this function is the i associated with the STORED
455  /// nodal dof value. For example, if we're only recovering the
456  /// velocity derivatives then i=0 is associated with du/dx
457  std::pair<unsigned,unsigned>
458  recovered_dof_to_container_id(const unsigned& i) const
459  {
460  // Create a pair to store the output
461  std::pair<unsigned,unsigned> container_id;
462 
463  // Find which derivative to calculate
464  unsigned derivative_index=i;
465 
466  // Variable to store the index of the vector (initialise the value to -1
467  // so we know whether or not the value has been set)
468  int vector_index=-1;
469 
470  // Get the number of vorticity derivatives to recover
471  unsigned n_vort_derivs=Maximum_order_of_vorticity_derivative;
472 
473  // Get the number of vorticity derivatives to recover
474  unsigned n_veloc_derivs=Maximum_order_of_velocity_derivative;
475 
476  // Loop over the entries of the vector
477  for (unsigned i=0;i<n_vort_derivs+1;i++)
478  {
479  // Get the number of partial derivatives of the vorticity
480  if (derivative_index<npartial_derivative(i))
481  {
482  // We're on the i-th vector
483  vector_index=i;
484 
485  // We're done
486  break;
487  }
488  else
489  {
490  // Decrease the value of derivative_index
491  derivative_index-=npartial_derivative(i);
492  }
493  } // for (unsigned i=0;i<n_vort_derivs+1;i++)
494 
495  // If the vector_index variable value hasn't been found yet
496  if (vector_index==-1)
497  {
498  // Loop over the entries of the vector
499  for (unsigned i=1;i<n_veloc_derivs+1;i++)
500  {
501  // Get the number of partial derivatives of the vorticity
502  if (derivative_index<2*npartial_derivative(i))
503  {
504  // We're on the i-th vector
505  vector_index=n_vort_derivs+i;
506 
507  // We're done
508  break;
509  }
510  else
511  {
512  // Decrease the value of derivative_index
513  derivative_index-=2*npartial_derivative(i);
514  }
515  } // for (unsigned i=1;i<n_veloc_derivs+1;i++)
516  } // if (vector_index==-1)
517 
518 #ifdef PARANOID
519  // Sanity check: if the value hasn't been set yet, something's wrong
520  if (vector_index==-1)
521  {
522  // Create an ostringstream object to create an error message
523  std::ostringstream error_message_stream;
524 
525  // Create an error message
526  error_message_stream << "Value of vector_index has not been set. "
527  << "Something's wrong!";
528 
529  // Throw an error
530  throw OomphLibError(error_message_stream.str(),
531  OOMPH_CURRENT_FUNCTION,
532  OOMPH_EXCEPTION_LOCATION);
533  }
534 #endif
535 
536  // Assign the first entry of container_id
537  container_id.first=vector_index;
538 
539  // Assign the second entry of container_id
540  container_id.second=derivative_index;
541 
542  // We'll never get here but need to return something
543  return container_id;
544  } // End of recovered_dof_to_container_id

546  /// \short Given the STORED dof number, this function returns the global
547  /// recovered number. For example, if we only want to recover the velocity
548  /// derivatives then the stored dof number of du/dy is 1 (as 0 is associated
549  /// with du/dx). The global recovered number is 11 (as there are currently
550  /// 10 vorticity derivatives that can be recovered and du/dy is the second
551  /// velocity derivative we can recover).
552  unsigned stored_dof_to_recoverable_dof(const unsigned& i) const
553  {
554  // Get the ID in the storage associated with i-th recovered dof
555  std::pair<unsigned,unsigned> id=recovered_dof_to_container_id(i-N_dim-1);
556 
557  // Vector entry index
558  unsigned vector_index=id.first;
559 
560  // Derivative index
561  unsigned deriv_index=id.second;
562 
563  // The index we want
564  unsigned index=0;
565 
566  // Maximum possible order of vorticity derivatives that can be recovered
567  unsigned max_vort_recov=Maximum_order_of_recoverable_vorticity_derivatives;
568 
569  // If we're dealing with a vorticity derivative
570  if (vector_index<max_vort_recov+1)
571  {
572  // Holds the number of derivatives of lower order
573  unsigned n_prev_deriv=0;
574 
575  // Loop over the derivative orders
576  for (unsigned j=0;j<vector_index;j++)
577  {
578  // Increment n_prev_deriv
579  n_prev_deriv+=npartial_derivative(j);
580  }
581 
582  // Update the value of index
583  index+=n_prev_deriv+deriv_index;
584  }
585  // We're dealing with derivatives of the velocity
586  else
587  {
588  // Holds the number of vorticity derivatives
589  unsigned n_prev_deriv=0;
590 
591  // Loop over the derivative orders
592  for (unsigned j=0;j<max_vort_recov+1;j++)
593  {
594  // Increment n_prev_deriv
595  n_prev_deriv+=npartial_derivative(j);
596  }
597 
598  // What is the user-chosen maximum vorticity derivative to recover?
599  unsigned n_vort_deriv=Maximum_order_of_vorticity_derivative;
600 
601  // Loop over the derivative orders
602  for (unsigned j=1;j<vector_index-n_vort_deriv;j++)
603  {
604  // Increment n_prev_deriv
605  n_prev_deriv+=2*npartial_derivative(j);
606  }
607 
608  // Update the value of index
609  index+=n_prev_deriv+deriv_index;
610  } // if (vector_index<max_vort_recov)
611 
612  // Return the value of index
613  return index;
614  } // End of stored_dof_to_recoverable_dof
