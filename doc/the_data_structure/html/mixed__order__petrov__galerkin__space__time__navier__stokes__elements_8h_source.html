210  /// Get residuals and Jacobian of Robin boundary conditions in pressure
211  /// advection diffusion problem in Fp preconditoner
212  /// NOTE: Care has to be taken here as fluxes are spatially dependent and
213  /// not temporally dependent but the template elements are space-time
214  /// elements!
215  //============================================================================
216  template<class ELEMENT>
217  void FpPressureAdvDiffRobinBCMixedOrderSpaceTimeElement<ELEMENT>::
218  fill_in_generic_residual_contribution_fp_press_adv_diff_robin_bc(
219  Vector<double>& residuals,
220  DenseMatrix<double>& jacobian,
221  const unsigned& flag)
222  {
223  // Throw a warning
224  throw OomphLibError("You shouldn't be using this just yet!",
225  OOMPH_CURRENT_FUNCTION,
226  OOMPH_EXCEPTION_LOCATION);
227 
228  // Get the dimension of the element
229  // DRAIG: Should be 2 as bulk (space-time) element is 3D...
230  unsigned my_dim=this->dim();
231 
232  // Storage for local coordinates in FaceElement
233  Vector<double> s(my_dim,0.0);
234 
235  // Storage for local coordinates in the associated bulk element
236  Vector<double> s_bulk(my_dim+1,0.0);
237 
238  // Storage for outer unit normal
239  // DRAIG: Need to be careful here...
240  Vector<double> unit_normal(my_dim+1,0.0);
241 
242  // Storage for velocity in bulk element
243  // DRAIG: Need to be careful here...
244  Vector<double> velocity(my_dim,0.0);
245 
246  // Set the value of n_intpt
247  unsigned n_intpt=this->integral_pt()->nweight();
248 
249  // Integer to store local equation number
250  int local_eqn=0;
251 
252  // Integer to store local unknown number
253  int local_unknown=0;
254 
255  // Get upcast bulk element
256  ELEMENT* bulk_el_pt=dynamic_cast<ELEMENT*>(this->bulk_element_pt());
257 
258  // Find out how many pressure dofs there are in the bulk element
259  unsigned n_pres=bulk_el_pt->npres_nst();
260 
261  // Get the Reynolds number from the bulk element
262  double re=bulk_el_pt->re();
263 
264  // Set up memory for pressure shape functions
265  Shape psip(n_pres);
266 
267  // Set up memory for pressure test functions
268  Shape testp(n_pres);
269 
270  // Loop over the integration points
271  for (unsigned ipt=0;ipt<n_intpt;ipt++)
272  {
273  // Get the integral weight
274  double w=this->integral_pt()->weight(ipt);
275 
276  // Assign values of local coordinate in FaceElement
277  for (unsigned i=0;i<my_dim;i++)
278  {
279  // Get the i-th local coordinate at this knot point
280  s[i]=this->integral_pt()->knot(ipt,i);
281  }
282 
283  // Find corresponding coordinate in the bulk element
284  s_bulk=this->local_coordinate_in_bulk(s);
285 
286  // Get outer unit normal
287  // DRAIG: Make sure the normal is calculated properly; i.e.
288  // the normal needs to be calculated in the spatial direction
289  // and NOT in the temporal direction!!!
290  this->outer_unit_normal(ipt,unit_normal);
291 
292  // Get velocity in bulk element
293  bulk_el_pt->interpolated_u_nst(s_bulk,velocity);
294 
295  // Get normal component of velocity
296  double flux=0.0;
297 
298  // Loop over the velocity components
299  for (unsigned i=0;i<my_dim;i++)
300  {
301  // Update the flux quantity
302  flux+=velocity[i]*unit_normal[i];
303  }
304 
305  // Modify BC: If outflow (flux>0) apply Neumann condition instead
306  if (flux>0.0)
307  {
308  // Apply no flux condition
309  flux=0.0;
310  }
311 
312  // Get the pressure at these local coordinates
313  double interpolated_press=bulk_el_pt->interpolated_p_nst(s_bulk);
314 
315  // Call the pressure shape and test functions in bulk element
316  bulk_el_pt->pshape_nst(s_bulk,psip,testp);
317 
318  // Find the Jacobian of the mapping within the FaceElement
319  double J=this->J_eulerian(s);
320 
321  // Premultiply the weights and the Jacobian
322  double W=w*J;
323 
324  // Loop over the pressure shape functions in bulk
325  // (wasteful but they'll be zero on the boundary)
326  for (unsigned l=0;l<n_pres;l++)
327  {
328  // Get the local equation number associated with this pressure dof
329  local_eqn=bulk_el_pt->p_local_eqn(l);
330 
331  // If it's not a boundary condition
332  if (local_eqn>=0)
333  {
334  // Update the residuals
335  residuals[local_eqn]-=re*flux*interpolated_press*testp[l]*W;
336 
337  // If the Jacobian needs to be computed too
338  if (flag)
339  {
340  // Loop over the shape functions in bulk
341  for (unsigned l2=0;l2<n_pres;l2++)
342  {
343  // Get the equation number associated with this pressure dof
344  local_unknown=bulk_el_pt->p_local_eqn(l2);
345 
346  // If it's not a boundary condition
347  if (local_unknown>=0)
348  {
349  // Update the appropriate Jacobian entry
350  jacobian(local_eqn,local_unknown)-=re*flux*psip[l2]*testp[l]*W;
351  }
352  } // for (unsigned l2=0;l2<n_pres;l2++)
353  } // if (flag)
354  } // if (local_eqn>=0)
355  } // for (unsigned l=0;l<n_pres;l++)
356  } // for (unsigned ipt=0;ipt<n_intpt;ipt++)
357  } // End of fill_in_generic_residual_contribution_fp_press_adv_diff_robin_bc

445  /// A class for elements that solve the Cartesian Navier-Stokes
446  /// equations, templated by the dimension DIM.
447  /// This contains the generic maths -- any concrete implementation must
448  /// be derived from this.
449  ///
450  /// We're solving:
451  ///
452  /// \f$ { Re \left( St \frac{\partial u_i}{\partial t}+
453  /// (u_j-u_j^{M}) \frac{\partial u_i}{\partial x_j} \right)=
454  /// -\frac{\partial p}{\partial x_i} -R_\rho B_i(x_j) -
455  /// \frac{Re}{Fr} G_i +
456  /// \frac{\partial }{\partial x_j} \left[ R_\mu \left(
457  /// \frac{\partial u_i}{\partial x_j}+
458  /// \frac{\partial u_j}{\partial x_i} \right) \right] } \f$
459  ///
460  /// and
461  ///
462  /// \f$ { \frac{\partial u_i}{\partial x_i}=Q } \f$
463  ///
464  /// We also provide all functions required to use this element
465  /// in FSI problems, by deriving it from the FSIFluidElement base
466  /// class.
467  ///
468  /// --------------------
469  /// SPACE-TIME ELEMENTS:
470  /// --------------------
471  /// The space-time extension is written ONLY for the 2D Navier-Stokes
472  /// equations. The result is a 3D problem (x,y,t) to be solved on a
473  /// 3D mesh. The template parameter DIM now corresponds to the
474  /// dimension of the space-time problem (i.e. DIM=3 for the 2D flow).
475  //======================================================================
