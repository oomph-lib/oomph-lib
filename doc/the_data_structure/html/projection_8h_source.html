236  /// \short Residual for the projection step. Flag indicates if we
237  /// want the Jacobian (1) or not (0). Virtual so it can be
238  /// overloaded if necessary
239  virtual void residual_for_projection(Vector<double> &residuals,
240  DenseMatrix<double> &jacobian,
241  const unsigned& flag)
242  {
243  //Am I a solid element
244  SolidFiniteElement* solid_el_pt =
245  dynamic_cast<SolidFiniteElement*>(this);
246 
247  unsigned n_dim=dim();
248 
249  //Allocate storage for local coordinates
250  Vector<double> s(n_dim);
251 
252  //Current field
253  unsigned fld=Projected_field;
254 
255  //Number of nodes
256  const unsigned n_node = this->nnode();
257  //Number of positional dofs
258  const unsigned n_position_type = this->nnodal_position_type();
259 
260  //Number of dof for current field
261  const unsigned n_value=nvalue_of_field(fld);
262 
263  //Set the value of n_intpt
264  const unsigned n_intpt = integral_pt()->nweight();
265 
266  //Loop over the integration points
267  for(unsigned ipt=0;ipt<n_intpt;ipt++)
268  {
269  // Get the local coordinates of Gauss point
270  for(unsigned i=0;i<n_dim;i++) s[i] = integral_pt()->knot(ipt,i);
271 
272  //Get the integral weight
273  double w = integral_pt()->weight(ipt);
274 
275  // Find same point in base mesh using external storage
276  FiniteElement* other_el_pt=0;
277  other_el_pt=this->external_element_pt(0,ipt);
278  Vector<double> other_s(n_dim);
279  other_s=this->external_element_local_coord(0,ipt);
280 
281  ProjectableElement<ELEMENT>* cast_el_pt =
282  dynamic_cast<ProjectableElement<ELEMENT>*>(other_el_pt);
283 
284  //Storage for the local equation and local unknown
285  int local_eqn=0, local_unknown=0;
286 
287  //Now set up the three different projection problems
288  switch(Projection_type)
289  {
290  case Lagrangian:
291  {
292  //If we don't have a solid element there is a problem
293  if(solid_el_pt==0)
294  {
295  throw OomphLibError(
296  "Trying to project Lagrangian coordinates in non-SolidElement\n",
297  OOMPH_CURRENT_FUNCTION,
298  OOMPH_EXCEPTION_LOCATION);
299  }
300 
301  //Find the position shape functions
302  Shape psi(n_node,n_position_type);
303  //Get the position shape functions
304  this->shape(s,psi);
305  //Get the jacobian
306  double J = this->J_eulerian(s);
307 
308  //Premultiply the weights and the Jacobian
309  double W = w*J;
310 
311  //Get the value of the current position of the 0th coordinate
312  //in the current element
313  //at the current time level (which is the unkown)
314  double interpolated_xi_proj = this->interpolated_x(s,0);
315 
316  //Get the Lagrangian position in the other element
317  double interpolated_xi_bar=
318  dynamic_cast<SolidFiniteElement*>(cast_el_pt)
319  ->interpolated_xi(other_s,Projected_lagrangian);
320 
321  //Now loop over the nodes and position dofs
322  for(unsigned l=0;l<n_node;++l)
323  {
324  //Loop over position unknowns
325  for(unsigned k=0;k<n_position_type;++k)
326  {
327  //The local equation is going to be the positional local equation
328  local_eqn = solid_el_pt->position_local_eqn(l,k,0);
329 
330  //Now assemble residuals
331  if(local_eqn >= 0)
332  {
333  //calculate residuals
334  residuals[local_eqn] +=
335  (interpolated_xi_proj - interpolated_xi_bar)*psi(l,k)*W;
336 
337  //Calculate the jacobian
338  if(flag==1)
339  {
340  for(unsigned l2=0;l2<n_node;l2++)
341  {
342  //Loop over position dofs
343  for(unsigned k2=0;k2<n_position_type;k2++)
344  {
345  local_unknown =
346  solid_el_pt->position_local_eqn(l2,k2,0);
347  if(local_unknown >= 0)
348  {
349  jacobian(local_eqn,local_unknown)
350  += psi(l2,k2)*psi(l,k)*W;
351  }
352  }
353  }
354  } //end of jacobian
355  }
356  }
357  }
358  } //End of Lagrangian coordinate case
359 
360  break;
361 
362  //Now the coordinate history case
363  case Coordinate:
364  {
365  //Find the position shape functions
366  Shape psi(n_node,n_position_type);
367  //Get the position shape functions
368  this->shape(s,psi);
369  //Get the jacobian
370  double J = this->J_eulerian(s);
371 
372  //Premultiply the weights and the Jacobian
373  double W = w*J;
374 
375  //Get the value of the current position in the current element
376  //at the current time level (which is the unkown)
377  double interpolated_x_proj = 0.0;
378  //If we are a solid element read it out directly from the data
379  if(solid_el_pt!=0)
380  {
381  interpolated_x_proj = this->interpolated_x(s,Projected_coordinate);
382  }
383  //Otherwise it's the field value at the current time
384  else
385  {
386  interpolated_x_proj = this->get_field(0,fld,s);
387  }
388 
389  //Get the position in the other element
390  double interpolated_x_bar=
391  cast_el_pt->interpolated_x(Time_level_for_projection,
392  other_s,Projected_coordinate);
393 
394  //Now loop over the nodes and position dofs
395  for(unsigned l=0;l<n_node;++l)
396  {
397  //Loop over position unknowns
398  for(unsigned k=0;k<n_position_type;++k)
399  {
400  //If I'm a solid element
401  if(solid_el_pt!=0)
402  {
403  //The local equation is going to be the positional local equation
404  local_eqn =
405  solid_el_pt->position_local_eqn(l,k,Projected_coordinate);
406  }
407  //Otherwise just pick the local unknown of a field to
408  //project into
409  else
410  {
411  //Complain if using generalised position types
412  //but this is not a solid element, because the storage
413  //is then not clear
414  if(n_position_type!=1)
415  {
416  throw OomphLibError(
417  "Trying to project generalised positions not in SolidElement\n",
418  OOMPH_CURRENT_FUNCTION,
419  OOMPH_EXCEPTION_LOCATION);
420  }
421  local_eqn = local_equation(fld,l);
422  }
423 
424  //Now assemble residuals
425  if(local_eqn >= 0)
426  {
427  //calculate residuals
428  residuals[local_eqn] +=
429  (interpolated_x_proj - interpolated_x_bar)*psi(l,k)*W;
430 
431  //Calculate the jacobian
432  if(flag==1)
433  {
434  for(unsigned l2=0;l2<n_node;l2++)
435  {
436  //Loop over position dofs
437  for(unsigned k2=0;k2<n_position_type;k2++)
438  {
439  //If I'm a solid element
440  if(solid_el_pt!=0)
441  {
442  local_unknown = solid_el_pt
443  ->position_local_eqn(l2,k2,Projected_coordinate);
444  }
445  else
446  {
447  local_unknown = local_equation(fld,l2);
448  }
449 
450  if(local_unknown >= 0)
451  {
452  jacobian(local_eqn,local_unknown)
453  += psi(l2,k2)*psi(l,k)*W;
454  }
455  }
456  }
457  } //end of jacobian
458  }
459  }
460  }
461  } //End of coordinate case
462  break;
463 
464  //Now the value case
465  case Value:
466  {
467  //Field shape function
468  Shape psi(n_value);
469 
470  //Calculate jacobian and shape functions for that field
471  double J=jacobian_and_shape_of_field(fld,s,psi);
472 
473  //Premultiply the weights and the Jacobian
474  double W = w*J;
475 
476  //Value of field in current element at current time level
477  //(the unknown)
478  unsigned now=0;
479  double interpolated_value_proj = this->get_field(now,fld,s);
480 
481  //Value of the interpolation of element located in base mesh
482  double interpolated_value_bar =
483  cast_el_pt->get_field(Time_level_for_projection,fld,other_s);
484 
485  //Loop over dofs of field fld
486  for(unsigned l=0;l<n_value;l++)
487  {
488  local_eqn = local_equation(fld,l);
489  if(local_eqn >= 0)
490  {
491  //calculate residuals
492  residuals[local_eqn] +=
493  (interpolated_value_proj - interpolated_value_bar)*psi[l]*W;
494 
495  //Calculate the jacobian
496  if(flag==1)
497  {
498  for(unsigned l2=0;l2<n_value;l2++)
499  {
500  local_unknown = local_equation(fld,l2);
501  if(local_unknown >= 0)
502  {
503  jacobian(local_eqn,local_unknown)
504  += psi[l2]*psi[l]*W;
505  }
506  }
507  } //end of jacobian
508  }
509  }
510  }
511  break;
512 
513  default:
514  throw OomphLibError(
515  "Wrong type specificied in Projection_type. This should never happen\n",
516  OOMPH_CURRENT_FUNCTION,
517  OOMPH_EXCEPTION_LOCATION);
518  } //End of the switch statement
519 
520  }//End of loop over ipt
521 
522  }//End of residual_for_projection function
