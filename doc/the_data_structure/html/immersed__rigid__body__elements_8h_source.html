43 // DOXYERROR: I removed a \f from the formula \f$\mbox{\boldmath$F$} =
44 // \mbox{\boldmath$F$}^{*}/\mu U\f$ below because it broke doxygen. Someone
45 // who knows the maths should check that this still makes sense.
46 /// Class that solves the equations of motion for a general
47 /// two-dimensional rigid body subject to a particular imposed force and
48 /// torque distribution and immersed within an external fluid. The body's
49 /// position is entirely specified by the location of its centre of mass,
50 /// \f$\mbox{\boldmath$X$}\f$, and a single angle, \f$\phi\f$, that represents
51 /// a possible rotation. The equations of motion are then simply Newton's
52 /// second law for the conservation of linear momentum in two directions and
53 /// angular momentum about the single possible axis of rotation.
54 ///
55 /// The non-dimensionalisation is based on the viscous scales of
56 /// the surrounding fluid, in which case, the governing equations are
57 /// \f[ Re St^{2} M \frac{\mbox{d}^{2} \mbox{\boldmath$X$}}{\mbox{d} t^{2}}
58 /// = \mbox{\boldmath$F$} + \oint \mbox{\boldmath$t$} \mbox{d} s
59 /// \quad\mbox{and}\quad Re St^{2} I \frac{\mbox{d}^{2}\Phi}{\mbox{d} t^{2}}
60 /// = \mbox{\boldmath$T$} + \oint \mbox{\boldmath$q$} \mbox{d} s,
61 /// \f]
62 /// where \f$\mbox{\boldmath$F$} = \mbox{\boldmath$F$}^{*}/\mu U\f$ is the
63 /// external force per unit length;
64 /// \f$\mbox{\boldmath$t$}\f$ is the net force per unit length
65 /// applied on the rigid body
66 /// by the surrounding fluid;
67 /// \f$\mbox{\boldmath$T$} = \mbox{\boldmath$T$}^{*}/(\mu UL)\f$
68 /// is the external torque per unit length; and \f$\mbox{\boldmath$q$}\f$ is
69 /// the net torque per unit length applied on the body by the fluid.
70 /// \f$M\f$ is a scaled mass the density ratio multiplied by a shape parameter
71 /// and \f$I\f$ is a scaled moment of inertia. Finally, \f$Re\f$ and \f$St\f$
72 /// are the Reynolds and Strouhal numbers of the surrounding fluid.
73 /// Note that these equations may be used without the external fluid,
74 /// in which case the non-dimensionalisation doesn't make a lot of sense,
75 /// but may be re-interpreted accordingly.
76 ///
77 /// A Data object whose three values
78 /// represent the x and y displacements of the body's centre of mass and
79 /// its rotation about the centre of mass may be passed in as external
80 /// data or, if not, it will be constructed internally.
81 ///
82 /// For general usage, an underlying geometric object must passed to the
83 /// constructor and the position will be determined by applying rigid body
84 /// motions to the underlying object based on the initial_centre_of_mass
85 /// and initial_phi (angle), which default to zero. If these defaults are
86 /// not suitable, the values must be set externally. In addition a mass
87 /// and moment of inertia should also be set externally.
88 ///
89 /// If added to a mesh in the Problem (in its incarnation as a
90 /// GeneralisedElement) the displacement/rotation of the body
91 /// is computed in response to (i) user-specifiable applied forces
92 /// and a torque and (ii) the net drag (and associated torque) from
93 /// a mesh of elements that can exert a drag onto the body (typically
94 /// Navier-Stokes FaceElements that apply a viscous drag to an
95 /// immersed body, represented by the body.)
96 //=====================================================================

326 /// Update the positions of the nodes in fluid elements
327 /// adjacent to the rigid body, defined as being elements in the
328 /// drag mesh.
329 inline void node_update_adjacent_fluid_elements()
330 {
331 // If there is no drag mesh then do nothing
332 if (Drag_mesh_pt == 0)
333 {
334 return;
335 }
336 // Otherwise update the elements adjacent to the rigid body
337 //(all the elements adjacent to those in the drag mesh)
338 else
339 {
340 unsigned nel = Drag_mesh_pt->nelement();
341 for (unsigned e = 0; e < nel; e++)
342 {
343 dynamic_cast<FaceElement*>(Drag_mesh_pt->element_pt(e))
344 ->bulk_element_pt()
345 ->node_update();
346 }
347 }
348 }

500 /// Helper function to adjust the position in
501 /// response to changes in position and angle of the solid
502 /// about the centre of mass
503 inline void apply_rigid_body_motion(const unsigned& t,
504 const Vector<double>& initial_x,
505 Vector<double>& r) const
506 {
507 // Scale relative to the centre of mass
508 double X = initial_x[0] - Initial_centre_of_mass[0];
509 double Y = initial_x[1] - Initial_centre_of_mass[1];
510
511 // Find the original angle and radius
512 double phi_orig = atan2(Y, X);
513 double r_orig = sqrt(X * X + Y * Y);
514
515 // Updated position vector
516 r[0] = Initial_centre_of_mass[0] +
517 this->Centre_displacement_data_pt->value(t, 0);
518
519 r[1] = Initial_centre_of_mass[1] +
520 this->Centre_displacement_data_pt->value(t, 1);
521
522 // Add in the rotation terms if there are to be included
523 if (Include_geometric_rotation)
524 {
525 r[0] += r_orig *
526 cos(phi_orig + this->Centre_displacement_data_pt->value(t, 2));
527 r[1] += r_orig *
528 sin(phi_orig + this->Centre_displacement_data_pt->value(t, 2));
529 }
530 else
531 {
532 r[0] += r_orig * cos(phi_orig);
533 r[1] += r_orig * sin(phi_orig);
534 }
535 }

653 /// Class upgrading a TriangleMeshPolygon to a "hole" for use during
654 /// triangle mesh generation. For mesh generation purposes, the main (and
655 /// only) addition to the base class is the provision of the coordinates of a
656 /// hole inside the polygon. To faciliate the movement of the "hole" through
657 /// the domain we also provide a Data object whose three values represent the
658 /// x and y displacements of its centre of gravity and the polygon's rotation
659 /// about its centre of gravity. If added to a mesh in the Problem (in its
660 /// incarnation as a GeneralisedElement) the displacement/rotation of the
661 /// polygon is computed in response to (i) user-specifiable applied forces and
662 /// a torque and (ii) the net drag (and associated torque) from a mesh of
663 /// elements that can exert a drag onto the polygon (typically Navier-Stokes
664 /// FaceElements that apply a viscous drag to an immersed body, represented by
665 /// the polygon.)

715 /// Get the initial position of the polygon
716 void get_initial_position(const Vector<double>& xi, Vector<double>& r) const
717 {
718 // Find the number of polylines (boundaries)
719 unsigned n_poly = this->npolyline();
720
721 // The boundary coordinate will be contiguous from polyline to
722 // polyline and each polyline will have the scaled arclength coordinate
723 // in the range 0->1.
724
725 // Find the maximum coordinate
726 double zeta_max = Zeta_vertex[n_poly - 1].back();
727
728 // If we are above the maximum complain
729 if (xi[0] > zeta_max)
730 {
731 std::ostringstream error_message;
732 error_message << "Value of intrinsic coordinate " << xi[0]
733 << "greater than maximum " << zeta_max << "\n";
734 throw OomphLibError(error_message.str(),
735 "TriangleMeshPolygon::position()",
736 OOMPH_EXCEPTION_LOCATION);
737 }
738
739 // If equal to the maximum, return the final vertex
740 if (xi[0] == zeta_max)
741 {
742 unsigned n_vertex = this->polyline_pt(n_poly - 1)->nvertex();
743 r = this->polyline_pt(n_poly - 1)->vertex_coordinate(n_vertex - 1);
744 return;
745 }
746
747 // Otherwise
748
749 // Find out which polyline we are in
750 // If we've got here this should be less than n_poly
751 unsigned p = static_cast<unsigned>(floor(xi[0]));
752
753 // If we are "above" the last polyline then throw an error
754 // This should have been caught by the trap above
755 if (p >= n_poly)
756 {
757 std::ostringstream error_message;
758 error_message
759 << "Something has gone wrong.\n"
760 << "The integer part of the input intrinsic coordinate is " << p
761 << "\nwhich is equal to or greater than the number of polylines, "
762 << n_poly << ".\n"
763 << "This should have triggered an earlier error\n";
764
765
766 throw OomphLibError(error_message.str(),
767 OOMPH_CURRENT_FUNCTION,
768 OOMPH_EXCEPTION_LOCATION);
769 }
770
771 // Cache the appropriate polyline
772 TriangleMeshPolyLine* const line_pt = this->polyline_pt(p);
773
774 // If we are at the first vertex in the polyline, return it
775 if (xi[0] == Zeta_vertex[p][0])
776 {
777 r = line_pt->vertex_coordinate(0);
778 return;
779 }
780
781 // Otherwise loop over the other points to find the appropriate
782 // segment
783
784 // Find the number of vertices in the chosen polyline
785 unsigned n_vertex = line_pt->nvertex();
786 // Now start from the first node in the appropriate polyline and loop up
787 for (unsigned v = 1; v < n_vertex; v++)
788 {
789 // First time that zeta falls below the vertex coordinate
790 // we have something
791 if (xi[0] < Zeta_vertex[p][v])
792 {
793 double fraction = (xi[0] - Zeta_vertex[p][v - 1]) /
794 (Zeta_vertex[p][v] - Zeta_vertex[p][v - 1]);
795 Vector<double> first = line_pt->vertex_coordinate(v - 1);
796 Vector<double> last = line_pt->vertex_coordinate(v);
797 r.resize(2);
798 for (unsigned i = 0; i < 2; i++)
799 {
800 r[i] = first[i] + fraction * (last[i] - first[i]);
801 }
802 return;
803 }
804 }
805 }
806
807
808 /// Helper function to assign the values of the (scaled) arc-length
809 /// to each node of each polyline. The direction will be the natural
810 /// order of the vertices within the polyline.
811 void assign_zeta()
812 {
813 // Find the number of polylines
814 unsigned n_poly = this->npolyline();
815
816 // Allocate appropriate storage for the zeta values
817 Zeta_vertex.resize(n_poly);
818
819 // Temporary storage for the vertex coordinates
820 Vector<double> vertex_coord_first;
821 Vector<double> vertex_coord_next;
822
823 // Set the initial value of zeta
824 double zeta_offset = 0.0;
825
826 // Loop over the polylines
827 for (unsigned p = 0; p < n_poly; ++p)
828 {
829 // Cache the pointer to the polyline
830 TriangleMeshPolyLine* const line_pt = this->polyline_pt(p);
831
832 // Find the number of vertices in the polyline
833 unsigned n_vertex = line_pt->nvertex();
834
835 // Allocate storage and set initial value
836 Zeta_vertex[p].resize(n_vertex);
837 Zeta_vertex[p][0] = 0.0;
838
839 // Loop over the vertices in the polyline and calculate the length
840 // between each for use as the intrinsic coordinate
841 vertex_coord_first = line_pt->vertex_coordinate(0);
842 for (unsigned v = 1; v < n_vertex; v++)
843 {
844 vertex_coord_next = line_pt->vertex_coordinate(v);
845 double length =
846 sqrt(pow(vertex_coord_next[0] - vertex_coord_first[0], 2.0) +
847 pow(vertex_coord_next[1] - vertex_coord_first[1], 2.0));
848 Zeta_vertex[p][v] = Zeta_vertex[p][v - 1] + length;
849 vertex_coord_first = vertex_coord_next;
850 }
851
852 // Now scale the length to unity and add the offset
853 Zeta_vertex[p][0] += zeta_offset;
854 for (unsigned v = 1; v < n_vertex; v++)
855 {
856 Zeta_vertex[p][v] /= Zeta_vertex[p][n_vertex - 1];
857 Zeta_vertex[p][v] += zeta_offset;
858 }
859 zeta_offset += 1.0;
860 } // End of loop over polylines
861 }
862
863 /// Vector of intrisic coordinate values at the nodes
864 Vector<Vector<double>> Zeta_vertex;
865 };
