44 // DOXYERROR: I removed a \f from the formula \f$\mbox{\boldmath$F$} = \mbox{\boldmath$F$}^{*}/\mu U\f$ below because it broke doxygen. Someone who knows the maths should check that this still makes sense.
45 /// Class that solves the equations of motion for a general
46 /// two-dimensional rigid body subject to a particular imposed force and torque
47 /// distribution and immersed within an external fluid.
48 /// The body's position is entirely specified by the location of its
49 /// centre of mass, \f$\mbox{\boldmath$X$}\f$,
50 /// and a single angle, \f$\phi\f$, that represents a possible rotation.
51 /// The equations of motion are then simply Newton's second law for the
52 /// conservation of linear momentum in two directions and angular
53 /// momentum about the single possible axis of rotation.
54 ///
55 /// The non-dimensionalisation is based on the viscous scales of
56 /// the surrounding fluid, in which case, the governing equations are
57 /// \f[ Re St^{2} M \frac{\mbox{d}^{2} \mbox{\boldmath$X$}}{\mbox{d} t^{2}}
58 /// = \mbox{\boldmath$F$} + \oint \mbox{\boldmath$t$} \mbox{d} s
59 /// \quad\mbox{and}\quad Re St^{2} I \frac{\mbox{d}^{2}\Phi}{\mbox{d} t^{2}}
60 /// = \mbox{\boldmath$T$} + \oint \mbox{\boldmath$q$} \mbox{d} s,
61 /// \f]
62 /// where \f$\mbox{\boldmath$F$} = \mbox{\boldmath$F$}^{*}/\mu U\f$ is the
63 /// external force per unit length;
64 /// \f$\mbox{\boldmath$t$}\f$ is the net force per unit length
65 /// applied on the rigid body
66 /// by the surrounding fluid;
67 /// \f$\mbox{\boldmath$T$} = \mbox{\boldmath$T$}^{*}/(\mu UL)\f$
68 /// is the external torque per unit length; and \f$\mbox{\boldmath$q$}\f$ is
69 /// the net torque per unit length applied on the body by the fluid.
70 /// \f$M\f$ is a scaled mass the density ratio multiplied by a shape parameter
71 /// and \f$I\f$ is a scaled moment of inertia. Finally, \f$Re\f$ and \f$St\f$
72 /// are the Reynolds and Strouhal numbers of the surrounding fluid.
73 /// Note that these equations may be used without the external fluid,
74 /// in which case the non-dimensionalisation doesn't make a lot of sense,
75 /// but may be re-interpreted accordingly.
76 ///
77 /// A Data object whose three values
78 /// represent the x and y displacements of the body's centre of mass and
79 /// its rotation about the centre of mass may be passed in as external
80 /// data or, if not, it will be constructed internally.
81 ///
82 /// For general usage, an underlying geometric object must passed to the
83 /// constructor and the position will be determined by applying rigid body
84 /// motions to the underlying object based on the initial_centre_of_mass
85 /// and initial_phi (angle), which default to zero. If these defaults are
86 /// not suitable, the values must be set externally. In addition a mass
87 /// and moment of inertia should also be set externally.
88 ///
89 /// If added to a mesh in the Problem (in its incarnation as a
90 /// GeneralisedElement) the displacement/rotation of the body
91 /// is computed in response to (i) user-specifiable applied forces
92 /// and a torque and (ii) the net drag (and associated torque) from
93 /// a mesh of elements that can exert a drag onto the body (typically
94 /// Navier-Stokes FaceElements that apply a viscous drag to an
95 /// immersed body, represented by the body.)

410  /// \short Helper function to adjust the position in
411  /// response to changes in position and angle of the solid
412  /// about the centre of mass
413  inline void apply_rigid_body_motion(const unsigned &t,
414  const Vector<double> &initial_x,
415  Vector<double> &r) const
416  {
417  //Scale relative to the centre of mass
418  double X = initial_x[0] - Initial_centre_of_mass[0];
419  double Y = initial_x[1] - Initial_centre_of_mass[1];
420 
421  //Find the original angle and radius
422  double phi_orig=atan2(Y,X);
423  double r_orig=sqrt(X*X+Y*Y);
424 
425  // Updated position vector
426  r[0] = Initial_centre_of_mass[0] +
427  this->Centre_displacement_data_pt->value(t,0);
428 
429  r[1] = Initial_centre_of_mass[1] +
430  this->Centre_displacement_data_pt->value(t,1);
431 
432  //Add in the rotation terms if there are to be included
433  if(Include_geometric_rotation)
434  {
435  r[0] +=
436  r_orig*cos(phi_orig + this->Centre_displacement_data_pt->value(t,2));
437  r[1] +=
438  r_orig*sin(phi_orig + this->Centre_displacement_data_pt->value(t,2));
439  }
440  else
441  {
442  r[0] += r_orig*cos(phi_orig);
443  r[1] += r_orig*sin(phi_orig);
444  }
445  }

638  /// \short Get the initial position of the polygon
639  void get_initial_position(const Vector<double> &xi, Vector<double> &r) const
640  {
641  //Find the number of polylines (boundaries)
642  unsigned n_poly = this->npolyline();
643 
644  //The boundary coordinate will be contiguous from polyline to
645  //polyline and each polyline will have the scaled arclength coordinate
646  //in the range 0->1.
647 
648  //Find the maximum coordinate
649  double zeta_max = Zeta_vertex[n_poly-1].back();
650 
651  //If we are above the maximum complain
652  if(xi[0] > zeta_max)
653  {
654  std::ostringstream error_message;
655  error_message << "Value of intrinsic coordinate " << xi[0] <<
656  "greater than maximum " << zeta_max << "\n";
657  throw
658  OomphLibError(error_message.str(),
659  "TriangleMeshPolygon::position()",
660  OOMPH_EXCEPTION_LOCATION);
661  }
662 
663  //If equal to the maximum, return the final vertex
664  if(xi[0] == zeta_max)
665  {
666  unsigned n_vertex = this->polyline_pt(n_poly-1)->nvertex();
667  r = this->polyline_pt(n_poly-1)->vertex_coordinate(n_vertex-1);
668  return;
669  }
670 
671  //Otherwise
672 
673  //Find out which polyline we are in
674  //If we've got here this should be less than n_poly
675  unsigned p = static_cast<unsigned> (floor(xi[0]));
676 
677  //If we are "above" the last polyline then throw an error
678  //This should have been caught by the trap above
679  if(p>=n_poly)
680  {
681  std::ostringstream error_message;
682  error_message
683  << "Something has gone wrong.\n"
684  << "The integer part of the input intrinsic coordinate is "
685  << p <<
686  "\nwhich is equal to or greater than the number of polylines, "
687  << n_poly << ".\n"
688  << "This should have triggered an earlier error\n";
689 
690 
691  throw OomphLibError(error_message.str(),
692  OOMPH_CURRENT_FUNCTION,
693  OOMPH_EXCEPTION_LOCATION);
694  }
695 
696  //Cache the appropriate polyline
697  TriangleMeshPolyLine* const line_pt = this->polyline_pt(p);
698 
699  //If we are at the first vertex in the polyline, return it
700  if(xi[0] == Zeta_vertex[p][0])
701  {
702  r = line_pt->vertex_coordinate(0);
703  return;
704  }
705 
706  //Otherwise loop over the other points to find the appropriate
707  //segment
708 
709  //Find the number of vertices in the chosen polyline
710  unsigned n_vertex = line_pt->nvertex();
711  //Now start from the first node in the appropriate polyline and loop up
712  for(unsigned v=1;v<n_vertex;v++)
713  {
714  //First time that zeta falls below the vertex coordinate
715  //we have something
716  if(xi[0] < Zeta_vertex[p][v])
717  {
718  double fraction = (xi[0] - Zeta_vertex[p][v-1])/
719  (Zeta_vertex[p][v] - Zeta_vertex[p][v-1]);
720  Vector<double> first = line_pt->vertex_coordinate(v-1);
721  Vector<double> last = line_pt->vertex_coordinate(v);
722  r.resize(2);
723  for(unsigned i=0;i<2;i++)
724  {
725  r[i] = first[i] + fraction*(last[i] - first[i]);
726  }
727  return;
728  }
729  }
730  }

733  /// Helper function to assign the values of the (scaled) arc-length
734  /// to each node of each polyline. The direction will be the natural
735  /// order of the vertices within the polyline.
736  void assign_zeta()
737  {
738  //Find the number of polylines
739  unsigned n_poly = this->npolyline();
740 
741  //Allocate appropriate storage for the zeta values
742  Zeta_vertex.resize(n_poly);
743 
744  //Temporary storage for the vertex coordinates
745  Vector<double> vertex_coord_first;
746  Vector<double> vertex_coord_next;
747 
748  //Set the initial value of zeta
749  double zeta_offset = 0.0;
750 
751  //Loop over the polylines
752  for(unsigned p=0;p<n_poly;++p)
753  {
754  //Cache the pointer to the polyline
755  TriangleMeshPolyLine* const line_pt = this->polyline_pt(p);
756 
757  //Find the number of vertices in the polyline
758  unsigned n_vertex = line_pt->nvertex();
759 
760  //Allocate storage and set initial value
761  Zeta_vertex[p].resize(n_vertex);
762  Zeta_vertex[p][0] = 0.0;
763 
764  //Loop over the vertices in the polyline and calculate the length
765  //between each for use as the intrinsic coordinate
766  vertex_coord_first = line_pt->vertex_coordinate(0);
767  for(unsigned v=1;v<n_vertex;v++)
768  {
769  vertex_coord_next = line_pt->vertex_coordinate(v);
770  double length =
771  sqrt(pow(vertex_coord_next[0] - vertex_coord_first[0],2.0)
772  + pow(vertex_coord_next[1] - vertex_coord_first[1],2.0));
773  Zeta_vertex[p][v] = Zeta_vertex[p][v-1] + length;
774  vertex_coord_first = vertex_coord_next;
775  }
776 
777  //Now scale the length to unity and add the offset
778  Zeta_vertex[p][0] += zeta_offset;
779  for(unsigned v=1;v<n_vertex;v++)
780  {
781  Zeta_vertex[p][v] /= Zeta_vertex[p][n_vertex-1];
782  Zeta_vertex[p][v] += zeta_offset;
783  }
784  zeta_offset += 1.0;
785  } //End of loop over polylines
786  }
787 
788  /// \short Vector of intrisic coordinate values at the nodes
789  Vector<Vector<double> > Zeta_vertex;
790 
791 };
