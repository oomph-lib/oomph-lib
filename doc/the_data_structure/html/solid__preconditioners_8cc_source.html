503  /// Apply preconditioner to r.
504 //=======================================================================
505  void PressureBasedSolidLSCPreconditioner::
506  preconditioner_solve(const DoubleVector &r, DoubleVector &z)
507  {
508 
509 #ifdef PARANOID
510  if (Preconditioner_has_been_setup==false)
511  {
512  std::ostringstream error_message;
513  error_message << "setup must be called before using preconditioner_solve";
514  throw OomphLibError(
515  error_message.str(),
516  OOMPH_CURRENT_FUNCTION,
517  OOMPH_EXCEPTION_LOCATION);
518  }
519  if (z.built())
520  {
521  if (z.nrow() != r.nrow())
522  {
523  std::ostringstream error_message;
524  error_message << "The vectors z and r must have the same number of "
525  << "of global rows";
526  throw OomphLibError(
527  error_message.str(),
528  OOMPH_CURRENT_FUNCTION,
529  OOMPH_EXCEPTION_LOCATION);
530  }
531  }
532 #endif
533 
534  double t_start_overall = TimingHelpers::timer();
535  double t_start = TimingHelpers::timer();
536  double t_end=0;
537 
538  // if z is not setup then give it the same distribution
539  if (!z.built())
540  {
541  z.build(r.distribution_pt(),0.0);
542  }
543 
544  // Step 1 - apply approximate Schur inverse to pressure unknowns (block 1)
545  // -----------------------------------------------------------------------
546 
547  // Working vectors
548  DoubleVector temp_vec;
549  DoubleVector another_temp_vec;
550 
551  // Copy pressure values from residual vector to temp_vec:
552  // Loop over all entries in the global vector (this one
553  // includes displacement/position and pressure dofs in some random fashion)
554  this->get_block_vector(1,r,temp_vec);
555 
556 
557  if(Doc_time)
558  {
559  t_end=TimingHelpers::timer();
560  oomph_info << "LSC prec solve: Time for get block vector: "
561  << t_end-t_start << std::endl;
562  t_start=TimingHelpers::timer();
563  }
564 
565  // NOTE: The vector temp_vec now contains the vector r_p.
566 
567  // Solve first pressure Poisson system
568 #ifdef PARANOID
569  // check a solver has been set
570  if (P_preconditioner_pt==0)
571  {
572  std::ostringstream error_message;
573  error_message << "P_preconditioner_pt has not been set.";
574  throw OomphLibError(
575  error_message.str(),
576  OOMPH_CURRENT_FUNCTION,
577  OOMPH_EXCEPTION_LOCATION);
578  }
579 #endif
580 
581  // use some Preconditioner's preconditioner_solve function
582  P_preconditioner_pt->preconditioner_solve(temp_vec, another_temp_vec);
583 
584 
585  if(Doc_time)
586  {
587  t_end=TimingHelpers::timer();
588  oomph_info << "LSC prec solve: First P solve [nrow="
589  << P_preconditioner_pt->nrow() << "]: "
590  << t_end-t_start << std::endl;
591  t_start=TimingHelpers::timer();
592  }
593 
594 
595  // NOTE: The vector another_temp_vec now contains the vector P^{-1} r_p
596 
597  // Multiply another_temp_vec by matrix E and stick the result into temp_vec
598  temp_vec.clear();
599  if (Form_BFBt_product)
600  {
601  E_mat_vec_pt->multiply(another_temp_vec, temp_vec);
602  }
603  else
604  {
605  QBt_mat_vec_pt->multiply(another_temp_vec, temp_vec);
606  another_temp_vec.clear();
607  F_mat_vec_pt->multiply(temp_vec,another_temp_vec);
608  temp_vec.clear();
609  QBt_mat_vec_pt->multiply_transpose(another_temp_vec, temp_vec);
610  }
611 
612 
613  if(Doc_time)
614  {
615  t_end=TimingHelpers::timer();
616  oomph_info << "LSC prec solve: E matrix vector product: "
