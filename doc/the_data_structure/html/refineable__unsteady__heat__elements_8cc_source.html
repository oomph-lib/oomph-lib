34 /// Add element's contribution to the elemental
35 /// residual vector and/or Jacobian matrix.
36 /// flag=1: compute both
37 /// flag=0: compute only residual vector
38 //========================================================================
39 template<unsigned DIM>
40 void RefineableUnsteadyHeatEquations<DIM>::
41 fill_in_generic_residual_contribution_ust_heat(Vector<double> &residuals,
42  DenseMatrix<double> &jacobian,
43  unsigned flag)
44 {
45 
46 //Find out how many nodes there are in the element
47 unsigned n_node = nnode();
48 
49 // Get continuous time from timestepper of first node
50 double time=node_pt(0)->time_stepper_pt()->time_pt()->time();
51 
52 //Find the index at which the unknown is stored
53  unsigned u_nodal_index = this->u_index_ust_heat();
54 
55 //Set up memory for the shape and test functions
56  Shape psi(n_node), test(n_node);
57  DShape dpsidx(n_node,DIM), dtestdx(n_node,DIM);
58 
59 //Set the value of n_intpt
60 unsigned n_intpt = integral_pt()->nweight();
61 
62 //Set the Vector to hold local coordinates
63 Vector<double> s(DIM);
64 
65  //Get Alpha and beta parameters number
66  double alpha_local = this->alpha();
67  double beta_local = this->beta();
68 
69 //Integers used to store the local equation number and local unknown
70 //indices for the residuals and jacobians
71 int local_eqn=0, local_unknown=0;
72 
73 //Local storage for pointers to hang info objects
74  HangInfo *hang_info_pt=0, *hang_info2_pt=0;
75 
76 //Local variable to determine the ALE stuff
77  bool ALE_is_disabled_flag = this->ALE_is_disabled;
78 
79 //Loop over the integration points
80 for(unsigned ipt=0;ipt<n_intpt;ipt++)
81 {
82 
83  //Assign values of s
84  for(unsigned i=0;i<DIM;i++) s[i] = integral_pt()->knot(ipt,i);
85 
86  //Get the integral weight
87  double w = integral_pt()->weight(ipt);
88 
89  //Call the derivatives of the shape and test functions
90  double J = this->
91  dshape_and_dtest_eulerian_at_knot_ust_heat(ipt,psi,dpsidx,test,dtestdx);
92 
93  //Premultiply the weights and the Jacobian
94  double W = w*J;
95 
96  //Calculate local values of the function
97  double dudt=0.0;
98  double interpolated_u=0.0;
99 
100  //This needs to be a Vector to be ANSI C++, initialise to zero
101  Vector<double> interpolated_x(DIM,0.0);
102  Vector<double> interpolated_dudx(DIM,0.0);
103  Vector<double> mesh_velocity(DIM,0.0);
104 
105 
106  //Calculate function value and derivatives:
107  //-----------------------------------------
108 
109  // Loop over nodes
110  for(unsigned l=0;l<n_node;l++)
111  {
112  //Get the value of the unknown at the node
113  double u_value = this->nodal_value(l,u_nodal_index);
114  interpolated_u += u_value*psi(l);
115  dudt += this->du_dt_ust_heat(l)*psi(l);
116  // Loop over directions
117  for(unsigned j=0;j<DIM;j++)
118  {
119  interpolated_x[j] += nodal_position(l,j)*psi(l);
120  interpolated_dudx[j] += u_value*dpsidx(l,j);
121  }
122  }
123 
124  if (!ALE_is_disabled_flag)
125  {
126  for(unsigned l=0;l<n_node;l++)
127  {
128  // Loop over directions
129  for(unsigned j=0;j<DIM;j++)
130  {
131  mesh_velocity[j] += dnodal_position_dt(l,j)*psi(l);
132  }
133  }
134  }
135 
136  //Get body force
137  double source;
138  this->get_source_ust_heat(time,ipt,interpolated_x,source);
139 
140 
141  // Assemble residuals and Jacobian
142  //================================
143 
144  // Loop over the nodes for the test functions
145  //-------------------------------------------
146  for(unsigned l=0;l<n_node;l++)
147  {
148  //Local variables to store the number of master nodes and
149  //the weight associated with the shape function if the node is hanging
150  unsigned n_master=1; double hang_weight=1.0;
151  //Local bool (is the node hanging)
152  bool is_node_hanging = this->node_pt(l)->is_hanging();
153 
154 
155  //If the node is hanging, get the number of master nodes
156  if(is_node_hanging)
157  {
158  hang_info_pt = this->node_pt(l)->hanging_pt();
159  n_master = hang_info_pt->nmaster();
160  }
161  //Otherwise there is just one master node, the node itself
162  else
163  {
164  n_master = 1;
165  }
166 
167  //Loop over the number of master nodes
168  for(unsigned m=0;m<n_master;m++)
169  {
170  //Get the local equation number and hang_weight
171  //If the node is hanging
172  if(is_node_hanging)
173  {
174  //Read out the local equation from the master node
175  local_eqn = this->local_hang_eqn(hang_info_pt->master_node_pt(m),
176  u_nodal_index);
177  //Read out the weight from the master node
178  hang_weight = hang_info_pt->master_weight(m);
179  }
180  //If the node is not hanging
181  else
182  {
183  //The local equation number comes from the node itself
184  local_eqn = this->nodal_local_eqn(l,u_nodal_index);
185  //The hang weight is one
186  hang_weight = 1.0;
187  }
188 
189  //If the nodal equation is not a boundary condition
190  if(local_eqn >= 0)
191  {
192  // Add body force/source term and time derivative
193  residuals[local_eqn]+=(alpha_local*dudt + source)*test(l)*W*hang_weight;
194 
195  // Mesh velocity and Laplace operator itself
196  for(unsigned k=0;k<DIM;k++)
197  {
198  double tmp=dtestdx(l,k)*beta_local;
199  if (!ALE_is_disabled_flag) tmp -= alpha_local*mesh_velocity[k]*test(l);
200  residuals[local_eqn] += interpolated_dudx[k]*tmp*W*hang_weight;
201  }
202 
203  // Calculate the Jacobian
204  if(flag)
205  {
206  //Local variables to store the number of master nodes
207  //and the weights associated with each hanging node
208  unsigned n_master2=1; double hang_weight2=1.0;
209  //Loop over the nodes for the variables
210  for(unsigned l2=0;l2<n_node;l2++)
211  {
212  //Local bool (is the node hanging)
213  bool is_node2_hanging = this->node_pt(l2)->is_hanging();
214  //If the node is hanging, get the number of master nodes
215  if(is_node2_hanging)
216  {
217  hang_info2_pt = this->node_pt(l2)->hanging_pt();
218  n_master2 = hang_info2_pt->nmaster();
219  }
220  //Otherwise there is one master node, the node itself
221  else
222  {
223  n_master2 = 1;
224  }
225 
226  //Loop over the master nodes
227  for(unsigned m2=0;m2<n_master2;m2++)
228  {
229  //Get the local unknown and weight
230  //If the node is hanging
231  if(is_node2_hanging)
232  {
233  //Read out the local unknown from the master node
234  local_unknown =
235  this->local_hang_eqn(hang_info2_pt->master_node_pt(m2),
236  u_nodal_index);
237  //Read out the hanging weight from the master node
238  hang_weight2 = hang_info2_pt->master_weight(m2);
239  }
240  //If the node is not hanging
241  else
242  {
243  //The local unknown number comes from the node
244  local_unknown = this->nodal_local_eqn(l2,u_nodal_index);
245  //The hang weight is one
246  hang_weight2 = 1.0;
247  }
248 
249  //If the unknown is not pinned
250  if(local_unknown >= 0)
251  {
252  //Add contribution to Elemental Matrix
253  // Mass matrix
254  jacobian(local_eqn,local_unknown)
255  += alpha_local*test(l)*psi(l2)*
256  this->node_pt(l2)->time_stepper_pt()->weight(1,0)
257  *W*hang_weight*hang_weight2;
258 
259  // Laplace operator and mesh veloc bit
260  for(unsigned k=0;k<DIM;k++)
261  {
262  double tmp=dtestdx(l,k)*beta_local;
263  if(!ALE_is_disabled_flag)
264  {
265  tmp -=
266  alpha_local*mesh_velocity[k]*test(l);
267  }
268  jacobian(local_eqn,local_unknown) +=
269  dpsidx(l2,k)*tmp*W*hang_weight*hang_weight2;
270  }
271  }
272  } //End of loop over master nodes
273  }
274  } //End of Jacobian calculation
275  }
276  } //End of loop over master nodes for residuals
277  } //End of loop over nodes
278 
279 } // End of loop over integration points

282 }
