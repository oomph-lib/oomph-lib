33  /// \short Setup the matrix vector product operator.
34  /// WARNING: This class is wrapper to Trilinos Epetra matrix vector
35  /// multiply methods, if Trilinos is not installed then this class will
36  /// function as expected, but there will be no computational speed gain.
37  /// By default the Epetra_CrsMatrix::multiply(...) are employed.
38  /// The optional argument col_dist_pt is the distribution of:
39  /// x if using multiply(...) or y if using multiply_transpose(...)
40  /// where this is A x = y. By default, this is assumed to the uniformly
41  /// distributed based on matrix_pt->ncol().
42  //============================================================================
43  void MatrixVectorProduct::setup(CRDoubleMatrix* matrix_pt,
44  const LinearAlgebraDistribution* col_dist_pt)
45  {
46  // clean memory
47  this->clean_up_memory();
48 
49  // (re)build distribution_pt
50  this->build_distribution(matrix_pt->distribution_pt());
51 
52  // store the number of columns
53  Ncol = matrix_pt->ncol();
54 
55  // determine whether we are using trilinos
56  Using_trilinos=false;
57 #ifdef OOMPH_HAS_TRILINOS
58 #ifdef OOMPH_HAS_MPI
59  if (MPI_Helpers::mpi_has_been_initialised())
60  {
61  Using_trilinos=true;
62  }
63 #else
64  Using_trilinos=true;
65 #endif
66 #endif
67 
68  // create the column distribution map, if a distribution is not provided,
69  // create a uniformly distributed based on matrix_pt->ncol().
70  // Otherwise, use the provided distribution.
71  if(col_dist_pt == 0)
72  {
73  Column_distribution_pt = new LinearAlgebraDistribution
74  (matrix_pt->distribution_pt()->communicator_pt(),
75  matrix_pt->ncol(),matrix_pt->distribution_pt()->distributed());
76  }
77  else
78  {
79  Column_distribution_pt = new LinearAlgebraDistribution(col_dist_pt);
80  }
81 
82  // setup the operator
83  if (Using_trilinos)
84  {
85 #ifdef OOMPH_HAS_TRILINOS
86  double t_start = TimingHelpers::timer();
87  Epetra_matrix_pt =
88  TrilinosEpetraHelpers::create_distributed_epetra_matrix
89  (matrix_pt,Column_distribution_pt);
90  double t_end = TimingHelpers::timer();
91  oomph_info << "Time to build epetra matrix [sec] : "
92  << t_end - t_start << std::endl;
93 #endif
94  }
95  else
96  {
97  double t_start = TimingHelpers::timer();
98  Oomph_matrix_pt = new CRDoubleMatrix(*matrix_pt);
99  double t_end = TimingHelpers::timer();
100  oomph_info << "Time to copy CRDoubleMatrix [sec] : "
101  << t_end - t_start << std::endl;
102  }
103  }

106  /// \short Apply the operator to the vector x and return the result in
107  /// the vector y
108  //============================================================================
109  void MatrixVectorProduct::multiply(const DoubleVector& x,
110  DoubleVector& y) const
111  {
112 #ifdef PARANOID
113  // check that the distribution of x is setup
114  if (!x.built())
115  {
116  std::ostringstream error_message_stream;
117  error_message_stream
118  << "The distribution of the vector x must be setup";
119  throw OomphLibError(error_message_stream.str(),
120  OOMPH_CURRENT_FUNCTION,
121  OOMPH_EXCEPTION_LOCATION);
122  }
123 
124  // Check to see if the distribution of the matrix column is the same as the
125  // distribution of the vector to be operated on.
126  if (*this->Column_distribution_pt != *x.distribution_pt())
127  {
128  std::ostringstream error_message_stream;
129  error_message_stream
130  << "The distribution of the x Vector is not the same as"
131  << " the column distribution.";
132  throw OomphLibError(error_message_stream.str(),
133  OOMPH_CURRENT_FUNCTION,
134  OOMPH_EXCEPTION_LOCATION);
135  }
136 
137  // if y is setup then it should have the same distribution as
138  // the matrix used to set this up.
139  if (y.built())
140  {
141  if (!(*y.distribution_pt() == *this->distribution_pt()))
142  {
143  std::ostringstream error_message_stream;
144  error_message_stream
145  << "The y vector is setup and therefore must have the same "
146  << "distribution as the matrix used to set up the MatrixVectorProduct";
147  throw OomphLibError(error_message_stream.str(),
148  OOMPH_CURRENT_FUNCTION,
149  OOMPH_EXCEPTION_LOCATION);
150  }
151  }
152 #endif
153 
154  // if y is not setup then setup the distribution
155  if (!y.built())
156  {
157  // Resize and initialize the solution vector
158  y.build(this->distribution_pt(),0.0);
159  }
160 
161  // apply the operator
162  if (Using_trilinos)
163  {
164 #ifdef OOMPH_HAS_TRILINOS
165  trilinos_multiply_helper(x,y);
166 #endif
167  }
168  else
169  {
170  Oomph_matrix_pt->multiply(x,y);
171  }
172  }

193  if (*this->distribution_pt() != *x.distribution_pt())
194  {
195  std::ostringstream error_message_stream;

197  << "This class assumes that the y vector has a uniform "

200  OOMPH_CURRENT_FUNCTION,
201  OOMPH_EXCEPTION_LOCATION);
202  }
203  // if y is setup then it should have the same distribution as x
204  if (y.built())
205  {
206  if (!(*y.distribution_pt() == *this->Column_distribution_pt))
207  {
208  std::ostringstream error_message_stream;
209  error_message_stream
210  << "The y vector is setup and therefore must have the same "
211  << "distribution as the vector x";
212  throw OomphLibError(error_message_stream.str(),
213  OOMPH_CURRENT_FUNCTION,
214  OOMPH_EXCEPTION_LOCATION);
215  }
216  }
217 #endif
218 
219  // if y is not setup then setup the distribution
220  if (!y.built())
221  {
222  // Resize and initialize the solution vector
223  y.build(this->Column_distribution_pt,0.0);
224  }
225 
226  // apply the transpose operator
227  if (Using_trilinos)
228  {
229 #ifdef OOMPH_HAS_TRILINOS
230  trilinos_multiply_transpose_helper(x,y);
231 #endif
232  }
233  else
