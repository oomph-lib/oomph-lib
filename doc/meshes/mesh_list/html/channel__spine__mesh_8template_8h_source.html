118  /// \short General node update function implements pure virtual function
119  /// defined in SpineMesh base class and performs specific node update
120  /// actions: along vertical spines
121  virtual void spine_node_update(SpineNode* spine_node_pt)
122  {
123  // Get spine node's fraction along the spine
124  double W = spine_node_pt->fraction();
125 
126  // Get local coordinates
127  Vector<double> s_wall(1);
128  s_wall[0] = spine_node_pt->spine_pt()->geom_parameter(0);
129 
130  // Get position vector to wall
131  Vector<double> position(2);
132  spine_node_pt->spine_pt()->geom_object_pt(0)->position(s_wall,position);
133 
134  //Set the value of y
135  spine_node_pt->x(1) = this->Ymin + W*position[1];
136  }

138  /// \short Return the value of the x-coordinate at the node given by the
139  /// local node number (xnode, ynode) in the element (xelement,yelement).
140  /// The description is in a "psudeo" two-dimensional coordinate system,
141  /// so the range of xelement is [0,Nx-1], yelement is [0,Ny-1], and
142  /// that of xnode and ynode is [0,Np-1]. The default is to return
143  /// nodes that are equally spaced in the x coodinate.
144  virtual double x_spacing_function(unsigned xelement, unsigned xnode,
145  unsigned yelement, unsigned ynode)
146  {
147  // Calculate the values of equal increments in nodal values in left region
148  double xstep1 = Lx0/((this->Np-1)*Nx0);
149  // Calculate the values of equal increments in nodal values in centre region
150  double xstep2 = Lx1/((this->Np-1)*Nx1);
151  // Calculate the values of equal increments in nodal values in right region
152  double xstep3 = Lx2/((this->Np-1)*Nx2);
153 
154  // left region
155  if (xelement<Nx0)
156  {
157  //Return the appropriate value
158  return (this->Xmin + xstep1*((this->Np-1)*xelement + xnode));
159  }
160  // centre region
161  else if (xelement<Nx0+Nx1)
162  {
163  //Return the appropriate value
164  return (Lx0 + xstep2*((this->Np-1)*(xelement-Nx0) + xnode));
165  }
166  // right region
167  else if (xelement<Nx0+Nx1+Nx2)
168  {
169  //Return the appropriate value
170  return (Lx0+Lx1 + xstep3*((this->Np-1)*(xelement-Nx0-Nx1) + xnode));
171  }
172  else
173  {
174  throw OomphLibError("Should not have got here",
175  OOMPH_CURRENT_FUNCTION,
176  OOMPH_EXCEPTION_LOCATION);
177  }
178  // Dummy return to keep compiler from barking
179  return 0.0;
180  }
