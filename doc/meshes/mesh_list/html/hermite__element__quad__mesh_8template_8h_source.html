69  /// \short Mesh Constructor (for a uniformly spaced mesh). Takes the following
70  /// arguments : nx : number of elements in x direction;
71  /// ny : number of elements in y direction;
72  /// domain : topologically rectangular domain;
73  /// periodic_in_x : flag specifying if the mesh is periodic in

82

85

87  Nelement.resize(2);
88  Nelement[0] = nx;
89  Nelement[1] = ny;

157  /// \short returns the macro element position of the node that is the x-th
158  /// node along from the LHS and the y-th node up from the lower edge
159  void macro_coordinate_position(const unsigned& node_num_x,
160  const unsigned& node_num_y,
161  Vector<double>& macro_element_position)
162  {
163  // compute macro element position in uniformly spaced mesh
164  macro_element_position[0] = 2*node_num_x/double(Nelement[0])-1;
165  macro_element_position[1] = 2*node_num_y/double(Nelement[1])-1;
166 
167  // if a non unform spacing function is provided
168  if (Node_spacing_fn != 0)
169  {
170  Vector<double> temp(macro_element_position);
171  (*Node_spacing_fn)(temp,macro_element_position);
172  }
173  }
