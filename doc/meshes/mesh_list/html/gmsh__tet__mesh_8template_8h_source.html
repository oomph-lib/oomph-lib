268  class TetEdge
269  {
270  public:
271 
272  /// \short Constructor: Pass two vertices, identified by their indices
273  /// Edge "direction" is from lower vertex to higher vertex id so
274  /// can compare if we're dealing with the same one...
275  TetEdge(const unsigned& vertex1, const unsigned& vertex2)
276  {
277  if (vertex1>vertex2)
278  {
279  Reversed=true;
280  Vertex_pair=std::make_pair(vertex2,vertex1);
281  }
282  else if (vertex1<vertex2)
283  {
284  Reversed=false;
285  Vertex_pair=std::make_pair(vertex1,vertex2);
286  }
287  else
288  {
289  throw OomphLibError(
290  "Muppet! You can't build an edge from one vertex to the same vertex!",
291  OOMPH_CURRENT_FUNCTION,
292  OOMPH_EXCEPTION_LOCATION);
293  }
294  }
295 
296 
297  /// First vertex id
298  unsigned first_vertex_id() const
299  {
300  return Vertex_pair.first;
301  }
302 
303  /// Second vertex id
304  unsigned second_vertex_id() const
305  {
306  return Vertex_pair.second;
307  }
308 
309 
310  /// \short Edge is reversed in the sense that vertex1 actually has a higher
311  /// id than vertex2 (when specified in the constructor)
312  bool is_reversed() const
313  {
314  return Reversed;
315  }
316 
317  /// \short Comparison operator: Edges are identical if their sorted
318  /// (and therefore possibly reversed) vertex ids agree
319  bool operator ==(const TetEdge& tet_edge) const
320  {
321  return ((tet_edge.first_vertex_id()==Vertex_pair.first)&&
322  (tet_edge.second_vertex_id()==Vertex_pair.second));
323  }
324 
325  /// \short Comparison operator. Lexicographic comparison based on
326  /// vertex ids
327  bool operator < (const TetEdge& tet_edge) const
328  {
329  if ((tet_edge.first_vertex_id()==Vertex_pair.first)&&
330  (tet_edge.second_vertex_id()==Vertex_pair.second))
331  {
332  return false;
333  }
334  else
335  {
336  if (tet_edge.first_vertex_id()==Vertex_pair.first)
337  {
338  return (tet_edge.second_vertex_id()<Vertex_pair.second);
339  }
340  else
341  {
342  return (tet_edge.first_vertex_id()<Vertex_pair.first);
343  }
344  }
345  }

456  void create_mesh_from_msh_file()
457  {
458  // Create filename from stem
459  std::string mesh_file_name=Gmsh_parameters_pt->
460  geo_and_msh_file_stem() + ".msh";
461 
462 // Keep around if we ever write a version where name is passed in.
463 /* // Check extension */
464 /* #ifdef PARANOID */
465 /* std::string mesh_file_stem= */
466 /* mesh_file_name.substr(0,mesh_file_name.length()-4); */
467 /* std::string test=mesh_file_stem+".msh"; */
468 /* if (test!=mesh_file_name) */
469 /* { */

472 /* throw OomphLibError(error_msg, */
473 /* OOMPH_CURRENT_FUNCTION, */
474 /* OOMPH_EXCEPTION_LOCATION); */
475 /* } */
476 /* #endif */
477 
478  // Open wide...
479  std::ifstream mesh_file(mesh_file_name.c_str(),std::ios_base::in);
480 
481 // Check that the file actually opened correctly
482 #ifdef PARANOID
483  if(!mesh_file.is_open())
484  {
485  std::string error_msg("Failed to open mesh file: ");
486  error_msg += "\"" + mesh_file_name + "\".";
487  throw OomphLibError(error_msg,
488  OOMPH_CURRENT_FUNCTION,
489  OOMPH_EXCEPTION_LOCATION);
490  }
491 #endif
492 
493  // First line: Must be "$MeshFormat"
494  //----------------------------------
495  std::string line;
496  mesh_file >> line;
497  if (line!="$MeshFormat")
498  {
499  std::string error_msg
500  ("First line has to contain the string \"$MeshFormat\"; ");
501  error_msg += " yours contains: " + line;
502  throw OomphLibError(error_msg,
503  OOMPH_CURRENT_FUNCTION,
504  OOMPH_EXCEPTION_LOCATION);
505  }
506 
507  // Rest of line
508  mesh_file >> line;
509 
510 
511  // Nodes
512  //------
513 
514  // Now keep reading until we find the nodes
515  bool keep_going=true;
516  while (keep_going)
517  {
518  std::string line;
519  mesh_file >> line;
520  if (line=="$Nodes")
521  {
522  keep_going=false;
523  }
524  }
525  unsigned nnod=0;
526  mesh_file >> nnod;
527  std::map<unsigned,Vector<double> > node_coordinate;
528  for (unsigned j=0;j<nnod;j++)
529  {
530  unsigned node_number=0;
531  mesh_file >> node_number;
532 
533  // Read rest of line word by word
534  std::string s;
535  std::getline(mesh_file, s);
536  std::istringstream iss(s);
537  std::string sub;
538  while(iss >> sub)
539  {
540  node_coordinate[node_number].push_back(atof(sub.c_str()));
541  }
542  }
543 
544  mesh_file >> line;
545  if (line!="$EndNodes")
546  {
547  std::string error_msg("Line has to contain the string \"$EndNodes\"; ");
548  error_msg += " yours contains: " + line;
549  throw OomphLibError(error_msg,
550  OOMPH_CURRENT_FUNCTION,
551  OOMPH_EXCEPTION_LOCATION);
552  }
553 
554  // Rewind
555  mesh_file.clear();
556  mesh_file.seekg(0);
557 
558  mesh_file >> line;
559  if (line!="$MeshFormat")
560  {
561  std::string error_msg
562  ("First line has to contain the string \"$MeshFormat\"; ");
563  error_msg += " yours contains: " + line;
564  throw OomphLibError(error_msg,
565  OOMPH_CURRENT_FUNCTION,
566  OOMPH_EXCEPTION_LOCATION);
567  }
568 
569  // Rest of line
570  mesh_file >> line;
571 
572  // Storage for boundaries the nodes are on
573  std::map<unsigned,std::set<unsigned> > one_based_boundaries_of_node;
574 
575  // Elements
576  //---------
577 
578  // node number = boundary_node[one_based_bound_id][...]
579  std::map<unsigned,std::set<unsigned> > boundary_node;
580 
581  // element number = region_element[one_based_region_id][...]
582  std::map<unsigned,std::set<unsigned> > region_element;
583 
584  // Now keep reading until we find the nodes
585  keep_going=true;
586  while (keep_going)
587  {
588  std::string line;
589  mesh_file >> line;
590  if (line=="$Elements")
591  {
592  keep_going=false;
593  }
594  }
595 
596 
597  unsigned highest_one_based_boundary_id=0;
598  unsigned n_tet_el=0;
599 
600  unsigned nel=0;
601  mesh_file >> nel;
602  std::map<unsigned,Vector<unsigned> > element_node_index;
603  for (unsigned e=0;e<nel;e++)
604  {
605  // For each element the msh file provides:
606  //
607  // elm-number elm-type number-of-tags < tag > ... node-number-list
608  //
609  // By default, the first tag is the number of the physical entity to
610  // which the element belongs; the second is the number of the elementary
611  // geometrical entity to which the element belongs; the third is the
612  // number of mesh partitions to which the element belongs, followed by
613  // the partition ids (negative partition ids indicate ghost cells). A
614  // zero tag is equivalent to no tag. Gmsh and most codes using the MSH 2
615  // format require at least the first two tags (physical and elementary
616  // tags).
617  unsigned el_number=0;
618  mesh_file >> el_number;
619 
620 
621  unsigned el_type=0;
622  mesh_file >> el_type;
623 
624  switch(el_type)
625  {
626  case 1:
627  //oomph_info << "Line element" << std::endl;
628  break;
629 
630  case 2:
631  //oomph_info << "Triangle" << std::endl;
632  break;
633 
634  case 4:
635  n_tet_el++;
636  //oomph_info << "Tet" << std::endl;
637  break;
638 
639  case 15:
640  //oomph_info << "Point" << std::endl;
641  break;
642 
643  default:
644  std:: string error_msg("Can't handle element type: ");
645  error_msg += oomph::StringConversion::to_string(el_type);
646  throw OomphLibError(error_msg,
647  OOMPH_CURRENT_FUNCTION,
648  OOMPH_EXCEPTION_LOCATION);
649  }
650 
651  unsigned ntags;
652  mesh_file >> ntags;
653 
654  // Gmesh produces at least two flags:
655 
656  // Physical tag = what we use as boundary or region IDs.
657  int physical_tag=0;
658  mesh_file >> physical_tag;
659 
660  // Geometric tag; not something we use
661  int geom_tag=0;
662  mesh_file >> geom_tag;
663 
664  Vector<int> other_tags;
665  for (unsigned i=2;i<ntags;i++)
666  {
667  int tag=0;
668  mesh_file >> tag;
669  other_tags.push_back(tag);
670  }
671 
672 
673  // Now read the rest: node numbers
674  // https://stackoverflow.com/questions/16991002/getting-a-line-from-a-file-and-then-reading-word-by-word-c
675  std::string s;
676  std::getline(mesh_file,s);
677  std::istringstream iss(s);
678  std::string sub;
679  Vector<int> other_ints;
680  while(iss >> sub)
681  {
682  other_ints.push_back(atoi(sub.c_str()));
683  }
684  unsigned n_el_nod=other_ints.size();
685  for (unsigned j=0;j<n_el_nod;j++)
686  {
687  unsigned node_number=unsigned(other_ints[j]);
688  element_node_index[el_number].push_back(node_number);
689 
690  // If the element is a triangle, add node to boundary
691  // lookup scheme (if tag is not zero, i.e. hasn't been specified)
692  if (el_type==2)
693  {
694  if (physical_tag!=0)
695  {
696  boundary_node[unsigned(physical_tag)].insert(node_number);
697  one_based_boundaries_of_node[node_number].insert(
698  unsigned(physical_tag));
699  if (unsigned(physical_tag)>highest_one_based_boundary_id)
700  {
701  highest_one_based_boundary_id=physical_tag;
702  }
703  }
704  }
705  }
706  // If it's a bulk element (tet) and the physical tag (region id)
707  // is not equal to 0 add it to region list
708  if (el_type==4)
709  {
