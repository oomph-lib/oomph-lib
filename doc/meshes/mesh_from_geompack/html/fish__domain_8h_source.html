479 void FishDomain::r_upper_fin_N(const unsigned& t,
480  const Vector<double>& zeta,
481  Vector<double>& r)
482 {
483  // Right end of fish back
484  Vector<double> x(1);
485  x[0]=Xi_tail;
486  Vector<double> r_fish(2);
487  Back_pt->position(t,x,r_fish);
488 
489  // Top end of fin
490  Vector<double> r_fin(2);
491  r_fin[0]=X_fin;
492  r_fin[1]=Y_fin;
493 
494 
495  // Straight line along upper fin
496  r[0]=r_fish[0]+(r_fin[0]-r_fish[0])*0.5*(zeta[0]+1.0);
497  r[1]=r_fish[1]+(r_fin[1]-r_fish[1])*0.5*(zeta[0]+1.0);
498 
499 
500 
501 }
