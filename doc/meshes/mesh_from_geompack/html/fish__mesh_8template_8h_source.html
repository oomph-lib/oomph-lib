194 /// Refineable fish shaped mesh with MacroElement-based node update.
195 /// The fish's back is represented by a specified geometric object.
196 /// Some or all of the geometric Data in that geometric object
197 /// may contain unknowns in the global Problem. The dependency
198 /// on these unknowns is taken into account when setting up
199 /// the Jacobian matrix of the elements. For this purpose,
200 /// the element (whose type is specified by the template parameter)
201 /// must inherit from MacroElementNodeUpdateElementBase.
202 //========================================================================
203 template<class ELEMENT>
204 class MacroElementNodeUpdateRefineableFishMesh :
205 public virtual MacroElementNodeUpdateMesh,
206 public virtual RefineableFishMesh<ELEMENT>
207 {
208 
209 public:
210 
211  /// \short Constructor: Pass pointer GeomObject that defines
212  /// the fish's back and pointer to timestepper
213  /// (defaults to (Steady) default timestepper defined in Mesh).
214  MacroElementNodeUpdateRefineableFishMesh(GeomObject* back_pt,
215  TimeStepper* time_stepper_pt=&Mesh::Default_TimeStepper) :
216  FishMesh<ELEMENT>(back_pt,time_stepper_pt),
217  RefineableFishMesh<ELEMENT>(time_stepper_pt)
218  {
219 #ifdef PARANOID
220  ELEMENT* el_pt=new ELEMENT;
221  if (dynamic_cast<MacroElementNodeUpdateElementBase*>(el_pt)==0)
222  {
223  std::ostringstream error_message;
224  error_message
225  << "Base class for ELEMENT in "
226  << "MacroElementNodeUpdateRefineableFishMesh needs"
227  << "to be of type MacroElementNodeUpdateElement!\n";
228  error_message << "Whereas it is: typeid(el_pt).name()"
229  << typeid(el_pt).name()
230  << std::endl;
231 
232  std::string function_name =
233  "MacroElementNodeUpdateRefineableFishMesh::\n";
234  function_name +=
235  "MacroElementNodeUpdateRefineableFishMesh()";
236 
237  throw OomphLibError(error_message.str(),
238  OOMPH_CURRENT_FUNCTION,
239  OOMPH_EXCEPTION_LOCATION);
240  }
241  delete el_pt;
242 #endif
243 
244 
245  // Setup all the information that's required for MacroElement-based
246  // node update: Tell the elements that their geometry depends on the
247  // fishback geometric object
248  unsigned n_element = this->nelement();
249  for(unsigned i=0;i<n_element;i++)
250  {
251  // Upcast from FiniteElement to the present element
252  ELEMENT *el_pt = dynamic_cast<ELEMENT*>(this->element_pt(i));
253 
254 #ifdef PARANOID
255  // Check if cast is successful
256  MacroElementNodeUpdateElementBase* m_el_pt=dynamic_cast<
257  MacroElementNodeUpdateElementBase*>(el_pt);
258  if (m_el_pt==0)
259  {
260  std::ostringstream error_message;
261  error_message
262  << "Failed to upcast to MacroElementNodeUpdateElementBase\n";
263  error_message
264  << "Element must be derived from MacroElementNodeUpdateElementBase\n";
265  error_message << "but it is of type " << typeid(el_pt).name();
266 
267  std::string function_name =
268  "MacroElementNodeUpdateRefineableFishMesh::\n";
269  function_name += "MacroElementNodeUpdateRefinableFishMesh()";
270 
271  throw OomphLibError(error_message.str(),
272  OOMPH_CURRENT_FUNCTION,
273  OOMPH_EXCEPTION_LOCATION);
274  }
275 #endif
276  // There's just one GeomObject
277  Vector<GeomObject*> geom_object_pt(1);
278  geom_object_pt[0] = this->Back_pt;
279 
280  // Tell the element which geom objects its macro-element-based
281  // node update depends on
282  el_pt->set_node_update_info(geom_object_pt);
283  }
284 
285  // Add the geometric object(s) for the wall to the mesh's storage
286  Vector<GeomObject*> geom_object_pt(1);
287  geom_object_pt[0] = this->Back_pt;
288  MacroElementNodeUpdateMesh::set_geom_object_vector_pt(geom_object_pt);
289 
290  // Fill in the domain pointer to the mesh's storage in the base class
291  MacroElementNodeUpdateMesh::macro_domain_pt()=this->domain_pt();
292 
293  }

377  /// \short Update nodal position at time level t (t=0: present;
378  /// t>0: previous)
379  void algebraic_node_update(const unsigned& t, AlgebraicNode*& node_pt)
380  {
381  // Update with the update function for the node's first (default)
382  // node update fct
383  unsigned id=node_pt->node_update_fct_id();
384 
385  // Upper/lower body
386  if ((id==this->Lower_body)||(id==this->Upper_body))
387  {
388  node_update_in_body(t,node_pt);
389  }
390  // Upper/lower fin
391  else if ((id==this->Lower_fin)||(id==this->Upper_fin))
392  {
393  node_update_in_fin(t,node_pt);
394  }
395  else
396  {
397  std::ostringstream error_message;
398  error_message << "The node update fct id is "
399  << id << ", but it should only be one of "
400  << this->Lower_body << ", "
401  << this->Upper_body << ", "
402  << this->Lower_fin << " or "
403  << this->Upper_fin << std::endl;
404  std::string function_name = "AlgebraicFishMesh::algebraic_node_update()";
405 
406  throw OomphLibError(error_message.str(),
407  OOMPH_CURRENT_FUNCTION,
408  OOMPH_EXCEPTION_LOCATION);
409  }
410  }

440  /// \short Update the geometric references that are used
441  /// to update node after mesh adaptation.
442  /// We're assuming that the GeomObject that specifies
443  /// the fish back does not have sub-objects, therefore
444  /// no update is required -- all reference
445  /// values can simply be scaled. We simply
446  /// paranoid-check that this is actually the case,
447  /// by checking if locate_zeta() returns the
448  /// original data.
449  void update_node_update(AlgebraicNode*& node_pt)
450  {
451 #ifdef PARANOID
452 
453  // Get the start and end Lagrangian coordinates on the
454  // wall from the domain:
455 
456  /// Start coordinate on wall (near nose)
457  double xi_nose=this->Domain_pt->xi_nose();
458 
459  /// End coordinate on wall (near tail)
460  double xi_tail=this->Domain_pt->xi_tail();
461 
462  /// Check halfway along the object
463  Vector<double> zeta(1);
464  zeta[0]=0.5*(xi_nose+xi_tail);
465 
466  Vector<double> s(1);
467  GeomObject* geom_obj_pt=0;
468  this->Back_pt->locate_zeta(zeta,geom_obj_pt,s);
469 
470  if ((geom_obj_pt!=this->Back_pt)||(s[0]!=zeta[0]))
471  {
472  std::ostringstream error_message;
473  error_message
474  << "AlgebraicFishMesh only works with GeomObjects\n"
475  << "that do not contain sub-elements (e.g. GeomObjects\n"
476  << "that represent a wall finite element mesh!\n"
477  << "Back_pt : " << this->Back_pt << std::endl
478  << "geom_obj_pt: " << geom_obj_pt << std::endl
479  << "s[0] : " << s[0] << std::endl
480  << "zeta[0] : " << zeta[0] << std::endl;
481 
482  throw OomphLibError(error_message.str(),
483  OOMPH_CURRENT_FUNCTION,
484  OOMPH_EXCEPTION_LOCATION);
485  }
486 #endif
487  }
