64 
65  // Loop over all elements
66  unsigned nel=this->nelement();
67  for (unsigned e=0;e<nel;e++)
68  {
69  // Try to cast to FiniteElement
70  FiniteElement* el_pt = dynamic_cast<FiniteElement*>(this->element_pt(e));
71 
72  // Set macro element pointer
73  el_pt->set_macro_elem_pt(Domain_pt->macro_element_pt(e));
74  }
75 
76  // Update node coordinates with macroelement coordinates,
77  // updating solid coordinates too.
78  this->node_update(true);
79 
80  // Setup boundary coordinates on inner boundary (boundary 1)
81  unsigned b=1;
82  unsigned nnod=this->nboundary_node(b);
83  for (unsigned j=0;j<nnod;j++)
84  {
85  // Pointer to node
86  Node* nod_pt=this->boundary_node_pt(b,j);
87 
88  // Get the Eulerian coordinates

92 
93  // Polar angle

95

100  nod_pt->set_coordinates_on_boundary(b,zeta);
