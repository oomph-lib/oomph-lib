129  /// \short General node update function implements pure virtual function
130  /// defined in SpineMesh base class and performs specific update
131  /// actions, depending on the node update fct id stored for each node.
132  void spine_node_update(SpineNode* spine_node_pt)
133  {
134 
135  unsigned id=spine_node_pt->node_update_fct_id();
136  switch(id)
137  {
138  case 0:
139  spine_node_update_film_lower(spine_node_pt);
140  break;
141 
142  case 1:
143  spine_node_update_horizontal_transition_lower(spine_node_pt);
144  break;
145 
146  case 2:
147  spine_node_update_vertical_transition_lower(spine_node_pt);
148  break;
149 
150  case 3:
151  spine_node_update_vertical_transition_upper(spine_node_pt);
152  break;
153 
154  case 4:
155  spine_node_update_horizontal_transition_upper(spine_node_pt);
156  break;
157 
158  case 5:
159  spine_node_update_film_upper(spine_node_pt);
160  break;
161 
162  case 6:
163  spine_node_update_channel(spine_node_pt);
164  break;
165 
166  default:
167  std::ostringstream error_message;
168  error_message << "Incorrect spine update id " << id << std::endl;
169 
170  throw OomphLibError(error_message.str(),
171  OOMPH_CURRENT_FUNCTION,
172  OOMPH_EXCEPTION_LOCATION);
173  }
174  }

216  /// \short Update function for the horizontal transitition region in the
217  /// lower part of the domain: Spine points from wall to origin
218  void spine_node_update_horizontal_transition_lower(SpineNode *spine_node_pt)
219  {
220  //Get fraction along the spine
221  double w = spine_node_pt->fraction();
222  //Get spine height
223  double h = spine_node_pt->h();
224 
225  // Get wall coordinate
226  Vector<double> s_lo(1);
227  s_lo[0] = spine_node_pt->spine_pt()->geom_parameter(0);
228 
229  // Get position vector to wall
230  Vector<double> r_wall_lo(2);
231  spine_node_pt->spine_pt()->geom_object_pt(0)->position(s_lo,r_wall_lo);
232 
233  // Get the spine centre
234  Vector<double> s_transition_lo(1), s_transition_up(1);
235  s_transition_lo[0] = spine_node_pt->spine_pt()->geom_parameter(1);
236  s_transition_up[0] = spine_node_pt->spine_pt()->geom_parameter(2);
237  Vector<double> r_transition_lo(2), r_transition_up(2);
238  spine_node_pt->spine_pt()->geom_object_pt(1)->
239  position(s_transition_lo,r_transition_lo);
240  spine_node_pt->spine_pt()->geom_object_pt(2)->
241  position(s_transition_up,r_transition_up);
242 
243  Vector<double> spine_centre(2);
244  //Horizontal position is always halfway between the walls
245  spine_centre[0] = 0.5*(r_transition_lo[0] + r_transition_up[0]);
246  //Vertical spine centre is given by fraction between the walls
247  spine_centre[1] = r_transition_lo[1] +
248  spine_centre_fraction()*(r_transition_up[1] - r_transition_lo[1]);
249 
250  // Get vector twoards spine origin
251  Vector<double> N(2);
252  N[0] = spine_centre[0] - r_wall_lo[0];
253  N[1] = spine_centre[1] - r_wall_lo[1];
254  double inv_length=1.0/sqrt(N[0]*N[0]+N[1]*N[1]);
255  //Update the nodal position
256  spine_node_pt->x(0) = r_wall_lo[0] + w*h*N[0]*inv_length;
257  spine_node_pt->x(1) = r_wall_lo[1] + w*h*N[1]*inv_length;
258  }
