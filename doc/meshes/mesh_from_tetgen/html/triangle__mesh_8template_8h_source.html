360 /// Triangle mesh build with the help of the scaffold mesh coming
361 /// from the triangle mesh generator Triangle.
362 /// http://www.cs.cmu.edu/~quake/triangle.html
363 //======================================================================
364  template<class ELEMENT>
365  class TriangleMesh : public virtual TriangleMeshBase
366  {
367  public:
368 
369  /// \short Empty constructor
370  TriangleMesh()
371  {
372 #ifdef OOMPH_HAS_TRIANGLE_LIB
373  // Using this constructor no Triangulateio object is built
374  Triangulateio_exists=false;
375  // By default allow the automatic creation of vertices along the
376  // boundaries by Triangle
377  this->Allow_automatic_creation_of_vertices_on_boundaries = true;
378 #ifdef OOMPH_HAS_MPI
379  // Initialize the flag to indicate this is the first time to
380  // compute the holes left by the halo elements
381  First_time_compute_holes_left_by_halo_elements = true;
382 #endif // #ifdef OOMPH_HAS_MPI
383 
384 #endif
385 
386  // Mesh can only be built with 2D Telements.
387  MeshChecker::assert_geometric_element<TElementGeometricBase,ELEMENT>(2);
388  }
389 
390  /// \short Constructor with the input files
391  TriangleMesh(const std::string& node_file_name,
392  const std::string& element_file_name,
393  const std::string& poly_file_name,
394  TimeStepper* time_stepper_pt=
395  &Mesh::Default_TimeStepper,
396  const bool &allow_automatic_creation_of_vertices_on_boundaries=true)
397  {
398  // Mesh can only be built with 2D Telements.
399  MeshChecker::assert_geometric_element<TElementGeometricBase,ELEMENT>(2);
400 
401  // Initialize the value for allowing creation of points on boundaries
402  this->Allow_automatic_creation_of_vertices_on_boundaries=
403  allow_automatic_creation_of_vertices_on_boundaries;
404 
405 #ifdef OOMPH_HAS_MPI
406  // Initialize the flag to indicate this is the first time to
407  // compute the holes left by the halo elements
408  First_time_compute_holes_left_by_halo_elements=true;
409 #endif // #ifdef OOMPH_HAS_MPI
410 
411  // Store Timestepper used to build elements
412  Time_stepper_pt=time_stepper_pt;
413 
414  // Check if we should use attributes. This is set to true if the .poly file
415  // specifies regions
416  bool should_use_attributes=false;
417 
418 #ifdef OOMPH_HAS_TRIANGLE_LIB
419  // Using this constructor build the triangulatio
420  TriangleHelper::create_triangulateio_from_polyfiles(node_file_name,
421  element_file_name,
422  poly_file_name,
423  Triangulateio,
424  should_use_attributes);
425 
426  // Record that the triangulateio object has been created
427  Triangulateio_exists=true;
428 #endif
429 
430  // Store the attributes
431  Use_attributes=should_use_attributes;
432 
433  // Build scaffold
434  this->Tmp_mesh_pt= new
435  TriangleScaffoldMesh(node_file_name,
436  element_file_name,
437  poly_file_name);
438 
439  // Convert mesh from scaffold to actual mesh
440  build_from_scaffold(time_stepper_pt,should_use_attributes);
441 
442  // kill the scaffold
443  delete this->Tmp_mesh_pt;
444  this->Tmp_mesh_pt=0;
445 
446  // Setup boundary coordinates for boundaries
447  unsigned nb=nboundary();
448  for (unsigned b=0;b<nb;b++)
449  {
450  this->template setup_boundary_coordinates<ELEMENT>(b);
451  }
452  }
453 
454  protected:
455 
456 #ifdef OOMPH_HAS_TRIANGLE_LIB
457 
458  public:
459 
460  /// \short Build mesh, based on the specifications on
461  /// TriangleMeshParameters
462  TriangleMesh(TriangleMeshParameters &triangle_mesh_parameters,
463  TimeStepper* time_stepper_pt=&Mesh::Default_TimeStepper)
464  {
465 
466  // Store the region target areas
467  Regions_areas=triangle_mesh_parameters.target_area_for_region();
468 
469  // Mesh can only be built with 2D Telements.
470  MeshChecker::assert_geometric_element<TElementGeometricBase,ELEMENT>(2);
471 
472  // Initialize the value for allowing creation of points on boundaries
473  this->Allow_automatic_creation_of_vertices_on_boundaries=
474  triangle_mesh_parameters.is_automatic_creation_of_vertices_on_boundaries_allowed();
475 
476  // Store Timestepper used to build elements
477  Time_stepper_pt=time_stepper_pt;
478 
479 #ifdef OOMPH_HAS_MPI
480  // Initialize the flag to indicate this is the first time to
481  // compute the holes left by the halo elements
482  First_time_compute_holes_left_by_halo_elements = true;
483 #endif // #ifdef OOMPH_HAS_MPI
484 
485  // ********************************************************************
486  // First part - Get polylines representations
487  // ********************************************************************
488 
489  // Create the polyline representation of all the boundaries and
490  // then create the mesh by calling to "generic_constructor()"
491 
492  // Initialise highest boundary id
493  unsigned max_boundary_id = 0;
494 
495  // *****************************************************************
496  // Part 1.1 - Outer boundary
497  // *****************************************************************
498  // Get the representation of the outer boundaries from the
499  // TriangleMeshParameters object
500  Vector<TriangleMeshClosedCurve *> outer_boundary_pt =
501  triangle_mesh_parameters.outer_boundary_pt();
502 
503 #ifdef PARANOID
504  // Verify that the outer_boundary_object_pt has been set
505  if (outer_boundary_pt.size()==0)
506  {
507  std::stringstream error_message;
508  error_message
509  << "There are no outer boundaries defined.\n"
510  << "Verify that you have specified the outer boundaries in the\n"
511  << "Triangle_mesh_parameter object\n\n";
512  throw OomphLibError(error_message.str(),
513  OOMPH_CURRENT_FUNCTION,
514  OOMPH_EXCEPTION_LOCATION);
515  } // if (outer_boundary_pt!=0)
516 #endif
517 
518  // Find the number of outer closed curves
519  unsigned n_outer_boundaries = outer_boundary_pt.size();
520 
521  // Create the storage for the polygons that define the outer
522  // boundaries
523  Vector<TriangleMeshPolygon*> outer_boundary_polygon_pt(n_outer_boundaries);
524 
525  // Loop over the number of outer boundaries
526  for(unsigned i=0;i<n_outer_boundaries;++i)
527  {
528  // Get the polygon representation and compute the max boundary_id on
529  // each outer polygon. Does nothing (i.e. just returns a pointer to
530  // the outer boundary that was input) if the outer boundary is
531  // already a polygon
532  outer_boundary_polygon_pt[i] =
533  closed_curve_to_polygon_helper(outer_boundary_pt[i], max_boundary_id);
534  }
535 
536  // *****************************************************************
537  // Part 1.2 - Internal closed boundaries (possible holes)
538  // *****************************************************************
539  // Get the representation of the internal closed boundaries from the
540  // TriangleMeshParameters object
541  Vector<TriangleMeshClosedCurve *> internal_closed_curve_pt =
542  triangle_mesh_parameters.internal_closed_curve_pt();
543 
544  // Find the number of internal closed curves
545  unsigned n_internal_closed_curves = internal_closed_curve_pt.size();
546 
547  // Create the storage for the polygons that define the internal closed
548  // boundaries (again nothing happens (as above) if an internal closed
549  // curve is already a polygon)
550  Vector<TriangleMeshPolygon*> internal_polygon_pt(n_internal_closed_curves);
551 
552  // Loop over the number of internal closed curves
553  for(unsigned i=0;i<n_internal_closed_curves;++i)
554  {
555  // Get the polygon representation and compute the max boundary_id on
556  // each internal polygon
557  internal_polygon_pt[i] =
558  closed_curve_to_polygon_helper(
559  internal_closed_curve_pt[i], max_boundary_id);
560  }
561 
562  // *****************************************************************
563  // Part 1.3 - Internal open boundaries
564  // *****************************************************************
565  // Get the representation of open boundaries from the
566  // TriangleMeshParameteres object
567  Vector<TriangleMeshOpenCurve*> internal_open_curve_pt =
568  triangle_mesh_parameters.internal_open_curves_pt();
569 
570  //Find the number of internal open curves
571  unsigned n_internal_open_curves = internal_open_curve_pt.size();
572 
573  // Create the storage for the polylines that define the open boundaries
574  Vector<TriangleMeshOpenCurve*> internal_open_curve_poly_pt(
575  n_internal_open_curves);
576 
577  // Loop over the number of internal open curves
578  for (unsigned i = 0; i < n_internal_open_curves; i++)
579  {
580  // Get the open polyline representation and compute the max boundary_id
581  // on each open polyline (again, nothing happens if there are curve
582  // sections on the current internal open curve)
583  internal_open_curve_poly_pt[i] =
584  create_open_curve_with_polyline_helper(
585  internal_open_curve_pt[i], max_boundary_id);
586  }
587 
588  // ********************************************************************
589  // Second part - Get associated geom objects and coordinate limits
590  // ********************************************************************
591 
592  // ***************************************************************
593  // Part 2.1 Outer boundary
594  // ***************************************************************
595  for (unsigned i = 0; i < n_outer_boundaries; i++)
596  {
597  set_geom_objects_and_coordinate_limits_for_close_curve(
598  outer_boundary_pt[i]);
599  }
600 
601  // ***************************************************************
602  // Part 2.2 - Internal closed boundaries (possible holes)
603  // ***************************************************************
604  for (unsigned i = 0; i < n_internal_closed_curves; i++)
605  {
606  set_geom_objects_and_coordinate_limits_for_close_curve(
607  internal_closed_curve_pt[i]);
608  }
609 
610  // ********************************************************************
611  // Part 2.3 - Internal open boundaries
612  // ********************************************************************
613  for (unsigned i = 0; i < n_internal_open_curves; i++)
614  {
615  set_geom_objects_and_coordinate_limits_for_open_curve(
616  internal_open_curve_pt[i]);
617  }
618 
619  // ********************************************************************
620  // Third part - Creates the TriangulateIO object by calling the
621  // "generic_constructor()" function
622  // ********************************************************************
623  // Get all the other parameters from the TriangleMeshParameters object
624  // The maximum element area
625  const double element_area =
626  triangle_mesh_parameters.element_area();
627 
628  // The holes coordinates
629  Vector<Vector<double> > extra_holes_coordinates =
630  triangle_mesh_parameters.extra_holes_coordinates();
631 
632  // The regions coordinates
633  std::map<unsigned, Vector<double> > regions =
634  triangle_mesh_parameters.regions_coordinates();
635 
636  // If we use regions then we use attributes
637  const bool use_attributes = triangle_mesh_parameters.is_use_attributes();
638 
639  const bool refine_boundary =
640  triangle_mesh_parameters.is_boundary_refinement_allowed();
641 
642  const bool refine_internal_boundary =
643  triangle_mesh_parameters.is_internal_boundary_refinement_allowed();
644 
645  if(!refine_internal_boundary && refine_boundary)
646  {
647  std::ostringstream error_stream;
648  error_stream
649  <<
650  "You have specified that Triangle may refine the outer boundary, but\n"
651  <<
652  "not internal boundaries. Triangle does not support this combination.\n"
653  <<
654  "If you do not want Triangle to refine internal boundaries, it can't\n"
655  <<
656  "refine outer boundaries either!\n"
657  << "Please either disable all boundary refinement\n"
658  << "(call TriangleMeshParameters::disable_boundary_refinement()\n"
659  << "or enable internal boundary refinement (the default)\n";
660 
661  throw OomphLibError(error_stream.str().c_str(),
662  OOMPH_CURRENT_FUNCTION,
663  OOMPH_EXCEPTION_LOCATION);

2104  /// \short 2D cross product of OA and OB vectors, i.e. z-component of their 3D
2105  /// cross product.
2106  /// Returns a positive value, if OAB makes a counter-clockwise turn,
2107  /// negative for clockwise turn, and zero if the points are collinear.
2108  coord2_t cross(const Point &O, const Point &A, const Point &B)

2110  return (A.x - O.x) * (B.y - O.y) - (A.y - O.y) * (B.x - O.x);

2113  /// \short Returns a list of points on the convex hull in counter-clockwise order.
2114  /// Note: the last point in the returned list is the same as the first one.
2115  std::vector<Point> convex_hull(std::vector<Point> P)
2116  {
2117  int n = P.size(), k = 0;
2118  std::vector<Point> H(2*n);
2119 
2120  // Sort points lexicographically
2121  std::sort(P.begin(), P.end());
2122 
2123  // Build lower hull
2124  for (int i = 0; i < n; ++i) {
2125  while (k >= 2 && cross(H[k-2], H[k-1], P[i]) <= 0) k--;
2126  H[k++] = P[i];
2127  }
2128 
2129  // Build upper hull
2130  for (int i = n-2, t = k+1; i >= 0; i--) {
2131  while (k >= t && cross(H[k-2], H[k-1], P[i]) <= 0) k--;
2132  H[k++] = P[i];
2133  }
2134 
2135  H.resize(k);
2136  return H;
2137  }
