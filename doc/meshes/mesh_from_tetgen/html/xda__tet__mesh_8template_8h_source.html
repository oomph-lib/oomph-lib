60  /// \short Constructor: Pass name of xda file. Note that
61  /// all boundary elements get their own ID -- this is required for
62  /// FSI problems. In this case; the vector containing the oomph-lib
63  /// boundary IDs of all oomph-lib boundaries that collectively form
64  /// a given boundary as specified in the xda input file can be
65  /// obtained from the access function oomph_lib_boundary_ids(...).
66  /// Timestepper defaults to steady pseudo-timestepper.
67  XdaTetMesh(const std::string xda_file_name,
68  TimeStepper* time_stepper_pt=
69  &Mesh::Default_TimeStepper);
70 
71  /// \short Setup boundary coordinate on boundary b while is
72  /// temporarily flattened to simplex faces. Boundary coordinates are the
73  /// x-y coordinates in the plane of that boundary with the
74  /// x-axis along the line from the (lexicographically)
75  /// "lower left" to the "upper right" node. The y axis
76  /// is obtained by taking the cross-product of the positive
77  /// x direction with the outer unit normal computed by
78  /// the face elements (or its negative if switch_normal is set
79  /// to true).
80  void setup_boundary_coordinates(const unsigned& b,
81  const bool& switch_normal)
82  {
83  std::ofstream outfile;
84  setup_boundary_coordinates(b,switch_normal,outfile);
85  }
