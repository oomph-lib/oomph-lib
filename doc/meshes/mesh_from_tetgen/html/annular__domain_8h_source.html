121 /// \short Vector representation of the imacro-th macro element
122 /// boundary idirect (N/S/W/E) at time level t
123 /// (t=0: present; t>0: previous): f(s)
124 //=================================================================
125  void AnnularDomain::macro_element_boundary(
126  const unsigned& t,
127  const unsigned& imacro,
128  const unsigned& idirect,
129  const Vector<double>& s,
130  Vector<double>& f)
131  {
132 
133  using namespace QuadTreeNames;
134 
135  // Get coordinates of macro element
136  unsigned i_theta=imacro%Ntheta;
137  unsigned i_r=(imacro-i_theta)/Ntheta;
138 
139  // Angle and radius limits
140  double theta_lo=Azimuthal_fraction*2.0*MathematicalConstants::Pi*
141  double(i_theta)/double(Ntheta);
142 
143  double theta_hi=Azimuthal_fraction*2.0*MathematicalConstants::Pi*
144  double(i_theta+1)/double(Ntheta);
145 
146  // Revert direction (convoluted -- don't ask. It mirrors what happens
147  // in the mesh...
148  theta_lo=-MathematicalConstants::Pi+
149  Azimuthal_fraction*2.0*MathematicalConstants::Pi-theta_lo;
150  theta_hi=-MathematicalConstants::Pi+
151  Azimuthal_fraction*2.0*MathematicalConstants::Pi-theta_hi;
152 
153  double r_lo=Inner_radius+Thickness*double(i_r)/double(Nr);
154  double r_hi=Inner_radius+Thickness*double(i_r+1)/double(Nr);
155 
156  // Actual radius and angle
157  double r=0.0;
158  double theta=0.0;
159 
160  // Which direction?
161  switch(idirect)
162  {
163  case N:
164 
165  theta=theta_lo+0.5*(s[0]+1.0)*(theta_hi-theta_lo);
166  r=r_hi;
167 
168  break;
169 
170  case S:
171 
172  theta=theta_lo+0.5*(s[0]+1.0)*(theta_hi-theta_lo);
173  r=r_lo;
174 
175  break;
176 
177  case W:
178 
179  theta=theta_lo;
180  r=r_lo+0.5*(s[0]+1.0)*(r_hi-r_lo);
181 
182  break;
183 
184  case E:
185 
186  theta=theta_hi;
187  r=r_lo+0.5*(s[0]+1.0)*(r_hi-r_lo);
188 
189  break;
190 
191  default:
192  std::ostringstream error_stream;
193  error_stream << "idirect is " << idirect
194  << " not one of N, S, W, E" << std::endl;
195 
196  throw OomphLibError(
197  error_stream.str(),
198  OOMPH_CURRENT_FUNCTION,
199  OOMPH_EXCEPTION_LOCATION);
200  }
201 
202  f[0]=r*cos(theta+Phi);
203  f[1]=r*sin(theta+Phi);
204 }
205 
206 }
