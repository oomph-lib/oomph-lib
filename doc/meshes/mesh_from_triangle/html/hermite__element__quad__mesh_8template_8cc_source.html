45 /// \short sets the generalised position of the node (i.e. - x_i, dx_i/ds_0,
46 /// dx_i/ds_1 & d2x_i/ds_0ds_1 for i = 1,2). Takes the x,y coordinates of the
47 /// node from which its position can be determined.
48 //=============================================================================
49 template<class ELEMENT>
50 void HermiteQuadMesh<ELEMENT>::set_position_of_node(const unsigned& node_num_x,
51  const unsigned& node_num_y,
52  Node* node_pt)
53 {
54 
55  // get the generalised macro element position of the node
56  DenseMatrix<double> m_gen(4,2);
57  generalised_macro_element_position_of_node(node_num_x,node_num_y,m_gen);
58 
59  // copy the macro element coordinates
60  Vector<double> node_macro_position(2);
61  node_macro_position[0] = m_gen(0,0);
62  node_macro_position[1] = m_gen(0,1);
63 
64  // get the global position of the nodes
65  Vector<double> x_node(2);
66  Domain_pt->macro_element_pt(0)->macro_map(node_macro_position,x_node);
67 
68  // get the jacobian of the mapping from macro to eulerian coordinates
69  DenseMatrix<double> jacobian_MX(2,2);
70  Domain_pt->macro_element_pt(0)->
71  assemble_macro_to_eulerian_jacobian(node_macro_position,jacobian_MX);
72 
73 
74  // get the jacobian2 of the mapping from macro to eulerian coordinates
75  DenseMatrix<double> jacobian2_MX(3,2);
76  Domain_pt->macro_element_pt(0)->
77  assemble_macro_to_eulerian_jacobian2(node_macro_position,jacobian2_MX);
78 
79  // set x_0
80  node_pt->x_gen(0,0) = x_node[0];
81  // set x_1
82  node_pt->x_gen(0,1) = x_node[1];
83 
84  // set dxi/dsj for i = 0,1 and j = 0,1
85  // computation : dxi/dsj = dm0/dsj*dxi/dm0 + dm1/dsj*dxi/dm1
86  for (unsigned i = 0; i < 2; i++)
87  {
88  for (unsigned j = 0; j < 2; j++)
89  {
90  node_pt->x_gen(j+1,i) = m_gen(j+1,0)*jacobian_MX(0,i)
91  + m_gen(j+1,1)*jacobian_MX(1,i);
92  }
93  }
94 
95  // set d2xi/ds0ds1 for i = 0,1
96  // d2xi/ds0ds1 = d2m0/ds0ds1*dxi/dm0
97  // + d2m1/ds0ds1*dxi/dm1
98  // + (dm0/ds1*d2xi/dm0^2 + dm1/ds1*d2xi/dm0dm1)*dm0/ds0
99  // + (dm0/ds1*d2xi/dm0dm1 + dm1/ds1*d2xi/dm1^2)*dm1/ds0
100  for (unsigned i = 0; i < 2; i++)
101  {
102  node_pt->x_gen(3,i) = m_gen(3,0)*jacobian_MX(0,i)
103  + m_gen(3,1)*jacobian_MX(1,i)
104  + m_gen(1,0)*(m_gen(2,0)*jacobian2_MX(0,i) +
105  m_gen(2,1)*jacobian2_MX(2,i))
106  + m_gen(1,1)*(m_gen(2,0)*jacobian2_MX(2,i) +
107  m_gen(2,1)*jacobian2_MX(1,i));
108  }
109 }

197 /// \short computes the generalised position of the node at position
198 /// (node_num_x, node_num_y) in the macro element coordinate scheme.
199 /// index 0 of m_gen : 0 - m_i

204 template<class ELEMENT>
205 void HermiteQuadMesh<ELEMENT>::generalised_macro_element_position_of_node
206 (const unsigned& node_num_x,const unsigned& node_num_y,
207  DenseMatrix<double>& m_gen)
208 {
209  // obtain position of node
210  Vector<double> s_macro_N(2);
211  macro_coordinate_position(node_num_x,node_num_y,s_macro_N);
212 
213  // set position of node in macro element coordinates
214  m_gen(0,0) = s_macro_N[0];
215  m_gen(0,1) = s_macro_N[1];
216 
217  // if on left hand edge
218  if (node_num_x == 0)
219  {
220 
221  // first dm0ds0 and dm1ds0
222 
223  // get position of right node
224  Vector<double> s_macro_R(2);
225  macro_coordinate_position(node_num_x+1,node_num_y,s_macro_R);
226 
227  // compute dm0ds0
228  m_gen(1,0) = 0.5*(s_macro_R[0]-s_macro_N[0]);
229 
230  // compute dm1ds0
231  m_gen(1,1) = 0.5*(s_macro_R[1]-s_macro_N[1]);
232 
233  // now d2m0/ds0ds1 and d2m1/ds0ds1
234 
235  // if lower left hand corner
236  if (node_num_y == 0)
237  {
238 
239  // get position of upper right node
240  Vector<double> s_macro_UR(2);
241  macro_coordinate_position(node_num_x+1,node_num_y+1,s_macro_UR);
242 
243  // get position of upper node
244  Vector<double> s_macro_U(2);
245  macro_coordinate_position(node_num_x,node_num_y+1,s_macro_U);
246 
247  // compute d2m0/ds0ds1
248  m_gen(3,0) = 0.5 * (0.5*(s_macro_UR[0] - s_macro_U[0]) - m_gen(1,0));
249 
250  // compute d2m1/ds0ds1
251  m_gen(3,1) = 0.5 * (0.5*(s_macro_UR[1] - s_macro_U[1]) - m_gen(1,1));
252  }
253 
254  // else if upper left hand corner
255  else if (node_num_y == Nelement[1])
256  {
257

260  macro_coordinate_position(node_num_x+1,node_num_y-1,s_macro_DR);
261 
262  // get position of lower node

264  macro_coordinate_position(node_num_x,node_num_y-1,s_macro_D);
265 
266  // compute d2m0/ds0ds1
267  m_gen(3,0) = 0.5 * (m_gen(1,0) - 0.5*(s_macro_DR[0]-s_macro_D[0]));
268 
269  // compute d2m0/ds0ds1
270  m_gen(3,1) = 0.5 * (m_gen(1,1) - 0.5*(s_macro_DR[1]-s_macro_D[1]));
271  }
272 
273  // else left hand edge (not corner)
274  else
275  {
276 
277  // get position of lower right node
278  Vector<double> s_macro_DR(2);
279  macro_coordinate_position(node_num_x+1,node_num_y-1,s_macro_DR);
280 
281  // get position of lower node
282  Vector<double> s_macro_D(2);
283  macro_coordinate_position(node_num_x,node_num_y-1,s_macro_D);
284 
285  // get position of upper right node
286  Vector<double> s_macro_UR(2);
287  macro_coordinate_position(node_num_x+1,node_num_y+1,s_macro_UR);

584  s_macro_UL[0]-s_macro_L[0]);

592  else
593  m_gen(3,1) = 0;
