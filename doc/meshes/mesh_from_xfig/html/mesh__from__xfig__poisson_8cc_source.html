51 /// Parameter for steepness of step
52 double Alpha;
53
54 /// Parameter for angle of step
55 double Beta;
56
57 /// Shift of step along x-direction
58 double X_shift=8.0;
59
60 /// Flag for zero TanhSolution
61 bool TanhSoln=false;
62
63 /// Exact solution as a Vector
64 void get_exact_u(const Vector<double>& x, Vector<double>& u)
65 {
66 if (TanhSoln)
67 {
68 u[0]=tanh(1.0-Alpha*(Beta*(x[0]-X_shift)-x[1]));
69 }
70 else
71 {
72 u[0]=0.0;
73 }
74 }
75
76
77 /// Exact solution as a scalar
78 void get_exact_u(const Vector<double>& x, double& u)
79 {
80 if (TanhSoln)
81 {
82 u=tanh(1.0-Alpha*(Beta*(x[0]-X_shift)-x[1]));
83 }
84 else
85 {
86 u=0.0;
87 }
88 }
89
90
91 /// Source function to make it an exact solution
92 void get_source(const Vector<double>& x, double& source)
93 {
94 if (TanhSoln)
95 {
96 source = 2.0*tanh(-1.0+Alpha*(Beta*(x[0]-X_shift)-x[1]))*
97 (1.0-pow(tanh(-1.0+Alpha*(Beta*(x[0]-X_shift)-x[1])),2.0))*
98 Alpha*Alpha*Beta*Beta+2.0*tanh(-1.0+Alpha*(Beta*(x[0]-X_shift)-x[1]))*
99 (1.0-pow(tanh(-1.0+Alpha*(Beta*(x[0]-X_shift)-x[1])),2.0))*Alpha*Alpha;
100 }
101 else
102 {
103 source=10.0;
104 }
105 }
106
107}

119template<class ELEMENT>
120class PoissonProblem : public Problem
121{

123public:

126 /// Constructor: Pass pointer to source function and names of
127 /// two triangle input files
128 PoissonProblem(PoissonEquations<2>::PoissonSourceFctPt source_fct_pt,
129 const string& node_file_name,
130 const string& element_file_name,
131 const string& poly_file_name);
132
133 /// Destructor (empty)
134 ~PoissonProblem(){}
135
136
137 /// Update the problem specs before solve: (Re)set boundary conditions
138 void actions_before_newton_solve()
139 {
140 //Loop over the boundaries
141 unsigned num_bound = mesh_pt()->nboundary();
142 for(unsigned ibound=0;ibound<num_bound;ibound++)
143 {
144 // Loop over the nodes on boundary
145 unsigned num_nod=mesh_pt()->nboundary_node(ibound);
146 for (unsigned inod=0;inod<num_nod;inod++)
147 {
148 Node* nod_pt=mesh_pt()->boundary_node_pt(ibound,inod);
149 double u;
150 Vector<double> x(2);
151 x[0]=nod_pt->x(0);
152 x[1]=nod_pt->x(1);
153 TanhSolnForPoisson::get_exact_u(x,u);
154 nod_pt->set_value(0,u);
155 }
156 }
157 }
158
159 /// Update the problem specs before solve (empty)
160 void actions_after_newton_solve()
161 {}
162
163
164 /// Access function for the specific mesh
165 TriangleMesh<ELEMENT>* mesh_pt()
166 {
167 return dynamic_cast<TriangleMesh<ELEMENT>*>(Problem::mesh_pt());
168 }
169
170 /// Doc the solution
171 void doc_solution(DocInfo& doc_info);
172
173private:
174
175 /// Pointer to source function
176 PoissonEquations<2>::PoissonSourceFctPt Source_fct_pt;
177
178};

182//========================================================================
183/// Constructor for Poisson problem
184//========================================================================
185template<class ELEMENT>
186PoissonProblem<ELEMENT>::
187 PoissonProblem(PoissonEquations<2>::PoissonSourceFctPt source_fct_pt,
188 const string& node_file_name,
189 const string& element_file_name,
190 const string& poly_file_name)
191 : Source_fct_pt(source_fct_pt)
192{
193
194 // Setup parameters for exact tanh solution
195
196 // Steepness of step
197 TanhSolnForPoisson::Alpha=1.0;
198
199 // Orientation of step
200 TanhSolnForPoisson::Beta=1.4;
201
202 //Create mesh
203 Problem::mesh_pt() = new TriangleMesh<ELEMENT>(node_file_name,
204 element_file_name,
205 poly_file_name);
206
207 // Set the boundary conditions for this problem: All nodes are
208 // free by default -- just pin the ones that have Dirichlet conditions
209 // here.
210 unsigned num_bound = mesh_pt()->nboundary();
211 for(unsigned ibound=0;ibound<num_bound;ibound++)
212 {
213 unsigned num_nod= mesh_pt()->nboundary_node(ibound);
214 for (unsigned inod=0;inod<num_nod;inod++)
215 {
216 mesh_pt()->boundary_node_pt(ibound,inod)->pin(0);
217 }
218 }
219
220 // Complete the build of all elements so they are fully functional
221
222 //Find number of elements in mesh
223 unsigned n_element = mesh_pt()->nelement();
224
225 // Loop over the elements to set up element-specific
226 // things that cannot be handled by constructor
227 for(unsigned i=0;i<n_element;i++)
228 {
229 // Upcast from GeneralElement to the present element
230 ELEMENT* el_pt = dynamic_cast<ELEMENT*>(mesh_pt()->element_pt(i));
231
232 //Set the source function pointer
233 el_pt->source_fct_pt() = Source_fct_pt;
234 }
235
236 // Setup equation numbering scheme
237 cout <<"Number of equations: " << assign_eqn_numbers() << std::endl;
238
239}

243//========================================================================
244/// Doc the solution
245//========================================================================
246template<class ELEMENT>
247void PoissonProblem<ELEMENT>::doc_solution(DocInfo& doc_info)
248{
249
250 ofstream some_file;
251 char filename[100];
252
253 // Number of plot points
254 unsigned npts;
255 npts=5;
256
257
258
259 // Output boundaries
260 //------------------
261 sprintf(filename,"%s/boundaries.dat",doc_info.directory().c_str());
262 some_file.open(filename);
263 mesh_pt()->output_boundaries(some_file);
264 some_file.close();
265
266
267 // Output solution
268 //----------------
269 sprintf(filename,"%s/soln%i.dat",doc_info.directory().c_str(),
270 doc_info.number());
271 some_file.open(filename);
272 mesh_pt()->output(some_file,npts);
273 some_file.close();
274
275 // Exact solution available?
276 if (TanhSolnForPoisson::TanhSoln)
277 {
278
279
280 // Output exact solution
281 //----------------------
282 sprintf(filename,"%s/exact_soln%i.dat",doc_info.directory().c_str(),
283 doc_info.number());
284 some_file.open(filename);
285 mesh_pt()->output_fct(some_file,npts,TanhSolnForPoisson::get_exact_u);
286 some_file.close();
287
288
289 // Doc error
290 //----------
291 double error,norm;
292 sprintf(filename,"%s/error%i.dat",doc_info.directory().c_str(),
293 doc_info.number());
294 some_file.open(filename);
295 mesh_pt()->compute_error(some_file,TanhSolnForPoisson::get_exact_u,
296 error,norm);
297 some_file.close();
298 cout << "error: " << sqrt(error) << std::endl;
299 cout << "norm : " << sqrt(norm) << std::endl << std::endl;
300
301 }
302
303 }

313{
314 // Store command line arguments
315 CommandLineArgs::setup(argc,argv);
316
317 // Check number of command line arguments: Need exactly four.
318 if (argc!=4)
319 {
320 std::string error_message =
321 "Wrong number of command line arguments.\n";
322 error_message +=
323 "Must specify the following file names \n";
324 error_message +=
325 "filename.node then filename.ele then filename.poly\n";
326
327 throw OomphLibError(error_message,
328 OOMPH_CURRENT_FUNCTION,
329 OOMPH_EXCEPTION_LOCATION);
330 }
331
332
333
334 // Convert arguments to strings that specify the input file names
335 string node_file_name(argv[1]);
336 string element_file_name(argv[2]);
337 string poly_file_name(argv[3]);
338
339 // Label for output
340 DocInfo doc_info;
341
342 // Output directory
343 doc_info.set_directory("RESLT");

345
346
347 // Linear Elements
348 //----------------
349 {
350 cout << std::endl << "Linear elements" << std::endl;
351 cout << "===============" << std::endl << std::endl;
352
353 //Set up the problem
354 PoissonProblem<TPoissonElement<2,2> >
355 problem(&TanhSolnForPoisson::get_source,
356 node_file_name,
357 element_file_name,
358 poly_file_name);
359
360 // Use TanhSolution?
361 TanhSolnForPoisson::TanhSoln=true;
362
363 // Solve once with forcing that makes TanhSolution an exact solution
364 // and once with constant source fct and zero bc
365 unsigned nstep=2;

368 // Solve the problem
