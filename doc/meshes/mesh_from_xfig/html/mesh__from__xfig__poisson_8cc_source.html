51  /// Parameter for steepness of step
52  double Alpha;
53 
54  /// Parameter for angle of step
55  double Beta;

57  /// Shift of step along x-direction
58  double X_shift=8.0;

60  /// Flag for zero TanhSolution
61  bool TanhSoln=false;

63  /// Exact solution as a Vector
64  void get_exact_u(const Vector<double>& x, Vector<double>& u)
65  {
66  if (TanhSoln)
67  {
68  u[0]=tanh(1.0-Alpha*(Beta*(x[0]-X_shift)-x[1]));
69  }
70  else
71  {
72  u[0]=0.0;
73  }
74  }
75 
76 
77  /// Exact solution as a scalar
78  void get_exact_u(const Vector<double>& x, double& u)
79  {
80  if (TanhSoln)
81  {
82  u=tanh(1.0-Alpha*(Beta*(x[0]-X_shift)-x[1]));
83  }
84  else
85  {
86  u=0.0;
87  }
88  }
89 
90 
91  /// Source function to make it an exact solution
92  void get_source(const Vector<double>& x, double& source)
93  {
94  if (TanhSoln)
95  {
96  source = 2.0*tanh(-1.0+Alpha*(Beta*(x[0]-X_shift)-x[1]))*
97  (1.0-pow(tanh(-1.0+Alpha*(Beta*(x[0]-X_shift)-x[1])),2.0))*
98  Alpha*Alpha*Beta*Beta+2.0*tanh(-1.0+Alpha*(Beta*(x[0]-X_shift)-x[1]))*
99  (1.0-pow(tanh(-1.0+Alpha*(Beta*(x[0]-X_shift)-x[1])),2.0))*Alpha*Alpha;
100  }
101  else
102  {
103  source=10.0;
104  }
105  }

137  /// Update the problem specs before solve: (Re)set boundary conditions
138  void actions_before_newton_solve()
139  {
140  //Loop over the boundaries
141  unsigned num_bound = mesh_pt()->nboundary();
142  for(unsigned ibound=0;ibound<num_bound;ibound++)
143  {
144  // Loop over the nodes on boundary
145  unsigned num_nod=mesh_pt()->nboundary_node(ibound);
146  for (unsigned inod=0;inod<num_nod;inod++)
147  {
148  Node* nod_pt=mesh_pt()->boundary_node_pt(ibound,inod);
149  double u;

151  x[0]=nod_pt->x(0);
152  x[1]=nod_pt->x(1);
153  TanhSolnForPoisson::get_exact_u(x,u);
154  nod_pt->set_value(0,u);
155  }

185 template<class ELEMENT>
186 PoissonProblem<ELEMENT>::
187  PoissonProblem(PoissonEquations<2>::PoissonSourceFctPt source_fct_pt,
188  const string& node_file_name,
189  const string& element_file_name,
190  const string& poly_file_name)
191  : Source_fct_pt(source_fct_pt)
192 {
193 
194  // Setup parameters for exact tanh solution
195 
196  // Steepness of step
197  TanhSolnForPoisson::Alpha=1.0;
198 
199  // Orientation of step
200  TanhSolnForPoisson::Beta=1.4;
201 
202  //Create mesh
203  Problem::mesh_pt() = new TriangleMesh<ELEMENT>(node_file_name,
204  element_file_name,
205  poly_file_name);
206 
207  // Set the boundary conditions for this problem: All nodes are
208  // free by default -- just pin the ones that have Dirichlet conditions
209  // here.
210  unsigned num_bound = mesh_pt()->nboundary();
211  for(unsigned ibound=0;ibound<num_bound;ibound++)
212  {
213  unsigned num_nod= mesh_pt()->nboundary_node(ibound);
214  for (unsigned inod=0;inod<num_nod;inod++)
215  {
216  mesh_pt()->boundary_node_pt(ibound,inod)->pin(0);
217  }
218  }
219 
220  // Complete the build of all elements so they are fully functional
221 
222  //Find number of elements in mesh
223  unsigned n_element = mesh_pt()->nelement();
224 
225  // Loop over the elements to set up element-specific
226  // things that cannot be handled by constructor
227  for(unsigned i=0;i<n_element;i++)
228  {
229  // Upcast from GeneralElement to the present element
230  ELEMENT* el_pt = dynamic_cast<ELEMENT*>(mesh_pt()->element_pt(i));
231 
232  //Set the source function pointer
233  el_pt->source_fct_pt() = Source_fct_pt;
234  }
235 
236  // Setup equation numbering scheme
237  cout <<"Number of equations: " << assign_eqn_numbers() << std::endl;
238
