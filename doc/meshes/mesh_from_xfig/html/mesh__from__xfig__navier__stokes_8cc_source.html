79  /// \short Update the problem specs before solve.
80  /// Re-set velocity boundary conditions just to be on the safe side...
81  void actions_before_newton_solve()
82  {
83  // Find max. and min y-coordinate at inflow
84  double y_min=1.0e20;
85  double y_max=-1.0e20;
86  unsigned ibound=0;
87  unsigned num_nod= mesh_pt()->nboundary_node(ibound);
88  for (unsigned inod=0;inod<num_nod;inod++)
89  {
90  double y=mesh_pt()->boundary_node_pt(ibound,inod)->x(1);
91  if (y>y_max)
92  {
93  y_max=y;
94  }
95  if (y<y_min)
96  {
97  y_min=y;
98  }
99  }
100 
101  // Loop over all boundaries
102  for (unsigned ibound=0;ibound<mesh_pt()->nboundary();ibound++)
103  {
104  unsigned num_nod= mesh_pt()->nboundary_node(ibound);
105  for (unsigned inod=0;inod<num_nod;inod++)
106  {
107  // Flow towards the north east on outer boundary (boundary 1)
108  if (ibound==1)
109  {
110  mesh_pt()->boundary_node_pt(ibound,inod)->set_value(0,-1.0);
111  mesh_pt()->boundary_node_pt(ibound,inod)->set_value(1,-1.0);
112  }
113  // Zero flow elsewhere
114  else
115  {
116  mesh_pt()->boundary_node_pt(ibound,inod)->set_value(0,0.0);
117  mesh_pt()->boundary_node_pt(ibound,inod)->set_value(1,0.0);
118  }
119  }
120  }
121  } // end_of_actions_before_newton_solve

137 /// Constructor for FlowPastBox problem. Pass filenames for mesh
138 //========================================================================
139 template<class ELEMENT>
140 FlowPastBoxProblem<ELEMENT>::FlowPastBoxProblem(
141  const string& node_file_name,
142  const string& element_file_name,
143  const string& poly_file_name)
144 {
145 
146  Problem::Max_residuals=1000.0;
147 
148  //Create mesh
149  Problem::mesh_pt() = new TriangleMesh<ELEMENT>(node_file_name,
150  element_file_name,
151  poly_file_name);
152 
153  // Set the boundary conditions for this problem: All nodes are
154  // free by default -- just pin the ones that have Dirichlet conditions
155  // here.
156  unsigned num_bound = mesh_pt()->nboundary();
157  for(unsigned ibound=0;ibound<num_bound;ibound++)
158  {
159  unsigned num_nod= mesh_pt()->nboundary_node(ibound);
160  for (unsigned inod=0;inod<num_nod;inod++)
161  {
162  // Pin velocity everywhere
163  mesh_pt()->boundary_node_pt(ibound,inod)->pin(0);
164  mesh_pt()->boundary_node_pt(ibound,inod)->pin(1);
165  }
166  } // end loop over boundaries
167 
168 
169  // Pin the zero-th pressure dof in the zero-th element
170  // and set it to zero
171  dynamic_cast<ELEMENT*>(mesh_pt()->element_pt(0))->
172  fix_pressure(0,0.0);
173 
174 
175  // Complete the build of all elements so they are fully functional
176 
177  //Find number of elements in mesh
178  unsigned n_element = mesh_pt()->nelement();
179 
180  // Loop over the elements to set up element-specific
181  // things that cannot be handled by constructor
182  for(unsigned e=0;e<n_element;e++)
183  {
184  // Upcast from GeneralisedElement to the present element
185  ELEMENT* el_pt = dynamic_cast<ELEMENT*>(mesh_pt()->element_pt(e));
186 
187  //Set the Reynolds number
188  el_pt->re_pt() = &Global_Physical_Variables::Re;
189  } // end loop over elements
190 
191 
192  // Setup equation numbering scheme
193  cout <<"Number of equations: " << assign_eqn_numbers() << std::endl;
194 
195 } // end_of_constructor
