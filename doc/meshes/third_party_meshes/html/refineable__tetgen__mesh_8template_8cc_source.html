45  /// \short Helper function that updates the input faceted surface
46  /// by using the flattened elements from FaceMesh(es) that are
47  /// constructed for the boundaries associated with the segments of the
48  /// polygon.
49  template <class ELEMENT>
50  void RefineableTetgenMesh<ELEMENT>::update_faceted_surface_using_face_mesh(
51  TetMeshFacetedSurface* &faceted_surface_pt)
52  {
53  //The easiest thing to do is simply to update the
54  //positions of the key control nodes, leaving the connectivity alone,
55  //but that doesn't allow for any surface remeshing
56 
57  ///List of vertex nodes
58  std::list<Node*> new_vertex_nodes;
59  //List of facets and boundary ids
60  std::list<std::pair<Vector<unsigned>, unsigned> > new_facet_list;
61 
62  //Loop over the number of old facets
63  unsigned n_old_facet = faceted_surface_pt->nfacet();
64  for(unsigned f=0;f<n_old_facet;f++)
65  {
66  //Get the boundary id of the facet. Need to subtract one,
67  //which is confusing now I think about it.
68  //ALH: Should fix this.
69  unsigned bound = faceted_surface_pt->one_based_facet_boundary_id(f) -1;
70 
71  // Create a face mesh adjacent to the fluid mesh's bound-th boundary.
72  // The face mesh consists of FaceElements that may also be
73  // interpreted as GeomObjects
74  Mesh* face_mesh_pt = new Mesh;
75  this->template build_face_mesh
76  <ELEMENT,FaceElementAsGeomObject>(bound, face_mesh_pt);
77 
78  // Loop over these new face elements and tell them the boundary number
79  // from the bulk fluid mesh -- this is required to they can
80  // get access to the boundary coordinates!
81  unsigned n_face_element = face_mesh_pt->nelement();
82  for(unsigned e=0;e<n_face_element;e++)
83  {
84  //Cast the element pointer to the correct thing!
85  FaceElementAsGeomObject<ELEMENT>* el_pt=
86  dynamic_cast<FaceElementAsGeomObject<ELEMENT>*>
87  (face_mesh_pt->element_pt(e));
88 
89  // Set bulk boundary number
90  el_pt->set_boundary_number_in_bulk_mesh(bound);
91  }
92 
93  //In order to set up the new faceted representation
94  //Need to know the positions of the corner nodes
95  //and the connectivity
96 
97  //Storage for the connectivity information
98  Vector<unsigned> new_local_facet(3);
99 
100  //Now we have the face mesh loop over the face elements and
101  //print out the end points
102  for(unsigned e=0;e<n_face_element;++e)
103  {
104  //Cache pointer to the element
105  FiniteElement const *elem_pt = face_mesh_pt->finite_element_pt(e);
106 
107  //Just use the three primary (corner) nodes to define the facet
108  unsigned n_vertex_node = 3;
109  for(unsigned n=0;n<n_vertex_node;n++)
110  {
111  //Cache the pointer to the node
112  Node* const nod_pt = elem_pt->node_pt(n);
113  //If the vertex node is in the list return the number
114  unsigned counter=0; bool found_it=false;
115  for(std::list<Node*>::iterator it = new_vertex_nodes.begin();
116  it!=new_vertex_nodes.end();++it,++counter)
117  {
118  //If we have found the node then store it
119  if(*it==nod_pt)
120  {
121  new_local_facet[n] = counter;
122  found_it=true;
123  break;
124  }
125  }
126 
127  //If we haven't found it
128  //then add the node to the list and fill in the counter
129  if(!found_it)
130  {
131  new_vertex_nodes.push_back(nod_pt);
132  //We know where it is
133  new_local_facet[n] = counter;
134  }
135  }
136 
137  //Add the new facet connectivity to the list
138  new_facet_list.push_back(std::make_pair(new_local_facet,bound+1));
139  }
140  } //End of loop over old facets
141 
142 
143  //Probably want the surface mesh in a better data structure so
144  //that we can perform refinement or coarsening on it
145  //i.e. want neighbours, edge flips all that fun stuff
146  //That will go here!
147 
148  //Now we need to put the information into the appropriate data structures
149  //for the Surface
150 
151  //Now sort out the facet nodes
152  unsigned n_facet_vertex = new_vertex_nodes.size();
153  //Vector<Vector<double> > facet_point(n_facet_vertex);
154  Vector<Node*> facet_nodes_pt(n_facet_vertex);
155  unsigned count=0;
156  for(std::list<Node*>::iterator it=new_vertex_nodes.begin();
157  it!=new_vertex_nodes.end();++it)
158  {
159  facet_nodes_pt[count] = *it;
160  //facet_point[count].resize(3);
161  // for(unsigned i=0;i<3;i++)
162  // {
163  // facet_point[count][i] = (*it)->x(i);
164  // }
165  ++count;
166  }
167 
168  //And also the facets
169  unsigned n_facet = new_facet_list.size();
170  Vector<Vector<unsigned> > new_facet(n_facet);
171  Vector<unsigned> new_facet_boundary_id(n_facet);
172  count=0;
173  for(std::list<std::pair<Vector<unsigned>, unsigned> >::iterator
174  it=new_facet_list.begin();it!=new_facet_list.end();++it)
175  {
176  new_facet[count] = (*it).first;
177  new_facet_boundary_id[count] = (*it).second;
178  ++count;
179  }
180 
181  for(unsigned f=0;f<n_facet;f++)
182  {
183  for(unsigned i=0;i<3;i++)
184  {
185  oomph_info << new_facet[f][i] << " " ;
186  }
187  oomph_info << " : ";
188  oomph_info << new_facet_boundary_id[f] << "\n";
189  }
190 
191  //Now just create the new boundary
192  delete faceted_surface_pt;
193  faceted_surface_pt = new TetMeshFacetedClosedSurfaceForRemesh(
194  facet_nodes_pt,new_facet,new_facet_boundary_id);
195 
196  //Also need to setup the reverse look-up scheme again
197  this->setup_reverse_lookup_schemes_for_faceted_surface(
198  faceted_surface_pt);
199 
200  //Take average to get a new hole position (Won't always work)
201  Vector<double> inner_point(3,0.0);
202  for(unsigned n=0;n<n_facet_vertex;n++)
203  {
204  for(unsigned i=0;i<3;i++)
205  {
206  inner_point[i] += facet_nodes_pt[n]->x(i);
207  }
208  }
209 
210  for(unsigned i=0;i<3;i++) {inner_point[i] /= n_facet_vertex;}
211 
212  //Now set the hole if required
213  dynamic_cast<TetMeshFacetedClosedSurface*>(faceted_surface_pt)->set_hole_for_tetgen(inner_point);
214 
215  }

239  /// \short Snap the boundary nodes onto any curvilinear boundaries
240  template<class ELEMENT>
241  void RefineableTetgenMesh<ELEMENT>::snap_nodes_onto_boundary(
242  RefineableTetgenMesh<ELEMENT>* &new_mesh_pt,
243  const unsigned &b)
244  {
245  // Quick return
246  if (!Boundary_coordinate_exists[b])
247  {
248  oomph_info << "Not snapping nodes on boundary " << b
249  << " because no boundary coordinate has been set up.\n";
250  return;
251  }
252 
253  //Firstly we set the boundary coordinates of the new nodes
254  //In case the mapping between the geometric object's intrinsic coordiante
255  //and the arc-length coordinate is nonlinear.
256  //This is only an approximation,
257  //but it will ensure that the nodes that were input to triangle will
258  //retain exactly the same boundary coordinates and
259  //then linear interpolation
260  //is used between those values for any newly created nodes.
261 
262 
263  // Create a face mesh adjacent to the fluid mesh's b-th boundary.
264  // The face mesh consists of FaceElements that may also be
265  // interpreted as GeomObjects
266  Mesh* face_mesh_pt = new Mesh;
267  this->template build_face_mesh
268  <ELEMENT,FaceElementAsGeomObject>(b, face_mesh_pt);
269 
270  // Loop over these new face elements and tell them the boundary number
271  // from the bulk fluid mesh -- this is required to they can
272  // get access to the boundary coordinates!
273  unsigned n_face_element = face_mesh_pt->nelement();
274  for(unsigned e=0;e<n_face_element;e++)
275  {
276  //Cast the element pointer to the correct thing!
277  FaceElementAsGeomObject<ELEMENT>* el_pt=
278  dynamic_cast<FaceElementAsGeomObject<ELEMENT>*>
279  (face_mesh_pt->element_pt(e));
280 
281  // Set bulk boundary number
282  el_pt->set_boundary_number_in_bulk_mesh(b);
283  }
284 
285 
286  //Compare face meshes
287  {
288  Mesh* new_face_mesh_pt = new Mesh;
289  new_mesh_pt->template build_face_mesh
290  <ELEMENT,FaceElementAsGeomObject>(b, new_face_mesh_pt);
291 
292  std::ostringstream filestring;
293  filestring << "old_mesh_boundary" << b << ".dat";
294  face_mesh_pt->output(filestring.str().c_str());
295  filestring.clear();
296  filestring << "new_mesh_boundary" << b << ".dat";
297  new_face_mesh_pt->output(filestring.str().c_str());
298 
299  Vector<double> b_coo(2);
300  std::cout << "OLD---\n";
301  //Now let's look at the boundary coordinates
302  unsigned n_tmp_node = this->nboundary_node(b);
303  for(unsigned n=0;n<n_tmp_node;++n)
304  {
305  Node* const nod_pt = this->boundary_node_pt(b,n);
306  nod_pt->get_coordinates_on_boundary(b,b_coo);
307  std::cout << nod_pt->x(0) << " "
308  << nod_pt->x(1) << " "
309  << nod_pt->x(2) << " "
310  << b_coo[0] << " " << b_coo[1] << "\n";
311  }
312 
313  std::cout << "NEW-----------\n";
314  //Now let's look at the boundary coordinates
315  n_tmp_node = new_mesh_pt->nboundary_node(b);
316  for(unsigned n=0;n<n_tmp_node;++n)
317  {
318  Node* const nod_pt = new_mesh_pt->boundary_node_pt(b,n);
319  nod_pt->get_coordinates_on_boundary(b,b_coo);
320  std::cout << nod_pt->x(0) << " "
321  << nod_pt->x(1) << " "
322  << nod_pt->x(2) << " "
323  << b_coo[0] << " " << b_coo[1] << "\n";
324  }
325 
326 
327  }
328 
329 
330  //Now make the mesh as geometric object
331  MeshAsGeomObject* mesh_geom_obj_pt = new MeshAsGeomObject(face_mesh_pt);
332 
333  //Now assign the new nodes positions based on the old meshes
334  //potentially curvilinear boundary (its geom object incarnation)
335  Vector<double> new_x(3);
336  Vector<double> b_coord(2);
337  unsigned n_new_boundary_node = new_mesh_pt->nboundary_node(b);
338  for(unsigned n=0;n<n_new_boundary_node;n++)
339  {
340  //Get the boundary coordinate of all new nodes
341  Node* const nod_pt = new_mesh_pt->boundary_node_pt(b,n);
342  nod_pt->get_coordinates_on_boundary(b,b_coord);
343  //Let's find boundary coordinates of the new node
344  mesh_geom_obj_pt->position(b_coord,new_x);
345  //Now snap to the boundary
346  for(unsigned i=0;i<3;i++)
347  {
348  nod_pt->x(i) = new_x[i];
349  }
350  }
351 
352 
353  //Delete the allocated memory for the geometric object and face mesh
354  delete mesh_geom_obj_pt;
355  face_mesh_pt->flush_element_and_node_storage();
356  delete face_mesh_pt;
357 
358  //Loop over the elements adjacent to the boundary
359  unsigned n_element = new_mesh_pt->nboundary_element(b);
360  if(n_element > 0)
361  {
362  //Make a dummy simplex element
363  TElement<3,2> dummy_four_node_element;
364  //Make a dummy quadratic element
365  TElement<3,3> dummy_ten_node_element;
366  Vector<double> s(3);
367  Vector<double> x_new(3);
368  for(unsigned n=0;n<4;n++) {dummy_four_node_element.construct_node(n);}
369  for(unsigned n=0;n<10;n++) {dummy_ten_node_element.construct_node(n);}
370  for(unsigned e=0;e<n_element;e++)
371  {
372  //Cache the element pointer
373  ELEMENT* elem_pt = dynamic_cast<ELEMENT*>(
374  new_mesh_pt->boundary_element_pt(b,e));
375 
376  //Find the number of nodes
377  const unsigned n_node = elem_pt->nnode();
378  //Only do something if not simplex element
379  if(n_node > 4)
380  {
381  //Copy the nodes into the dummy
382  for(unsigned n=0;n<4;n++)
383  {
384  for(unsigned i=0;i<3;i++)
385  {
386  dummy_four_node_element.node_pt(n)->x(i) =
387  elem_pt->node_pt(n)->x(i);
388  }
389  }
390 
391  //Now sort out the mid-side nodes
392  for(unsigned n=4;n<10;n++)
393  {
394  //If it's not on a boundary then reset to be interpolated
395  //from the simplex
396  if(!elem_pt->node_pt(n)->is_on_boundary())
397  {
398  elem_pt->local_coordinate_of_node(n,s);
399  dummy_four_node_element.interpolated_x(s,x_new);
400  for(unsigned i=0;i<3;i++) {elem_pt->node_pt(n)->x(i) = x_new[i];}
401  }
402  }
403  }
404 
405  //If we have more than 10 nodes interpolate from the quadratic shape
406  if(n_node > 10)
407  {
408  //Copy the nodes into the dummy
409  for(unsigned n=0;n<10;n++)
410  {
411  for(unsigned i=0;i<3;i++)
412  {
413  dummy_ten_node_element.node_pt(n)->x(i) =
414  elem_pt->node_pt(n)->x(i);
415  }
416  }
417 
418  //Now sort out the mid-face and central nodes
419  for(unsigned n=10;n<n_node;n++)
420  {
421  //If it's not on a boundary then reset to be interpolated
422  //from the simplex
423  if(!elem_pt->node_pt(n)->is_on_boundary())
424  {
425  elem_pt->local_coordinate_of_node(n,s);
426  dummy_ten_node_element.interpolated_x(s,x_new);
427  for(unsigned i=0;i<3;i++) {elem_pt->node_pt(n)->x(i) = x_new[i];}
428  }
429  }
430  }
431  }
432  } //End of fix up of elements
433  }

444 void RefineableTetgenMesh<ELEMENT>::adapt(const Vector<double>& elem_error)
445  {
446

448  double t_start=0.0;
449  //###################################
450  t_start=TimingHelpers::timer();
451  //###################################
452

454  Vector<double> target_size(elem_error.size());
455  double max_edge_ratio=this->compute_volume_target(elem_error,
456  target_size);
457  // Get maximum target volume
458  unsigned n=target_size.size();
459  double max_size=0.0;
460  double min_size=DBL_MAX;
461  for (unsigned e=0;e<n;e++)
462  {
463  if (target_size[e]>max_size) max_size=target_size[e];
464  if (target_size[e]<min_size) min_size=target_size[e];

470  << this->Nrefined << std::endl;
471  oomph_info << "Number of elements to be unrefined "
472  << this->Nunrefined << std::endl;
473  oomph_info << "Max edge ratio "<< max_edge_ratio << std::endl;

524  }

535

537  //----------------------------------------
538  // with volume set by maximum required volume
539  //---------------------------------------

553  }
554  else

557  (this->Outer_boundary_pt,
558  this->Internal_surface_pt,
559  max_size,
560  this->Time_stepper_pt,
561  this->Use_attributes,
562  this->Corner_elements_must_be_split);
563  }
564 
565 
566  //##################################################################
567  oomph_info << "adapt: Time for building temp mesh : "
568  << TimingHelpers::timer()-t_start
569  << " sec " << std::endl;
570  //##################################################################
571 
572  // tmp_new_mesh_pt->output("pre_mesh_nodes_snapped_0.dat");
573 
574  // //Move the nodes on the new boundary onto the
575  // //old curvilinear boundary
576  // //If the boundary is straight this will do precisely nothing
577  // //but will be somewhat inefficient

593
