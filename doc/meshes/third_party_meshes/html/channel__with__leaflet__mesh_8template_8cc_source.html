40 /// Constructor: Pass pointer to GeomObject that represents the leaflet,
41 /// the length of the domain to left and right of the leaflet, the
42 /// height of the leaflet and the overall height of the channel,
43 /// the number of element columns to the left and right of the leaflet,
44 /// the number of rows of elements from the bottom of the channel to
45 /// the end of the leaflet, the number of rows of elements above the
46 /// end of the leaflet. Timestepper defaults to Steady default
47 /// Timestepper defined in the Mesh base class
48 //===================================================================
49 template <class ELEMENT>
50 ChannelWithLeafletMesh<ELEMENT>::ChannelWithLeafletMesh(
51  GeomObject* leaflet_pt,
52  const double& lleft,
53  const double& lright,
54  const double& hleaflet,
55  const double& htot,
56  const unsigned& nleft,
57  const unsigned& nright,
58  const unsigned& ny1,
59  const unsigned& ny2,
60  TimeStepper* time_stepper_pt)
61  : SimpleRectangularQuadMesh<ELEMENT>(nright+nleft,ny1+ny2,lright+lleft,
62  htot,time_stepper_pt)
63 {
64  // Mesh can only be built with 2D Qelements.
65  MeshChecker::assert_geometric_element<QElementGeometricBase,ELEMENT>(2);
66 
67  // Copy pointer to leaflet into private storage
68  Leaflet_pt = leaflet_pt;
69 
70  //Create the ChannelWithLeafletDomain with the wall
71  //represented by the geometric object
72  Domain_pt = new ChannelWithLeafletDomain(leaflet_pt, lleft, lright,
73  hleaflet, htot,nleft,
74  nright,ny1,ny2);
75 
76 
77  // Total number of (macro/finite)elements
78  unsigned nmacro = (ny1+ny2)*(nleft+nright);
79 
80  // Loop over all elements and set macro element pointer
81  for (unsigned e=0;e<nmacro;e++)
82  {
83  // Get pointer to finite element
84  FiniteElement* el_pt=this->finite_element_pt(e);
85 
86  // Set pointer to macro element
87  el_pt->set_macro_elem_pt(this->Domain_pt->macro_element_pt(e));
88  }
89 
90  // Update the nodal positions corresponding to their
91  // macro element representations.
92  this->node_update();
93 
94  //Change the numbers of boundaries
95  this->set_nboundary(7);
96 
97  //Remove the nodes of boundary 0
98  this->remove_boundary_nodes(0);
99 
100  // Get the number of nodes along the element edge from first element
101  unsigned nnode_1d=this->finite_element_pt(0)->nnode_1d();
102 
103  //Boundary 0 will be before the wall, boundary 6 after it
104  for(unsigned e=0;e<nleft;e++)
105  {
106  for (unsigned i=0;i<nnode_1d;i++)
107  {
108  Node* nod_pt=this->finite_element_pt(e)->node_pt(i);
109  //do not add the last node : it will go to boundary 6
110  if( e!=nleft-1 || i!=2)
111  {
112  this->add_boundary_node(0, nod_pt);
113  }
114  }
115  }
116 
117  for(unsigned e=nleft;e<nleft+nright;e++)
118  {
119  for (unsigned i=0;i<nnode_1d;i++)
120  {
121  Node* nod_pt=this->finite_element_pt(e)->node_pt(i);
122  this->add_boundary_node(6, nod_pt);
123  }
124  }
125 
126  // Vector of Lagrangian coordinates used as boundary coordinate
127  Vector<double> zeta(1);
128 
129  //Set the wall as a boundary
130  for (unsigned k=0;k<ny1;k++)
131  {
132  unsigned e = (nleft+nright)*k + nleft-1;
133  for (unsigned i=0;i<nnode_1d;i++)
134  {
135  Node* nod_pt=this->finite_element_pt(e)->node_pt((i+1)*nnode_1d-1);
136  this->convert_to_boundary_node(nod_pt);
137  this->add_boundary_node(4, nod_pt);
138 
139  //Set coordinates
140  zeta[0]= double(k)*hleaflet/double(ny1)+
141  double(i)*hleaflet/double(ny1)/double(nnode_1d-1);
142  nod_pt->set_coordinates_on_boundary(4,zeta);
143  }
144  }
145 
146 
147  //Duplicate the nodes of the wall and assign then as a boundary.
148  //This will make one boundary for the east of the elements at the
149  //left of the wall, and one for the west of the elements at the right
150  //of the wall.
151  //This is necessary to use TaylorHoodElements, because it will allow
152  //a discontinuity of the pressure accross the wall.
153  //We separate the lower element from the rest as we do not want to
154  //add the same node twice to the boundary, and the upper element as its
155  //upper node must be the same one than the node of the upper element
156  //at the right of the wall
157 
158  //Lower element
159  unsigned e = nleft -1;
160 
161  // Allocate storage for newly created node outside
162  // so we can refer to the most recently created one below.
163  Node* nod_pt=0;
164 
165  //duplicate the east nodes and add them to the 6th boundary
166  //Add the first node to the 0th boundary (horizontal)
167  for (unsigned i=0;i<nnode_1d;i++)
168  {
169  nod_pt=this->finite_element_pt(e)->
170  construct_boundary_node((i+1)*nnode_1d-1,time_stepper_pt);
171  this->add_boundary_node(5, nod_pt);
172  if(i==0){this->add_boundary_node(0, nod_pt);}
173  this->add_node_pt(nod_pt);
174  //Set coordinates
175  zeta[0]= i*hleaflet/double(ny1)/double(nnode_1d-1);
176  nod_pt->set_coordinates_on_boundary(5,zeta);
177  }
178 
179 
180  //Other elements just at the left of the wall
181  for (unsigned k=1;k<ny1;k++)
182  {
183  e = (nleft+nright)*k + nleft-1;
184 
185  //add the upper node of the previous element
186  this->finite_element_pt(e)->node_pt(nnode_1d-1)=nod_pt;
187  this->add_boundary_node(5,nod_pt);
188  //Set coordinates
189  zeta[0]= k*hleaflet/double(ny1);
190  nod_pt->set_coordinates_on_boundary(5,zeta);
191 
192  //Loop over other nodes on element's eastern edge
193  for (unsigned i=1;i<nnode_1d;i++)
194  {
195  // Don't duplicate the node at the very top of the "obstacle"
196  if( (k==ny1-1)&&(i==nnode_1d-1) )
197  {
198  // Get the node but don't do anything else
199  nod_pt=this->finite_element_pt(e)->node_pt(nnode_1d*nnode_1d-1);
200  }
201  else
202  {
203  // Overwrite the node with a boundary node
204  nod_pt=this->finite_element_pt(e)->
205  construct_boundary_node((i+1)*nnode_1d-1,time_stepper_pt);
206  this->add_node_pt(nod_pt);
207  }
208 
209  // Add node to boundary
210  this->add_boundary_node(5, nod_pt);
211  //Set coordinates
212  zeta[0]= double(k)*hleaflet/double(ny1)+
213  double(i)*hleaflet/double(ny1)/double(nnode_1d-1);
214  nod_pt->set_coordinates_on_boundary(5,zeta);
215 
216  }
217  }
218 
219  this->node_update();
220 
221  // Re-setup lookup scheme that establishes which elements are located
222  // on the mesh boundaries (doesn't need to be wiped)
223  this->setup_boundary_element_info();
224 
225  //We have parametrised boundary 4 and 5
226  this->Boundary_coordinate_exists[4] = true;
227  this->Boundary_coordinate_exists[5] = true;
228 
229 }//end of constructor

238 /// Setup algebraic node update. Leaflet is
239 /// assumed to be in its undeformed (straight vertical) position!
240 //=================================================================
241 template<class ELEMENT>
242 void AlgebraicChannelWithLeafletMesh<ELEMENT>::setup_algebraic_node_update()
243 {
244 
245  // Extract some reference lengths from the Domain.
246  double hleaflet= this->domain_pt()->hleaflet();
247  double htot = this->domain_pt()->htot();
248  double lleft= this->domain_pt()->lleft();
249  double lright=this->domain_pt()->lright();
250 
251  // Loop over all nodes in mesh
252  unsigned nnod=this->nnode();
253  for (unsigned j=0;j<nnod;j++)
254  {
255  // Get pointer to node
256  AlgebraicNode* nod_pt=node_pt(j);
257 
258  // Get coordinates
259  double x=nod_pt->x(0);
260  double y=nod_pt->x(1);
261 
262  // Quick check to know if the wall is in its undeformed position
263  // It actually checks that the top of the wall is in (x_0,hleaflet)
264  Vector<double> zeta(1);
265  Vector<double> r(2);
266  zeta[0]=Hleaflet;
267  this->Leaflet_pt->position(zeta,r);
268  if( (r[0]!=X_0)||(r[1]!=hleaflet) )
269  {
270  std::ostringstream error_stream;
271  error_stream
272  << "Wall must be in its undeformed position when\n"
273  << "algebraic node update information is set up!\n "
274  << r[0] << " " << X_0 << " " << r[1] << " " << hleaflet
275  << std::endl;
276  throw OomphLibError(
277  error_stream.str(),
278  OOMPH_CURRENT_FUNCTION,
279  OOMPH_EXCEPTION_LOCATION);
280  }
281 
282 
283  // The update function requires four parameters:
284  Vector<double> ref_value(4);
285 
286  // Part I
287  if( (x<=X_0)&&(y<=hleaflet) )
288  {
289  // First reference value: y0
290  ref_value[0]=y;
291 
292  // zeta coordinate on wall : fourth reference value
293  //(needed for adaptativity)
294  Vector<double> zeta(1);
295  zeta[0]= y;
296  ref_value[3]=zeta[0];
297 
298  //Sub-geomobject corresponding to the zeta coordinate on the wall
299  GeomObject* geom_obj_pt;
300  Vector<double> s(1);
301  this->Leaflet_pt->locate_zeta(zeta,geom_obj_pt,s);
302 
303  //Create vector of geomobject for add_node_update_info()
304  Vector<GeomObject*> geom_object_pt(1);
305  geom_object_pt[0]=geom_obj_pt;
306 
307  // Second reference value: Reference local coordinate
308  // in the sub-geomobject
309  ref_value[1]=s[0];
310 
311  // Third reference value:fraction of the horizontal line
312  //between the edge and the wall
313  ref_value[2]=(lleft+x-X_0)/lleft;
314 
315 
316  // Setup algebraic update for node: Pass update information
317  // to AlgebraicNode:
318  nod_pt->add_node_update_info(
319  1, // ID
320  this, // mesh
321  geom_object_pt, // vector of geom objects
322  ref_value); // vector of ref. values
323  }
324  //Part II
325  if((x>=X_0)&&(y<=hleaflet) )
326  {
327  // First reference value: y0
328  ref_value[0]=y;
329 
330  // zeta coordinate on wall: fourth reference value
331  //(needed for adaptativity)
332  Vector<double> zeta(1);
333  zeta[0]= y;
334  ref_value[3]=zeta[0];
335 
336  //Sub-geomobject corresponding to the zeta coordinate on the wall
337  GeomObject* geom_obj_pt;
338  Vector<double> s(1);
339  this->Leaflet_pt->locate_zeta(zeta,geom_obj_pt,s);
340 
341  //Create vector of geomobject for add_node_update_info()
342  Vector<GeomObject*> geom_object_pt(1);
343  geom_object_pt[0]=geom_obj_pt;
344 
345  // Second reference value: Reference local coordinate
346  // in the sub-geomobject
347  ref_value[1]=s[0];
348 
349  // Third reference value:fraction of the horizontal line
350  //between the edge and the wall
351  ref_value[2]=(x-X_0)/lright;
352 
353  // Setup algebraic update for node: Pass update information
354  // to AlgebraicNode:
355  nod_pt->add_node_update_info(
356  2, // ID
357  this, // mesh
358  geom_object_pt, // vector of geom objects
359  ref_value); // vector of ref. values
360  }
361  //Part III
362  if((x<=X_0)&&(y>=hleaflet) )
363  {
364  // First reference value: y0
365  ref_value[0]=y;
366 
367  // Second reference value: zeta coordinate on the middle line
368  ref_value[1]= (y-hleaflet)/(htot-hleaflet);
369 
370  // Third reference value:fraction of the horizontal line
371  //between the edge and the middle line
372  ref_value[2]= (lleft+x-X_0)/lleft;
373 
374  // geomobject
375  Vector<GeomObject*> geom_object_pt(1);
376  geom_object_pt[0]=this->Leaflet_pt;
377 
378  // Setup algebraic update for node: Pass update information
379  // to AlgebraicNode:
380  nod_pt->add_node_update_info(
381  3, // ID
382  this, // mesh
383  geom_object_pt, // vector of geom objects
384  ref_value); // vector of ref. values
385  }
386  //Part IV
387  if((x>=X_0)&&(y>=hleaflet) )
388  {
389  // First reference value: y0
390  ref_value[0]=y;
391 
392  // Second reference value: zeta coordinate on wall
393  ref_value[1]= (y-hleaflet)/(htot-hleaflet);
394 
395  // Third reference value:fraction of the horizontal line
396  //between the edge and the wall
397  ref_value[2]=(x-X_0)/lright;
398 
399  // geomobject
400  Vector<GeomObject*> geom_object_pt(1);
401  geom_object_pt[0]=this->Leaflet_pt;
402 
403  // Setup algebraic update for node: Pass update information
404  // to AlgebraicNode:
405  nod_pt->add_node_update_info(
406  4, // ID
407  this, // mesh
408  geom_object_pt, // vector of geom objects
409  ref_value); // vector of ref. values
410  }
411  }
412 
413 } //end of setup_algebraic_node_update

418 /// Perform algebraic node update
419 //=================================================================
420 template<class ELEMENT>
421 void AlgebraicChannelWithLeafletMesh<ELEMENT>::algebraic_node_update(
422  const unsigned& t, AlgebraicNode*& node_pt)
423 {
424  unsigned id=node_pt->node_update_fct_id();
425 
426  switch(id)
427  {
428  case 1:
429  node_update_I(t,node_pt);
430  break;
431 
432  case 2:
433  node_update_II(t,node_pt);
434  break;
435 
436  case 3:
437  node_update_III(t,node_pt);
438  break;
439 
440  case 4:
441  node_update_IV(t,node_pt);
442  break;
443 
444  default:
445  std::ostringstream error_message;
446  error_message << "The node update fct id is "
447  << id << ", but it should only be one of "
448  << 1 << ", "
449  << 2 << ", "
450  << 3 << " or "
451  << 4<< std::endl;
452  std::string function_name =
453  "AlgebraicChannelWithLeafletMesh::algebraic_node_update()";
454 
455  throw OomphLibError(error_message.str(),
456  OOMPH_CURRENT_FUNCTION,
457  OOMPH_EXCEPTION_LOCATION);
458  }
459 
460 }//end of algebraic_node_update()

465 //=================================================================
466 template<class ELEMENT>
467 void AlgebraicChannelWithLeafletMesh<ELEMENT>::node_update_I(
468  const unsigned&t, AlgebraicNode*& node_pt)
469 {
470  //relevant data of the domain for part I
471  double lleft=this->domain_pt()->lleft();
472 
473  // Extract reference values for update by copy construction
474  Vector<double> ref_value(node_pt->vector_ref_value());
475 
476  // Extract geometric objects for update by copy construction
477  Vector<GeomObject*> geom_object_pt(node_pt->vector_geom_object_pt());
478 
479  // Pointer to wall geom object
480  GeomObject* leaflet_pt=geom_object_pt[0];
481 
482  //Coordinates of the steady node on the left boundary
483  //corresponding to the current node
484  double y0 = ref_value[0];
485  double x0 = -lleft+X_0;
486 
487  // second reference value: local coordinate on wall
488  Vector<double> s(1);
489  s[0]=ref_value[1];
490 
491 
492  // Get position vector to wall at timestep t
493  Vector<double> r_wall(2);
494  leaflet_pt->position(t,s,r_wall);
495 
496 
497  //Third reference value : fraction of the horizontal line
498  //between the edge and the wall
499  double r = ref_value[2];
500 
501 
502  // Assign new nodal coordinates
503  node_pt->x(t,0)= x0+ r*(r_wall[0]-x0);
504  node_pt->x(t,1)= y0+ r*(r_wall[1]-y0);
505 }

553 template<class ELEMENT>
554 void AlgebraicChannelWithLeafletMesh<ELEMENT>::slanted_bound_up(
555  const unsigned& t,
556  const Vector<double>& zeta,
557  Vector<double>& r)
558 {
559 ///\short Coordinates of the point on the boundary beetween the upper
560 /// and the lower part, in the same column, at the east.
561  double htot = this->domain_pt()->htot();
562 
563  Vector<double> xi(1);
564  xi[0]=Hleaflet;
565 
566  Vector<double> r_join(2);
567 
568  this->Leaflet_pt->position(t,xi,r_join);
569 
570  r[0] = r_join[0] + zeta[0]*( X_0-r_join[0]);
571  r[1] = r_join[1] + zeta[0]*( htot-r_join[1]);
572 }

653 template<class ELEMENT>
654 void RefineableAlgebraicChannelWithLeafletMesh<ELEMENT>::update_node_update(
655  AlgebraicNode*& node_pt)
656 {
657 
658  //Extract ID
659  unsigned id=node_pt->node_update_fct_id();
660
