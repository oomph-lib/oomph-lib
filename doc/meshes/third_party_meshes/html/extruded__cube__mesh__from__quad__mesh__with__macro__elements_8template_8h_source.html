97  /// \short Mesh class that takes a 2D mesh consisting of quadrilateral
98  /// elements and "extrudes" it in the z-direction.
99  //=======================================================================
100  template <class ELEMENT>
101  class ExtrudedCubeMeshFromQuadMesh : public virtual BrickMeshBase
102  {
103 
104  public:
105 
106  /// \short Constructor: Pass a mesh consisting of quad elements,
107  /// specify the number of elements in the z direction, and the
108  /// corresponding length in this direction. Assumes that the back
109  /// lower left corner is located at (0,0,0). Timestepper defaults
110  /// to the Steady timestepper.
111  ExtrudedCubeMeshFromQuadMesh(QuadMeshBase* quad_mesh_pt,
112  const unsigned& nz,
113  const double& lz,
114  TimeStepper* time_stepper_pt=
115  &Mesh::Default_TimeStepper) :
116  Nz(nz), Zmin(0.0), Zmax(lz)
117  {
118  // Call the generic build function
119  build_mesh(quad_mesh_pt,time_stepper_pt);
120  }
121 
122 
123  /// \short Constructor: Pass a mesh consisting of quad elements,
124  /// specify the number of elements in the z direction, and the
125  /// corresponding minimum and maximum z-value of the mesh. Again,
126  /// timestepper defaults to Steady.
127  ExtrudedCubeMeshFromQuadMesh(QuadMeshBase* quad_mesh_pt,
128  const unsigned& nz,
129  const double& zmin,
130  const double& zmax,
131  TimeStepper* time_stepper_pt=
132  &Mesh::Default_TimeStepper) :
133  Nz(nz), Zmin(zmin), Zmax(zmax)
134  {
135  // Call the generic mesh constructor
136  build_mesh(quad_mesh_pt,time_stepper_pt);
137  }
138 
139 
140  /// \short Destructor: If the underlying spatial domain was made up of
141  /// any macro elements then we will have created an extruded domain
142  /// which in turn creates extruded macro elements. As we're responsible
143  /// for creating the domain, we need to kill it and it'll kill the
144  /// extruded macro elements...
145  ~ExtrudedCubeMeshFromQuadMesh()
146  {
147  // Are there any extruded domains?
148  if (Extruded_domain_pt.size()>0)
149  {
150  // How many extruded domains did we create?
151  unsigned n_extruded_domain=Extruded_domain_pt.size();
152 
153  // Loop over the extruded domains
154  for (unsigned i=0; i<n_extruded_domain; i++)
155  {
156  // Delete the i-th extruded domain
157  delete Extruded_domain_pt[i];
158 
159  // Make it a null pointer
160  Extruded_domain_pt[i]=0;
161  }
162  } // if (Extruded_domain_pt.size()>0)
163  } // End of ExtrudedCubeMeshFromQuadMesh
164 
165 
166  /// \short Return the value of the z-coordinate at the node given by the
167  /// local node number, znode.
168  virtual double z_spacing_function(const unsigned& z_element,
169  const unsigned& z_node) const
170  {
171  // Calculate the values of equal increments in nodal values
172  double z_step=(Zmax-Zmin)/((N_node_1d-1)*Nz);
173 
174  // Return the appropriate value
175  return (Zmin+z_step*((N_node_1d-1)*z_element+z_node));
176  } // End of z_spacing_function
177 
178 
179  /// \short Get all the boundary information of an element using the
180  /// input (quad_mesh_pt) mesh. If the element lies on a boundary then
181  /// the user will be given the corresponding boundary index and the
182  /// index of the face of quad_el_pt attached to the boundary. If the
183  /// element does NOT lie on any boundaries, this function simply
184  /// returns a vector of size zero.
185  Vector<std::pair<unsigned,int> >
186  get_element_boundary_information(QuadMeshBase* quad_mesh_pt,
187  FiniteElement* quad_el_pt);
188 
189  /// Access function for number of elements in z-direction (const version)
190  const unsigned& nz() const
191  {
192  // Return the value of the appropriate private variable
193  return Nz;
194  } // End of nz
195 
196  protected:
197 
198  /// The number of nodes in each direction
199  unsigned N_node_1d;
200 
201  /// Number of elements in the z-direction
202  unsigned Nz;
203 
204  /// Minimum value of z coordinate
205  double Zmin;
206 
207  /// Maximum value of z coordinate
208  double Zmax;
209 
210  /// Vector of pointers to extruded domain objects
211  Vector<ExtrudedDomain*> Extruded_domain_pt;
212 
213  /// Generic mesh construction function: contains all the hard work
214  void build_mesh(QuadMeshBase* quad_mesh_pt,
215  TimeStepper* time_stepper_pt=&Mesh::Default_TimeStepper);
216  };
