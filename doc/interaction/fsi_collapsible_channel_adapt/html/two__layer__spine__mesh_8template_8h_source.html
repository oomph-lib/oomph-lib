39 /// Two-layer spine mesh class derived from standard 2D mesh.
40 /// The mesh contains two layers of spinified fluid elements (of type ELEMENT;
41 /// e.g SpineElement<QCrouzeixRaviartElement<2>).
42 ///
43 /// This mesh paritions the elements into those above and below a
44 /// notional interface and relabels boundaries so that the mesh is as follows
45 /// 3
46 /// ---------------------------------------
47 /// | |
48 /// 4 | | 2
49 /// | 6 |
50 /// ---------------------------------------
51 /// | |
52 /// 5 | | 1
53 /// | |
54 /// ---------------------------------------
55 /// 0
56 /// Update information for the nodes in response to changes in spine length
57 /// is given, but additional equations must be specified in order to
58 /// completely specify the problem.
59 //======================================================================
60 template <class ELEMENT>

154  /// \short General node update function implements pure virtual function
155  /// defined in SpineMesh base class and performs specific update
156  /// actions, depending on the node update fct id stored for each node.
157  void spine_node_update(SpineNode* spine_node_pt)
158  {
159  unsigned id=spine_node_pt->node_update_fct_id();
160  switch(id)
161  {
162  case 0:
163  spine_node_update_lower(spine_node_pt);
164  break;
165 
166  case 1:
167  spine_node_update_upper(spine_node_pt);
168  break;
169 
170  default:
171  std::ostringstream error_message;
172  error_message << "Unknown id passed to spine_node_update " << id
173  << std::endl;
174  throw OomphLibError(error_message.str(),
175  OOMPH_CURRENT_FUNCTION,
176  OOMPH_EXCEPTION_LOCATION);
177  }

224  //Get fraction along the spine
225  double W = spine_node_pt->fraction();
226  //Get spine height
227  double H = spine_node_pt->h();
228  //Set the value of y
229  spine_node_pt->x(1) = this->Ymin + W*H;
230  }

235  {
236  //Get fraction alon the spine
237  double W = spine_node_pt->fraction();
238 
239  //Get spine height
240  double H = spine_node_pt->h();
241 
242  //Set the value of y
243  spine_node_pt->x(1) = (this->Ymin+H) + W *(this->Ymax - (this->Ymin+H) );
244  }
