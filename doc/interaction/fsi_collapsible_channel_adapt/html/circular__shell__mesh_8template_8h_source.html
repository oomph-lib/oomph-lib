47 /// A 2D solid mesh for (topologically) circular cylindrical shells.
48 /// The shell is represented by two Lagrangian coordinates that correspond

169  /// Default axial scaling fct
170  double piecewise_linear_axial_bl_stretching_fct(const double& xi)
171  {
172  // Length of shell
173  double lx=this->Xmax-this->Xmin;
174 
175  // Old axial extent of the elements spanning the boundary layer
176  double old_delta_bl=double(Nx_bl)*lx/double(this->Nx);
177 
178  double tmp_xi=xi;
179  if (xi<old_delta_bl)
180  {
181  tmp_xi=xi*Delta_bl/old_delta_bl;
182  }
183  else if (xi<(lx-old_delta_bl))
184  {
185  tmp_xi=Delta_bl+(xi-old_delta_bl)/(lx-2.0*old_delta_bl)*
186  (lx-2.0*Delta_bl);
187  }
188  else
189  {
190  double end_x=lx-Delta_bl;
191  tmp_xi=end_x+(xi-(lx-old_delta_bl))/old_delta_bl*Delta_bl;
192  }
193  return tmp_xi;
194  }
