111  /// \short Helper function to interpolate linearly between the
112  /// "right" and "left" points; \f$ s \in [-1,1] \f$
113  void linear_interpolate(Vector<double> left, Vector<double> right,
114  const double &s, Vector<double> &f)
115  {
116  for(unsigned i=0;i<2;i++)
117  {
118  f[i] = left[i] + (right[i] - left[i])*0.5*(s+1.0);
119  }
120  }

154  //Macro element 0, is is immediately left of the cylinder
155  case 0:
156

161  Cylinder_pt->position(time,xi,point_on_circle);

163  break;
164 
165  case S:

168  linear_interpolate(Lower_mid_left,point_on_circle,s[0],f);

173  break;
174 
175  case E:

178  break;
179 
180  default:

187  }
188 
189  break;
190

192  case 1:

196  case N:
197  linear_interpolate(Upper_mid_left,Upper_mid_right,s[0],f);
198  break;
199

232  {
233  case N:

268  switch(direction)
269  {
270  case N:
271  xi[0] = -3.0*atan(1.0) + 2.0*atan(1.0)*0.5*(1.0+s[0]);
272  Cylinder_pt->position(time,xi,f);
273  break;
274 
275  case S:
276  linear_interpolate(Lower_mid_left,Lower_mid_right,s[0],f);

279  case W:
