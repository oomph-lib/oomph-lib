37 /// Brick mesh layer built on top of a given tet mesh. Typically
38 /// used in FSI problems where the tet mesh is the fluid mesh and this
39 /// mesh acts as the solid mesh that surrounds the FSI interface.
40 //========================================================================
41  template<class ELEMENT>
42  class ThinLayerBrickOnTetMesh : public virtual BrickMeshBase
43  {
44 
45  public:
46 
47  /// \short Function pointer to function that specifies the wall thickness
48  /// as a fct of the coordinates of the inner surface
49  typedef void (*ThicknessFctPt)(const Vector<double>& x, double& h_thick);
50 
51  /// \short Constructor: Specify (quadratic) tet mesh, boundary IDs of
52  /// boundary on which the current mesh is to be erected (in an FSI context
53  /// this boundary tends to be the FSI boundary of the fluid mesh. Also
54  /// specify the uniform thickness of layer, and the number of element layers.
55  /// The vectors stored in in_out_boundary_ids contain the boundary
56  /// IDs of the other boundaries in the tet mesh. In an FSI context
57  /// these typically identify the in/outflow boundaries in the fluid
58  /// mesh. The boundary enumeration of the current mesh follows the
59  /// one of the underlying fluid mesh: The enumeration of the FSI boundary
60  /// matches (to enable the setup of the FSI matching); the "in/outflow"
61  /// faces in this mesh inherit the same enumeration as the in/outflow
62  /// faces in the underlying fluid mesh. Finally, the "outer" boundary
63  /// gets its own boundary ID.
64  /// Timestepper defaults to steady pseudo-timestepper.
65  ThinLayerBrickOnTetMesh(Mesh* tet_mesh_pt,
66  const Vector<unsigned>& boundary_ids,
67  ThicknessFctPt thickness_fct_pt,
68  const unsigned& nlayer,
69  const Vector<Vector<unsigned> >& in_out_boundary_id,
70  TimeStepper* time_stepper_pt=
71  &Mesh::Default_TimeStepper);
72 
73 
74  /// \short Access functions to the Vector of oomph-lib boundary ids
75  /// that make up boundary on which the mesh was erected (typically
76  /// the FSI interface in an FSI problem)
77  Vector<unsigned> fsi_boundary_id()
78  {
79  return FSI_boundary_id;
80  }
81 
82  /// \short Boundary ID of the "outer" surface -- in an FSI context
83  /// this is the non-wetted tube surface at a distance h_thick from
84  /// the FSI surface
85  unsigned outer_boundary_id()
86  {
87  return Outer_boundary_id;
88  }
89 
90  /// \short Access function to the vector containing the ids of the oomph-lib
91  /// mesh boundaries that make up the specified in/outflow boundaries
92  /// as specified in constructor.
93  Vector<unsigned> in_out_boundary_id(const unsigned& boundary_id)
94  {
95  return In_out_boundary_id[boundary_id];
96  }
97 
98 
99  private:
100 
101  /// \short Vector of oomph-lib boundary ids
102  /// that make up boundary on which the mesh was erected (typically
103  /// the FSI interface in an FSI problem)
104  Vector<unsigned> FSI_boundary_id;
105 
106  /// \short Boundary ID of the "outer" surface -- the non-wetted
107  /// tube surface at a distance h_thick from the FSI surface
108  unsigned Outer_boundary_id;
109 
110  /// \short Vector of vectors containing the ids of the oomph-lib
111  /// mesh boundaries that make up the specified in/outflow boundaries
112  Vector<Vector<unsigned> > In_out_boundary_id;
113 
114  /// \short Function pointer to function that specifies the wall thickness
115  /// as a fct of the coordinates of the inner surface
116  ThicknessFctPt Thickness_fct_pt;
117 
118  };
