50 /// Refineable fish mesh upgraded to become a solid mesh
51 //=========================================================================
52 template<class ELEMENT>
53 class ElasticFishMesh : public virtual RefineableFishMesh<ELEMENT>,
54  public virtual SolidMesh
55 {
56 
57 public:
58 
59  /// \short Constructor: Build underlying adaptive fish mesh and then
60  /// set current Eulerian coordinates to be the Lagrangian ones.
61  /// Pass pointer to geometric objects that specify the
62  /// fish's back in the "current" and "undeformed" configurations,
63  /// and pointer to timestepper (defaults to Static)
64  // Note: FishMesh is virtual base and its constructor is automatically
65  // called first! --> this is where we need to build the mesh;
66  // the constructors of the derived meshes don't call the
67  // base constructor again and simply add the extra functionality.
68  ElasticFishMesh(GeomObject* back_pt, GeomObject* undeformed_back_pt,
69  TimeStepper* time_stepper_pt=&Mesh::Default_TimeStepper) :
70  FishMesh<ELEMENT>(back_pt,time_stepper_pt),
71  RefineableFishMesh<ELEMENT>(back_pt,time_stepper_pt)
72  {
73  // Mesh has been built, adaptivity etc has been set up -->
74  // assign the Lagrangian coordinates so that the current
75  // configuration becomes the stress-free initial configuration
76  set_lagrangian_nodal_coordinates();
77 
78  // Build "undeformed" domain: This is a "deep" copy of the
79  // Domain that we used to create set the Eulerian coordinates
80  // in the initial mesh -- the original domain (accessible via
81  // the private member data Domain_pt) will be used to update
82  // the position of the boundary nodes; the copy that we're
83  // creating here will be used to determine the Lagrangian coordinates
84  // of any newly created SolidNodes during mesh refinement
85  double xi_nose = this->Domain_pt->xi_nose();
86  double xi_tail = this->Domain_pt->xi_tail();
87  Undeformed_domain_pt=new FishDomain(undeformed_back_pt,xi_nose,xi_tail);
88 
89  // Loop over all elements and set the undeformed macro element pointer
90  unsigned n_element=this->nelement();
91  for (unsigned e=0;e<n_element;e++)
92  {
93  // Get pointer to full element type
94  ELEMENT* el_pt=dynamic_cast<ELEMENT*>(this->element_pt(e));
95 
96  // Set pointer to macro element so the curvlinear boundaries
97  // of the undeformed mesh/domain get picked up during adaptive
98  // mesh refinement
99  el_pt->set_undeformed_macro_elem_pt(
100  Undeformed_domain_pt->macro_element_pt(e));
101 
102  // Use MacroElement representation for
103  // Lagrangian coordinates of newly created
104  // nodes
105  el_pt->enable_use_of_undeformed_macro_element_for_new_lagrangian_coords();
106  }
107  }
108 
109 
110  /// Destructor: Kill "undeformed" Domain
111  virtual ~ElasticFishMesh()
112  {
113  delete Undeformed_domain_pt;
114  }
115 
116 
117 private:
118 
119  /// Pointer to "undeformed" Domain -- used to determine the
120  /// Lagrangian coordinates of any newly created SolidNodes during
121  /// Mesh refinement
122  Domain* Undeformed_domain_pt;
123 
124 };

178 /// Boundary-driven elastic deformation of fish-shaped domain.
179 //======================================================================
180 template<class ELEMENT>
181 class ElasticFishProblem : public Problem
182 {
183 
184 public:
185 
186  /// Constructor:
187  ElasticFishProblem();
188 
189  /// Run simulation.
190  void run();
191 
192  /// Access function for the mesh
193  ElasticFishMesh<ELEMENT>* mesh_pt()
194  {return dynamic_cast<ElasticFishMesh<ELEMENT>*>(Problem::mesh_pt());}
195 
196  /// Doc the solution
197  void doc_solution(DocInfo& doc_info);
198 
199  /// Update function (empty)
200  void actions_after_newton_solve() {}
201 
202  /// \short Update before solve: We're dealing with a static problem so
203  /// the nodal positions before the next solve merely serve as
204  /// initial conditions. For meshes that are very strongly refined
205  /// near the boundary, the update of the displacement boundary
206  /// conditions (which only moves the SolidNodes *on* the boundary),
207  /// can lead to strongly distorted meshes. This can cause the
208  /// Newton method to fail --> the overall method is actually more robust
209  /// if we use the nodal positions as determined by the Domain/MacroElement-
210  /// based mesh update as initial guesses.
211  void actions_before_newton_solve()
212  {
213  bool update_all_solid_nodes=true;
214  mesh_pt()->node_update(update_all_solid_nodes);
215  }
216 
217  /// Update after adapt: Pin all redundant pressure nodes (if required)
218  void actions_after_adapt()
219  {
220  // Pin the redundant solid pressures (if any)
221  PVDEquationsBase<2>::pin_redundant_nodal_solid_pressures(
222  mesh_pt()->element_pt());
223  }
224 
225 
226 private:
227 
228  // Geometric object that represents the deformable fish back
229  Circle* Fish_back_pt;
230 
231 };

236 template<class ELEMENT>
237 ElasticFishProblem<ELEMENT>::ElasticFishProblem()
238 {
239 
240  // Set coordinates and radius for the circle that will become the fish back
241  double x_c=0.5;
242  double y_c=0.0;
243  double r_back=1.0;
244 
245  // Build geometric object that will become the deformable fish back
246  //GeomObject* fish_back_pt=new ElasticFishBackElement(x_c,y_c,r_back);
247  Fish_back_pt=new Circle(x_c,y_c,r_back);
248 
249  // Build geometric object that specifies the fish back in the
250  // undeformed configuration (basically a deep copy of the previous one)
251  GeomObject* undeformed_fish_back_pt=new Circle(x_c,y_c,r_back);
252 
253  // Build fish mesh with geometric objects that specify the deformable
254  // and undeformed fish back
255  Problem::mesh_pt()=new ElasticFishMesh<ELEMENT>(Fish_back_pt,
256  undeformed_fish_back_pt);
257 
258  // Set error estimator
259  Z2ErrorEstimator* error_estimator_pt=new Z2ErrorEstimator;
260  mesh_pt()->spatial_error_estimator_pt()=error_estimator_pt;
261 
262  // Change/doc targets for mesh adaptation
263  mesh_pt()->max_permitted_error()=0.05;
264  mesh_pt()->min_permitted_error()=0.005;
265  mesh_pt()->doc_adaptivity_targets(cout);
266 
267  // Pin all nodal positions apart from those on the tail
268  unsigned num_bound = mesh_pt()->nboundary();
269  for(unsigned ibound=0;ibound<num_bound;ibound++)
270  {
271  if (ibound!=2)
272  {
273  unsigned num_nod=mesh_pt()->nboundary_node(ibound);
274  for (unsigned inod=0;inod<num_nod;inod++)
275  {
276  for (unsigned i=0;i<2;i++)
277  {
278  mesh_pt()->boundary_node_pt(ibound,inod)->pin_position(i);
279  }
280  }
281  }
282  }
283 
284  //Loop over the elements in the mesh to set parameters/function pointers
285  unsigned n_element =mesh_pt()->nelement();
286  for(unsigned i=0;i<n_element;i++)
287  {
288  //Cast to a solid element
289  ELEMENT *el_pt = dynamic_cast<ELEMENT*>(mesh_pt()->element_pt(i));
290 
291  // Set the constitutive law
292  el_pt->constitutive_law_pt() =
293  Global_Physical_Variables::Constitutive_law_pt;
294 
295  // Set the body force
296  el_pt->body_force_fct_pt()=Global_Physical_Variables::body_force;
297 
298  }
299 
300  // Pin the redundant solid pressures (if any)
301  PVDEquationsBase<2>::pin_redundant_nodal_solid_pressures(
302  mesh_pt()->element_pt());
303 
304  //Attach the boundary conditions to the mesh
305  cout << assign_eqn_numbers() << std::endl;
306 
307  // Refine the problem uniformly (this automatically passes the
308  // function pointers/parameters to the finer elements
309  refine_uniformly();
310 
311  // The non-pinned positions of the newly SolidNodes will have been
312  // determined by interpolation. Update all solid nodes based on
313  // the Mesh's Domain/MacroElement representation.
314  bool update_all_solid_nodes=true;
315  mesh_pt()->node_update(update_all_solid_nodes);
316 
317  // Now set the Eulerian equal to the Lagrangian coordinates
318  mesh_pt()->set_lagrangian_nodal_coordinates();
319 
320 }

356

362 
363  // Step number
364  doc_info.number()=0;

367

370 
371  //Parameter incrementation
372  unsigned nstep=5;
373  for(unsigned i=0;i<nstep;i++)
374  {
375  //Solve the problem with Newton's method, allowing for up to 5
376  //rounds of adaptation
377  newton_solve(5);
378 
379  // Doc solution
380  doc_solution(doc_info);
381  doc_info.number()++;
382 
383  // Increment width
