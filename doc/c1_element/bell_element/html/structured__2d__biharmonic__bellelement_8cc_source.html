210  /// \short Return FE representation of function value unknown(s)
211  /// at local coordinate s
212  inline Vector<double> interpolated_u_biharmonic(const Vector<double> &s) const
213  {
214  //Find number of position dofs
215  const unsigned n_position_type = this->nnodal_position_type();
216 
217  //Get the index at which the unknown is stored
218  const unsigned u_nodal_index = u_index_biharmonic();
219 
220  //Local c1-shape funtion
221  Shape psi(3,n_position_type);
222  DShape dpsidxi(3,n_position_type,DIM);
223  DShape d2psidxi(3,n_position_type,3);
224  Vector<double> interpolated_xi(DIM,0.0);
225 
226  //Initialise value of u
227  Vector<double> interpolated_u(u_nodal_index,0.0);
228 
229  //Find values of shape function
230  this->d2basis_local(s,psi,dpsidxi,d2psidxi);
231 
232  //Interpolated unknown
233  for(unsigned l=0;l<3;l++)
234  {
235  for(unsigned k=0;k<n_position_type;k++)
236  {
237  interpolated_u[0] += this->nodal_value(l,k)*psi(l,k);
238  interpolated_u[1] += this->nodal_value(l,k)*dpsidxi(l,k,0);
239  interpolated_u[2] += this->nodal_value(l,k)*dpsidxi(l,k,1);
240  interpolated_u[3] += this->nodal_value(l,k)*d2psidxi(l,k,0);
241  interpolated_u[4] += this->nodal_value(l,k)*d2psidxi(l,k,1);
242  interpolated_u[5] += this->nodal_value(l,k)*d2psidxi(l,k,2);
243  }
244  } // End of varible loop
245 
246  return(interpolated_u);
247  }

526 template <unsigned DIM, unsigned NNODE_1D>
527 void MyBiharmonicEquations<DIM,NNODE_1D>::
528 fill_in_generic_residual_contribution(Vector<double> &residuals,
529  DenseMatrix<double> &jacobian,
530  const unsigned& flag)
531 {
532  //Find out how many nodes there are
533  const unsigned n_node = this->nnode();
534  //Find out how many nodes positional dofs there are
535  unsigned n_position_type = this->nnodal_position_type();
536 
537  //Set up memory for the shape and test functions
538  Shape psi(3,n_position_type), test(n_node);
539  DShape dpsidxi(3,n_position_type,DIM), dtestdxi(n_node,DIM);
540  DShape d2psidxi(3,n_position_type,3), d2testdxi(n_node,3);
541 
542  //Set the value of n_intpt
543  const unsigned n_intpt = this->integral_pt()->nweight();
544 
545  //Integers to store the local equation and unknown numbers
546  int local_eqn=0, local_unknown=0;
547 
548  //Loop over the integration points
549  for(unsigned ipt=0;ipt<n_intpt;ipt++)
550  {
551  //Get the integral weight
552  double w = this->integral_pt()->weight(ipt);
553 
554  //Call the derivatives of the shape and test functions for normal direction
555  double J = d2shape_and_d2test_eulerian_at_knot_biharmonic(ipt,psi,dpsidxi,d2psidxi,test,dtestdxi,d2testdxi);
556 
557  //Premultiply the weights and the Jacobian
558  double W = w*J;
559 
560  //Calculate values of unknown
561  Vector<double> interpolated_u(1,0.0);
562  DenseMatrix<double> interpolated_dudxi(1,DIM,0.0);
563  DenseMatrix<double> interpolated_d2udxi(1,3,0.0);
564 
565  //Calculate local values of unknown
566  //Allocate and initialise to zero
567  Vector<double> interpolated_x(DIM,0.0);
568  Vector<double> s(2);
569  s[0] = this->integral_pt()->knot(ipt,0);
570  s[1] = this->integral_pt()->knot(ipt,1);
571  this->my_interpolated_x(s,interpolated_x);
572 
573  //Calculate function value and derivatives
574  //-----------------------------------------
575  // Loop over nodes
576  for(unsigned l=0;l<3;l++)
577  {
578  for(unsigned k=0;k<n_position_type;k++)
579  {
580  //Get the nodal value of the unknown
581  double u_value = this->raw_nodal_value(l,k);
582  interpolated_u[0] += u_value*psi(l,k);
583  // Loop over directions
584  for(unsigned j=0;j<DIM;j++)
585  {
586  interpolated_dudxi(0,j) += u_value*dpsidxi(l,k,j);
587  }
588  for(unsigned j=0;j<3;j++)
589  {
590  interpolated_d2udxi(0,j) += u_value*d2psidxi(l,k,j);
591  }
592  }
593  }
594 
595  //Get source function
596  //-------------------
597  double source;
598  get_source_function(ipt,interpolated_x,source);
599  // Assemble residuals and Jacobian
600  //--------------------------------
601  for(unsigned l=0;l<3;l++)
602  {
603  // Loop over the test functions
604  for(unsigned k=0;k<n_position_type;k++)
605  {
606  //Get the local equation
607  local_eqn = this->nodal_local_eqn(l,k);
608  //IF it's not a boundary condition
609  if(local_eqn >= 0)
610  {
611  // Add body force/source term here
612  residuals[local_eqn] -= source*psi(l,k)*W;
613 
614  residuals[local_eqn] += interpolated_d2udxi(0,0)*d2psidxi(l,k,0)*W + interpolated_d2udxi(0,1)*d2psidxi(l,k,1)*W + interpolated_d2udxi(0,1)*d2psidxi(l,k,0)*W + interpolated_d2udxi(0,0)*d2psidxi(l,k,1)*W;
615 
616  //-----------------------
617  if(flag)
618  {
619  //Loop over the velocity shape functions again
620  for(unsigned l2=0;l2<3;l2++)
621  {
622  // Loop over position dofs
