565 template <unsigned DIM, unsigned NNODE_1D>
566 void MyBiharmonicEquations<DIM,NNODE_1D>::
567 fill_in_generic_residual_contribution_biharmonic(Vector<double> &residuals,
568  DenseMatrix<double> &jacobian,
569  const unsigned& flag)
570 {
571  //Find out how many nodes there are
572  const unsigned n_node = this->nnode();
573  //Find out how many nodes positional dofs there are
574  unsigned n_position_type = this->nnodal_position_type();
575 
576  //Set up memory for the shape and test functions
577  Shape psi(n_node,n_position_type), test(n_node);
578  DShape dpsidxi(n_node,n_position_type,DIM), dtestdxi(n_node,DIM);
579  DShape d2psidxi(n_node,n_position_type,3), d2testdxi(n_node,3);
580 
581  //Set the value of n_intpt
582  const unsigned n_intpt = this->integral_pt()->nweight();
583 
584  //Integers to store the local equation and unknown numbers
585  int local_eqn=0, local_unknown=0;
586 
587  //Loop over the integration points
588  for(unsigned ipt=0;ipt<n_intpt;ipt++)
589  {
590  //Get the integral weight
591  double w = this->integral_pt()->weight(ipt);
592 
593  //Call the derivatives of the shape and test functions for the unknown
594  double J = d2shape_and_d2test_eulerian_at_knot_biharmonic(ipt,psi,dpsidxi,d2psidxi,test,dtestdxi,d2testdxi);
595 
596  //Premultiply the weights and the Jacobian
597  double W = w*J;
598 
599  //Calculate values of unknown
600  Vector<double> interpolated_u(1,0.0);
601  DenseMatrix<double> interpolated_dudxi(1,DIM,0.0);
602  DenseMatrix<double> interpolated_d2udxi(1,3,0.0);
603 
604  //Allocate and initialise to zero
605  Vector<double> interpolated_x(DIM,0.0);
606 
607  Vector<double> s(2);
608  s[0] = this->integral_pt()->knot(ipt,0);
609  s[1] = this->integral_pt()->knot(ipt,1);
610  this->my_interpolated_x(s,interpolated_x);
611 
612  //Calculate function value and derivatives
613  //-----------------------------------------
614  Vector<double> tt(2), xi(2);
615  Vector<double> exact_soln(6);
616  DenseMatrix<double> M(21,36,0.0);
617  this->my_interpolated_u_normal1(s,interpolated_u,interpolated_dudxi,interpolated_d2udxi,M);
618 
619  //Get source function
620  //-------------------
621  double source;
622  get_source_function(ipt,interpolated_x,source);
623 
624  //Check whether this element is a boundary element or not
625  unsigned bd_element = this->is_boundary_element();
