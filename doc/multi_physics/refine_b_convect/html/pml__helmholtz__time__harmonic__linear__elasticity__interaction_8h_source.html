40 /// A class for elements that allow the imposition of an applied traction
41 /// in the equations of time-harmonic linear elasticity from a
42 /// PMLHelmholtz potential (interpreted as a displacement potential
43 /// for the fluid in a quasi-steady, linearised FSI problem.)
44 /// The geometrical information can be read from the FaceGeometry<ELEMENT>
45 /// class and thus, we can be generic enough without the need to have
46 /// a separate equations class.
47 //======================================================================

51  public virtual FaceElement,

176  /// \short Output function: Plot traction etc at Gauss points
177  /// nplot is ignored.
178  void output(std::ostream &outfile, const unsigned &n_plot)
179  {
180  // Dimension
181  unsigned n_dim = this->nodal_dimension();
182 
183  // Storage for traction
184  Vector<std::complex<double> > traction(n_dim);
185 
186  // Get FSI parameter
187  const double q_value=q();
188 
189  outfile << "ZONE\n";
190 
191  //Set the value of n_intpt
192  unsigned n_intpt = integral_pt()->nweight();
193 
194  //Loop over the integration points
195  for(unsigned ipt=0;ipt<n_intpt;ipt++)
196  {
197  Vector<double> s_int(n_dim-1);
198  for (unsigned i=0;i<n_dim-1;i++)
199  {
200  s_int[i]=integral_pt()->knot(ipt,i);
201  }
202 
203  //Get the outer unit normal
204  Vector<double> interpolated_normal(n_dim);
205  outer_unit_normal(ipt,interpolated_normal);
206 
207  // Boundary coordinate
208  Vector<double> zeta(1);
209  interpolated_zeta(s_int,zeta);
210 
211  // Get external element for potential
212  HELMHOLTZ_BULK_ELEMENT* ext_el_pt=
213  dynamic_cast<HELMHOLTZ_BULK_ELEMENT*>(external_element_pt(0,ipt));
214  Vector<double> s_ext(external_element_local_coord(0,ipt));
215  std::complex<double> u_helmholtz=
216  ext_el_pt->interpolated_u_pml_helmholtz(s_ext);
217 
218  // Traction: Pressure is proportional to POSITIVE potential
219  ext_el_pt->interpolated_u_pml_helmholtz(s_ext);
220  traction[0]=-q_value*interpolated_normal[0]*u_helmholtz;
221  traction[1]=-q_value*interpolated_normal[1]*u_helmholtz;
222 
223  outfile << ext_el_pt->interpolated_x(s_ext,0) << " "
224  << ext_el_pt->interpolated_x(s_ext,1) << " "
225  << traction[0].real() << " "
226  << traction[1].real() << " "
227  << traction[0].imag() << " "
228  << traction[1].imag() << " "
229  << interpolated_normal[0] << " "
230  << interpolated_normal[1] << " "
231  << u_helmholtz.real() << " "
232  << u_helmholtz.imag() << " "
233  << interpolated_x(s_int,0) << " "
234  << interpolated_x(s_int,1) << " "
235  << sqrt(pow(ext_el_pt->interpolated_x(s_ext,0)-

237  pow(ext_el_pt->interpolated_x(s_ext,1)-
238  interpolated_x(s_int,1),2)) << " "
239  << zeta[0] << std::endl;
240  }

261  /// \short Compute the element's contribution to the integral of
262  /// the flux over the artificial boundary. Also output the
263  /// flux in the specified output file if it's open.
264  std::complex<double> global_flux_contribution_from_solid
265  (std::ofstream& outfile)
266  {
267  // pointer to the corresponding elasticity bulk element
268  ELASTICITY_BULK_ELEMENT* bulk_elem_pt =
269  dynamic_cast<ELASTICITY_BULK_ELEMENT*>(this->bulk_element_pt());
270 
271  //get the dim of the bulk and local nodes
272  const unsigned bulk_dim= bulk_elem_pt->dim();
273  const unsigned local_dim=this->dim();
274 
275  //Set up memory for the outer unit normal
276  Vector<double> unit_normal(bulk_dim);
277 
278  //Set the value of n_intpt
279  const unsigned n_intpt = integral_pt()->nweight();
280 
281  //Set the Vector to hold local coordinates
282  Vector<double> s(local_dim);
283  std::complex<double> flux(0.0,0.0);
284  Vector<std::complex<double> > u(bulk_dim);
285 
286  // Output?
287  if (outfile.is_open())
288  {
289  outfile << "ZONE\n";
290  }
291 
292  //Loop over the integration points
293  //--------------------------------
294  for(unsigned ipt=0;ipt<n_intpt;ipt++)
295  {
296  //Assign values of s
297  for(unsigned i=0;i<local_dim;i++)
298  {
299  s[i] = integral_pt()->knot(ipt,i);
300  }
301  //get the outer_unit_ext vector
302  this->outer_unit_normal(s,unit_normal);
303 
304  //Get the integral weight
305  double w = integral_pt()->weight(ipt);
306 
307  // Get jacobian of mapping
308  double J=J_eulerian(s);
309 
310  //Premultiply the weights and the Jacobian
311  double W = w*J;
312 
313  // Get local coordinates in bulk element by copy construction
314  Vector<double> s_bulk(local_coordinate_in_bulk(s));
315 
316  // Get the displacement from the bulk element
317  bulk_elem_pt->interpolated_u_time_harmonic_linear_elasticity(s_bulk,u);
318 
319  // Compute normal displacement u.n
320  std::complex<double> u_dot_n(0.0,0.0);
321  for(unsigned i=0;i<bulk_dim;i++)
322  {
323  u_dot_n += u[i]*unit_normal[i];
324  }
325 
326  // Output?
327  if (outfile.is_open())
328  {
329  Vector<double> x(bulk_dim);
330  interpolated_x(s,x);
331  outfile << x[0] << " "
332  << x[1] << " "
333  << u_dot_n.real() << " "
334  << u_dot_n.imag() << "\n";
335  }
336 
337  // ...add to integral
338  flux+=u_dot_n*W;
339  }
340 
341  return flux;
342  }

354  /// \short Compute the element's contribution to the integral of
355  /// the flux over the artificial boundary. Also output the
356  /// flux in the specified output file if it's open.
357  std::complex<double> global_flux_contribution_from_helmholtz
358  (std::ofstream& outfile)
359  {
360  // Get the dim of this element
361  const unsigned local_dim=this->dim();
362 
363  //Set the value of n_intpt
364  const unsigned n_intpt = integral_pt()->nweight();
365 
366  //Set the Vector to hold local coordinates
367  Vector<double> s(local_dim);
368  std::complex<double> flux(0.0,0.0);
369 
370  // Output?
371  if (outfile.is_open())
372  {
373  outfile << "ZONE\n";
374  }
375 
376  //Loop over the integration points
377  //--------------------------------
378  for(unsigned ipt=0;ipt<n_intpt;ipt++)
379  {
380  //Assign values of s
381  for(unsigned i=0;i<local_dim;i++)
382  {
383  s[i] = integral_pt()->knot(ipt,i);
384  }
385 
386  // Get local coordinates in bulk element by copy construction
387  Vector<double> s_bulk(local_coordinate_in_bulk(s));
388 
389  // Get external element for potential
390  HELMHOLTZ_BULK_ELEMENT* ext_el_pt=
391  dynamic_cast<HELMHOLTZ_BULK_ELEMENT*>(external_element_pt(0,ipt));
392 
393  // number of nodes in ext element
394  unsigned nnode_ext=ext_el_pt->nnode();
395 
396  // get the dim of the external nodes
397  const unsigned ext_dim=ext_el_pt->dim();
398 
399  // Set up memory for the external shape function
400  Shape psi_ext(nnode_ext);
401  DShape dpsi_ext_dx(nnode_ext,ext_dim);
402 
403  //Call the derivatives of the shape functions
404  //in the external element -- must do this via s because this point
405  //is not an integration point the bulk element!
406  (void) ext_el_pt->dshape_eulerian(s_bulk,psi_ext,dpsi_ext_dx);
407 
408  //Set up memory for the outer unit normal
409  Vector<double> unit_normal(ext_dim);
410 
411  //get the outer_unit_ext vector
412  this->outer_unit_normal(s,unit_normal);
413 
414  //Get the integral weight
415  double w = integral_pt()->weight(ipt);
416 
417  // Get jacobian of mapping
418  double J=J_eulerian(s);
419 
420  //Premultiply the weights and the Jacobian
421  double W = w*J;
422 
423  // Global gradient
424  Vector<std::complex <double> >
425  interpolated_dphidx(ext_dim,std::complex<double>(0.0,0.0));
426  for(unsigned l=0;l<nnode_ext;l++)
427  {
428  //Get the nodal value of the helmholtz unknown
429  const std::complex<double> phi_value(
430  ext_el_pt->nodal_value(l,ext_el_pt->u_index_helmholtz().real()),
431  ext_el_pt->nodal_value(l,ext_el_pt->u_index_helmholtz().imag()));
432 
433  //Loop over directions
434  for(unsigned i=0;i<ext_dim;i++)
435  {
436  interpolated_dphidx[i] += phi_value*dpsi_ext_dx(l,i);
437  }
438  }
439 
440  //define dphi_dn
441  std::complex<double> dphi_dn(0.0,0.0);
442  for(unsigned i=0;i<ext_dim;i++)
443  {
444  dphi_dn += interpolated_dphidx[i]*unit_normal[i];
445  }
446 
447 
448 #ifdef PARANOID
449  double max_permitted_discrepancy=1.0e-10;
450  double diff=sqrt(pow(ext_el_pt->interpolated_x(s_bulk,0)-
451  interpolated_x(s,0),2)+
452  pow(ext_el_pt->interpolated_x(s_bulk,1)-
453  interpolated_x(s,1),2));
454  if (diff>max_permitted_discrepancy)
455  {
456  std::ostringstream error_stream;
457  error_stream
458  << "Position computed by external el and face element differ by "
459  << diff << " which is more than the acceptable tolerance "
460  << max_permitted_discrepancy << std::endl;
461  throw OomphLibError(
462  error_stream.str(),
463  OOMPH_CURRENT_FUNCTION,
464  OOMPH_EXCEPTION_LOCATION);
465  }
466 #endif
467 
468  // Output?
469  if (outfile.is_open())
470  {
471  Vector<double> x(ext_dim);
472  interpolated_x(s,x);
473  outfile << x[0] << " "
474  << x[1] << " "
475  << dphi_dn.real() << " "
476  << dphi_dn.imag() << " "
477  << "\n";
478  }
479 
480  // ...add to integral
481  flux+=dphi_dn*W;
482  }
483 
484  return flux;
485  }

507 //=====================================================================
508 /// Return the residuals
509 //=====================================================================
510  template <class ELASTICITY_BULK_ELEMENT, class HELMHOLTZ_BULK_ELEMENT>
511  void TimeHarmonicLinElastLoadedByPMLHelmholtzPressureBCElement<
512  ELASTICITY_BULK_ELEMENT, HELMHOLTZ_BULK_ELEMENT>::
513  fill_in_contribution_to_residuals_helmholtz_traction(
514  Vector<double> &residuals)
515  {
516 
517  //Find out how many nodes there are
518  unsigned n_node = nnode();
519 
520 #ifdef PARANOID
521  //Find out how many positional dofs there are
522  unsigned n_position_type = this->nnodal_position_type();
523  if(n_position_type != 1)
524  {
525  throw OomphLibError(
526  "LinearElasticity is not yet implemented for more than one position type.",
527  OOMPH_CURRENT_FUNCTION,
528  OOMPH_EXCEPTION_LOCATION);
529  }
530 #endif
531 
532  //Find out the dimension of the node
533  const unsigned n_dim = this->nodal_dimension();
534 
535  //Cache the nodal indices at which the displacement components are stored
536  std::complex<unsigned> u_nodal_index[n_dim];
537  for(unsigned i=0;i<n_dim;i++)
538  {
539  u_nodal_index[i] =
540  this->U_index_time_harmonic_linear_elasticity_helmholtz_traction[i];
541  }
542 
543  //Integer to hold the local equation number
544  int local_eqn=0;
545 
546  //Set up memory for the shape functions
547  Shape psi(n_node);
548  DShape dpsids(n_node,n_dim-1);
549 
550  // Get FSI parameter
551  const double q_value=q();
552 
553  // Storage for traction
554  Vector<std::complex<double> > traction(2);
555 
556  //Set the value of n_intpt
557  unsigned n_intpt = integral_pt()->nweight();
558 
559  //Loop over the integration points
560  for(unsigned ipt=0;ipt<n_intpt;ipt++)
561  {
562  //Get the integral weight
563  double w = integral_pt()->weight(ipt);
564 
565  //Only need to call the local derivatives
566  dshape_local_at_knot(ipt,psi,dpsids);
567 
568  //Calculate the coordinates
569  Vector<double> interpolated_x(n_dim,0.0);
570 
571  //Also calculate the surface tangent vectors
572  DenseMatrix<double> interpolated_A(n_dim-1,n_dim,0.0);
573 
574  //Calculate displacements and derivatives
575  for(unsigned l=0;l<n_node;l++)
576  {
577  //Loop over directions
578  for(unsigned i=0;i<n_dim;i++)
579  {
580  //Calculate the Eulerian coords
581  const double x_local = nodal_position(l,i);
582  interpolated_x[i] += x_local*psi(l);
583 
584  //Loop over LOCAL derivative directions, to calculate the tangent(s)
585  for(unsigned j=0;j<n_dim-1;j++)
586  {
587  interpolated_A(j,i) += x_local*dpsids(l,j);
588  }
589  }
590  }
591 
592  //Now find the local metric tensor from the tangent Vectors
593  DenseMatrix<double> A(n_dim-1);
594  for(unsigned i=0;i<n_dim-1;i++)
595  {
596  for(unsigned j=0;j<n_dim-1;j++)
597  {
598  //Initialise surface metric tensor to zero
599  A(i,j) = 0.0;
600 
601  //Take the dot product
602  for(unsigned k=0;k<n_dim;k++)
603  {
604  A(i,j) += interpolated_A(i,k)*interpolated_A(j,k);
605  }
606  }
607  }
608 
609  //Get the outer unit normal
610  Vector<double> interpolated_normal(n_dim);
611  outer_unit_normal(ipt,interpolated_normal);
612 
613  //Find the determinant of the metric tensor
614  double Adet =0.0;
615  switch(n_dim)
616  {
617  case 2:
618  Adet = A(0,0);
619  break;
620  case 3:
621  Adet = A(0,0)*A(1,1) - A(0,1)*A(1,0);
622  break;
623  default:
624  throw
625  OomphLibError(
626  "Wrong dimension in TimeHarmonicLinElastLoadedByPressureElement",
627  OOMPH_CURRENT_FUNCTION,
628  OOMPH_EXCEPTION_LOCATION);
629  }
630 
631  //Premultiply the weights and the square-root of the determinant of
632  //the metric tensor
633  double W = w*sqrt(Adet);
634 
635  // Get bulk element for potential
636  HELMHOLTZ_BULK_ELEMENT* ext_el_pt=
637  dynamic_cast<HELMHOLTZ_BULK_ELEMENT*>(external_element_pt(0,ipt));
638  Vector<double> s_ext(external_element_local_coord(0,ipt));
639 
640  // Traction: Pressure is proportional to POSITIVE potential
641  std::complex<double> u_helmholtz=ext_el_pt->

1102  U_index_helmholtz_from_displacement;
