42  /// \short Preconditioner for FSI problems with pseudo-elastic fluid node
43  /// updates.
44  /// Note:
45  /// NavierStokesSchurComplementPreconditioner is applied to the Navier Stokes
46  /// subsidiary system.
47  /// Default solid preconditioner is SuperLUPreconditioner.
48  /// \b Enumeration of Elastic DOF types in the Pseudo-Elastic Elements
49  /// The method get_dof_types_for_unknowns() must be implemented such that
50  /// DOFs subject be Lagrange multiplier and DOFs NOT subject to Lagrange
51  /// multiplier have different labels. For example in a 3D problem there are
52  /// 6 DOF types and the following labelling must be implemented:
53  /// 0 - x displacement (without lagr mult traction)
54  /// 1 - y displacement (without lagr mult traction)
55  /// 2 - z displacement (without lagr mult traction)
56  /// 3 - x displacement (with lagr mult traction)
57  /// 4 - y displacement (with lagr mult traction)
58  /// 5 - z displacement (with lagr mult traction)
59  //============================================================================
60  class PseudoElasticFSIPreconditioner :
61  public BlockPreconditioner<CRDoubleMatrix>
62  {
63 
64  public:
65 
66  /// \short constructor - just set defaults. Specify the spatial
67  /// dimension of the fluid and a (non-const) problem pointer needed for
68  /// the underlying NavierStokesSchurComplementPreconditioner.
69  PseudoElasticFSIPreconditioner(const unsigned& dim, Problem* problem_pt)
70  : Dim(dim)
71  {
72  Use_navier_stokes_schur_complement_preconditioner = true;
73 
74  // set the number of meshes
75  this->set_nmesh(3);
76 
77  // null pointers
78  Fluid_and_pseudo_elastic_mesh_pt = 0;
79  Solid_mesh_pt = 0;
80  Lagrange_multiplier_mesh_pt = 0;
81 
82  // create the pseudo solid preconditioner
83  Pseudo_elastic_preconditioner_pt =
84  new PseudoElasticPreconditioner();
85 
86  // using Schur complement preconditioner for NS
87  Navier_stokes_preconditioner_pt = new SuperLUPreconditioner;
88  Navier_stokes_schur_complement_preconditioner_pt
89  = new NavierStokesSchurComplementPreconditioner(problem_pt);
90 
91  // set defaults
92  Using_default_solid_preconditioner = true;
93 
94  // default super lu
95  Solid_preconditioner_pt = new SuperLUPreconditioner;
96 
97  // create the matrix vector product operatrs
98  Solid_fluid_matvec_pt = new MatrixVectorProduct;
99  Solid_pseudo_elastic_matvec_pt = new MatrixVectorProduct;
100  Fluid_pseudo_elastic_matvec_pt = new MatrixVectorProduct;
101  Lagrange_solid_matvec_pt = new MatrixVectorProduct;
102  }
103

105  virtual ~PseudoElasticFSIPreconditioner()
106  {
107  // clean the memory
108  this->clean_up_memory();

110  // delete the pseudo solid preconditioner

112 
113  // delete the navier stokes preconditioner
114  delete Navier_stokes_preconditioner_pt;
115  delete Navier_stokes_schur_complement_preconditioner_pt;
116 
117  // delete the solid preconditioner if default
118  if (Using_default_solid_preconditioner)
119  {
120  delete Solid_preconditioner_pt;
121  }
122 
123  // delete the matrix vector product operators
124  delete Fluid_pseudo_elastic_matvec_pt;
125  delete Solid_fluid_matvec_pt;
126  delete Solid_pseudo_elastic_matvec_pt;
127  delete Lagrange_solid_matvec_pt;
128  }
