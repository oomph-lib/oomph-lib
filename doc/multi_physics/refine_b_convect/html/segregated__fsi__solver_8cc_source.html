36 /// Extrapolate solid data and update fluid mesh. Extrapolation is based
37 /// on previous history values and is therefore only called in
38 /// time-dependent runs.
39 //============================================================================
40  void SegregatableFSIProblem::extrapolate_solid_data()
41  {
42  // Number of solid Data items:
43  unsigned n_data=Solid_data_pt.size();
44 
45  // Loop over all solid Data items:
46  for (unsigned i=0;i<n_data;i++)
47  {
48  // Number of values stored at this Data item:
49  unsigned n_value=Solid_data_pt[i]->nvalue();
50 
51  // Number of values stored at this Data item:
52  unsigned n_prev=Solid_data_pt[i]->time_stepper_pt()->nprev_values();
53 
54  // Can't use this extrapolation if we don't have at least two
55  // history values; may be able to do better for higher order
56  // timesteppers, though.
57  if (n_prev>=2)
58  {
59  // Extrapolate all values
60  for (unsigned k=0;k<n_value;k++)
61  {
62  // Linear extrapolation based on previous two values,
63  // assuming constant timestep.
64  double new_value=2.0*Solid_data_pt[i]->value(1,k)-
65  Solid_data_pt[i]->value(2,k);
66  *(Solid_data_pt[i]->value_pt(0,k))=new_value;
67  }
68  }
69  }
70 
71  // Update fluid node position in response to changes in wall shape
72  // (If fluid mesh was specified via non-NULL Fluid_mesh_pt
73  // this node update will only act on the fluid nodes).
74  if (Fluid_mesh_pt!=0)
75  {
76  Fluid_mesh_pt->node_update();
77  }
78  else
79  {
80  mesh_pt()->node_update();
81  }
82 
83  }

321 //============================================================================
322 /// Get rms of change in the solid dofs; the max. change of the
323 /// solid dofs and the rms norm of the solid dofs themselves.
324 /// Change is computed relative to the reference values stored when
325 /// store_solid_dofs() was last called.
326 //============================================================================
327  void SegregatableFSIProblem::get_solid_change(
328  double& rms_change, double& max_change, double& rms_norm)
329  {
330 
331  // Initialise
332  rms_change=0.0;
333  max_change=0.0;
334  rms_norm=0.0;
335 
336  // Counter for the number of values:
337  unsigned value_count=0;
338 
339  // Number of solid Data items:
340  unsigned n_data=Solid_data_pt.size();
341 
342  // Loop over all solid Data items:
343  for (unsigned i=0;i<n_data;i++)
344  {
345  // Number of values stored at this Data item:
346  unsigned n_value=Solid_data_pt[i]->nvalue();
347 
348  // Loop over all values
349  for (unsigned k=0;k<n_value;k++)
350  {
351 
352  // Change
353  double change=Previous_solid_value[value_count]-
354  Solid_data_pt[i]->value(k);
355 
356  // Max change?
357  if (std::fabs(change)>max_change) max_change=std::fabs(change);
358 
359  //Add square of change relative to previous value
360  rms_change+=pow(change,2);
361 
362  //Add square of previous value
363  rms_norm+=pow(Previous_solid_value[value_count],2);
364 
365  // Increment counter
366  value_count++;
367  }
368  }
369 
370  // Turn into rms:
371  rms_change=sqrt(rms_change/double(value_count));
372  rms_norm=sqrt(rms_norm/double(value_count));
373  }

378 //============================================================================
379 /// Under-relax solid, either by classical relaxation or by Irons & Tuck.
380 //============================================================================
381  void SegregatableFSIProblem::under_relax_solid()
382  {
383 
384  // Irons and Tuck extrapolation/relaxation; an extension of Aitken's method
385  //-------------------------------------------------------------------------
386  if (Use_irons_and_tuck_extrapolation)
387  {
388  double top=0.0;
389  double den=0.0;
390  double crit=0.0;
391 
392  // Counter for the number of values:
393  unsigned value_count=0;
394 
395  // Number of solid Data items:
396  unsigned n_data=Solid_data_pt.size();
397 
398  // Loop over all solid Data items:
399  for (unsigned i=0;i<n_data;i++)
400  {
401  // Number of values stored at this Data item:
402  unsigned n_value=Solid_data_pt[i]->nvalue();
403 
404  // Loop over all values
405  for (unsigned k=0;k<n_value;k++)
406  {
407  // Prediction from solid solver
408  double new_solid_value=Solid_data_pt[i]->value(k);
409 
410  // Previus value
411  double old_solid_value=Previous_solid_value[value_count];
412 
413  // Change
414  double change=old_solid_value-new_solid_value;
415 
416  // Change of change
417  double del2=Del_irons_and_tuck[value_count]-change;
418 
419  // Update change
420  Del_irons_and_tuck[value_count]=change;
421 
422  // Update top
423  top+=del2*change;
424 
425  // Update denominator
426  den+=del2*del2;
427 
428  // Update convergence criterion
429  crit+=std::fabs(change);
430 
431  // Increment counter
432  value_count++;
433  }
434  }
435 
436  // Update relaxation factor. The if buffers the case in which
437  // we haven't realised that we've converged (so that den=0).
438  // This can happen, e.g. if the convergence assessment is based on the
439  // global residual or during validation. In that case we
440  // obviously don't want any changes to the iterates.
441  if (den!=0.0)
442  {
443  double new_r=R_irons_and_tuck+(R_irons_and_tuck-1.0)*top/den;
444  R_irons_and_tuck=new_r;
445  }
446  else
447  {
448  R_irons_and_tuck=0.0;
449  }
450 
451  // Loop over all solid Data items for update
452  value_count=0;
453  for (unsigned i=0;i<n_data;i++)
454  {
455  // Number of values stored at this Data item:
456  unsigned n_value=Solid_data_pt[i]->nvalue();
457 
458  // Loop over all values
459  for (unsigned k=0;k<n_value;k++)
460  {
461  // Compute relaxed/extrapolated value
462  double new_value=Solid_data_pt[i]->value(k)+
463  R_irons_and_tuck*Del_irons_and_tuck[value_count];
464 
465  // Assign
466  Solid_data_pt[i]->set_value(k,new_value);
467 
468  // Increment counter
469  value_count++;
470  }
471  }
472  return;
473  }
474 
475  // Standard relaxation
476  //--------------------
477  else
478  {
479 
480 
481  // No relaxation: Can return immediately
482  if (Omega_relax==1.0) return;
483 
484  // Counter for the number of values:
485  unsigned value_count=0;
486 
487  // Number of solid Data items:
488  unsigned n_data=Solid_data_pt.size();
489 
490  // Loop over all solid Data items:
491  for (unsigned i=0;i<n_data;i++)
492  {
493  // Number of values stored at this Data item:
494  unsigned n_value=Solid_data_pt[i]->nvalue();
495 
496  // Loop over all values
497  for (unsigned k=0;k<n_value;k++)
498  {
499 
500  // Prediction from solid solver
501  double new_solid_value=Solid_data_pt[i]->value(k);
502 
503  // Previus value
504  double old_solid_value=Previous_solid_value[value_count];
505 
506  // Relax
507  Solid_data_pt[i]->set_value(k,Omega_relax*new_solid_value+
508  (1.0-Omega_relax)*old_solid_value);
509 
510  // Increment counter
511  value_count++;
512  }
513  }
514  }
515  }

518 /// Pointwise Aitken extrapolation for solid variables
519 //============================================================================
520  void SegregatableFSIProblem::pointwise_aitken_extrapolate()
521  {
522 
523  // Counter for the number of values:
524  unsigned value_count=0;
525 
526  // Number of solid Data items:
527  unsigned n_data=Solid_data_pt.size();
528 
529  // Loop over all solid Data items:
530  for (unsigned i=0;i<n_data;i++)
531  {
532  // Number of values stored at this Data item:
533  unsigned n_value=Solid_data_pt[i]->nvalue();
534 
535  // Loop over all values
536  for (unsigned k=0;k<n_value;k++)
537  {
538  // Shorthand
539  double v0=Pointwise_aitken_solid_value[value_count][0];
540  double v1=Pointwise_aitken_solid_value[value_count][1];
541  double v2=Pointwise_aitken_solid_value[value_count][2];
542 
543  double new_value=v2;
544 
545  double max_diff=std::max(std::fabs(v1-v0),std::fabs(v2-v1));
546  if (max_diff>1.0e-10)
547  {
548  new_value=v2-std::pow((v2-v1),int(2))/(v2-2.0*v1+v0);
549  }
550  Solid_data_pt[i]->set_value(k,new_value);
551 
552  // Increment counter
553  value_count++;
554  }
555  }
556 
557  // Reset the counter for the Aitken convergence check
558  // (setting counter to -1 forces three new genuine
559  // iterates to be computed).
560  Pointwise_aitken_counter=-1;
561 
562  }

565 //============================================================================
566 /// Segregated fixed point solver with optional pointwise Aitken acceleration
567 /// on selected solid dofs. Returns PicardConvergenceData object
568 /// that contains the vital stats of the iteration.
569 //============================================================================
570  PicardConvergenceData SegregatableFSIProblem::segregated_solve()
571  {
572  // Initialise timer for essential bits of code
573  reset_timer();
574  // Start timer for overall solve
575  clock_t t_total_start = clock();
576 
577  // If convergence is based on max. global residual we may as well
578  // document it...
579  bool get_max_global_residual=Doc_max_global_residual;
580  if(Convergence_criterion==Assess_convergence_based_on_max_global_residual)
581  {
582  get_max_global_residual=true;
583  }
584 
585  // Create object to doc convergence stats
586  PicardConvergenceData conv_data;
587 
588  // Initialise total time for computation of global residuals
589  double cpu_for_global_residual=0.0;
590 
591  //Update anything that needs updating
592  actions_before_segregated_solve();
593 
594  // Set flags to values that are appropriate if Picard iteration
595  // does not converge with Max_picard iterations
596  bool converged=false;
597  unsigned iter_taken=Max_picard;
598 
599  // This one will be overwritten during the convergence checks
600  double tol_achieved=0.0;
601 
602  // Store the current values of the solid dofs as reference values
603  // and for the pointwise Aitken extrapolation
604  store_solid_dofs();
605 
606  // Loop over Picard iterations
607  //----------------------------
608  for (unsigned iter=1;iter<=Max_picard;iter++)
609  {
610 
611  //Calculate the initial residuals
612  if(iter==1)
613  {
614  //Problem is always non-linear?
615  //Perform any actions before the convergence check
616  actions_before_segregated_convergence_check();
617 
618  double max_res = 0.0;
619  if (get_max_global_residual)
620  {
621  clock_t t_start = clock();
622  restore_solid_dofs();
623  restore_fluid_dofs();
624  rebuild_monolithic_mesh();
625  assign_eqn_numbers();
626  //This is now a full problem
627  Solve_type = Full_solve;
628  DoubleVector residual;
629  get_residuals(residual);
630  //Get maximum residuals using our own abscmp function
631  max_res = residual.max();
632  clock_t t_end = clock();
633  cpu_for_global_residual+=double(t_end-t_start)/CLOCKS_PER_SEC;
634  }
635 
636  oomph_info << "==================================================\n";
637  oomph_info << "Initial iteration : "
638  << 0 << std::endl;
639  oomph_info << "RMS change : "
640  << 0 << std::endl;
641  oomph_info << "Max. change : "
642  << 0 << std::endl;
643  oomph_info << "RMS norm : "
644  << 0 << std::endl;
645  if (get_max_global_residual)
646  {
647  oomph_info << "Max. global residual : "
648  << max_res << std::endl;
649  }
650  oomph_info << "==================================================\n\n";
651 
652  // Check for convergence, but this only makes sense
653  // for the maximum (rather than relative case)
654  if((Convergence_criterion==
655  Assess_convergence_based_on_max_global_residual) &&
656  (max_res < Convergence_tolerance))
657  {
658  oomph_info
659  << "\n\n\n////////////////////////////////////////////////////////"
660  << "\nPicard iteration converged after "
661  << 0 << " steps!" << std::endl
662  << "Convergence was based on max. residual of coupled eqns \n"
663  << "being less than " << Convergence_tolerance << "." << std::endl
664  << "////////////////////////////////////////////////////////\n\n\n"
665  << std::endl;
666 
667  //Converged, so bail out
668  // Number of iterations taken
669  iter_taken=0;
670 
671  // We have converged (overwrites default of false)
672  conv_data.set_solver_converged();
673 
674  // Break loop using a GO TO! This is the first (and hopefully
675  // the last) one in the whole of oomph-lib. Here it's
676  // it's actually the cleanest way to exit from these
677  // nested control structures
678  goto jump_out_of_picard;
679  }
680  }
681 
682  // Stage 1: Freeze wall and solve single-physics fluid problem
683  //------------------------------------------------------------
684  // Pin the solid dofs, restore the pinned status of the fluid dofs
685  // and re-assign the equation numbers
686  restore_fluid_dofs();
687  pin_solid_dofs();
688  use_only_fluid_elements();
689  assign_eqn_numbers();
690 
691  // Solve the fluid problem for the current wall geometry
692  oomph_info <<"\n\nDOING FLUID SOLVE\n\n";
693  //This is a fluid solve at the moment done by a newton solve
694  Solve_type=Fluid_solve;
695  newton_solve();
696 
697  // Stage 2: Freeze fluid variables and update wall shape
698  //------------------------------------------------------
699 
700  // Now restore the pinned status of the wall and pin the fluid
701  // dofs in anticipation of the wall update:
702  restore_solid_dofs();
703  pin_fluid_dofs();
704  use_only_solid_elements();
705  assign_eqn_numbers();
706 
707  // Solve the solid problem for the wall solution
708  oomph_info <<"\n\nDOING SOLID SOLVE\n\n";
709  //This is a solid solve, at the moment done by a newton method
710  Solve_type = Solid_solve;
711  newton_solve();
712 
713  // Under-relax, either by classical relaxation or by Irons & Tuck
714  // Note that we are under-relaxing before the convergence check
715  // because under-relaxtion may be required to acheieve any
716  // kind of convergence. If the convergence check is on the RELATIVE
717  // change of the residuals, however, then a small under-relaxation
718  // parameter will cause a false convergence. You have been warned!
719  under_relax_solid();
720 
721  // Stage 3: Convergence check (possibly again after pointwise Aitken
722  //------------------------------------------------------------------
723  // extrapolation)
724  //---------------
725  //Assume that we are not doing Aitken acceleration
726  Recheck_convergence_after_pointwise_aitken=false;
727  do
728  {
729  //Perform any actions before the convergence check
730  actions_before_segregated_convergence_check();
731
