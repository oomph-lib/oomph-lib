38 /// A class for elements that allow the imposition of an applied traction
39 /// in the equations of time-harmonic linear elasticity from a Helmholtz
40 /// potential (interpreted as a displacement potential for the fluid in a
41 /// quasi-steady, linearised FSI problem.)
42 /// The geometrical information can be read from the FaceGeometry<ELEMENT>
43 /// class and and thus, we can be generic enough without the need to have
44 /// a separate equations class.
45 //======================================================================

173  /// \short Output function: Plot traction etc at Gauss points
174  /// nplot is ignored.
175  void output(std::ostream &outfile, const unsigned &n_plot)
176  {
177  // Dimension
178  unsigned n_dim = this->nodal_dimension();
179 
180  // Storage for traction
181  Vector<std::complex<double> > traction(n_dim);
182 
183  // Get FSI parameter
184  const double q_value=q();
185 
186  outfile << "ZONE\n";
187 
188  //Set the value of n_intpt
189  unsigned n_intpt = integral_pt()->nweight();
190 
191  //Loop over the integration points
192  for(unsigned ipt=0;ipt<n_intpt;ipt++)
193  {
194  Vector<double> s_int(n_dim-1);
195  for (unsigned i=0;i<n_dim-1;i++)
196  {
197  s_int[i]=integral_pt()->knot(ipt,i);
198  }
199 
200  //Get the outer unit normal
201  Vector<double> interpolated_normal(n_dim);
202  outer_unit_normal(ipt,interpolated_normal);
203 
204  // Boundary coordinate
205  Vector<double> zeta(1);
206  interpolated_zeta(s_int,zeta);
207 
208  // Get bulk element for potential
209  HELMHOLTZ_BULK_ELEMENT* ext_el_pt=
210  dynamic_cast<HELMHOLTZ_BULK_ELEMENT*>(external_element_pt(0,ipt));
211  Vector<double> s_ext(external_element_local_coord(0,ipt));
212  std::complex<double> u_helmholtz=
213  ext_el_pt->interpolated_u_helmholtz(s_ext);
214 
215  // Traction: Pressure is proportional to POSITIVE potential
216  ext_el_pt->interpolated_u_helmholtz(s_ext);
217  traction[0]=-q_value*interpolated_normal[0]*u_helmholtz;
218  traction[1]=-q_value*interpolated_normal[1]*u_helmholtz;
219 
220  outfile << ext_el_pt->interpolated_x(s_ext,0) << " "
221  << ext_el_pt->interpolated_x(s_ext,1) << " "
222  << traction[0].real() << " "
223  << traction[1].real() << " "
224  << traction[0].imag() << " "
225  << traction[1].imag() << " "
226  << interpolated_normal[0] << " "
227  << interpolated_normal[1] << " "
228  << u_helmholtz.real() << " "
229  << u_helmholtz.imag() << " "
230  << interpolated_x(s_int,0) << " "
231  << interpolated_x(s_int,1) << " "
232 
233  << sqrt(pow(ext_el_pt->interpolated_x(s_ext,0)-
234  interpolated_x(s_int,0),2)+
235  pow(ext_el_pt->interpolated_x(s_ext,1)-
236  interpolated_x(s_int,1),2)) << " "
237  << zeta[0] << std::endl;
238  }
239  }

266 //=====================================================================
267 /// Return the residuals
268 //=====================================================================
269  template <class ELASTICITY_BULK_ELEMENT, class HELMHOLTZ_BULK_ELEMENT>
270  void TimeHarmonicLinElastLoadedByHelmholtzPressureBCElement<
271  ELASTICITY_BULK_ELEMENT, HELMHOLTZ_BULK_ELEMENT>::
272  fill_in_contribution_to_residuals_helmholtz_traction(
273  Vector<double> &residuals)
274  {
275 
276  //Find out how many nodes there are
277  unsigned n_node = nnode();
278 
279 #ifdef PARANOID
280  //Find out how many positional dofs there are
281  unsigned n_position_type = this->nnodal_position_type();
282  if(n_position_type != 1)
283  {
284  throw OomphLibError(
285  "LinearElasticity is not yet implemented for more than one position type.",
286  OOMPH_CURRENT_FUNCTION,
287  OOMPH_EXCEPTION_LOCATION);
288  }
289 #endif
290 
291  //Find out the dimension of the node
292  const unsigned n_dim = this->nodal_dimension();
293 
294  //Cache the nodal indices at which the displacement components are stored
295  std::vector<std::complex<unsigned> > u_nodal_index(n_dim);
296  for(unsigned i=0;i<n_dim;i++)
297  {
298  u_nodal_index[i] =
299  this->U_index_time_harmonic_linear_elasticity_helmholtz_traction[i];
300  }
301 
302  //Integer to hold the local equation number
303  int local_eqn=0;
304 
305  //Set up memory for the shape functions
306  Shape psi(n_node);
307  DShape dpsids(n_node,n_dim-1);
308 
309  // Get FSI parameter
310  const double q_value=q();
311 
312  // Storage for traction
313  Vector<std::complex<double> > traction(2);
314 
315  //Set the value of n_intpt
316  unsigned n_intpt = integral_pt()->nweight();
317 
318  //Loop over the integration points
319  for(unsigned ipt=0;ipt<n_intpt;ipt++)
320  {
321  //Get the integral weight
322  double w = integral_pt()->weight(ipt);
323 
324  //Only need to call the local derivatives
325  dshape_local_at_knot(ipt,psi,dpsids);
326 
327  //Calculate the coordinates
328  Vector<double> interpolated_x(n_dim,0.0);
329 
330  //Also calculate the surface tangent vectors
331  DenseMatrix<double> interpolated_A(n_dim-1,n_dim,0.0);
332 
333  //Calculate displacements and derivatives
334  for(unsigned l=0;l<n_node;l++)
335  {
336  //Loop over directions
337  for(unsigned i=0;i<n_dim;i++)
338  {
339  //Calculate the Eulerian coords
340  const double x_local = nodal_position(l,i);
341  interpolated_x[i] += x_local*psi(l);
342 
343  //Loop over LOCAL derivative directions, to calculate the tangent(s)
344  for(unsigned j=0;j<n_dim-1;j++)
345  {
346  interpolated_A(j,i) += x_local*dpsids(l,j);
347  }
348  }
349  }
350 
351  //Now find the local metric tensor from the tangent Vectors
352  DenseMatrix<double> A(n_dim-1);
353  for(unsigned i=0;i<n_dim-1;i++)
354  {
355  for(unsigned j=0;j<n_dim-1;j++)
356  {
357  //Initialise surface metric tensor to zero
358  A(i,j) = 0.0;
359 
360  //Take the dot product
361  for(unsigned k=0;k<n_dim;k++)
362  {
363  A(i,j) += interpolated_A(i,k)*interpolated_A(j,k);
364  }
365  }
366  }
367 
368  //Get the outer unit normal
369  Vector<double> interpolated_normal(n_dim);
370  outer_unit_normal(ipt,interpolated_normal);
371 
372  //Find the determinant of the metric tensor
373  double Adet =0.0;
374  switch(n_dim)
375  {
376  case 2:
377  Adet = A(0,0);
378  break;
379  case 3:
380  Adet = A(0,0)*A(1,1) - A(0,1)*A(1,0);
381  break;
382  default:
383  throw
384  OomphLibError(
385  "Wrong dimension in TimeHarmonicLinElastLoadedByPressureElement",
386  "TimeHarmonicLinElastLoadedByPressureElement::fill_in_contribution_to_residuals()",
387  OOMPH_EXCEPTION_LOCATION);
388  }
389 
390  //Premultiply the weights and the square-root of the determinant of
391  //the metric tensor
392  double W = w*sqrt(Adet);
393 
394  // Get bulk element for potential
395  HELMHOLTZ_BULK_ELEMENT* ext_el_pt=
396  dynamic_cast<HELMHOLTZ_BULK_ELEMENT*>(external_element_pt(0,ipt));
397  Vector<double> s_ext(external_element_local_coord(0,ipt));
398 
399  // Traction: Pressure is proportional to POSITIVE potential
400  std::complex<double> u_helmholtz=ext_el_pt->interpolated_u_helmholtz(s_ext);
401  traction[0]=-q_value*interpolated_normal[0]*u_helmholtz;
402  traction[1]=-q_value*interpolated_normal[1]*u_helmholtz;
403 
404  //Loop over the test functions, nodes of the element
405  for(unsigned l=0;l<n_node;l++)
406  {
407  //Loop over the displacement components
408  for(unsigned i=0;i<n_dim;i++)
409  {
410 
411  local_eqn = this->nodal_local_eqn(l,u_nodal_index[i].real());
412  /*IF it's not a boundary condition*/
413  if(local_eqn >= 0)
414  {
415  //Add the loading terms to the residuals
416  residuals[local_eqn] -= traction[i].real()*psi(l)*W;
417  }
418 
419  local_eqn = this->nodal_local_eqn(l,u_nodal_index[i].imag());
420  /*IF it's not a boundary condition*/
421  if(local_eqn >= 0)
422  {
423  //Add the loading terms to the residuals
424  residuals[local_eqn] -= traction[i].imag()*psi(l)*W;
425  }
426 
427  } //End of if not boundary condition
428  } //End of loop over shape functions
429  } //End of loop over integration points
430 
431  }

826 /// \short Helper function to compute the element's residual vector and
827 /// the Jacobian matrix.
828 //===========================================================================
829  template <class HELMHOLTZ_BULK_ELEMENT, class ELASTICITY_BULK_ELEMENT>
830  void HelmholtzFluxFromNormalDisplacementBCElement
831  <HELMHOLTZ_BULK_ELEMENT,ELASTICITY_BULK_ELEMENT>::
832  fill_in_generic_residual_contribution_helmholtz_flux_from_displacement(
833  Vector<double> &residuals, DenseMatrix<double> &jacobian,
834  const unsigned& flag)
835  {
836  //Find out how many nodes there are
837  const unsigned n_node = nnode();
838 
839  //Set up memory for the shape and test functions
840  Shape psif(n_node), testf(n_node);
841 
842  //Set the value of Nintpt
843  const unsigned n_intpt = integral_pt()->nweight();
844 
845  //Set the Vector to hold local coordinates
846  Vector<double> s(Dim-1);
847 
848  //Integers to hold the local equation and unknown numbers
849  int local_eqn=0;
850 
851  // Locally cache the index at which the variable is stored
852  const std::complex<unsigned> u_index_helmholtz =
853  U_index_helmholtz_from_displacement;
854 
855  //Loop over the integration points
856  //--------------------------------
857  for(unsigned ipt=0;ipt<n_intpt;ipt++)
858  {
859 
860  //Assign values of s
861  for(unsigned i=0;i<(Dim-1);i++) {s[i] = integral_pt()->knot(ipt,i);}
862 
863  //Get the integral weight
864  double w = integral_pt()->weight(ipt);
865 
866  //Find the shape and test functions and return the Jacobian
867  //of the mapping
868  double J = shape_and_test(s,psif,testf);
869 
870  //Premultiply the weights and the Jacobian
871  double W = w*J;
872 
873  //Need to find position to feed into flux function, initialise to zero
874  Vector<double> interpolated_x(Dim,0.0);
875 
876  //Calculate velocities and derivatives
877  for(unsigned l=0;l<n_node;l++)
878  {
879  //Loop over velocity components
880  for(unsigned i=0;i<Dim;i++)
881  {
882  interpolated_x[i] += nodal_position(l,i)*psif[l];
883  }
884  }
885 
886  //Get the outer unit normal
887  Vector<double> interpolated_normal(2);
888  outer_unit_normal(ipt,interpolated_normal);
889 
890  // Get displacements
891  ELASTICITY_BULK_ELEMENT* ext_el_pt=dynamic_cast<ELASTICITY_BULK_ELEMENT*>(
892  external_element_pt(0,ipt));
893  Vector<double> s_ext(external_element_local_coord(0,ipt));
894  Vector<std::complex<double> > displ(2);
895  ext_el_pt->interpolated_u_time_harmonic_linear_elasticity(s_ext,displ);
896 
897  // Convert into flux BC: This takes the dot product of the
898  // actual displacement with the flux element's own outer
899  // unit normal so the plus sign is OK.
900  std::complex<double> flux=(displ[0]*interpolated_normal[0]+
901  displ[1]*interpolated_normal[1]);
902 
903  //Now add to the appropriate equations
904 
905  //Loop over the test functions
906  for(unsigned l=0;l<n_node;l++)
907  {
908  local_eqn = nodal_local_eqn(l,u_index_helmholtz.real());
909  /*IF it's not a boundary condition*/
910  if(local_eqn >= 0)
911  {
912  //Add the prescribed flux terms
913  residuals[local_eqn] -= flux.real()*testf[l]*W;
914 
915  // Imposed traction doesn't depend upon the solution,
916  // --> the Jacobian is always zero, so no Jacobian
917  // terms are required
918  }
919 
920  local_eqn = nodal_local_eqn(l,u_index_helmholtz.imag());
921  /*IF it's not a boundary condition*/
922  if(local_eqn >= 0)
923  {
924  //Add the prescribed flux terms
925  residuals[local_eqn] -= flux.imag()*testf[l]*W;
926 
927  // Imposed traction doesn't depend upon the solution,
928  // --> the Jacobian is always zero, so no Jacobian
929  // terms are required
930  }
931  }
932  }
933  }
